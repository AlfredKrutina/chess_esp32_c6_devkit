219void button_scan_all(void) {
220 uint32_t current_time = esp_timer_get_time() / 1000;
221
222 // In simulation mode, simulate button presses
223 if (simulation_mode) {
224 // Simulate button presses every 5 seconds
225 if (current_time - last_simulation_time > 5000) {
226 last_simulation_time = current_time;
227
228 // Cycle through buttons
229 current_simulation_button =
230 (current_simulation_button + 1) % CHESS_BUTTON_COUNT;
231
232 // Simulate button press
233 button_simulate_press(current_simulation_button);
234
235 // Simulate button release after 200ms using non-blocking approach
236 // Store release time instead of blocking
237 button_simulate_release_time = current_time + 200;
238 }
239
240 // Check if it's time to release the simulated button
241 if (button_simulate_release_time > 0 &&
242 current_time >= button_simulate_release_time) {
243 button_simulate_release(current_simulation_button);
244 button_simulate_release_time = 0; // Reset release time
245 }
246
247 return;
248 }
249
250 // REAL HARDWARE BUTTON SCANNING
251 // Scan physical button pins and update button_states
252 //
253 // Physical buttons:
254 // - 4 promotion buttons (MATRIX_COL_0-3): button_id 0-3
255 // - 1 reset button (GPIO15): button_id 8
256 //
257 // LED indications (9 LEDs):
258 // - LED 64-67: White promotion (button_id 0-3 visual)
259 // - LED 68-71: Black promotion (button_id 4-7 visual)
260 // - LED 72: Reset (button_id 8)
261
262 for (int i = 0; i < CHESS_BUTTON_COUNT; i++) {
263 bool current_state = false;
264
265 if (i == 8) {
266 // Reset button (GPIO15) - active low
267 current_state = (gpio_get_level(BUTTON_RESET) == 0);
268
269 } else if (i >= 0 && i <= 3) {
270 // Promotion buttons (0-3): MATRIX_COL_0-3
271 // These pins are time-multiplexed with matrix columns
272 // During button scan window (20-25ms), they can be read as buttons
273 // Active low (pulled up, buttons pull to ground)
274 const gpio_num_t promotion_pins[] = {
275 BUTTON_QUEEN, // MATRIX_COL_0
276 BUTTON_ROOK, // MATRIX_COL_1
277 BUTTON_BISHOP, // MATRIX_COL_2
278 BUTTON_KNIGHT // MATRIX_COL_3
279 };
280 current_state = (gpio_get_level(promotion_pins[i]) == 0);
281
282 } else {
283 // button_id 4-7: No physical buttons (only LED indications)
284 current_state = false;
285 }
286
287 button_states[i] = current_state;
288 }
289}

291void button_simulate_press(uint8_t button_id) {
292 if (button_id >= CHESS_BUTTON_COUNT)
293 return;
294

299 button_press_time[button_id] = esp_timer_get_time() / 1000;
300 button_long_press_sent[button_id] = false;
301
302 // Send button press event
303 button_send_event(button_id, BUTTON_EVENT_PRESS, 0);
304
305 // Update button LED feedback
306 button_update_led_feedback(button_id, true);

309void button_simulate_release(uint8_t button_id) {
310 if (button_id >= CHESS_BUTTON_COUNT)
311 return;
312
313 uint32_t press_duration =
314 esp_timer_get_time() / 1000 - button_press_time[button_id];
315
316 ESP_LOGI(TAG, "Simulating button release: %s (ID: %d, duration: %lu ms)",
317 button_names[button_id], button_id, press_duration);
318
319 button_states[button_id] = false;
320 button_release_time[button_id] = esp_timer_get_time() / 1000;
321
322 // Check for long press
323 if (press_duration >= BUTTON_LONG_PRESS_MS &&
324 !button_long_press_sent[button_id]) {
325 button_send_event(button_id, BUTTON_EVENT_LONG_PRESS, press_duration);
326 button_long_press_sent[button_id] = true;
327 }
328
329 // Send button release event
330 button_send_event(button_id, BUTTON_EVENT_RELEASE, press_duration);
331
332 // Update button LED feedback
333 button_update_led_feedback(button_id, false);
334
335 // Check for double press
336 button_check_double_press(button_id);
337}

343void button_process_events(void) {
344 uint32_t current_time = esp_timer_get_time() / 1000;
345
346 // Process button state changes
347 for (int i = 0; i < CHESS_BUTTON_COUNT; i++) {
348 if (button_states[i] != button_previous[i]) {
349 if (button_states[i]) {
350 // Button pressed
351 button_handle_press(i);
352 } else {
353 // Button released
354 button_handle_release(i);
355 }
356
357 button_previous[i] = button_states[i];
358 }
359
360 // Check for long press
361 if (button_states[i] && !button_long_press_sent[i]) {
362 uint32_t press_duration = current_time - button_press_time[i];
363 if (press_duration >= BUTTON_LONG_PRESS_MS) {
364 button_send_event(i, BUTTON_EVENT_LONG_PRESS, press_duration);
365 button_long_press_sent[i] = true;
366 }
367 }
368 }
369}

417void button_check_double_press(uint8_t button_id) {
418 if (button_id >= CHESS_BUTTON_COUNT)
419 return;
420
421 uint32_t current_time = esp_timer_get_time() / 1000;
422 uint32_t time_since_last_release =
423 current_time - button_release_time[button_id];
424
425 if (time_since_last_release <= BUTTON_DOUBLE_PRESS_MS) {
426 button_press_count[button_id]++;
427
428 if (button_press_count[button_id] == 2) {
429 ESP_LOGI(TAG, "Double press detected: %s (ID: %d)",
430 button_names[button_id], button_id);
431
432 // Send double press event
433 button_send_event(button_id, BUTTON_EVENT_DOUBLE_PRESS, 0);
434
435 // Reset press count
436 button_press_count[button_id] = 0;
437 }
438 } else {
439 // Reset press count if too much time has passed
440 button_press_count[button_id] = 1;
441 }

505 case 0: // Reset all buttons

508

510 button_print_status();
511 break;
512
513 case 2: // Test all buttons
514 button_test_all();
515 break;
516
517 default:
518 ESP_LOGW(TAG, "Unknown button command: %d", command);
519 break;
520 }
521 }

578void button_task_start(void *pvParameters) {
579 ESP_LOGI(TAG, "Button task started successfully");
580
581 // ✅ CRITICAL: Register with TWDT from within task
582 esp_err_t wdt_ret = esp_task_wdt_add(NULL);
583 if (wdt_ret != ESP_OK && wdt_ret != ESP_ERR_INVALID_ARG) {
584 ESP_LOGE(TAG, "Failed to register Button task with TWDT: %s",
585 esp_err_to_name(wdt_ret));
586 } else {
587 ESP_LOGI(TAG, "✅ Button task registered with TWDT");
588 }
589
590 ESP_LOGI(TAG, "Features:");
591 ESP_LOGI(TAG, " • 9 button handling (promotion + reset)");
592 ESP_LOGI(TAG, " • Button debouncing and event detection");
593 ESP_LOGI(TAG, " • LED feedback for button states");
594 ESP_LOGI(TAG, " • Long press and double press detection");
595 ESP_LOGI(TAG, " • Simulation mode (no HW required)");
596 ESP_LOGI(TAG, " • 5ms scan cycle");
597
598 task_running = true;
599
600 // Initialize button states
601 button_reset_all();
602
603 // Main task loop
604 uint32_t loop_count = 0;
605 TickType_t last_wake_time = xTaskGetTickCount();
606
607 for (;;) {
608 // CRITICAL: Reset watchdog for button task in every iteration
609 esp_err_t wdt_ret = button_task_wdt_reset_safe();
610 if (wdt_ret != ESP_OK && wdt_ret != ESP_ERR_NOT_FOUND) {
611 // Task not registered with TWDT yet - this is normal during startup
612 }
613
614 // Watchdog logging every 1000 iterations (5 seconds)
615 if (loop_count % 1000 == 0) {
616 ESP_LOGI(TAG, "Button Task Watchdog: loop=%d, heap=%zu", loop_count,
617 esp_get_free_heap_size());
618 }
619
620 // Process button commands
621 button_process_commands();
622
623 // Button scanning is now handled by FreeRTOS timer
624 // This task only processes commands and events
625 button_process_events();
626
627 // Periodic status update - reduced frequency for cleaner UART
628 if (loop_count % 100000 == 0) { // Every 100000 loops (500 seconds)
629 ESP_LOGI(TAG, "Button Task Status: loop=%d, simulation_button=%d",
630 loop_count, current_simulation_button);
631 }
632
633 loop_count++;
634
635 // Wait for next cycle (5ms)
636 vTaskDelayUntil(&last_wake_time, pdMS_TO_TICKS(5));
637 }
