41esp_err_t error_system_init(const error_system_config_t* config) {

86esp_err_t error_show_visual(visual_error_type_t error_type, const chess_move_t* failed_move) {
87 if (!system_initialized) {
88 return ESP_ERR_INVALID_STATE;
89 }
90
91 if (error_type >= ERROR_VISUAL_COUNT) {
92 return ESP_ERR_INVALID_ARG;
93 }
94
95 // Check if we can display more errors
96 if (active_error_count >= current_config.max_concurrent_errors) {
97 ESP_LOGW(TAG, "Maximum concurrent errors reached, ignoring new error");
98 return ESP_ERR_NO_MEM;
99 }
100
101 // Clear previous error indicators
102 error_clear_all_indicators();
103
104 // Get error color
105 uint8_t r, g, b;
106 esp_err_t color_result = error_get_color_for_type(error_type, &r, &g, &b);
107 if (color_result != ESP_OK) {
108 ESP_LOGE(TAG, "Failed to get color for error type %d", error_type);
109 return color_result;
110 }
111
112 // Get LED positions for error
113 uint8_t led_positions[8];
114 uint8_t led_count = 0;
115
116 if (failed_move) {
117 esp_err_t pos_result = error_get_led_positions_for_move(failed_move, led_positions, &led_count);
118 if (pos_result != ESP_OK) {
119 ESP_LOGE(TAG, "Failed to get LED positions for move");
120 return pos_result;
121 }
122 } else {
123 // Generic error - flash first 8 board LEDs only (array size is 8!)
124 // ✅ CRITICAL BUG FIX: Was i < 64 causing buffer overflow!
125 for (int i = 0; i < 8; i++) {
126 led_positions[i] = i;
127 }
128 led_count = 8; // ✅ FIXED: Only 8 LEDs, not 64
129 }
130
131 // Store error information
132 last_error.error_type = error_type;
133 last_error.error_led_count = led_count;
134 memcpy(last_error.error_led_positions, led_positions, led_count);
135 last_error.display_duration_ms = current_config.flash_duration_ms * current_config.flash_count;
136
137 // Set error messages
138 strncpy(last_error.user_message, error_get_message_for_type(error_type), sizeof(last_error.user_message) - 1);
139 last_error.user_message[sizeof(last_error.user_message) - 1] = '\0';
140
141 if (current_config.enable_recovery_hints) {
142 strncpy(last_error.recovery_hint, error_get_hint_for_type(error_type), sizeof(last_error.recovery_hint) - 1);
143 last_error.recovery_hint[sizeof(last_error.recovery_hint) - 1] = '\0';
144 } else {
145 last_error.recovery_hint[0] = '\0';
146 }
147
148 // Display error visually
149 error_flash_leds(led_positions, led_count, r, g, b);
150
151 displaying_error = true;
152 active_error_count++;
153
154 ESP_LOGI(TAG, "Displaying error: %s", last_error.user_message);
155 if (strlen(last_error.recovery_hint) > 0) {
156 ESP_LOGI(TAG, "Recovery hint: %s", last_error.recovery_hint);
157 }
158
159 return ESP_OK;
160}

162esp_err_t error_show_blocking_path(uint8_t from_row, uint8_t from_col,
163 uint8_t to_row, uint8_t to_col) {
164 if (!system_initialized) {
165 return ESP_ERR_INVALID_STATE;
166 }
167
168 // Get LED positions for source and destination
169 uint8_t from_led_positions[8], to_led_positions[8];
170 uint8_t from_count = 0, to_count = 0;
171
172 esp_err_t from_result = error_get_led_positions_for_square(from_row, from_col, from_led_positions, &from_count);
173 esp_err_t to_result = error_get_led_positions_for_square(to_row, to_col, to_led_positions, &to_count);
174
175 if (from_result != ESP_OK || to_result != ESP_OK) {
176 return ESP_ERR_INVALID_ARG;
177 }
178
179 // Flash source in red, destination in orange
180 error_flash_leds(from_led_positions, from_count, 255, 0, 0); // Red for source
181 error_flash_leds(to_led_positions, to_count, 255, 165, 0); // Orange for destination
182
183 // Store error information
184 last_error.error_type = ERROR_VISUAL_PIECE_BLOCKING;
185 last_error.error_led_count = from_count + to_count;
186 memcpy(last_error.error_led_positions, from_led_positions, from_count);
187 memcpy(last_error.error_led_positions + from_count, to_led_positions, to_count);
188
189 strncpy(last_error.user_message, "Path is blocked by another piece", sizeof(last_error.user_message) - 1);
190 strncpy(last_error.recovery_hint, "Remove the blocking piece or choose a different path", sizeof(last_error.recovery_hint) - 1);
191
192 displaying_error = true;
193 active_error_count++;
194
195 ESP_LOGI(TAG, "Showing blocking path error: %s", last_error.user_message);
196
197 return ESP_OK;
198}

454esp_err_t error_get_led_positions_for_move(const chess_move_t* move, uint8_t* led_positions, uint8_t* count) {
455 if (!move || !led_positions || !count) {
456 return ESP_ERR_INVALID_ARG;
457 }
458
459 // Get LED positions for source and destination squares
460 uint8_t from_positions[8], to_positions[8];
461 uint8_t from_count = 0, to_count = 0;
462
463 esp_err_t from_result = error_get_led_positions_for_square(move->from_row, move->from_col, from_positions, &from_count);
464 esp_err_t to_result = error_get_led_positions_for_square(move->to_row, move->to_col, to_positions, &to_count);
465
466 if (from_result != ESP_OK || to_result != ESP_OK) {
467 return ESP_ERR_INVALID_ARG;
468 }
469
470 // Combine positions
471 *count = from_count + to_count;
472 memcpy(led_positions, from_positions, from_count);
473 memcpy(led_positions + from_count, to_positions, to_count);
474
475 return ESP_OK;
476}

478esp_err_t error_get_led_positions_for_square(uint8_t row, uint8_t col, uint8_t* led_positions, uint8_t* count) {
479 if (!led_positions || !count) {
480 return ESP_ERR_INVALID_ARG;

484 return ESP_ERR_INVALID_ARG;

486

489
490 led_positions[0] = led_index;
491 *count = 1;
492
493 return ESP_OK;
494}

513static void error_flash_leds(uint8_t* led_positions, uint8_t count, uint8_t r, uint8_t g, uint8_t b) {
514 for (uint8_t flash = 0; flash < current_config.flash_count; flash++) {
515 // Flash on
516 for (uint8_t i = 0; i < count; i++) {
517 led_set_pixel_layer(LED_LAYER_ERROR, led_positions[i], r, g, b);
518 }
519 led_force_full_update();
520
521 // Flash off
522 for (uint8_t i = 0; i < count; i++) {
523 led_set_pixel_layer(LED_LAYER_ERROR, led_positions[i], 0, 0, 0);
524 }
525 led_force_full_update();
526 }
