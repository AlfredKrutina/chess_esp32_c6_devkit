37static void format_time_mmss(char *buffer, size_t size, uint32_t time_ms) {
38 uint32_t total_seconds = time_ms / 1000;
39 uint32_t minutes = total_seconds / 60;
40 uint32_t seconds = total_seconds % 60;
41 snprintf(buffer, size, "%02u:%02u", (unsigned int)minutes,
42 (unsigned int)seconds);
43}

48command_result_t uart_cmd_timer(const char *args) {
49 (void)args; // Unused
50 SAFE_WDT_RESET();
51
52 chess_timer_t timer_state;
53 if (timer_get_state(&timer_state) != ESP_OK) {
54 uart_send_error("❌ Failed to get timer state");
55 return CMD_ERROR_SYSTEM_ERROR;
56 }
57
58 uart_send_colored_line(COLOR_INFO, "⏱️ Timer State");
59 uart_send_formatted(
60 "═══════════════════════════════════════════════════════════════");
61
62 // Time control info
63 if (timer_state.config.type == TIME_CONTROL_NONE) {
64 uart_send_formatted("⏸️ Time Control: None (No time limit)");
65 uart_send_line("");
66 return CMD_SUCCESS;
67 }
68
69 uart_send_formatted("📋 Time Control: %s", timer_state.config.name);
70 // Description is a const char* - check if it's not empty (skip NULL check as
71 // it's always set in TIME_CONTROLS array)
72 if (timer_state.config.description[0] != '\0') {
73 uart_send_formatted(" (%s)", timer_state.config.description);
74 }
75
76 // Format increment info
77 if (timer_state.config.increment_ms > 0) {
78 char inc_str[32];
79 format_time_mmss(inc_str, sizeof(inc_str), timer_state.config.increment_ms);
80 uart_send_formatted(" Increment: %s per move", inc_str);
81 }
82 uart_send_line("");
83
84 // Player times
85 char white_time_str[16], black_time_str[16];
86 format_time_mmss(white_time_str, sizeof(white_time_str),
87 timer_state.white_time_ms);
88 format_time_mmss(black_time_str, sizeof(black_time_str),
89 timer_state.black_time_ms);
90
91 // Determine active player indicator
92 const char *white_indicator =
93 timer_state.is_white_turn && timer_state.timer_running ? "⏱️ " : " ";
94 const char *black_indicator =
95 !timer_state.is_white_turn && timer_state.timer_running ? "⏱️ " : " ";
96
97 uart_send_formatted("%s White: %s", white_indicator, white_time_str);
98 if (timer_state.is_white_turn && timer_state.timer_running) {
99 uart_send_formatted(" (running)");
100 }
101 uart_send_line("");
102
103 uart_send_formatted("%s Black: %s", black_indicator, black_time_str);
104 if (!timer_state.is_white_turn && timer_state.timer_running) {
105 uart_send_formatted(" (running)");
106 }
107 uart_send_line("");
108
109 // Status
110 if (timer_state.game_paused) {
111 uart_send_colored_line(COLOR_WARNING, "⏸️ Timer: PAUSED");
112 } else if (timer_state.timer_running) {
113 uart_send_formatted("▶️ Timer: RUNNING");
114 } else {
115 uart_send_formatted("⏸️ Timer: STOPPED");
116 }
117
118 if (timer_state.time_expired) {
119 uart_send_colored_line(COLOR_ERROR, "⚠️ TIME EXPIRED!");
120 }
121
122 // Statistics
123 if (timer_state.total_moves > 0) {
124 uart_send_formatted("📊 Total moves: %lu", timer_state.total_moves);
125 if (timer_state.avg_move_time_ms > 0) {
126 char avg_time_str[16];
127 format_time_mmss(avg_time_str, sizeof(avg_time_str),
128 timer_state.avg_move_time_ms);
129 uart_send_formatted(" | Avg move time: %s", avg_time_str);
130 }
131 uart_send_line("");
132 }
133
134 uart_send_line("");
135 return CMD_SUCCESS;
136}

143
144 // If no args or "options", show all available time controls
145 if (!args || strlen(args) == 0 || strcmp(args, "options") == 0 ||
146 strcmp(args, "OPTIONS") == 0 || strcmp(args, "list") == 0 ||
147 strcmp(args, "LIST") == 0) {
148
149 uart_send_colored_line(COLOR_INFO, "⏱️ Available Time Controls");
150 uart_send_formatted(
151 "═══════════════════════════════════════════════════════════════");
152
153 time_control_config_t controls[16];
154 uint32_t count = timer_get_available_controls(controls, 16);
155
156 for (uint32_t i = 0; i < count; i++) {
157 char time_str[32], increment_str[16];
158
159 // Format time
160 uint32_t total_sec = controls[i].initial_time_ms / 1000;
161 uint32_t minutes = total_sec / 60;
162 uint32_t seconds = total_sec % 60;
163 if (minutes >= 60) {
164 unsigned long hours = (unsigned long)(minutes / 60);
165 unsigned long mins = (unsigned long)(minutes % 60);
166 snprintf(time_str, sizeof(time_str), "%luh %02lum", hours, mins);
167 } else if (seconds > 0) {
168 unsigned long mins = (unsigned long)minutes;
169 unsigned long secs = (unsigned long)seconds;
170 snprintf(time_str, sizeof(time_str), "%lum %02lus", mins, secs);
171 } else {
172 unsigned long mins = (unsigned long)minutes;
173 snprintf(time_str, sizeof(time_str), "%lum", mins);
174 }
175
176 // Format increment
177 if (controls[i].increment_ms > 0) {
178 unsigned long inc_sec =
179 (unsigned long)(controls[i].increment_ms / 1000);
180 snprintf(increment_str, sizeof(increment_str), "+%lus", inc_sec);
181 } else {
182 snprintf(increment_str, sizeof(increment_str), "+0s");
183 }
184
185 // Display with format: ID: Name (time + increment) - Description
186 const char *speed_indicator = controls[i].is_fast ? "⚡" : "🕐";
187
188 uart_send_formatted("%2lu: %s %s (%s %s)", i, speed_indicator,
189 controls[i].name, time_str, increment_str);
190 if (strlen(controls[i].description) > 0) {
191 uart_send_formatted(" %s", controls[i].description);
192 }
193 }
194
195 uart_send_formatted("");
196 uart_send_formatted("Usage:");
197 uart_send_formatted(
198 " TIMER_CONFIG <0-14> - Set predefined time control");
199 uart_send_formatted(" TIMER_CONFIG custom <min> <inc> - Set custom time "
200 "(1-180 min, 0-60s increment)");
201 uart_send_formatted("");
202 uart_send_formatted("Examples:");
203 uart_send_formatted(" TIMER_CONFIG 1 - Set Bullet 1+0");
204 uart_send_formatted(" TIMER_CONFIG 8 - Set Rapid 10+0");
205 uart_send_formatted(
206 " TIMER_CONFIG custom 15 5 - Set custom 15min + 5s increment");
207 uart_send_formatted("");
208
209 return CMD_SUCCESS;
210 }
211
212 if (game_command_queue == NULL) {
213 uart_send_error("❌ Game task not available");
214 return CMD_ERROR_SYSTEM_ERROR;
215 }
216
217 chess_move_command_t cmd = (chess_move_command_t){0};
218 cmd.type = GAME_CMD_SET_TIME_CONTROL;
219
220 char arg1[32], arg2[32], arg3[32];
221 int parsed = sscanf(args, "%31s %31s %31s", arg1, arg2, arg3);
222
223 if (strcmp(arg1, "custom") == 0 || strcmp(arg1, "CUSTOM") == 0) {
224 if (parsed < 3) {
225 uart_send_error(
226 "❌ Usage: TIMER_CONFIG custom <minutes> <increment_sec>");
227 return CMD_ERROR_INVALID_SYNTAX;
228 }
229 cmd.timer_data.timer_config.time_control_type = TIME_CONTROL_CUSTOM;
230 cmd.timer_data.timer_config.custom_minutes = (uint32_t)atoi(arg2);
231 cmd.timer_data.timer_config.custom_increment = (uint32_t)atoi(arg3);
232
233 // Validate custom values
234 if (cmd.timer_data.timer_config.custom_minutes < 1 ||
235 cmd.timer_data.timer_config.custom_minutes > 180) {
236 uart_send_error("❌ Minutes must be between 1 and 180");
237 return CMD_ERROR_INVALID_PARAMETER;
238 }
239 if (cmd.timer_data.timer_config.custom_increment > 60) {
240 uart_send_error("❌ Increment must be between 0 and 60 seconds");
241 return CMD_ERROR_INVALID_PARAMETER;
242 }
243 } else {
244 int type = atoi(arg1);
245 if (type < 0 || type >= TIME_CONTROL_MAX) {
246 char error_msg[128];
247 snprintf(error_msg, sizeof(error_msg),
248 "❌ Invalid type. Use 0-%d or 'custom'. Use 'TIMER_CONFIG "
249 "options' to see all.",
250 TIME_CONTROL_MAX - 1);
251 uart_send_error(error_msg);
252 return CMD_ERROR_INVALID_SYNTAX;
253 }
254 cmd.timer_data.timer_config.time_control_type = (uint8_t)type;
255 if (type == TIME_CONTROL_CUSTOM && parsed >= 3) {
256 cmd.timer_data.timer_config.custom_minutes = (uint32_t)atoi(arg2);
257 cmd.timer_data.timer_config.custom_increment = (uint32_t)atoi(arg3);
258 }
259 }
260
261 if (xQueueSend(game_command_queue, &cmd, pdMS_TO_TICKS(100)) == pdTRUE) {
262 uart_send_formatted("✅ Time control update sent");
263 } else {
264 uart_send_error("❌ Failed to send time control");
265 return CMD_ERROR_SYSTEM_ERROR;
266 }
267
268 return CMD_SUCCESS;
269}

358command_result_t uart_cmd_wifi(const char *args) {
359 SAFE_WDT_RESET();
360
361 if (!args || strlen(args) == 0) {
362 uart_send_error("❌ Usage: WIFI <ssid> <password>");
363 uart_send_formatted(" Example: WIFI MyHomeWiFi MyPassword123");
364 uart_send_formatted(" Example: WIFI \"My Home WiFi\" \"My Password\"");
365 return CMD_ERROR_INVALID_SYNTAX;
366 }
367
368 char ssid[33] = {0};
369 char password[65] = {0};
370
371 // Parsovat argumenty - podporovat uvozovky
372 const char *p = args;
373 while (*p == ' ')
374 p++; // Skip leading spaces
375
376 // Parsovat SSID
377 if (*p == '"') {
378 // SSID v uvozovkach
379 p++;
380 const char *end = strchr(p, '"');
381 if (end == NULL) {
382 uart_send_error("❌ Missing closing quote for SSID");
383 return CMD_ERROR_INVALID_SYNTAX;
384 }
385 size_t len = end - p;
386 if (len >= sizeof(ssid))
387 len = sizeof(ssid) - 1;
388 strncpy(ssid, p, len);
389 ssid[len] = '\0';
390 p = end + 1;
391 } else {
392 // SSID bez uvozovek - do prvni mezery
393 const char *space = strchr(p, ' ');
394 if (space == NULL) {
395 uart_send_error("❌ Missing password");
396 return CMD_ERROR_INVALID_SYNTAX;
397 }
398 size_t len = space - p;
399 if (len >= sizeof(ssid))
400 len = sizeof(ssid) - 1;
401 strncpy(ssid, p, len);
402 ssid[len] = '\0';
403 p = space;
404 }
405
406 // Skip spaces
407 while (*p == ' ')
408 p++;
409
410 // Parsovat password
411 if (*p == '"') {
412 // Password v uvozovkach
413 p++;
414 const char *end = strchr(p, '"');
415 if (end == NULL) {
416 uart_send_error("❌ Missing closing quote for password");
417 return CMD_ERROR_INVALID_SYNTAX;
418 }
419 size_t len = end - p;
420 if (len >= sizeof(password))
421 len = sizeof(password) - 1;
422 strncpy(password, p, len);
423 password[len] = '\0';
424 } else {
425 // Password bez uvozovek - zbytek retezce
426 size_t len = strlen(p);
427 if (len >= sizeof(password))
428 len = sizeof(password) - 1;
429 strncpy(password, p, len);
430 password[len] = '\0';
431 }
432
433 // Validovat
434 if (strlen(ssid) == 0 || strlen(ssid) > 32) {
435 uart_send_error("❌ SSID too long (max 32 characters)");
436 return CMD_ERROR_INVALID_PARAMETER;
437 }
438
439 if (strlen(password) == 0 || strlen(password) > 64) {
440 uart_send_error("❌ Password too long (max 64 characters)");
441 return CMD_ERROR_INVALID_PARAMETER;
442 }
443
444 // Ulozit do NVS
445 esp_err_t ret = wifi_save_config_to_nvs(ssid, password);
446 if (ret != ESP_OK) {
447 uart_send_error("❌ Failed to save WiFi configuration to NVS");
448 uart_send_formatted(" Error: %s", esp_err_to_name(ret));
449 return CMD_ERROR_SYSTEM_ERROR;
450 }
451
452 uart_send_formatted("✅ WiFi configuration saved");
453 uart_send_formatted(" SSID: %s", ssid);
454 uart_send_formatted(" Use 'WIFI_CONNECT' to connect");
455
456 return CMD_SUCCESS;
457}

770command_result_t uart_cmd_mqtt_config(const char *args) {
771 if (!args || strlen(args) == 0) {
772 uart_send_error(
773 "❌ Usage: MQTT_CONFIG <host> [port] [username] [password]");
774 uart_send_formatted(" Example: MQTT_CONFIG broker.example.com 1883");
775 uart_send_formatted(
776 " Example: MQTT_CONFIG broker.example.com 1883 user pass");
777 return CMD_ERROR_INVALID_SYNTAX;
778 }
779
780 SAFE_WDT_RESET();
781
782 // Parse arguments
783 char host[128] = {0};
784 uint16_t port = 1883; // Default
785 char username[64] = {0};
786 char password[64] = {0};
787
788 int parsed =
789 sscanf(args, "%127s %hu %63s %63s", host, &port, username, password);
790 if (parsed < 1) {
791 uart_send_error("❌ Failed to parse arguments");
792 uart_send_formatted(
793 " Usage: MQTT_CONFIG <host> [port] [username] [password]");
794 return CMD_ERROR_INVALID_PARAMETER;
795 }
796
797 /* uint16_t is always <= 65535; only 0 is invalid */
798 if (port == 0) {
799 uart_send_error("❌ Invalid port (must be 1-65535)");
800 return CMD_ERROR_INVALID_PARAMETER;
801 }
802
803 // Save to NVS
804 const char *username_ptr = (strlen(username) > 0) ? username : NULL;
805 const char *password_ptr = (strlen(password) > 0) ? password : NULL;
806
807 esp_err_t ret =
808 mqtt_save_config_to_nvs(host, port, username_ptr, password_ptr);
809 if (ret != ESP_OK) {
810 uart_send_error("❌ Failed to save MQTT configuration");
811 uart_send_formatted(" Error: %s", esp_err_to_name(ret));
812 return CMD_ERROR_SYSTEM_ERROR;
813 }
814
815 uart_send_formatted("✅ MQTT configuration saved");
816 uart_send_formatted(" Host: %s", host);
817 uart_send_formatted(" Port: %d", port);
818 uart_send_formatted(" Username: %s",
819 username_ptr ? username_ptr : "(none)");
820 uart_send_formatted(" Password: %s", password_ptr ? "***" : "(none)");
821 uart_send_formatted("");
822
823 // Reinit MQTT client if WiFi STA is connected
824 extern bool wifi_is_sta_connected(void);
825 if (wifi_is_sta_connected()) {
826 uart_send_formatted(
827 "🔄 Reinitializing MQTT client with new configuration...");
828 esp_err_t reinit_ret = ha_light_reinit_mqtt();

830 uart_send_formatted("✅ MQTT client reinicialized");

832 uart_send_formatted("⚠️ Failed to reinit MQTT client: %s",
833 esp_err_to_name(reinit_ret));
834 uart_send_formatted(" Client will reconnect on next WiFi connection");
835 }
836 } else {
837 uart_send_formatted("ℹ️ MQTT client will reconnect with new configuration "
838 "on next WiFi connection");
839 }

841 return CMD_SUCCESS;
842}
