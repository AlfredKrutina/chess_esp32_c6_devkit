65static bool game_fen_expected_occupancy_64(const char *fen, uint8_t exp[64]) {
66 memset(exp, 0, 64);
67 if (fen == NULL) {
68 return false;
69 }
70 int row = 7;
71 int col = 0;
72 const char *p = fen;
73 while (*p != '\0' && *p != ' ') {
74 if (*p == '/') {
75 if (col != 8) {
76 return false;
77 }
78 row--;
79 col = 0;
80 p++;
81 continue;
82 }
83 if (*p >= '1' && *p <= '8') {
84 col += (*p - '0');
85 if (col > 8) {
86 return false;
87 }
88 p++;
89 continue;
90 }
91 if (row < 0 || row > 7 || col < 0 || col > 7) {
92 return false;
93 }
94 exp[row * 8 + col] = 1;
95 col++;
96 p++;
97 }
98 return (row == 0 && col == 8 && *p == ' ');
99}

105bool game_physical_matrix_matches_fen(const char *fen) {
106 uint8_t exp[64];
107 if (!game_fen_expected_occupancy_64(fen, exp)) {
108 return false;
109 }
110 uint8_t phys[64];
111 matrix_get_state(phys);
112 for (int i = 0; i < 64; i++) {
113 uint8_t want = exp[i] ? (uint8_t)1 : (uint8_t)0;
114 uint8_t got = phys[i] ? (uint8_t)1 : (uint8_t)0;
115 if (want != got) {
116 return false;
117 }
118 }
119 return true;
120}

143bool game_puzzle_start(uint8_t puzzle_id) {
144 const game_puzzle_definition_t *puzzle = game_get_puzzle_definition(puzzle_id);
145 if (puzzle == NULL) {
146 return false;
147 }
148
149 game_reset_game();
150
151 player_t fen_player = PLAYER_WHITE;
152 if (!game_load_position_from_fen(puzzle->fen, &fen_player)) {
153 ESP_LOGE(TAG, "puzzle: failed to load FEN for id=%u", (unsigned)puzzle_id);
154 return false;
155 }
156
157 uint8_t from_row = 0, from_col = 0, to_row = 0, to_col = 0;
158 if (!convert_notation_to_coords(puzzle->solution_from, &from_row, &from_col) ||
159 !convert_notation_to_coords(puzzle->solution_to, &to_row, &to_col)) {
160 ESP_LOGE(TAG, "puzzle: invalid solution notation for id=%u",
161 (unsigned)puzzle_id);
162 return false;
163 }
164
165 current_player = fen_player;
166 current_game_state = GAME_STATE_ACTIVE;
167 game_active = true;
168 game_start_time = esp_timer_get_time() / 1000;
169 last_move_time = game_start_time;
170 board_setup_tutorial_active = false;
171 puzzle_setup_active = false;
172 puzzle_setup_id = 0;
173 puzzle_active = true;
174 puzzle_active_id = puzzle_id;
175 puzzle_solution_from_row = from_row;
176 puzzle_solution_from_col = from_col;
177 puzzle_solution_to_row = to_row;
178 puzzle_solution_to_col = to_col;
179 puzzle_feedback = PUZZLE_FEEDBACK_NONE;
180 led_clear_board_only();
181 game_highlight_movable_pieces();
182 STAGING_LOGI(TAG, "puzzle: started id=%u difficulty=%u",
183 (unsigned)puzzle->id, (unsigned)puzzle->difficulty);
184 return true;
185}
