21move_error_t game_is_valid_move(const chess_move_t *move) {
22 if (!move) {
23 return MOVE_ERROR_INVALID_MOVE_STRUCTURE;
24 }
25
26 if (!game_is_valid_position(move->from_row, move->from_col) ||
27 !game_is_valid_position(move->to_row, move->to_col)) {
28 return MOVE_ERROR_OUT_OF_BOUNDS;
29 }
30
31 if (!game_active) {
32 return MOVE_ERROR_GAME_NOT_ACTIVE;
33 }
34
35 // Check if source position has a piece
36 piece_t source_piece = game_get_piece(move->from_row, move->from_col);
37
38 // Pokud je source_piece prázdné (král je zvednutý během resignation timeru),
39 // použít move->piece místo source_piece
40 if (source_piece == PIECE_EMPTY) {
41 // Pokud je resignation timer aktivní a move->piece je král, použít
42 // move->piece
43 if (resignation_state.active && move->piece != PIECE_EMPTY &&
44 (move->piece == PIECE_WHITE_KING || move->piece == PIECE_BLACK_KING) &&
45 resignation_state.king_row == move->from_row &&
46 resignation_state.king_col == move->from_col) {
47 source_piece = move->piece;
48 ESP_LOGI(TAG,
49 "🏰 game_is_valid_move: Using move->piece (%d) because king is "
50 "lifted (resignation timer active)",
51 move->piece);
52 } else {
53 return MOVE_ERROR_NO_PIECE;
54 }
55 }
56
57 // Check if source piece belongs to current player
58 if ((current_player == PLAYER_WHITE && !game_is_white_piece(source_piece)) ||
59 (current_player == PLAYER_BLACK && !game_is_black_piece(source_piece))) {
60 return MOVE_ERROR_WRONG_COLOR;
61 }
62
63 // Check if destination is occupied by own piece
64 piece_t dest_piece = game_get_piece(move->to_row, move->to_col);
65 if (dest_piece != PIECE_EMPTY &&
66 game_is_same_color(source_piece, dest_piece)) {
67 return MOVE_ERROR_DESTINATION_OCCUPIED;
68 }
69
70 // KONTROLA ROŠÁDY V PROGRESS - PŘED validací pohybu figurky
71 // Pokud je to správný rošáda tah věže, přeskočit standardní validaci pohybu
72 // figurky (která by jinak zjistila, že cesta je blokovaná králem)
73 bool is_castling_rook_move = false;
74 if (castling_state.in_progress) {
75 ESP_LOGI(TAG, "🏰 game_is_valid_move: Castling in progress - checking if "
76 "this is the expected rook move");
77 ESP_LOGI(TAG, "🏰 Expected: from %c%d to %c%d, actual: from %c%d to %c%d",
78 'a' + castling_state.rook_from_col,
79 castling_state.rook_from_row + 1, 'a' + castling_state.rook_to_col,
80 castling_state.rook_to_row + 1, 'a' + move->from_col,
81 move->from_row + 1, 'a' + move->to_col, move->to_row + 1);
82
83 // Očekáváme tah věže
84 if (move->from_row != castling_state.rook_from_row ||
85 move->from_col != castling_state.rook_from_col) {
86 ESP_LOGE(TAG, "❌ Castling in progress - expected rook move from %c%d",
87 'a' + castling_state.rook_from_col,
88 castling_state.rook_from_row + 1);
89 return MOVE_ERROR_CASTLING_BLOCKED; // Použijeme existující error code
90 }
91
92 if (move->to_row != castling_state.rook_to_row ||
93 move->to_col != castling_state.rook_to_col) {
94 ESP_LOGE(
95 TAG, "❌ Incorrect rook destination during castling - expected %c%d",
96 'a' + castling_state.rook_to_col, castling_state.rook_to_row + 1);
97 return MOVE_ERROR_CASTLING_BLOCKED; // Použijeme existující error code
98 }
99
100 // Toto je správný rošáda tah věže - přeskočit standardní validaci pohybu
101 // figurky
102 is_castling_rook_move = true;
103 ESP_LOGI(TAG, "✅ Correct rook move for castling completion - skipping "
104 "standard piece validation (path would be blocked by king)");
105 }
106
107 // Validate move based on piece type (POUZE pokud NENÍ rošáda tah věže)
108 if (!is_castling_rook_move) {
109 move_error_t piece_error =
110 game_validate_piece_move_enhanced(move, source_piece);
111 if (piece_error != MOVE_ERROR_NONE) {
112 return piece_error;
113 }
114 } else {
115 ESP_LOGI(TAG,
116 "🏰 Skipping standard piece validation for castling rook move");
117 }
118
119 // Check if move would leave king in check
120 if (game_would_move_leave_king_in_check(move)) {
121 return MOVE_ERROR_KING_IN_CHECK;
122 }
123
124 return MOVE_ERROR_NONE;
125}

198move_error_t game_validate_pawn_move_enhanced(const chess_move_t *move,
199 piece_t piece) {
200 int row_diff = move->to_row - move->from_row;
201 int col_diff = move->to_col - move->from_col;
202 int abs_col_diff = abs(col_diff);
203
204 bool is_white = game_is_white_piece(piece);
205 int direction = is_white ? 1 : -1;
206 int start_row = is_white ? 1 : 6;
207
208 // Logovani validace pesce
209 char from_sq[3], to_sq[3];
210 game_coords_to_square(move->from_row, move->from_col, from_sq);
211 game_coords_to_square(move->to_row, move->to_col, to_sq);
212 piece_t dest_piece = game_get_piece(move->to_row, move->to_col);
213
214 // Forward move
215 if (col_diff == 0) {
216 // Check if moving backwards (invalid for pawns)
217 if (row_diff * direction < 0) {
218 ESP_LOGD(TAG, "❌ Pawn %s→%s: cannot move backwards", from_sq, to_sq);
219 return MOVE_ERROR_INVALID_PATTERN;
220 }
221
222 // Single square move
223 if (row_diff == direction) {
224 if (game_is_empty(move->to_row, move->to_col)) {
225 return MOVE_ERROR_NONE;
226 } else {
227 ESP_LOGD(TAG, "🔍 Pawn %s->%s: blocked by piece", from_sq, to_sq);
228 return MOVE_ERROR_BLOCKED_PATH;
229 }
230 }
231
232 // Double square move from starting position
233 if (row_diff == 2 * direction && move->from_row == start_row) {
234 if (game_is_empty(move->from_row + direction, move->from_col) &&
235 game_is_empty(move->to_row, move->to_col)) {
236 return MOVE_ERROR_NONE;
237 } else {
238 ESP_LOGD(TAG, "🔍 Pawn %s->%s: blocked path double move", from_sq,
239 to_sq);
240 return MOVE_ERROR_BLOCKED_PATH;
241 }
242 }
243
244 ESP_LOGD(TAG, "🔍 Pawn %s->%s: invalid forward distance", from_sq, to_sq);
245 return MOVE_ERROR_INVALID_PATTERN;
246 }
247
248 // Capture move (diagonal)
249 if (abs_col_diff == 1 && row_diff == direction) {
250 if (dest_piece != PIECE_EMPTY && !game_is_same_color(piece, dest_piece)) {
251 ESP_LOGD(TAG, "✅ Pawn %s→%s: valid capture of %s", from_sq, to_sq,
252 game_get_piece_name(dest_piece));
253 return MOVE_ERROR_NONE;
254 }
255
256 // Check for en passant
257 if (game_is_en_passant_possible(move)) {
258 ESP_LOGD(TAG, "✅ Pawn %s→%s: valid en passant", from_sq, to_sq);
259 return MOVE_ERROR_NONE;
260 }
261
262 if (dest_piece == PIECE_EMPTY) {
263 ESP_LOGD(TAG, "❌ Pawn %s→%s: diagonal to empty square (not en passant)",
264 from_sq, to_sq);
265 } else {
266 ESP_LOGD(TAG, "❌ Pawn %s→%s: trying to capture own piece", from_sq,
267 to_sq);
268 }
269 return MOVE_ERROR_INVALID_PATTERN;
270 }
271
272 ESP_LOGD(TAG,
273 "❌ Pawn %s→%s: invalid move pattern (col_diff=%d, row_diff=%d)",
274 from_sq, to_sq, col_diff, row_diff);
275 return MOVE_ERROR_INVALID_PATTERN;
276}

321move_error_t game_validate_bishop_move_enhanced(const chess_move_t *move) {
322 int row_diff = move->to_row - move->from_row;
323 int col_diff = move->to_col - move->from_col;
324 int abs_row_diff = abs(row_diff);
325 int abs_col_diff = abs(col_diff);
326
327 // Bishop moves diagonally
328 if (abs_row_diff != abs_col_diff) {
329 return MOVE_ERROR_INVALID_PATTERN;
330 }
331
332 // Check if path is blocked
333 // Původně: while (...&& ...) - končilo předčasně
334 // Nově: while (...|| ...) - kontroluje až do cíle
335 int row_step = (row_diff > 0) ? 1 : -1;
336 int col_step = (col_diff > 0) ? 1 : -1;
337
338 int current_row = move->from_row + row_step;
339 int current_col = move->from_col + col_step;
340
341 // SAFETY: Prevent infinite loop (max 8 squares on diagonal)
342 int steps = 0;
343 while ((current_row != move->to_row || current_col != move->to_col) &&
344 steps < 8) {
345 if (!game_is_empty(current_row, current_col)) {
346 ESP_LOGD(TAG, "🔍 Bishop blocked at %c%d", 'a' + current_col,
347 current_row + 1);
348 return MOVE_ERROR_BLOCKED_PATH;
349 }
350 current_row += row_step;
351 current_col += col_step;
352 steps++;
353 }
354
355 // Safety check: if loop ended without reaching target, something is wrong
356 if (steps >= 8 &&
357 (current_row != move->to_row || current_col != move->to_col)) {
358 ESP_LOGE(TAG, "🔍 Bishop validation error: loop exceeded max steps");
359 return MOVE_ERROR_INVALID_PATTERN;
360 }
361
362 return MOVE_ERROR_NONE;
363}

370move_error_t game_validate_rook_move_enhanced(const chess_move_t *move) {
371 int row_diff = move->to_row - move->from_row;
372 int col_diff = move->to_col - move->from_col;
373
374 // Rook moves horizontally or vertically
375 if (row_diff != 0 && col_diff != 0) {
376 return MOVE_ERROR_INVALID_PATTERN;
377 }
378
379 // Check if path is blocked
380 if (row_diff == 0) {
381 // Horizontal move
382 int col_step = (col_diff > 0) ? 1 : -1;
383 int current_col = move->from_col + col_step;
384
385 while (current_col != move->to_col) {
386 if (!game_is_empty(move->from_row, current_col)) {
387 return MOVE_ERROR_BLOCKED_PATH;
388 }
389 current_col += col_step;
390 }
391 } else {
392 // Vertical move
393 int row_step = (row_diff > 0) ? 1 : -1;
394 int current_row = move->from_row + row_step;
395
396 while (current_row != move->to_row) {
397 if (!game_is_empty(current_row, move->from_col)) {
398 return MOVE_ERROR_BLOCKED_PATH;
399 }
400 current_row += row_step;
401 }
402 }
403
404 return MOVE_ERROR_NONE;
405}

495bool game_would_move_leave_king_in_check(const chess_move_t *move) {
496 if (!move)
497 return true; // Safety check
498
499 // Save original board state
500 piece_t original_from_piece = board[move->from_row][move->from_col];
501 piece_t original_to_piece = board[move->to_row][move->to_col];
502
503 // Pokud je from_piece prázdné (král je zvednutý během resignation timeru),
504 // použít move->piece místo original_from_piece
505 if (original_from_piece == PIECE_EMPTY && move->piece != PIECE_EMPTY) {
506 original_from_piece = move->piece;
507 ESP_LOGD(TAG,
508 "🏰 game_would_move_leave_king_in_check: Using move->piece (%d) "
509 "because from square is empty (king lifted)",
510 move->piece);
511 }
512
513 // Check if this is an en passant move
514 bool is_en_passant = game_is_en_passant_possible(move);
515 piece_t original_en_passant_piece = PIECE_EMPTY;
516 uint8_t en_passant_victim_row_local = 0;
517 uint8_t en_passant_victim_col_local = 0;
518
519 // If en passant, save the victim pawn position
520 if (is_en_passant) {
521 original_en_passant_piece =
522 board[en_passant_victim_row][en_passant_victim_col];
523 en_passant_victim_row_local = en_passant_victim_row;
524 en_passant_victim_col_local = en_passant_victim_col;
525 // Remove the victim pawn for simulation
526 board[en_passant_victim_row][en_passant_victim_col] = PIECE_EMPTY;
527 }
528
529 // Make the move temporarily
530 board[move->to_row][move->to_col] = original_from_piece;
531 board[move->from_row][move->from_col] = PIECE_EMPTY;
532
533 // Determine which player's king to check
534 player_t player =
535 (game_is_white_piece(original_from_piece)) ? PLAYER_WHITE : PLAYER_BLACK;
536
537 // Check if king is in check after this move
538 bool king_in_check = game_is_king_in_check(player);
539
540 // Undo the move
541 board[move->from_row][move->from_col] = original_from_piece;
542 board[move->to_row][move->to_col] = original_to_piece;
543
544 // Restore en passant victim if it was removed
545 if (is_en_passant) {
546 board[en_passant_victim_row_local][en_passant_victim_col_local] =
547 original_en_passant_piece;
548 }
549
550 return king_in_check;
551}

567bool game_is_en_passant_possible(const chess_move_t *move) {
568 if (!has_last_move) {
569 ESP_LOGD(TAG, "🔍 En passant: no last move");
570 return false;
571 }
572
573 // Check if the last move was a pawn moving two squares
574 piece_t last_piece = board[last_move_to_row][last_move_to_col];
575 bool last_was_pawn =
576 (last_piece == PIECE_WHITE_PAWN || last_piece == PIECE_BLACK_PAWN);
577
578 if (!last_was_pawn) {
579 ESP_LOGD(TAG, "🔍 En passant: last move was not pawn");
580 return false;
581 }
582
583 // Check if last move was a double square move
584 int last_row_diff = abs((int)last_move_to_row - (int)last_move_from_row);
585 if (last_row_diff != 2) {
586 ESP_LOGD(TAG, "🔍 En passant: last move was not 2 squares (diff=%d)",
587 last_row_diff);
588 return false;
589 }
590
591 // Zkontrolovat, zda útočící pěšec je na stejném řádku jako právě
592 // pohnutý pěšec
593 if (move->from_row != last_move_to_row) {
594 ESP_LOGD(TAG,
595 "🔍 En passant: attacking pawn not on same row (from_row=%d, "
596 "last_to_row=%d)",
597 move->from_row, last_move_to_row);
598 return false;
599 }
600
601 // Zkontrolovat, zda útočící pěšec je přesně vedle právě pohnutého
602 // pěšce
603 int col_diff = abs((int)move->from_col - (int)last_move_to_col);
604 if (col_diff != 1) {
605 ESP_LOGD(TAG, "🔍 En passant: pawn not adjacent (col_diff=%d)", col_diff);
606 return false;
607 }
608
609 // En passant pole je vždy uprostřed mezi from a to (prostý průměr)
610 int en_passant_target_row = (last_move_from_row + last_move_to_row) / 2;
611
612 ESP_LOGD(TAG,
613 "🔍 En passant check: last=%c%d→%c%d, target=%c%d, move=%c%d→%c%d",
614 'a' + last_move_from_col, last_move_from_row + 1,
615 'a' + last_move_to_col, last_move_to_row + 1, 'a' + last_move_to_col,
616 en_passant_target_row + 1, 'a' + move->from_col, move->from_row + 1,
617 'a' + move->to_col, move->to_row + 1);
618
619 if (move->to_row == en_passant_target_row &&
620 move->to_col == last_move_to_col) {
621 ESP_LOGD(TAG, "✅ En passant VALID!");
622 return true;
623 }
624
625 ESP_LOGD(TAG,
626 "❌ En passant: target mismatch (to_row=%d vs target=%d, to_col=%d "
627 "vs last_col=%d)",
628 move->to_row, en_passant_target_row, move->to_col, last_move_to_col);
629 return false;
630}

654move_error_t game_validate_castling(const chess_move_t *move) {
655 if (!move) {
656 return MOVE_ERROR_INVALID_MOVE_STRUCTURE;
657 }
658
659 piece_t piece = move->piece;
660 bool is_white = game_is_white_piece(piece);
661
662 // Check if it's a king
663 if (piece != PIECE_WHITE_KING && piece != PIECE_BLACK_KING) {
664 return MOVE_ERROR_INVALID_PATTERN;
665 }
666
667 // Check if king has moved
668 if ((is_white && white_king_moved) || (!is_white && black_king_moved)) {
669 ESP_LOGD(TAG, "❌ Castling: king has moved");
670 return MOVE_ERROR_CASTLING_BLOCKED;
671 }
672
673 // Pokud je resignation timer aktivní a král je zvednutý,
674 // použít uloženou pozici krále z resignation_state místo kontroly desky
675 int king_row = is_white ? 0 : 7;
676 int expected_king_col = 4; // e-file
677
678 // Pokud je resignation timer aktivní a král patří aktuálnímu hráči,
679 // použít uloženou pozici krále z resignation_state
680 if (resignation_state.active &&
681 resignation_state.player == (is_white ? PLAYER_WHITE : PLAYER_BLACK)) {
682 // Král je zvednutý - použít uloženou pozici z resignation_state
683 king_row = resignation_state.king_row;
684 expected_king_col = resignation_state.king_col;
685 ESP_LOGI(TAG,
686 "🏰 Castling validation: Using stored king position from "
687 "resignation_state: %c%d",
688 'a' + expected_king_col, king_row + 1);
689 }
690
691 // Kontrola, zda from_row/from_col odpovídá očekávané pozici krále
692 // (buď z desky, nebo z resignation_state)
693 if (move->from_row != king_row || move->from_col != expected_king_col) {
694 ESP_LOGD(
695 TAG,
696 "❌ Castling: king not in starting position (from %c%d, expected %c%d)",
697 'a' + move->from_col, move->from_row + 1, 'a' + expected_king_col,
698 king_row + 1);
699 return MOVE_ERROR_CASTLING_BLOCKED;
700 }
701
702 // Determine castling direction
703 bool is_kingside = (move->to_col == 6);
704 bool is_queenside = (move->to_col == 2);
705
706 if (!is_kingside && !is_queenside) {
707 return MOVE_ERROR_INVALID_PATTERN;
708 }
709
710 // Determine rook position
711 int rook_col = is_kingside ? 7 : 0;
712
713 // Check if rook exists at expected position
714 piece_t expected_rook = is_white ? PIECE_WHITE_ROOK : PIECE_BLACK_ROOK;
715 piece_t rook_piece = board[king_row][rook_col];
716 if (rook_piece != expected_rook) {
717 ESP_LOGD(TAG, "❌ Castling: rook not found at expected position %c%d",
718 'a' + rook_col, king_row + 1);
719 return MOVE_ERROR_CASTLING_BLOCKED;
720 }
721
722 // Check if corresponding rook has moved
723 if (is_white) {
724 if (is_kingside && white_rook_h_moved) {
725 ESP_LOGD(TAG, "❌ Castling: kingside rook has moved");
726 return MOVE_ERROR_CASTLING_BLOCKED;
727 }
728 if (is_queenside && white_rook_a_moved) {
729 ESP_LOGD(TAG, "❌ Castling: queenside rook has moved");
730 return MOVE_ERROR_CASTLING_BLOCKED;
731 }
732 } else {
733 if (is_kingside && black_rook_h_moved) {
734 ESP_LOGD(TAG, "❌ Castling: kingside rook has moved");
735 return MOVE_ERROR_CASTLING_BLOCKED;
736 }
737 if (is_queenside && black_rook_a_moved) {
738 ESP_LOGD(TAG, "❌ Castling: queenside rook has moved");
739 return MOVE_ERROR_CASTLING_BLOCKED;
740 }
741 }
742
743 // Check if squares between king and rook are empty
744 // rook_col již definováno výše
745 int start_col =
746 (move->from_col < rook_col) ? move->from_col + 1 : rook_col + 1;
747 int end_col = (move->from_col < rook_col) ? rook_col : move->from_col;
748
749 for (int col = start_col; col < end_col; col++) {
750 if (!game_is_empty(king_row, col)) {
751 return MOVE_ERROR_CASTLING_BLOCKED;
752 }
753 }
754
755 // Check if king is currently in check
756 if (game_is_king_in_check(is_white ? PLAYER_WHITE : PLAYER_BLACK)) {
757 return MOVE_ERROR_CASTLING_BLOCKED;
758 }
759
760 // Check if king would move through check or end in check
761 int step = is_kingside ? 1 : -1;
762 for (int col = move->from_col; col != move->to_col + step; col += step) {
763 if (col == move->from_col)
764 continue; // Skip starting position
765
766 // Create temporary move to test this square
767 chess_move_t temp_move = {.from_row = move->from_row,
768 .from_col = move->from_col,
769 .to_row = move->to_row,
770 .to_col = col,
771 .piece = piece,
772 .captured_piece = PIECE_EMPTY,
773 .timestamp = 0};
774
775 if (game_would_move_leave_king_in_check(&temp_move)) {
776 return MOVE_ERROR_CASTLING_BLOCKED;
777 }
778 }
779
780 return MOVE_ERROR_NONE;
781}
