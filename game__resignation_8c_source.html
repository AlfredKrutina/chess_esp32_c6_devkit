33static void resignation_update_button_leds(uint32_t elapsed_ms) {
34 const uint32_t ORANGE_START_MS = 4000; // Barva na oranžovou až po 4 s
35 if (elapsed_ms < ORANGE_START_MS)
36 return;
37
38 const uint8_t WHITE_BASE = 64;
39 const uint8_t BLACK_BASE = 68;
40
41 const uint32_t FADE_START_MS = 5000;
42 const uint32_t FADE_DURATION_MS = 1000;
43 const uint32_t LED_OFFSET_MS = 1000;
44
45 for (int i = 0; i < 4; i++) {
46 uint32_t led_start_time = FADE_START_MS + (i * LED_OFFSET_MS);
47 uint8_t brightness = 255;
48
49 if (elapsed_ms >= led_start_time) {
50 uint32_t led_elapsed = elapsed_ms - led_start_time;
51 if (led_elapsed >= FADE_DURATION_MS) {
52 brightness = 0;
53 } else {
54 brightness = 255 - (led_elapsed * 255 / FADE_DURATION_MS);
55 }
56 }
57
58 uint8_t r = brightness;
59 uint8_t g = brightness / 2;
60 uint8_t b = 0;
61 button_set_led_color(WHITE_BASE + i, r, g, b);
62 button_set_led_color(BLACK_BASE + i, r, g, b);
63 }
64}

92static void resignation_finalize_timeout(void) {
93 if (!resignation_state.active) {
94 return;
95 }
96
97 const char *player_name =
98 (resignation_state.player == PLAYER_WHITE) ? "White" : "Black";
99 const char *winner_name =
100 (resignation_state.player == PLAYER_WHITE) ? "Black" : "White";
101
102 ESP_LOGI(TAG, "🏳️ %s resigned! %s wins!", player_name, winner_name);
103
104 // Ukončit hru
105 current_game_state = GAME_STATE_FINISHED;
106 game_result = GAME_STATE_FINISHED;
107 current_result_type = (resignation_state.player == PLAYER_WHITE)
108 ? RESULT_BLACK_WINS

110 current_endgame_reason = ENDGAME_REASON_RESIGNATION;
111 game_active = false;
112

114 game_update_endgame_statistics(current_result_type);
115 endgame_report_requested = true;
116
117 timer_pause();
118
119 // Spustit victory animaci pro vítěze
120 player_t winner_player =

122 game_trigger_victory_animation(winner_player);
123
124 // Timeout = finalized cleanup (no cancel messages, do not restore king)
125 resignation_stop(true);
126}

131void resignation_tick(void) {
132 if (!resignation_state.active) {
133 return;
134 }
135
136 uint64_t now_ms = esp_timer_get_time() / 1000;
137 uint32_t elapsed_ms = (uint32_t)(now_ms - resignation_state.start_time_ms);
138
139 // Update button LEDs (fade effect) in safe task context
140 resignation_update_button_leds(elapsed_ms);
141
142 // Finalize after 10 seconds
143 if (elapsed_ms >= 10000) {
144 resignation_finalize_timeout();
145 }

151void resignation_start(player_t player, uint8_t row, uint8_t col) {
152 // Zastavit existující timer pokud běží (restart = tichý cleanup)
153 if (resignation_state.active) {
154 resignation_stop(true);
155 }
156
157 const char *player_name = (player == PLAYER_WHITE) ? "White" : "Black";
158 printf("\r\n⚠️ %s king lifted - hold for 10+ seconds to resign!\r\n",
159 player_name);
160 printf("👑 Resignation timer started for %s player\r\n", player_name);
161 printf("🎨 Button LED animation started (8 buttons, both sides)\r\n\r\n");
162
163 // Inicializovat stav
164 resignation_state.active = true;
165 resignation_state.player = player;
166 resignation_state.king_row = row;
167 resignation_state.king_col = col;
168 resignation_state.start_time_ms = esp_timer_get_time() / 1000;
169 resignation_state.last_countdown_sec = 10;
170
171 // Odstranit krále z boardu (jako normální pickup)
172 // Jinak drop kód očekává prázdné pole a spadne
173 board[row][col] = PIECE_EMPTY;
174 ESP_LOGI(TAG, "✅ King removed from board[%d][%d] for resignation", row, col);
175
176 // Kombinace červené (varování) a žluté (source square)
177 // Oranžovo-červená mix místo jen červené
178 led_set_pixel_safe(chess_pos_to_led_index(row, col), 255, 128, 0);
179
180 // Tlačítka se na oranžovou změní až po 4 s (v resignation_update_button_leds)
181 // – prvních 4 s zůstávají v původní barvě
182
183 // PRODUCTION STABILITY:
184 // Do NOT run resignation logic from FreeRTOS timer callbacks ("Tmr Svc").
185 // Handling is done in `resignation_tick()` from the main game_task loop.
186 resignation_state.main_timer = NULL;
187 resignation_state.animation_timer = NULL;
188}

195void resignation_stop(bool finalize) {
196 if (!resignation_state.active)
197 return;
198
199 if (!finalize) {
200 printf("✅ King placed - resignation cancelled!\r\n");
201 printf("🛑 Resignation timer stopped\r\n");
202 printf("🎨 Button LED animation stopped\r\n\r\n");
203 } else {
204 ESP_LOGI(TAG,
205 "🔄 [STAGING] Resignation finalized (cleanup, no cancel msgs)");
206 }
207
208 // Pouze při zrušení uživatelem: vrátit krále na původní pozici
209 if (!finalize) {
210 piece_t king_piece = (resignation_state.player == PLAYER_WHITE)
211 ? PIECE_WHITE_KING
212 : PIECE_BLACK_KING;
213
214 if (board[resignation_state.king_row][resignation_state.king_col] ==
215 PIECE_EMPTY) {
216 board[resignation_state.king_row][resignation_state.king_col] =
217 king_piece;
218 ESP_LOGI(TAG,
219 "✅ King returned to original position [%d][%d] (resignation "
220 "cancelled)",
221 resignation_state.king_row, resignation_state.king_col);
222 } else {
223 ESP_LOGW(
224 TAG,
225 "⚠️ Original king position [%d][%d] is occupied - drop processing "
226 "will handle it",
227 resignation_state.king_row, resignation_state.king_col);
228 }
229
230 led_set_pixel_safe(chess_pos_to_led_index(resignation_state.king_row,
231 resignation_state.king_col),
232 0, 0, 0);
233 }
234
235 // Zastavit a smazat timery
236 if (resignation_state.main_timer) {
237 xTimerStop(resignation_state.main_timer, 0);
238 xTimerDelete(resignation_state.main_timer, 0);
239 resignation_state.main_timer = NULL;
240 }
241
242 if (resignation_state.animation_timer) {
243 xTimerStop(resignation_state.animation_timer, 0);
244 xTimerDelete(resignation_state.animation_timer, 0);
245 resignation_state.animation_timer = NULL;
246 }
247
248 if (finalize) {
249 // Při dokončení rezignace (timeout) vypnout button LEDs
250 for (int i = 0; i < 8; i++) {
251 button_set_led_color(64 + i, 0, 0, 0);
252 }
253 } else {
254 // Při zrušení (král položen zpět) obnovit normální stav tlačítek
255 // (zelená/modrá podle promoce, LED 72 zelená)
256 game_check_promotion_needed();
257 }
258
259 // Reset stavu
260 resignation_state.active = false;
261}
