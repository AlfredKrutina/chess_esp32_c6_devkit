32static int hex_nibble(char c) {
33 if (c >= '0' && c <= '9') {
34 return (int)(c - '0');

37 return (int)(c - 'a') + 10;

45static bool parse_hex32(const char *hex, uint8_t *out32) {
46 if (hex == NULL) {
47 return false;
48 }
49 size_t len = strlen(hex);
50 if (len != 64) {
51 return false;
52 }
53 for (size_t i = 0; i < 32; i++) {
54 int hi = hex_nibble(hex[i * 2]);
55 int lo = hex_nibble(hex[i * 2 + 1]);
56 if (hi < 0 || lo < 0) {
57 return false;
58 }
59 out32[i] = (uint8_t)((unsigned)hi << 4 | (unsigned)lo);
60 }
61 return true;
62}

75 esp_fill_random(s_token, sizeof(s_token));

77 if (e != ESP_OK) {
78 ESP_LOGE(TAG, "nvs_set_blob token: %s", esp_err_to_name(e));
79 nvs_close(h);
80 return e;
81 }
82 e = nvs_commit(h);
83 if (e != ESP_OK) {
84 ESP_LOGE(TAG, "nvs_commit: %s", esp_err_to_name(e));
85 nvs_close(h);
86 return e;
87 }
88#if CHESS_DEBUG_MODE
89 ESP_LOGW(TAG, "[STAGING] Generated new board API token (use UART API_TOKEN)");

91 } else if (e != ESP_OK) {
92 ESP_LOGE(TAG, "nvs_get_blob token: %s", esp_err_to_name(e));
93 nvs_close(h);
94 return e;
95 }

97 nvs_close(h);
98 s_loaded = true;
99 return ESP_OK;

131static bool bearer_matches_request(httpd_req_t *req) {
132 if (!s_loaded || req == NULL) {
133 return false;
134 }
135
136 char auth[180];
137 if (httpd_req_get_hdr_value_str(req, "Authorization", auth, sizeof(auth)) !=
138 ESP_OK) {
139 return false;
140 }
141
142 char *p = auth;
143 while (*p == ' ' || *p == '\t') {
144 p++;
145 }
146 if (strncasecmp(p, "Bearer ", 7) != 0) {
147 return false;
148 }
149 p += 7;
150 while (*p == ' ' || *p == '\t') {
151 p++;
152 }
153
154 uint8_t candidate[32];
155 if (!parse_hex32(p, candidate)) {
156 return false;
157 }
158 return ct_memeq(candidate, s_token, sizeof(s_token));
159}
