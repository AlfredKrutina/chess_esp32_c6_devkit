68static esp_err_t http_send_piece_blob(httpd_req_t *req) {
69 const piece_blob_t *b = (const piece_blob_t *)req->user_ctx;
70 if (b == NULL || b->start == NULL || b->end == NULL || b->end <= b->start) {
71 httpd_resp_set_status(req, "500 Internal Server Error");
72 return httpd_resp_send(req, "Bad asset", HTTPD_RESP_USE_STRLEN);
73 }
74 size_t len = (size_t)(b->end - b->start);
75 httpd_resp_set_type(req, "image/png");
76 httpd_resp_set_hdr(req, "Cache-Control", "public, max-age=604800");
77 return httpd_resp_send(req, b->start, len);
78}

135esp_err_t chess_piece_register_http_uris(httpd_handle_t hd) {
136 if (hd == NULL) {
137 return ESP_ERR_INVALID_ARG;
138 }
139 for (size_t i = 0; i < sizeof(s_piece_routes) / sizeof(s_piece_routes[0]); i++) {
140 httpd_uri_t u = {.uri = s_piece_routes[i].uri,
141 .method = HTTP_GET,
142 .handler = http_send_piece_blob,
143 .user_ctx = (void *)s_piece_routes[i].blob};
144 esp_err_t e = httpd_register_uri_handler(hd, &u);
145 if (e != ESP_OK) {
146 ESP_LOGE(TAG_PIECE, "register %s failed: %s", s_piece_routes[i].uri,
147 esp_err_to_name(e));
148 return e;
149 }
150 }
151 ESP_LOGI(TAG_PIECE, "registered %zu piece PNG URIs",
152 sizeof(s_piece_routes) / sizeof(s_piece_routes[0]));
153 return ESP_OK;
154}
