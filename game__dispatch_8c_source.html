39static void game_send_board_to_uart(QueueHandle_t response_queue) {
40 if (response_queue == NULL) {
41 ESP_LOGW(TAG, "No response queue available for board data");
42 return;
43 }
44
45 // Reset WDT before starting operation
46 game_task_wdt_reset_safe();
47
48 // MEMORY OPTIMIZATION: Use streaming output instead of malloc(3072)
49 // This eliminates heap fragmentation and reduces memory usage
50 ESP_LOGI(TAG, "📡 Using streaming output for board display (no malloc)");
51
52 // Configure streaming for queue output
53 esp_err_t ret = streaming_set_queue_output(response_queue);
54 if (ret != ESP_OK) {
55 ESP_LOGE(TAG, "Failed to configure streaming output: %s",
56 esp_err_to_name(ret));
57 return;
58 }
59
60 // MEMORY OPTIMIZATION: Use streaming output instead of large buffer
61
62 // Stream board header
63 stream_writeln(" a b c d e f g h");
64 stream_writeln(" +---+---+---+---+---+---+---+---+");
65
66 // Stream board rows
67 for (int row = 7; row >= 0; row--) {
68 // Reset watchdog every row to prevent timeouts
69 game_task_wdt_reset_safe();
70
71 stream_printf("%d |", row + 1);
72
73 for (int col = 0; col < 8; col++) {
74 piece_t piece = board[row][col];
75 const char *symbol;
76
77 switch (piece) {
78 case PIECE_WHITE_PAWN:
79 symbol = "P";
80 break;
81 case PIECE_WHITE_KNIGHT:
82 symbol = "N";
83 break;
84 case PIECE_WHITE_BISHOP:
85 symbol = "B";
86 break;
87 case PIECE_WHITE_ROOK:
88 symbol = "R";
89 break;
90 case PIECE_WHITE_QUEEN:
91 symbol = "Q";
92 break;
93 case PIECE_WHITE_KING:
94 symbol = "K";
95 break;
96 case PIECE_BLACK_PAWN:
97 symbol = "p";
98 break;
99 case PIECE_BLACK_KNIGHT:
100 symbol = "n";
101 break;
102 case PIECE_BLACK_BISHOP:
103 symbol = "b";
104 break;
105 case PIECE_BLACK_ROOK:
106 symbol = "r";
107 break;
108 case PIECE_BLACK_QUEEN:
109 symbol = "q";
110 break;
111 case PIECE_BLACK_KING:
112 symbol = "k";
113 break;
114 default:
115 symbol = "·";
116 break;
117 }
118
119 stream_printf(" %s |", symbol);
120 }
121
122 stream_printf(" %d\n", row + 1);
123
124 if (row > 0) {
125 stream_writeln(" +---+---+---+---+---+---+---+---+");
126 }
127 }
128
129 // Stream board footer
130 stream_writeln(" +---+---+---+---+---+---+---+---+");
131 stream_writeln(" a b c d e f g h");
132
133 // Stream game status
134 stream_printf("\nCurrent player: %s\n",
135 current_player == PLAYER_WHITE ? "White" : "Black");
136
137 stream_printf("Move count: %" PRIu32 "\n", move_count);
138
139 // Stream helpful tip
140 stream_writeln("💡 Use 'UP <square>' to lift piece, 'DN <square>' to place");
141
142 // MEMORY OPTIMIZATION: Streaming output handles data transmission
143 // Send completion message to response queue
144 game_response_t completion_response = {
145 .type = GAME_RESPONSE_SUCCESS,
146 .command_type = GAME_CMD_SHOW_BOARD,
147 .error_code = 0,
148 .message = "Board display streaming completed successfully",
149 .timestamp = esp_timer_get_time() / 1000};
150 strcpy(completion_response.data, "streaming completed");
151
152 if (response_queue != NULL) {
153 xQueueSend(response_queue, &completion_response, pdMS_TO_TICKS(100));
154 }
155
156 // No need for manual queue management or large buffers
157 ESP_LOGI(TAG, "✅ Board display streaming completed successfully");
158
159 // Final WDT reset
160 game_task_wdt_reset_safe();
161}

190static bool game_undo_last_move_impl(void) {
191 if (history_index == 0) {
192 ESP_LOGW(TAG, "Undo: no moves in history");
193 return false;
194 }
195
196 uint32_t idx = history_index - 1;
197 chess_move_t m = move_history[idx];
198 move_type_t kind = move_history_kind[idx];
199
200 bool mover_white =
201 (m.piece >= PIECE_WHITE_PAWN && m.piece <= PIECE_WHITE_KING);
202 uint8_t fr = m.from_row;
203 uint8_t fc = m.from_col;
204 uint8_t tr = m.to_row;
205 uint8_t tc = m.to_col;
206
207 bool toggle_player_after = true;
208 if (kind == MOVE_TYPE_CASTLE_KING || kind == MOVE_TYPE_CASTLE_QUEEN) {
209 /* Rošáda: hráč se nemění po tahu krále; až po položení věže. Undo dokončené
210 * rošády musí zase přepnout na protihráče → toggle true jen když věž už byla
211 * v logice přesunutá na „mezilehlé“ pole. */
212 uint8_t rook_castle_col =
213 (tc > fc) ? (uint8_t)(tc - 1) : (uint8_t)(tc + 1);
214 piece_t rook_piece =
215 (m.piece == PIECE_WHITE_KING) ? PIECE_WHITE_ROOK : PIECE_BLACK_ROOK;
216 toggle_player_after = (board[fr][rook_castle_col] == rook_piece);
217 } else if (kind == MOVE_TYPE_PROMOTION) {
218 piece_t at_dest = board[tr][tc];
219 if ((at_dest == PIECE_WHITE_PAWN || at_dest == PIECE_BLACK_PAWN) &&
220 promotion_state.pending) {
221 toggle_player_after = false;
222 }
223 }
224
225 switch (kind) {
226 case MOVE_TYPE_EN_PASSANT: {
227 board[tr][tc] = PIECE_EMPTY;
228 board[fr][fc] = m.piece;
229 piece_t victim_pawn =
230 mover_white ? PIECE_BLACK_PAWN : PIECE_WHITE_PAWN;
231 board[fr][tc] = victim_pawn;
232 break;
233 }
234 case MOVE_TYPE_PROMOTION: {
235 piece_t at_dest = board[tr][tc];
236 if ((at_dest == PIECE_WHITE_PAWN || at_dest == PIECE_BLACK_PAWN) &&
237 promotion_state.pending) {
238 if (xSemaphoreTakeRecursive(promotion_mutex, pdMS_TO_TICKS(100)) ==
239 pdTRUE) {
240 promotion_state.pending = false;
241 promotion_state.square_row = 0;
242 promotion_state.square_col = 0;
243 xSemaphoreGiveRecursive(promotion_mutex);
244 }
245 }
246 board[tr][tc] = m.captured_piece;
247 board[fr][fc] = m.piece;
248 break;
249 }
250 default:
251 board[tr][tc] = m.captured_piece;
252 board[fr][fc] = m.piece;
253 break;
254 }
255
256 if (kind == MOVE_TYPE_CASTLE_KING || kind == MOVE_TYPE_CASTLE_QUEEN) {
257 uint8_t rook_home_col = (tc > fc) ? 7u : 0u;
258 uint8_t rook_castle_col =
259 (tc > fc) ? (uint8_t)(tc - 1) : (uint8_t)(tc + 1);
260 piece_t rook_piece =
261 (m.piece == PIECE_WHITE_KING) ? PIECE_WHITE_ROOK : PIECE_BLACK_ROOK;
262 if (board[fr][rook_castle_col] == rook_piece) {
263 board[fr][rook_castle_col] = PIECE_EMPTY;
264 board[fr][rook_home_col] = rook_piece;
265 }
266 }
267
268 if (kind == MOVE_TYPE_CAPTURE) {
269 undo_pop_last_capture_list(m.captured_piece, mover_white);
270 }
271
272 if (kind == MOVE_TYPE_CASTLE_KING || kind == MOVE_TYPE_CASTLE_QUEEN) {
273 castling_state.in_progress = false;
274 }
275
276 history_index--;
277 if (move_count > 0) {
278 move_count--;
279 }
280 if (advantage_history_count > 0) {
281 advantage_history_count--;
282 }
283
284 if (mover_white) {
285 if (white_moves_count > 0) {
286 white_moves_count--;
287 }
288 } else {
289 if (black_moves_count > 0) {
290 black_moves_count--;
291 }
292 }
293
294 if (toggle_player_after) {
295 current_player = (current_player == PLAYER_WHITE) ? PLAYER_BLACK
296 : PLAYER_WHITE;
297 }
298
299 if (history_index > 0) {
300 chess_move_t *p = &move_history[history_index - 1];
301 last_move_from_row = p->from_row;
302 last_move_from_col = p->from_col;
303 last_move_to_row = p->to_row;
304 last_move_to_col = p->to_col;
305 has_last_move = true;
306 } else {
307 has_last_move = false;
308 }
309
310 if (current_game_state == GAME_STATE_FINISHED) {
311 current_game_state = GAME_STATE_ACTIVE;
312 game_active = true;
313 }
314
315 piece_lifted = false;
316 lifted_piece_row = 0;
317 lifted_piece_col = 0;
318 lifted_piece = PIECE_EMPTY;
319
320 game_end_timer_move();
321 game_start_timer_move(current_player == PLAYER_WHITE);
322
323 game_matrix_guard_check_resync_after_restore();
324 if (!game_is_matrix_guard_active()) {
325 led_clear_board_only();
326 chess_policy_highlight_movable_if_enabled();
327 } else {
328 STAGING_LOGI(TAG,
329 "undo: matrix guard active — highlight squares until physical "
330 "board matches logic (conflicts=%u)",
331 (unsigned)game_get_matrix_guard_conflict_count());
332 }
333
334 game_snapshot_persist_after_valid_move();
335 game_bump_revision_and_notify();
336
337 ESP_LOGI(TAG, "Undo applied (kind=%d), side to move: %s", (int)kind,
338 current_player == PLAYER_WHITE ? "White" : "Black");
339 return true;
340}
