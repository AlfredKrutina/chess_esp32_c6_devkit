35void game_handle_invalid_move_smart(const chess_move_t *move,
36 move_error_t error) {
37 ESP_LOGI(TAG, "🚫 INVALID MOVE - Starting smart recovery");
38
39 if (!move) {
40 ESP_LOGE(TAG, "❌ Critical error: NULL move pointer in error handling");
41 return;
42 }
43
44 // 1. Červené bliknutí chybného tahu
45 uint8_t from_led = chess_pos_to_led_index(move->from_row, move->from_col);
46 uint8_t to_led = chess_pos_to_led_index(move->to_row, move->to_col);
47
48 // Flash error - 5 rychlých červených bliknutí
49 for (int i = 0; i < 5; i++) {
50 led_set_pixel_safe(from_led, 255, 0, 0); // Red
51 led_set_pixel_safe(to_led, 255, 0, 0); // Red
52 vTaskDelay(pdMS_TO_TICKS(100));
53 led_clear_board_only();
54 vTaskDelay(pdMS_TO_TICKS(100));
55 }
56
57 // 2. Rosvítit zeleně zdrojové pole (kde je figurka)
58 led_set_pixel_safe(from_led, 0, 255, 0); // Green - kde vzít figurku
59
60 // 3. Rosvítit červeně kolem nevalidního cíle
61 game_highlight_invalid_target_area(move->to_row, move->to_col);
62
63 // 4. Rosvítit validní tahy pro tuto figurku
64 game_highlight_valid_moves_for_piece(move->from_row, move->from_col);
65
66 // 5. KRITICKÉ: Neměnit hráče, nechat ho opravit tah
67 ESP_LOGI(TAG, "💡 Please return piece to correct square. Valid moves are "
68 "highlighted.");
69
70 // 6. Nastavit internal stav "waiting for correction"
71 error_recovery_state.waiting_for_move_correction = true;
72 error_recovery_state.invalid_row = move->from_row;
73 error_recovery_state.invalid_col = move->from_col;
74}

127void game_handle_invalid_move(move_error_t error, const chess_move_t *move) {
128 ESP_LOGI(TAG, "🚨 ENTERING game_handle_invalid_move - error: %d", error);
129
130 // KRITICKÁ OPRAVA: Bezpečnostní kontrola move pointeru
131 if (!move) {
132 ESP_LOGE(TAG, "❌ Critical error: NULL move pointer in error handling");
133 return;
134 }
135
136 // Kontrola validity coordinates před použitím
137 if (move->from_row >= 8 || move->from_col >= 8 || move->to_row >= 8 ||
138 move->to_col >= 8) {
139 ESP_LOGE(TAG, "❌ Critical error: Invalid coordinates in move structure");
140 return;
141 }
142
143 ESP_LOGI(TAG, "🚨 Invalid move detected - smart error handling");
144 ESP_LOGI(TAG, " Move: %c%d -> %c%d", 'a' + move->from_col,
145 move->from_row + 1, 'a' + move->to_col, move->to_row + 1);
146
147 // Remote / zastaralý UI může poslat tah z prázdného pole. Simulace „figurka je
148 // fyzicky na to“ kopíruje board[from] na board[to]; když je from prázdné,
149 // smaže se figurka na cíli (logicky „zmizí“). HW recovery v tomto případě
150 // nedává smysl — desku neměnit.
151 piece_t src_piece = board[move->from_row][move->from_col];
152 if (src_piece == PIECE_EMPTY) {
153 ESP_LOGW(TAG,
154 "[STAGING] game_handle_invalid_move: empty source square — "
155 "skipping board mutation and LED recovery (error=%d)",
156 (int)error);
157 return;
158 }
159
160 // Chytrý error handling podle požadavků
161
162 // 1. Nastavit červené pole na neplatné pozici
163 error_recovery_state.has_invalid_piece = true;
164 error_recovery_state.invalid_row = move->to_row;
165 error_recovery_state.invalid_col = move->to_col;
166 error_recovery_state.piece_type = board[move->from_row][move->from_col];
167
168 // 2. Zapamatovat si původní validní pozici (odkud byl tah)
169 error_recovery_state.original_valid_row = move->from_row;
170 error_recovery_state.original_valid_col = move->from_col;
171
172 // 3. Přesunout figurku na neplatnou pozici (simulace HW reality)
173 board[move->to_row][move->to_col] = board[move->from_row][move->from_col];
174 board[move->from_row][move->from_col] = PIECE_EMPTY;
175
176 // 4. JASNÉ VIZUÁLNÍ UPOZORNĚNÍ - červené pole + blikání pro upoutání
177 // pozornosti
178 led_clear_all_safe();
179 uint8_t invalid_led = chess_pos_to_led_index(move->to_row, move->to_col);
180
181 ESP_LOGI(TAG, "🚨 STARTING ERROR ANIMATION - RED BLINKING");
182
183 // Blikání pro upoutání pozornosti (krátké, non-blocking)
184 for (int blink = 0; blink < 4; blink++) {
185 led_set_pixel_safe(invalid_led, 255, 0, 0); // Červená
186 vTaskDelay(pdMS_TO_TICKS(150));
187 led_set_pixel_safe(invalid_led, 0, 0, 0); // Zhasnout
188 vTaskDelay(pdMS_TO_TICKS(150));
189 }
190
191 // Nechat červené pole trvale rozsvícené
192 led_set_pixel_safe(invalid_led, 255, 0, 0);
193
194 ESP_LOGI(TAG, "🔴 RED SQUARE ACTIVE: %c%d (piece must be lifted from here)",
195 'a' + move->to_col, move->to_row + 1);
196 ESP_LOGI(TAG,
197 "📍 RECOVERY TARGET: Valid moves will show from %c%d when piece is "
198 "lifted",
199 'a' + move->from_col, move->from_row + 1);
200 ESP_LOGI(TAG,
201 "💡 USER ACTION REQUIRED: Lift piece from red square to continue");
202
203 // KRITICKÁ OPRAVA: NON-BLOCKING error handling - žádné loops!
204 // Červené pole zůstane rozsvícené dokud se figurka nezvedne
205 // Recovery se řeší v game_process_pickup_command()
206
207 ESP_LOGI(TAG, "💡 Error recovery active - red square will stay lit until "
208 "piece is lifted");
209}

215void game_process_move_command(const void *move_cmd_ptr) {
216 typedef struct {
217 uint8_t command_type;
218 uint8_t from_row;
219 uint8_t from_col;
220 uint8_t to_row;
221 uint8_t to_col;
222 } move_command_t;
223
224 const move_command_t *move_cmd = (const move_command_t *)move_cmd_ptr;
225
226 ESP_LOGI(TAG, "Processing move: [%d,%d] -> [%d,%d]", move_cmd->from_row,
227 move_cmd->from_col, move_cmd->to_row, move_cmd->to_col);
228
229 // Validate coordinates
230 if (move_cmd->from_row >= 8 || move_cmd->from_col >= 8 ||
231 move_cmd->to_row >= 8 || move_cmd->to_col >= 8) {
232 ESP_LOGE(TAG, "Invalid coordinates: out of board range");
233 return;
234 }
235
236 // Get pieces at source and destination
237 piece_t from_piece = board[move_cmd->from_row][move_cmd->from_col];
238 piece_t to_piece = board[move_cmd->to_row][move_cmd->to_col];
239
240 // Check if source square has a piece
241 if (from_piece == PIECE_EMPTY) {
242 ESP_LOGE(TAG, "Invalid move: no piece at [%d,%d]", move_cmd->from_row,
243 move_cmd->from_col);
244 return;
245 }
246
247 // Check if piece belongs to current player
248 bool is_white_piece =
249 (from_piece >= PIECE_WHITE_PAWN && from_piece <= PIECE_WHITE_KING);
250 bool is_black_piece =
251 (from_piece >= PIECE_BLACK_PAWN && from_piece <= PIECE_BLACK_KING);
252
253 if ((current_player == PLAYER_WHITE && !is_white_piece) ||
254 (current_player == PLAYER_BLACK && !is_black_piece)) {
255 ESP_LOGE(TAG, "Invalid move: cannot move opponent's piece");
256 return;
257 }
258
259 // Check if destination is occupied by own piece
260 if (to_piece != PIECE_EMPTY) {
261 bool dest_is_white =
262 (to_piece >= PIECE_WHITE_PAWN && to_piece <= PIECE_WHITE_KING);
263 bool dest_is_black =
264 (to_piece >= PIECE_BLACK_PAWN && to_piece <= PIECE_BLACK_KING);
265
266 if ((current_player == PLAYER_WHITE && dest_is_white) ||
267 (current_player == PLAYER_BLACK && dest_is_black)) {
268 ESP_LOGE(TAG, "Invalid move: destination occupied by own piece");
269 return;
270 }
271 }
272
273 // TODO: Add proper chess move validation (piece-specific rules)
274 // For now, just allow any move that passes basic checks
275
276 // Create chess move structure for validation
277 chess_move_t chess_move = {.from_row = move_cmd->from_row,
278 .from_col = move_cmd->from_col,
279 .to_row = move_cmd->to_row,
280 .to_col = move_cmd->to_col,
281 .piece = from_piece,
282 .captured_piece = to_piece,
283 .timestamp = esp_timer_get_time() / 1000};
284
285 // Validate move using enhanced validation
286 move_error_t move_error = game_is_valid_move(&chess_move);
287 if (move_error != MOVE_ERROR_NONE) {
288 // NEW ERROR HANDLING: Don't return, handle invalid move properly
289 game_handle_invalid_move(move_error, &chess_move);
290 return;
291 }
292
293 // Use game_execute_move() instead of direct board
294 // manipulation This ensures all game state is properly updated (castling
295 // flags, en passant, etc.) and player is changed correctly
296 ESP_LOGI(TAG, "Executing move: %s piece from [%d,%d] to [%d,%d]",
297 (is_white_piece ? "White" : "Black"), move_cmd->from_row,
298 move_cmd->from_col, move_cmd->to_row, move_cmd->to_col);
299
300 // Execute the move using proper game_execute_move() function
301 if (!game_execute_move(&chess_move)) {
302 ESP_LOGE(TAG, "Failed to execute move from [%d,%d] to [%d,%d]",
303 move_cmd->from_row, move_cmd->from_col, move_cmd->to_row,
304 move_cmd->to_col);
305 // Note: move_cmd doesn't have response_queue, this is legacy function
306 return;
307 }
308
309 // HRÁČ SE UŽ ZMĚNIL V game_execute_move() - použít aktuálního hráče
310 player_t previous_player =
311 (current_player == PLAYER_WHITE) ? PLAYER_BLACK : PLAYER_WHITE;
312
313 // Spustit move path animaci PŘED změnou hráče (podle starého projektu)
314 uint8_t from_led =
315 chess_pos_to_led_index(move_cmd->from_row, move_cmd->from_col);
316 uint8_t to_led = chess_pos_to_led_index(move_cmd->to_row, move_cmd->to_col);
317 led_command_t move_path_cmd = {
318 .type = LED_CMD_ANIM_MOVE_PATH,
319 .led_index = from_led,
320 .red = 255,
321 .green = 255,
322 .blue = 0, // Yellow
323 .duration_ms = 1000,
324 .data = &to_led // Cílová pozice v data
325 };
326 led_execute_command_new(&move_path_cmd);
327
328 // STABILITY FIX: Animace běží asynchronně, neblokujeme zpracování
329 // Animace jsou spuštěny v led_task a běží nezávisle
330 // (vTaskDelay odstraněn pro lepší throughput)
331
332 // KRITICKÉ: Endgame kontrola PŘED player change animací!
333 // Pokud je endgame, player change se NESPOUŠTÍ
334 game_state_t end_game_result = game_check_end_game_conditions();
335 if (end_game_result == GAME_STATE_FINISHED) {
336 current_game_state = GAME_STATE_FINISHED;
337 game_active = false;
338
339 // Najít pozici krále vítěze pro endgame animaci
340 player_t winner =
341 previous_player; // Vítěz je předchozí hráč (ten, kdo udělal tah)
342 uint8_t king_pos = 28; // default e4
343 for (int i = 0; i < 64; i++) {
344 piece_t piece = board[i / 8][i % 8];
345 if ((winner == PLAYER_WHITE && piece == PIECE_WHITE_KING) ||
346 (winner == PLAYER_BLACK && piece == PIECE_BLACK_KING)) {
347 king_pos = i;
348 break;
349 }
350 }
351
352 ESP_LOGI(TAG,
353 "🎯 Game finished in move command! Starting endgame animation at "
354 "position %d",
355 king_pos);
356
357 // Spustit endgame animaci (wave z krále vítěze)
358 led_command_t endgame_cmd = {.type = LED_CMD_ANIM_ENDGAME,
359 .led_index = king_pos,
360 .red = 255,
361 .green = 255,
362 .blue = 0, // Yellow
363 .duration_ms = 0, // Endless
364 .data = NULL};
365 led_execute_command_new(&endgame_cmd);
366
367 ESP_LOGI(TAG,
368 "✅ Endgame animation started - player change animation SKIPPED");
369 } else {
370 // Není endgame - spustit player change animaci s NOVÝM hráčem
371 // (current_player)
372 uint8_t player_color =
373 (current_player == PLAYER_WHITE) ? 1 : 0; // 1=white, 0=black
374 led_command_t player_change_cmd = {
375 .type = LED_CMD_ANIM_PLAYER_CHANGE, // Použít
376 // ANIM_PLAYER_CHANGE místo
377 // PLAYER_CHANGE
378 .led_index = 0,
379 .red = 0,
380 .green = 0,
381 .blue = 0,
382 .duration_ms = 0,
383 .data = &player_color // Předat barvu NOVÉHO hráče
384 };
385 led_execute_command_new(&player_change_cmd);
386
387 // Zkontrolovat, zda je nový hráč v šachu
388 bool in_check = game_is_king_in_check(current_player);
389 if (in_check) {
390 // Najít pozici krále
391 int king_row = -1, king_col = -1;
392 piece_t king_piece = (current_player == PLAYER_WHITE) ? PIECE_WHITE_KING
393 : PIECE_BLACK_KING;
394 for (int row = 0; row < 8; row++) {
395 for (int col = 0; col < 8; col++) {
396 if (board[row][col] == king_piece) {
397 king_row = row;
398 king_col = col;
399 break;
400 }
401 }
402 if (king_row != -1)
403 break;
404 }
405
406 if (king_row != -1 && king_col != -1) {
407 if (game_led_guidance_show_check_anim()) {
408 uint8_t king_led_index = chess_pos_to_led_index(king_row, king_col);
409 // Spustit check animaci - růžové svícení na králi
410 led_command_t check_cmd = {.type = LED_CMD_ANIM_CHECK,
411 .led_index = king_led_index,
412 .red = 0,
413 .green = 0,
414 .blue = 0,
415 .duration_ms =
416 0, // Trvalé až do dalšího tahu
417 .data = NULL};
418 led_execute_command_new(&check_cmd);
419 ESP_LOGI(TAG, "⚠️ CHECK! %s is in check",
420 current_player == PLAYER_WHITE ? "White" : "Black");
421 }
422 }
423 }
424 }
425
426 // History and last_move tracking is already done in
427 // game_execute_move() - no need to duplicate here
428
429 // Track captured pieces
430 if (to_piece != PIECE_EMPTY) {
431 if (current_player == PLAYER_WHITE) {
432 // Black piece was captured by white
433 if (black_captured_index < 16) {
434 black_captured_pieces[black_captured_index++] = to_piece;
435 black_captured_count++;
436 }
437 } else {
438 // White piece was captured by black
439 if (white_captured_index < 16) {
440 white_captured_pieces[white_captured_index++] = to_piece;
441 white_captured_count++;
442 }
443 }
444 moves_without_capture = 0; // Reset counter on capture
445 } else {
446 moves_without_capture++; // Increment counter for non-capture moves
447 if (moves_without_capture > max_moves_without_capture) {
448 max_moves_without_capture = moves_without_capture;
449 }
450 }
451
452 // Update time statistics
453 uint32_t current_time = esp_timer_get_time() / 1000;
454 uint32_t move_time = current_time - last_move_time;
455
456 if (current_player == PLAYER_WHITE) {
457 // Black just moved
458 black_time_total += move_time;
459 black_moves_count++;
460 } else {
461 // White just moved
462 white_time_total += move_time;
463 white_moves_count++;
464 }
465
466 // Update position history for draw detection
467 game_add_position_to_history();
468
469 // Note: End-game conditions already checked earlier in this function (line
470 // 6612) If game finished, it was already handled with endgame animation
471
472 // Check if king is in check
473 if (game_is_king_in_check(current_player)) {
474 if (current_player == PLAYER_WHITE) {
475 white_checks++;
476 } else {
477 black_checks++;
478 }
479 ESP_LOGI(TAG, "⚠️ CHECK! %s king is under attack!",
480 (current_player == PLAYER_WHITE) ? "White" : "Black");
481 }
482
483 // Update last move time for next move
484 last_move_time = current_time;
485
486 // Display updated board (check auto-display setting)
487 // Note: Auto-display control is in UART task, board is always shown for now
488 game_print_board();
489
490 // Check for check/checkmate conditions
491 game_check_game_conditions();
492}

561 // ED: Find all pieces first, then animate all columns simultaneously

563 -1, -1, -1, -1}; // Row positions for each column
564 int curr_pieces[8] = {-1, -1, -1, -1,
565 -1, -1, -1, -1}; // Row positions for each column
566
567 // Find closest pieces for all columns
568 for (int col = 0; col < 8; col++) {
569 // Find closest piece of previous player (closest to opponent's side)

579 } else {

584 prev_pieces[col] = row;
585 break;
586 }
587 }
588 }
589
590 // Find closest piece of current player (closest to previous player's
591 // side)
592 if (current_player == PLAYER_WHITE) {
593 // Look from bottom (row 0) up to find white piece closest to black side
594 for (int row = 0; row < 8; row++) {
595 piece_t piece = board[row][col];
596 if (piece >= PIECE_WHITE_PAWN && piece <= PIECE_WHITE_KING) {
597 curr_pieces[col] = row;
598 break;
599 }
600 }
601 } else {
602 // Look from top (row 7) down to find black piece closest to white side
603 for (int row = 7; row >= 0; row--) {
604 piece_t piece = board[row][col];
605 if (piece >= PIECE_BLACK_PAWN && piece <= PIECE_BLACK_KING) {
606 curr_pieces[col] = row;
607 break;
608 }
609 }
610 }
611 }
612
613 // ENHANCED: Ultra-smooth player change animation with harmonious colors
