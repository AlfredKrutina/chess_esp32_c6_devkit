23static uint8_t hall_map_segment_field_to_square(unsigned seg, unsigned field) {
24 unsigned lr = field / 4;
25 unsigned lc = field % 4;
26 switch (seg) {
27 case 0:
28 return (uint8_t)(lr * 8 + lc);
29 case 1:
30 return (uint8_t)(lr * 8 + (4 + lc));
31 case 2:
32 return (uint8_t)((4 + lr) * 8 + (4 + lc));
33 case 3:
34 return (uint8_t)((4 + lr) * 8 + lc);
35 default:
36 return 0;
37 }
38}

40static bool hall_pair_occupied(uint16_t r0, uint16_t r1) {
41#if CONFIG_CHESS_HALL_DETECT_DIFF
42 int d = (int)r0 - (int)r1;
43 if (d < 0) {
44 d = -d;
45 }
46 return d >= CONFIG_CHESS_HALL_PAIR_DIFF_THRESHOLD;
47#else
48 uint16_t m = r0 > r1 ? r0 : r1;
49 return m >= CONFIG_CHESS_HALL_MAX_THRESHOLD;
50#endif
51}

103void hall_i2c_matrix_fill_state(uint8_t matrix_state[64]) {
104 if (!s_i2c_ready || matrix_state == NULL) {
105 return;
106 }
107
108 i2c_port_t port = (i2c_port_t)CONFIG_CHESS_HALL_I2C_PORT_NUM;
109 uint8_t reg = (uint8_t)CONFIG_CHESS_HALL_REG_START;
110 uint8_t buf[HALL_I2C_PAYLOAD_BYTES];
111 static uint32_t s_scan;
112
113 s_scan++;
114
115#if CONFIG_CHESS_HALL_LOG_INTERVAL_SCANS > 0
116 bool do_log = (s_scan % (uint32_t)CONFIG_CHESS_HALL_LOG_INTERVAL_SCANS) == 1;
117#else
118 bool do_log = false;
119#endif
120
121 /* Po přidání Kconfig položky může starý sdkconfig makro nemít — výchozí 4 segmenty. */
122#if defined(CONFIG_CHESS_HALL_SEGMENT_COUNT)
123 unsigned seg_max = (unsigned)CONFIG_CHESS_HALL_SEGMENT_COUNT;
124#else
125 unsigned seg_max = 4u;
126#endif
127 if (seg_max > 4u) {
128 seg_max = 4u;
129 }
130 if (seg_max < 1u) {
131 seg_max = 1u;
132 }
133
134 for (unsigned seg = seg_max; seg < 4u; seg++) {
135 for (unsigned field = 0; field < HALL_I2C_FIELDS_PER_SEGMENT; field++) {
136 uint8_t sq = hall_map_segment_field_to_square(seg, field);
137 matrix_state[sq] = 0;
138 }
139 }
140
141 for (unsigned seg = 0; seg < seg_max; seg++) {
142 uint8_t addr = segment_addr(seg);
143 esp_err_t err = i2c_master_write_read_device(
144 port, addr, &reg, 1, buf, sizeof(buf), pdMS_TO_TICKS(80));
145
146 if (err != ESP_OK) {
147 uint8_t bit = (uint8_t)(1u << seg);
148 if ((s_read_fail_warned_mask & bit) == 0) {
149 ESP_LOGW(TAG,
150 "segment %u addr 0x%02x read failed: %s "
151 "(další selhání stejného segmentu jen DEBUG)",
152 seg, addr, esp_err_to_name(err));
153 s_read_fail_warned_mask |= bit;
154 } else {
155 ESP_LOGD(TAG, "segment %u addr 0x%02x read failed: %s", seg, addr,
156 esp_err_to_name(err));
157 }
158 continue;
159 }
160
161 s_read_fail_warned_mask &= (uint8_t)~((uint8_t)(1u << seg));
162
163 for (unsigned field = 0; field < HALL_I2C_FIELDS_PER_SEGMENT; field++) {
164 unsigned off = field * sizeof(uint16_t) * HALL_I2C_SENSORS_PER_FIELD;
165 uint16_t r0 = hall_i2c_spec_read_le16(&buf[off]);
166 uint16_t r1 = hall_i2c_spec_read_le16(&buf[off + sizeof(uint16_t)]);
167 uint8_t sq = hall_map_segment_field_to_square(seg, field);
168 matrix_state[sq] = hall_pair_occupied(r0, r1) ? 1 : 0;
169 }
170
171 if (do_log) {
172 ESP_LOGI(TAG,
173 "[staging] seg%u addr=0x%02x first_pair raw=(%u,%u) occ=%u",
174 seg, addr, hall_i2c_spec_read_le16(&buf[0]),
175 hall_i2c_spec_read_le16(&buf[sizeof(uint16_t)]),
176 (unsigned)matrix_state[hall_map_segment_field_to_square(seg, 0)]);
177 }
178 }
179}
