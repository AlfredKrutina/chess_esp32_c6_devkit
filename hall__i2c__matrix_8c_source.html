38static uint8_t hall_map_segment_field_to_square(unsigned seg, unsigned field) {
39 unsigned lr = field / 4;
40 unsigned lc = field % 4;
41 switch (seg) {
42 case 0:
43 return (uint8_t)(lr * 8 + lc);
44 case 1:
45 return (uint8_t)(lr * 8 + (4 + lc));
46 case 2:
47 return (uint8_t)((4 + lr) * 8 + (4 + lc));
48 case 3:
49 return (uint8_t)((4 + lr) * 8 + lc);
50 default:
51 return 0;
52 }
53}

55static bool hall_pair_occupied(uint16_t r0, uint16_t r1) {
56#if CONFIG_CHESS_HALL_DETECT_DIFF
57 int d = (int)r0 - (int)r1;
58 if (d < 0) {
59 d = -d;
60 }
61 return d >= CONFIG_CHESS_HALL_PAIR_DIFF_THRESHOLD;
62#else
63 uint16_t m = r0 > r1 ? r0 : r1;
64 return m >= CONFIG_CHESS_HALL_MAX_THRESHOLD;
65#endif
66}

120void hall_i2c_matrix_fill_state(uint8_t matrix_state[64]) {
121 if (!s_i2c_ready || matrix_state == NULL) {
122 return;
123 }
124
125 i2c_port_t port = (i2c_port_t)CONFIG_CHESS_HALL_I2C_PORT_NUM;
126 uint8_t reg = (uint8_t)CONFIG_CHESS_HALL_REG_START;
127 uint8_t buf[HALL_I2C_PAYLOAD_BYTES];
128 static uint32_t s_scan;
129
130 s_scan++;
131
132#if CONFIG_CHESS_HALL_LOG_INTERVAL_SCANS > 0
133 bool do_log = (s_scan % (uint32_t)CONFIG_CHESS_HALL_LOG_INTERVAL_SCANS) == 1;
134#else
135 bool do_log = false;
136#endif
137
138 /* Po přidání Kconfig položky může starý sdkconfig makro nemít — výchozí 4 segmenty. */
139#if defined(CONFIG_CHESS_HALL_SEGMENT_COUNT)
140 unsigned seg_max = (unsigned)CONFIG_CHESS_HALL_SEGMENT_COUNT;
141#else
142 unsigned seg_max = 4u;
143#endif
144 if (seg_max > 4u) {
145 seg_max = 4u;
146 }
147 if (seg_max < 1u) {
148 seg_max = 1u;
149 }
150
151 for (unsigned seg = seg_max; seg < 4u; seg++) {
152 for (unsigned field = 0; field < HALL_I2C_FIELDS_PER_SEGMENT; field++) {
153 uint8_t sq = hall_map_segment_field_to_square(seg, field);
154 matrix_state[sq] = 0;
155 }
156 }
157
158 for (unsigned seg = 0; seg < seg_max; seg++) {
159 uint8_t addr = segment_addr(seg);
160 esp_err_t err = i2c_master_write_read_device(
161 port, addr, &reg, 1, buf, sizeof(buf), pdMS_TO_TICKS(80));
162
163 if (err != ESP_OK) {
164 uint8_t bit = (uint8_t)(1u << seg);
165 if ((s_read_fail_warned_mask & bit) == 0) {
166 ESP_LOGW(TAG,
167 "segment %u addr 0x%02x read failed: %s "
168 "(další selhání stejného segmentu jen DEBUG)",
169 seg, addr, esp_err_to_name(err));
170 s_read_fail_warned_mask |= bit;
171 } else {
172 ESP_LOGD(TAG, "segment %u addr 0x%02x read failed: %s", seg, addr,
173 esp_err_to_name(err));
174 }
175 continue;
176 }
177
178 s_read_fail_warned_mask &= (uint8_t)~((uint8_t)(1u << seg));
179
180 for (unsigned field = 0; field < HALL_I2C_FIELDS_PER_SEGMENT; field++) {
181 unsigned off = field * sizeof(uint16_t) * HALL_I2C_SENSORS_PER_FIELD;
182 uint16_t r0 = hall_i2c_spec_read_le16(&buf[off]);
183 uint16_t r1 = hall_i2c_spec_read_le16(&buf[off + sizeof(uint16_t)]);
184 uint8_t sq = hall_map_segment_field_to_square(seg, field);
185 matrix_state[sq] = hall_pair_occupied(r0, r1) ? 1 : 0;
186 }
187
188 if (do_log) {
189 ESP_LOGI(TAG,
190 "[staging] seg%u addr=0x%02x first_pair raw=(%u,%u) occ=%u",
191 seg, addr, hall_i2c_spec_read_le16(&buf[0]),
192 hall_i2c_spec_read_le16(&buf[sizeof(uint16_t)]),
193 (unsigned)matrix_state[hall_map_segment_field_to_square(seg, 0)]);
194 }
195 }
196}
