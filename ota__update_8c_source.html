58static void ota_worker_task(void *arg) {
59 char *url = (char *)arg;
60 if (url == NULL) {
61 xSemaphoreGive(s_ota_sem);
62 vTaskDelete(NULL);
63 return;
64 }
65
66 s_last_err[0] = '\0';
67 s_state = OTA_UI_DOWNLOADING;
68 s_percent = 0;
69
70 esp_http_client_config_t http_cfg = {
71 .url = url,
72 .timeout_ms = 120000,
73 .crt_bundle_attach = esp_crt_bundle_attach,
74 .keep_alive_enable = true,
75 };
76
77 esp_https_ota_config_t ota_cfg = {
78 .http_config = &http_cfg,
79 };
80
81#if CHESS_DEBUG_MODE
82 ESP_LOGI(TAG, "[STAGING] OTA start url=%s", url);
83#endif
84
85 esp_err_t err = esp_https_ota(&ota_cfg);
86 free(url);
87
88 if (err == ESP_OK) {
89 s_percent = 100;
90 s_state = OTA_UI_DONE;
91#if CHESS_DEBUG_MODE
92 ESP_LOGI(TAG, "[STAGING] OTA OK, restart");
93#endif
94 vTaskDelay(pdMS_TO_TICKS(500));
95 esp_restart();
96 }
97
98 s_state = OTA_UI_ERROR;
99 snprintf(s_last_err, sizeof(s_last_err), "%s", esp_err_to_name(err));
100 ESP_LOGE(TAG, "OTA failed: %s", esp_err_to_name(err));
101 xSemaphoreGive(s_ota_sem);
102 vTaskDelete(NULL);
103}

105static esp_err_t schedule_ota(const char *url) {
106 if (!wifi_is_sta_connected()) {
107 return ESP_ERR_INVALID_STATE;
108 }
109 if (!url_ok_https(url)) {
110 return ESP_ERR_INVALID_ARG;
111 }
112 if (xSemaphoreTake(s_ota_sem, 0) != pdTRUE) {
113 return ESP_ERR_INVALID_STATE;
114 }
115
116 size_t len = strlen(url);
117 char *copy = (char *)malloc(len + 1);
118 if (copy == NULL) {
119 xSemaphoreGive(s_ota_sem);
120 return ESP_ERR_NO_MEM;
121 }
122 memcpy(copy, url, len + 1);
123
124 BaseType_t ok =
125 xTaskCreate(ota_worker_task, "ota_https", 12288, copy, 5, NULL);
126 if (ok != pdPASS) {
127 free(copy);
128 xSemaphoreGive(s_ota_sem);
129 return ESP_FAIL;
130 }
131 return ESP_OK;
132}

171static esp_err_t http_post_ota(httpd_req_t *req) {
172 char body[520];
173 int rlen = httpd_req_recv(req, body, sizeof(body) - 1);
174 if (rlen <= 0) {
175 httpd_resp_set_status(req, "400 Bad Request");
176 httpd_resp_set_type(req, "application/json");
177 httpd_resp_send(req, "{\"ok\":false,\"error\":\"empty_body\"}", -1);
178 return ESP_OK;
179 }
180 body[rlen] = '\0';
181
182 cJSON *root = cJSON_Parse(body);
183 if (root == NULL) {
184 httpd_resp_set_status(req, "400 Bad Request");
185 httpd_resp_set_type(req, "application/json");
186 httpd_resp_send(req, "{\"ok\":false,\"error\":\"bad_json\"}", -1);
187 return ESP_OK;
188 }
189 cJSON *ju = cJSON_GetObjectItemCaseSensitive(root, "url");
190 const char *url =
191 (cJSON_IsString(ju) && ju->valuestring != NULL) ? ju->valuestring : NULL;
192 esp_err_t q = schedule_ota(url);
193 cJSON_Delete(root);
194
195 httpd_resp_set_type(req, "application/json");
196 if (q == ESP_OK) {
197 httpd_resp_set_status(req, "202 Accepted");
198 httpd_resp_send(req, "{\"ok\":true,\"accepted\":true}", -1);
199 return ESP_OK;
200 }
201 if (q == ESP_ERR_INVALID_STATE) {
202 httpd_resp_set_status(req, "409 Conflict");
203 httpd_resp_send(req,
204 "{\"ok\":false,\"error\":\"busy_or_no_wifi_sta\"}", -1);
205 return ESP_OK;
206 }
207 if (q == ESP_ERR_INVALID_ARG) {
208 httpd_resp_set_status(req, "400 Bad Request");
209 httpd_resp_send(req, "{\"ok\":false,\"error\":\"need_https_url\"}", -1);
210 return ESP_OK;
211 }
212 httpd_resp_set_status(req, "500 Internal Server Error");
213 httpd_resp_send(req, "{\"ok\":false,\"error\":\"ota_queue_failed\"}", -1);
214 return ESP_OK;
215}
