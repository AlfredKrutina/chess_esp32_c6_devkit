26bool game_is_king_in_check(player_t player) {
27 // Find king position
28 int king_row = -1, king_col = -1;
29 piece_t king_piece =
30 (player == PLAYER_WHITE) ? PIECE_WHITE_KING : PIECE_BLACK_KING;
31
32 for (int row = 0; row < 8; row++) {
33 for (int col = 0; col < 8; col++) {
34 if (board[row][col] == king_piece) {
35 king_row = row;
36 king_col = col;
37 break;
38 }
39 }
40 if (king_row != -1)
41 break;
42 }
43
44 if (king_row == -1)
45 return false; // King not found
46
47 // Check if any opponent piece can capture the king
48 piece_t opponent_pieces_start =
49 (player == PLAYER_WHITE) ? PIECE_BLACK_PAWN : PIECE_WHITE_PAWN;
50 piece_t opponent_pieces_end =
51 (player == PLAYER_WHITE) ? PIECE_BLACK_KING : PIECE_WHITE_KING;
52
53 for (int row = 0; row < 8; row++) {
54 for (int col = 0; col < 8; col++) {
55 piece_t piece = board[row][col];
56 if (piece >= opponent_pieces_start && piece <= opponent_pieces_end) {
57 // Create temporary move to check if it can capture king
58 chess_move_t temp_move = {.from_row = row,
59 .from_col = col,
60 .to_row = king_row,
61 .to_col = king_col,
62 .piece = piece,
63 .captured_piece = king_piece,
64 .timestamp = 0};
65
66 // Check if this move is valid (would capture king)
67 if (game_validate_piece_move_enhanced(&temp_move, piece) ==
68 MOVE_ERROR_NONE) {
69 return true; // King is in check
70 }
71 }
72 }
73 }
74
75 return false;
76}

104bool game_is_insufficient_material(void) {
105 int white_pieces = 0, black_pieces = 0;
106 int white_minors = 0, black_minors = 0; // Knights and bishops
107 bool white_has_bishop = false, black_has_bishop = false;
108 bool white_has_knight = false, black_has_knight = false;
109 bool white_bishop_color = false,
110 black_bishop_color = false; // false = light, true = dark
111
112 // Count pieces and analyze material
113 for (int row = 0; row < 8; row++) {
114 for (int col = 0; col < 8; col++) {
115 piece_t piece = board[row][col];
116
117 switch (piece) {
118 case PIECE_WHITE_PAWN:
119 case PIECE_WHITE_ROOK:
120 case PIECE_WHITE_QUEEN:
121 white_pieces++;
122 break;
123 case PIECE_WHITE_BISHOP:
124 white_pieces++;
125 white_minors++;
126 white_has_bishop = true;
127 white_bishop_color = ((row + col) % 2 == 0); // Light squares = false
128 break;
129 case PIECE_WHITE_KNIGHT:
130 white_pieces++;
131 white_minors++;
132 white_has_knight = true;
133 break;
134 case PIECE_BLACK_PAWN:
135 case PIECE_BLACK_ROOK:
136 case PIECE_BLACK_QUEEN:
137 black_pieces++;
138 break;
139 case PIECE_BLACK_BISHOP:
140 black_pieces++;
141 black_minors++;
142 black_has_bishop = true;
143 black_bishop_color = ((row + col) % 2 == 0); // Light squares = false
144 break;
145 case PIECE_BLACK_KNIGHT:
146 black_pieces++;
147 black_minors++;
148 black_has_knight = true;
149 break;
150 default:
151 break;
152 }
153 }
154 }
155
156 // King vs King - always insufficient
157 if (white_pieces == 0 && black_pieces == 0) {
158 return true;
159 }
160
161 // King + minor vs King - insufficient
162 if ((white_pieces == 1 && black_pieces == 0) ||
163 (white_pieces == 0 && black_pieces == 1)) {
164 return true;
165 }
166
167 // King + Bishop vs King + Bishop (same color) - insufficient
168 if (white_pieces == 1 && black_pieces == 1 && white_has_bishop &&
169 black_has_bishop && white_bishop_color == black_bishop_color) {
170 return true;
171 }
172
173 // King + Knight vs King + Knight - insufficient
174 if (white_pieces == 1 && black_pieces == 1 && white_has_knight &&
175 black_has_knight) {
176 return true;
177 }
178
179 // King + 2 Knights vs King - insufficient (very rare, but technically
180 // insufficient)
181 if ((white_pieces == 2 && black_pieces == 0 && white_minors == 2 &&
182 white_has_knight) ||
183 (white_pieces == 0 && black_pieces == 2 && black_minors == 2 &&
184 black_has_knight)) {
185 return true;
186 }
187
188 return false; // Sufficient material for checkmate
189}

219game_state_t game_check_end_game_conditions(void) {
220 // Check if current player is in check
221 bool in_check = game_is_king_in_check(current_player);
222
223 // Check if current player has legal moves
224 bool has_moves = game_has_legal_moves(current_player);
225
226 if (in_check && !has_moves) {
227 // Checkmate - určit typ podle posledního tahu
228 game_result = GAME_STATE_FINISHED;
229 current_result_type = (current_player == PLAYER_WHITE) ? RESULT_BLACK_WINS
230 : RESULT_WHITE_WINS;
231
232 // Rozlišit typ šachmatu podle posledního tahu
233 switch (last_move_type) {
234 case LAST_MOVE_EN_PASSANT:
235 current_endgame_reason = ENDGAME_REASON_CHECKMATE_EN_PASSANT;
236 ESP_LOGI(TAG, "🎯 CHECKMATE BY EN PASSANT! %s wins in %" PRIu32 " moves!",
237 (current_player == PLAYER_WHITE) ? "Black" : "White",
238 move_count);
239 break;
240 case LAST_MOVE_CASTLING:
241 current_endgame_reason = ENDGAME_REASON_CHECKMATE_CASTLING;
242 ESP_LOGI(TAG, "🎯 CHECKMATE BY CASTLING! %s wins in %" PRIu32 " moves!",
243 (current_player == PLAYER_WHITE) ? "Black" : "White",
244 move_count);
245 break;
246 case LAST_MOVE_PROMOTION:
247 current_endgame_reason = ENDGAME_REASON_CHECKMATE_PROMOTION;
248 ESP_LOGI(TAG, "🎯 CHECKMATE BY PROMOTION! %s wins in %" PRIu32 " moves!",
249 (current_player == PLAYER_WHITE) ? "Black" : "White",
250 move_count);
251 break;
252 case LAST_MOVE_DISCOVERED:
253 current_endgame_reason = ENDGAME_REASON_CHECKMATE_DISCOVERED;
254 ESP_LOGI(
255 TAG,
256 "🎯 CHECKMATE BY DISCOVERED CHECK! %s wins in %" PRIu32 " moves!",
257 (current_player == PLAYER_WHITE) ? "Black" : "White", move_count);
258 break;
259 default:
260 current_endgame_reason = ENDGAME_REASON_CHECKMATE;
261 ESP_LOGI(TAG, "🎯 CHECKMATE! %s wins in %" PRIu32 " moves!",
262 (current_player == PLAYER_WHITE) ? "Black" : "White",
263 move_count);
264 break;
265 }
266
267 game_update_endgame_statistics(current_result_type);
268 endgame_report_requested = true; // Požadavek na report
269
270 // Zastavit timer (hra skončila)
271 timer_pause();
272
273 // UNIFIED VICTORY ANIMATION
274 // Checkmate = Victory for current_player (who delivered mate? No wait.)
275 // Logic: if current_player is in check & no moves, they LOST.
276 // Winner is the OPPONENT.
277 player_t winner =
278 (current_player == PLAYER_WHITE) ? PLAYER_BLACK : PLAYER_WHITE;
279 game_trigger_victory_animation(winner);
280
281 return GAME_STATE_FINISHED;
282 } else if (!in_check && !has_moves) {
283 // Stalemate
284 game_result = GAME_STATE_FINISHED;
285 current_result_type = RESULT_DRAW_STALEMATE;
286 current_endgame_reason = ENDGAME_REASON_STALEMATE; // Označit jako stalemate
287 game_update_endgame_statistics(current_result_type);
288 endgame_report_requested = true; // Požadavek na report
289
290 // Zastavit timer (hra skončila)
291 timer_pause();
292
293 ESP_LOGI(TAG, "🤝 STALEMATE! Game drawn in %" PRIu32 " moves", move_count);
294 return GAME_STATE_FINISHED;
295 }
296
297 // BUG #13: Check for draw conditions
298 // 50-move rule: 50 plných tahů = 100 půltahů (half-moves/ply)
299 // moves_without_capture počítá půltahy, takže >= 100
300 if (moves_without_capture >= 100) {
301 game_result = GAME_STATE_FINISHED;
302 current_result_type = RESULT_DRAW_50_MOVE;
303 current_endgame_reason =
304 ENDGAME_REASON_50_MOVE; // Označit jako 50-move rule
305 game_update_endgame_statistics(current_result_type);
306 endgame_report_requested = true; // Požadavek na report
307
308 // Zastavit timer (hra skončila)
309 timer_pause();
310
311 ESP_LOGI(TAG, "🤝 DRAW! 50 moves without capture (50-move rule)");
312 return GAME_STATE_FINISHED;
313 }
314
315 if (game_is_position_repeated()) {
316 game_result = GAME_STATE_FINISHED;
317 current_result_type = RESULT_DRAW_REPETITION;
318 current_endgame_reason =
319 ENDGAME_REASON_REPETITION; // Označit jako threefold repetition
320 game_update_endgame_statistics(current_result_type);
321 endgame_report_requested = true; // Požadavek na report
322
323 // Zastavit timer (hra skončila)
324 timer_pause();
325
326 ESP_LOGI(TAG, "🤝 DRAW! Position repeated (draw by repetition)");
327 return GAME_STATE_FINISHED;
328 }
329
330 // Check for insufficient material
331 if (game_is_insufficient_material()) {
332 game_result = GAME_STATE_FINISHED;
333 current_result_type = RESULT_DRAW_INSUFFICIENT;
334 current_endgame_reason =
335 ENDGAME_REASON_INSUFFICIENT; // Označit jako insufficient material
336 game_update_endgame_statistics(current_result_type);
337 endgame_report_requested = true; // Požadavek na report
338
339 // Zastavit timer (hra skončila)
340 timer_pause();
341
342 ESP_LOGI(TAG, "🤝 DRAW! Insufficient material to checkmate");
343 return GAME_STATE_FINISHED;
344 }
345
346 return GAME_STATE_ACTIVE;
347}
