110static void ble_sanitize_cmd_token(const char *src, char *dst, size_t dst_cap) {
111 if (dst_cap == 0) {
112 return;
113 }
114 size_t j = 0;
115 for (size_t i = 0; src[i] != '\0' && j + 1 < dst_cap; i++) {
116 unsigned char c = (unsigned char)src[i];
117 if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') ||
118 (c >= '0' && c <= '9') || c == '_' || c == '-' || c == '.') {
119 dst[j++] = (char)c;
120 }
121 }
122 dst[j] = '\0';
123}

155void ble_task_notify_command_result(esp_err_t err, const char *json_body) {
156 char cmd_safe[48] = {0};
157 if (json_body != NULL) {
158 char raw[44] = {0};
159 if (web_server_ble_extract_cmd_for_ack(json_body, raw, sizeof(raw))) {
160 ble_sanitize_cmd_token(raw, cmd_safe, sizeof(cmd_safe));
161 }
162 }
163
164 const char *code = "internal_error";
165 const char *ok_str = "false";
166 const char *msg = "Interní chyba při zpracování příkazu.";
167
168 switch (err) {
169 case ESP_OK:
170 ok_str = "true";
171 code = "ok";
172 msg = "Příkaz přijat.";
173 break;
174 case ESP_ERR_NOT_SUPPORTED:
175 code = "unknown_command";
176 msg = "Deska tento příkaz nezná.";
177 break;
178 case ESP_ERR_NOT_FOUND:
179 code = "missing_cmd";
180 msg = "V JSON chybí pole cmd.";
181 break;
182 case ESP_ERR_INVALID_ARG:
183 code = "invalid_argument";
184 msg = "Neplatné parametry příkazu.";
185 break;
186 case ESP_ERR_INVALID_STATE:
187 code = "blocked";
188 msg = "Akce je zablokovaná (zámek webu nebo stav desky).";
189 break;
190 case ESP_ERR_NOT_FINISHED:
191 code = "tutorial_finish_conflict";
192 msg =
193 "Fyzická deska neodpovídá základnímu postavení (řádky 1–2 a 7–8 plné).";
194 break;
195 case ESP_FAIL:
196 code = "internal_error";
197 msg = "Interní chyba při zpracování příkazu.";
198 break;
199 default:
200 code = "error";
201 msg = "Chyba při zpracování příkazu.";
202 break;
203 }
204
205 char out[256];
206 int n = snprintf(out, sizeof(out),
207 "{\"channel\":\"cmd_ack\",\"ok\":%s,\"code\":\"%s\","
208 "\"cmd\":\"%s\",\"message\":\"%s\",\"esp\":%d}",
209 ok_str, code, cmd_safe, msg, (int)err);
210 if (n < 0 || (size_t)n >= sizeof(out)) {
211 ESP_LOGW(TAG, "cmd_ack JSON truncate");
212 }
213
214 if (err == ESP_OK) {
215 ESP_LOGD(TAG, "[cmd_ack] %s", out);
216 } else {
217 ESP_LOGI(TAG, "[cmd_ack] %s", out);
218 }
219
220 if (g_cmd_ack_val_handle == 0 ||
221 s_conn_handle == BLE_HS_CONN_HANDLE_NONE || !s_cmd_ack_notify_enabled) {
222 ESP_LOGD(TAG,
223 "cmd_ack notify skipped (h=%u conn=%d sub=%d)",
224 (unsigned)g_cmd_ack_val_handle, (int)s_conn_handle,
225 (int)s_cmd_ack_notify_enabled);
226 return;
227 }
228
229 size_t len = strlen(out);
230 struct os_mbuf *om = ble_hs_mbuf_from_flat(out, (uint16_t)len);
231 if (om == NULL) {
232 ESP_LOGW(TAG, "cmd_ack: mbuf alloc failed");

234 }
235 int rc = ble_gatts_notify_custom(s_conn_handle, g_cmd_ack_val_handle, om);
236 if (rc != 0) {
237 ESP_LOGW(TAG, "cmd_ack: notify_custom rc=%d", rc);
238 }

242 char sta_ip[16] = {0};

256 "\"ap_ip\":\"%s\","
257 "\"ap_ssid\":\"%s\","
258 "\"ap_active\":%s,"
259 "\"online\":%s"

262 sta_connected ? sta_ip : "",

264 ap_ip_lit,
265 ap_ssid_lit,
266 ap_on ? "true" : "false",
267 sta_connected ? "true" : "false");
268 return 0;

271static int czechmate_gatt_access(uint16_t conn_handle, uint16_t attr_handle,
272 struct ble_gatt_access_ctxt *ctxt, void *arg) {
273 (void)arg;
274 ESP_LOGD(TAG,
275 "[STAGING] GATT access conn=%u op=%d attr=%u snap_h=%u cmd_h=%u net_h=%u",
276 (unsigned)conn_handle, (int)ctxt->op, (unsigned)attr_handle,
277 (unsigned)g_snap_val_handle, (unsigned)g_cmd_val_handle,
278 (unsigned)g_net_val_handle);
279 switch (ctxt->op) {
280 case BLE_GATT_ACCESS_OP_READ_CHR:
281 if (attr_handle == g_snap_val_handle) {
282 char *snap = NULL;
283 size_t len = 0;
284 esp_err_t e = web_server_build_game_snapshot_json_shared(&snap, &len);
285 if (e != ESP_OK || len == 0 || snap == NULL) {
286 ESP_LOGW(TAG,
287 "[STAGING] BLE read snapshot: build failed (%s)",
288 esp_err_to_name(e));
289 return BLE_ATT_ERR_UNLIKELY;
290 }
291 if (len > 65535U) {
292 ESP_LOGW(TAG, "[STAGING] BLE read snapshot: JSON > 65535 B");
293 return BLE_ATT_ERR_UNLIKELY;
294 }
295 int rc = os_mbuf_append(ctxt->om, snap, (uint16_t)len);
296 if (rc != 0) {
297 ESP_LOGW(TAG, "[STAGING] os_mbuf_append failed rc=%d", rc);
298 return BLE_ATT_ERR_INSUFFICIENT_RES;
299 }
300 return 0;
301 }
302 if (attr_handle == g_net_val_handle) {
303 char net_json[384];
304 czechmate_build_network_json(net_json, sizeof(net_json));
305 size_t len = strlen(net_json);
306 int rc = os_mbuf_append(ctxt->om, net_json, (uint16_t)len);
307 if (rc != 0) {
308 ESP_LOGW(TAG, "[STAGING] os_mbuf_append network failed rc=%d", rc);
309 return BLE_ATT_ERR_INSUFFICIENT_RES;
310 }
311 ESP_LOGI(TAG, "[STAGING] BLE read network: %s", net_json);
312 return 0;
313 }
314 break;
315 case BLE_GATT_ACCESS_OP_WRITE_CHR:
316 if (attr_handle == g_cmd_val_handle) {
317 /*
318 * hint_highlight / hint_clear: synchronní web_server_ble_command_dispatch v rámci
319 * ATT write; po něm cmd_ack notify na jiné charakteristice. Snapshot notify jede
320 * zvlášť (game task → ble_task_push_snapshot_json) — jiný GATT handle; NimBLE
321 * serializuje notify; dlouhé snapshot chunky mohou zpoždět další notify, ale
322 * nesmí zahodit samotný CMD (ověřovat při ladění MTU a zatížení desky).
323 */
324 uint16_t om_len = OS_MBUF_PKTLEN(ctxt->om);
325 uint8_t tmp[768];
326 uint16_t n = om_len > sizeof(tmp) ? (uint16_t)sizeof(tmp) : om_len;
327 if (n > 0) {
328 os_mbuf_copydata(ctxt->om, 0, n, tmp);
329 }
330 /* Raw firmware chunky: magic OB + idx/total + payload (nezapisovat do JSON bufferu). */
331 if (n >= 7 && tmp[0] == 'O' && tmp[1] == 'B') {
332 if (!ble_task_conn_is_encrypted()) {
333 ESP_LOGW(TAG, "OTA BLE chunk rejected: link not encrypted");
334 return BLE_ATT_ERR_INSUFFICIENT_AUTHOR;
335 }
336 esp_err_t derr = ota_update_ble_feed_chunk(tmp, n);
337 static const char ack_chunk_bad[] =
338 "{\"cmd\":\"ota_ble_chunk\",\"ok\":false}";
339 /*
340 * Úspěšný chunk: neposílat cmd_ack notify — klient OTA nečeká na ACK,
341 * jen na ATT write response; notify každý chunk zahlcuje frontu na iOS.
342 */
343 if (derr != ESP_OK) {
344 ble_task_notify_command_result(derr, ack_chunk_bad);
345 }
346 if (derr == ESP_OK || derr == ESP_ERR_NOT_SUPPORTED) {
347 return 0;
348 }
349 if (derr == ESP_ERR_INVALID_STATE) {
350 ESP_LOGW(TAG, "OTA BLE chunk blocked: %s", esp_err_to_name(derr));
351 return BLE_ATT_ERR_INSUFFICIENT_AUTHOR;
352 }
353 if (derr == ESP_ERR_INVALID_ARG || derr == ESP_ERR_INVALID_SIZE) {
354 return BLE_ATT_ERR_INVALID_ATTR_VALUE_LEN;
355 }
356 ESP_LOGW(TAG, "OTA BLE chunk: %s", esp_err_to_name(derr));
357 return BLE_ATT_ERR_UNLIKELY;
358 }
359 /* Musí být v souladu s web_server_ble_command_dispatch (větší JSON: timer/virtual). */
360 if (om_len >= sizeof(s_ble_cmd_copy)) {
361 om_len = (uint16_t)(sizeof(s_ble_cmd_copy) - 1);
362 }
363 if (om_len > 0) {
364 os_mbuf_copydata(ctxt->om, 0, om_len, s_ble_cmd_copy);
365 }
366 s_ble_cmd_copy[om_len] = '\0';
367 esp_err_t derr =
368 web_server_ble_command_dispatch(s_ble_cmd_copy, om_len);
369 if (!web_server_ble_dispatch_custom_ack_was_sent()) {
370 ble_task_notify_command_result(derr, s_ble_cmd_copy);
371 }
372 if (derr == ESP_OK || derr == ESP_ERR_NOT_SUPPORTED) {
373 ESP_LOGD(TAG,
374 "[STAGING] CMD dispatch %s (%u B) — ATT write response OK",
375 esp_err_to_name(derr), (unsigned)om_len);
376 return 0;
377 }
378 if (derr == ESP_ERR_INVALID_STATE) {
379 ESP_LOGW(TAG, "CMD blocked: %s", esp_err_to_name(derr));
380 return BLE_ATT_ERR_INSUFFICIENT_AUTHOR;
381 }
382 if (derr == ESP_ERR_NOT_ALLOWED) {
383 ESP_LOGW(TAG, "CMD not allowed: %s", esp_err_to_name(derr));
384 return BLE_ATT_ERR_INSUFFICIENT_AUTHOR;
385 }
386 if (derr == ESP_ERR_NOT_FOUND) {
387 ESP_LOGW(TAG, "CMD missing field");
388 return BLE_ATT_ERR_ATTR_NOT_FOUND;
389 }
390 if (derr == ESP_ERR_INVALID_ARG) {
391 ESP_LOGW(TAG, "CMD invalid JSON/args");
392 return BLE_ATT_ERR_INVALID_ATTR_VALUE_LEN;
393 }
394 ESP_LOGW(TAG, "CMD dispatch: %s", esp_err_to_name(derr));
395 return BLE_ATT_ERR_UNLIKELY;
396 }
397 break;
398 default:
399 break;
400 }
401 return BLE_ATT_ERR_UNLIKELY;
402}

444static void czechmate_advertise(void) {
445 struct ble_gap_adv_params adv_params;
446 struct ble_hs_adv_fields fields;
447 struct ble_hs_adv_fields rsp_fields;
448 int rc;
449
450 ESP_LOGD(TAG, "[STAGING] czechmate_advertise: set ADV + scanRsp fields");
451
452 /*
453 * Primary ADV packet (max 31 B):
454 * Flags: 3 B (2 header + 1 value)
455 * UUID128: 18 B (2 header + 16 UUID)
456 * = 21 B — OK, pod limit 31 B.
457 * Name přesunuto do Scan Response (uvolní místo, iOS 13+ posílá ScanReq
458 * automaticky).
459 */
460 memset(&fields, 0, sizeof(fields));
461 fields.flags = BLE_HS_ADV_F_DISC_GEN | BLE_HS_ADV_F_BREDR_UNSUP;
462 fields.uuids128 = &czechmate_svc_uuid;
463 fields.num_uuids128 = 1;
464 fields.uuids128_is_complete = 1;
465
466 rc = ble_gap_adv_set_fields(&fields);
467 if (rc != 0) {
468 ESP_LOGE(TAG, "adv_set_fields rc=%d", rc);
469 return;
470 }
471
472 /* Scan Response: plné jméno (iOS ho čte jako
473 * CBAdvertisementDataLocalNameKey). */
474 memset(&rsp_fields, 0, sizeof(rsp_fields));
475 const char *name = "CZECHMATE";
476 rsp_fields.name = (uint8_t *)name;
477 rsp_fields.name_len = strlen(name);
478 rsp_fields.name_is_complete = 1;
479
480 rc = ble_gap_adv_rsp_set_fields(&rsp_fields);
481 if (rc != 0) {
482 /* Scan response není kritický — loguj ale pokračuj */
483 ESP_LOGW(TAG, "adv_rsp_set_fields rc=%d", rc);
484 }
485
486 memset(&adv_params, 0, sizeof(adv_params));
487 adv_params.conn_mode = BLE_GAP_CONN_MODE_UND;
488 adv_params.disc_mode = BLE_GAP_DISC_MODE_GEN;
489 rc = ble_gap_adv_start(BLE_OWN_ADDR_PUBLIC, NULL, BLE_HS_FOREVER, &adv_params,
490 czechmate_gap_event, NULL);
491 if (rc != 0) {
492 /* Fallback: RPA (random private address) — pokud ESP nemá platnou public BD
493 * addr. */
494 ESP_LOGW(TAG, "adv_start PUBLIC rc=%d — fallback na RPA", rc);
495 rc =
496 ble_gap_adv_start(BLE_OWN_ADDR_RPA_PUBLIC_DEFAULT, NULL, BLE_HS_FOREVER,
497 &adv_params, czechmate_gap_event, NULL);
498 if (rc != 0) {
499 ESP_LOGE(TAG, "adv_start RPA fallback rc=%d", rc);
500 return;
501 }
502 ESP_LOGD(TAG, "[STAGING] adv_start ok (RPA address)");
503 } else {
504 ESP_LOGD(TAG, "[STAGING] adv_start ok (PUBLIC address)");
505 }
506}

616 (void)arg;
617 ESP_LOGD(TAG, "[STAGING] GAP event type=%d", (int)event->type);
618 switch (event->type) {
619 case BLE_GAP_EVENT_CONNECT:

622 ESP_LOGI(TAG, "connected handle=%d", s_conn_handle);
623#if CONFIG_ESP_COEX_ENABLED

636 }
637#endif
638#if CONFIG_BT_NIMBLE_SECURITY_ENABLE
639 /* iOS: SMP dřív než centrál dokončí GATT discovery často končí disconnect 531 /
640 * enc_change status≠0. Aplikace typicky začíná discovery ~0,5–1,5 s po LINK —
641 * první ble_gap_security_initiate odložit (~1,8 s). */

655 (void)ble_npl_callout_reset(
656 &s_sec_retry_co, ble_npl_time_ms_to_ticks32(450));
657 }

660 int src = ble_gap_security_initiate(event->connect.conn_handle);
661 ESP_LOGI(TAG, "[STAGING] ble_gap_security_initiate(no_timer) rc=%d",
662 src);
663 }
664 }
665#endif

668 czechmate_gap_restart_advertising();
669 }
670 return 0;
671#if CONFIG_BT_NIMBLE_SECURITY_ENABLE
672 case BLE_GAP_EVENT_ENC_CHANGE:

694 }

697 ESP_LOGI(TAG,

700 (unsigned)event->pairing_complete.conn_handle);
701 return 0;
702 case BLE_GAP_EVENT_REPEAT_PAIRING: {
703 struct ble_gap_conn_desc desc;
704 int rc = ble_gap_conn_find(event->repeat_pairing.conn_handle, &desc);

712 ESP_LOGI(TAG, "[STAGING] PASSKEY_ACTION act=%u numcmp=%" PRIu32,

714 if (act == BLE_SM_IOACT_NONE) {
715 return 0;

720 case BLE_SM_IOACT_NUMCMP:
721 /* Headless periferie: uživatel ověří hodnotu na telefonu (iOS). */
722 io.numcmp_accept = 1;

727 case BLE_SM_IOACT_INPUT:

735 if (rc != 0) {
736 ESP_LOGW(TAG, "ble_sm_inject_io act=%u rc=%d", (unsigned)act, rc);

739 }
740#endif

742#if CONFIG_BT_NIMBLE_SECURITY_ENABLE

964void ble_task_push_snapshot_json(const uint8_t *data, size_t len) {
965 if (data == NULL || len == 0 || g_snap_val_handle == 0) {
966 ESP_LOGV(TAG, "push_snapshot: skip (no data or snap handle 0)");
967 return;
968 }
969 if (s_conn_handle == BLE_HS_CONN_HANDLE_NONE || !s_snap_notify_enabled) {
970 ESP_LOGV(TAG, "push_snapshot: skip (conn=%d notify=%d)", (int)s_conn_handle,
971 (int)s_snap_notify_enabled);
972 return;
973 }
974 uint16_t mtu = ble_att_mtu(s_conn_handle);
975 if (mtu < 27) {
976 mtu = 23;
977 }
978 size_t chunk_cap = (size_t)mtu - 7U;
979 if (chunk_cap < 8U) {
980 chunk_cap = 8U;
981 }
982 if (chunk_cap > SNAPSHOT_NOTIFY_CHUNK_CAP) {
983 chunk_cap = SNAPSHOT_NOTIFY_CHUNK_CAP;
984 }
985 ESP_LOGD(TAG, "[STAGING] push_snapshot: %u B → notify chunk(s) mtu=%u cap=%u",
986 (unsigned)len, (unsigned)mtu, (unsigned)chunk_cap);
987 size_t off = 0;
988 uint8_t part = 0;
989 uint32_t total_u = (uint32_t)((len + chunk_cap - 1) / chunk_cap);
990 if (total_u < 1U) {
991 total_u = 1U;
992 }
993 /* Protokol má part/total v uint8 — max 255 notify na jeden snapshot. */
994 if (total_u > 255U) {
995 size_t need = (len + 254U) / 255U;
996 if (need <= SNAPSHOT_NOTIFY_CHUNK_CAP && need + 7U <= (size_t)mtu) {
997 chunk_cap = need;
998 total_u = (uint32_t)((len + chunk_cap - 1) / chunk_cap);
999 }
1000 }
1001 if (total_u > 255U) {
1002 ESP_LOGE(TAG,
1003 "push_snapshot: %u B @ mtu=%u cap=%u → %u chunků (max 255); MTU "
1004 "ještě nebo moc malé",
1005 (unsigned)len, (unsigned)mtu, (unsigned)chunk_cap,
1006 (unsigned)total_u);
1007 return;
1008 }
1009 uint8_t total = (uint8_t)total_u;
1010 while (off < len) {
1011 /* Voláno typicky z web_server_task — dlouhá řada notify; TWDT jinak nestíhá. */
1012 (void)esp_task_wdt_reset();
1013 part++;
1014 size_t chunk = len - off;
1015 if (chunk > chunk_cap) {
1016 chunk = chunk_cap;
1017 }
1018 uint8_t pkt[4 + SNAPSHOT_NOTIFY_CHUNK_CAP];
1019 pkt[0] = 0x43;
1020 pkt[1] = 0x4D;
1021 pkt[2] = part;
1022 pkt[3] = total;
1023 memcpy(pkt + 4, data + off, chunk);
1024 struct os_mbuf *om = ble_hs_mbuf_from_flat(pkt, (uint16_t)(4 + chunk));
1025 if (om == NULL) {
1026 return;
1027 }
1028 int rc = ble_gatts_notify_custom(s_conn_handle, g_snap_val_handle, om);
1029 if (rc != 0) {
1030 ESP_LOGW(TAG, "notify_custom rc=%d", rc);
1031 return;
1032 }
1033 ESP_LOGD(TAG, "[STAGING] notify chunk %u/%u (%u B)", (unsigned)part,
1034 (unsigned)total, (unsigned)chunk);
1035 off += chunk;
1036 }
1037}
