102static bool opening_parse_uci(const char *uci, char *from, char *to) {
103 if (uci == NULL || from == NULL || to == NULL) {
104 return false;
105 }
106 size_t len = strlen(uci);
107 if (len < 4) {
108 return false;
109 }
110 from[0] = uci[0];
111 from[1] = uci[1];
112 from[2] = '\0';
113 to[0] = uci[2];
114 to[1] = uci[3];
115 to[2] = '\0';
116 return true;
117}

129 .led_index = from_led,
130 .red = 0,
131 .green = 120,
132 .blue = 255,
133 .duration_ms = 800,
134 .data = &to_led,
135 };
136 led_execute_command_new(&move_path_cmd);
137}

218bool game_opening_apply_uci(const char *uci) {
219 if (uci == NULL) {
220 return false;
221 }
222 char from[3] = {0};
223 char to[3] = {0};
224 if (!opening_parse_uci(uci, from, to)) {
225 return false;
226 }
227 uint8_t from_row = 0, from_col = 0, to_row = 0, to_col = 0;
228 if (!convert_notation_to_coords(from, &from_row, &from_col) ||
229 !convert_notation_to_coords(to, &to_row, &to_col)) {
230 return false;
231 }
232 piece_t piece = board[from_row][from_col];
233 if (piece == PIECE_EMPTY) {
234 return false;
235 }
236 chess_move_t move = {
237 .from_row = from_row,
238 .from_col = from_col,
239 .to_row = to_row,
240 .to_col = to_col,
241 .piece = piece,
242 .captured_piece = board[to_row][to_col],
243 .timestamp = esp_timer_get_time() / 1000,
244 };
245 if (game_is_valid_move(&move) != MOVE_ERROR_NONE) {
246 return false;
247 }
248 return game_execute_move(&move);
249}

265static bool opening_begin_opponent_physical_turn(void) {
266 if (opening_state.ply_index >= opening_state.line_uci_count) {
267 return false;
268 }
269 const char *uci = opening_state.line_uci[opening_state.ply_index];
270 opening_parse_uci(uci, opening_state.expected_from, opening_state.expected_to);
271 strncpy(opening_state.last_opponent_uci, uci,
272 sizeof(opening_state.last_opponent_uci) - 1);
273 opening_state.awaiting_opponent_physical = true;
274 opening_state.feedback = OPENING_FEEDBACK_OPPONENT_TURN;
275 opening_show_opponent_physical_led(false);
276 STAGING_LOGI(TAG, "opponent physical ply=%u uci=%s", (unsigned)opening_state.ply_index,
277 uci);
278 return true;
279}

281static void opening_run_until_player_turn(void) {
282 while (opening_state.active &&
283 opening_state.ply_index < opening_state.line_uci_count) {
284 if (opening_is_player_ply(opening_state.ply_index)) {
285 opening_prepare_player_turn();
286 return;
287 }
288
289 if (opening_state.opponent_mode == OPENING_OPPONENT_PHYSICAL) {
290 opening_begin_opponent_physical_turn();
291 return;
292 }
293
294 const char *uci = opening_state.line_uci[opening_state.ply_index];
295 char from[3], to[3];
296 opening_parse_uci(uci, from, to);
297 if (game_opening_apply_uci(uci)) {
298 strncpy(opening_state.last_opponent_uci, uci,
299 sizeof(opening_state.last_opponent_uci) - 1);
300 opening_play_move_led(from, to);
301 STAGING_LOGI(TAG, "virtual reply ply=%u uci=%s", (unsigned)opening_state.ply_index,
302 uci);
303 } else {
304 ESP_LOGE(TAG, "virtual uci failed: %s", uci);
305 break;
306 }
307 opening_state.ply_index++;
308 if (opening_should_checkpoint_after_ply(opening_state.ply_index - 1)) {
309 opening_state.awaiting_checkpoint_ack = true;
310 opening_state.feedback = OPENING_FEEDBACK_CHECKPOINT;
311 return;
312 }
313 }
314
315 if (opening_state.ply_index >= opening_state.line_uci_count) {
316 opening_state.active = false;
317 opening_state.feedback = OPENING_FEEDBACK_COMPLETE;
318 led_command_t end_cmd = {.type = LED_CMD_ANIM_ENDGAME, .duration_ms = 1500};
319 led_execute_command_new(&end_cmd);
320 STAGING_LOGI(TAG, "line complete id=%s", opening_state.line_id);
321 }
322}

369bool game_opening_validate_checkpoint_physical(void) {
370 uint8_t matrix[64] = {0};
371 matrix_get_state(matrix);
372 for (uint8_t i = 0; i < 64; i++) {
373 uint8_t row = i / 8;
374 uint8_t col = i % 8;

383bool game_opening_load_config(const char *line_id, const char *start_fen,
384 const char line_uci[][6], uint8_t line_uci_count,
385 const uint8_t *player_ply_indices,
386 uint8_t player_ply_count,
387 const uint8_t *checkpoint_ply_indices,
388 uint8_t checkpoint_count, uint8_t mode,
389 uint8_t player_side_white,
390 uint8_t opponent_mode) {
391 if (line_id == NULL || start_fen == NULL || line_uci == NULL ||
392 line_uci_count == 0 || line_uci_count > OPENING_MAX_PLIES ||
393 player_ply_count == 0 || player_ply_count > OPENING_MAX_PLAYER_PLIES) {
394 return false;
395 }
396 game_opening_cancel();
397 memset(&opening_state, 0, sizeof(opening_state));
398 strncpy(opening_state.line_id, line_id, sizeof(opening_state.line_id) - 1);
399 strncpy(opening_state.start_fen, start_fen, sizeof(opening_state.start_fen) - 1);
400 opening_state.line_uci_count = line_uci_count;
401 for (uint8_t i = 0; i < line_uci_count; i++) {
402 strncpy(opening_state.line_uci[i], line_uci[i], sizeof(opening_state.line_uci[i]) - 1);
403 }
404 opening_state.player_ply_count = player_ply_count;
405 for (uint8_t i = 0; i < player_ply_count; i++) {
406 opening_state.player_ply_indices[i] = player_ply_indices[i];
407 }
408 opening_state.checkpoint_count =
409 checkpoint_count > OPENING_MAX_CHECKPOINTS ? OPENING_MAX_CHECKPOINTS : checkpoint_count;
410 for (uint8_t i = 0; i < opening_state.checkpoint_count; i++) {
411 opening_state.checkpoint_ply_indices[i] = checkpoint_ply_indices[i];
412 }
413 opening_state.mode = (opening_mode_t)mode;
414 opening_state.opponent_mode =
415 opponent_mode == 1 ? OPENING_OPPONENT_PHYSICAL : OPENING_OPPONENT_VIRTUAL;
416 opening_state.player_side = player_side_white ? PLAYER_WHITE : PLAYER_BLACK;
417 opening_state.feedback = OPENING_FEEDBACK_NONE;
418 opening_state.wrong_move_count = 0;
419 opening_state.last_wrong_uci[0] = '\0';
420 opening_state.awaiting_opponent_physical = false;
421 opening_state.setup_phase = false;
422 STAGING_LOGI(TAG, "load line=%s plies=%u opponent=%s", line_id,
423 (unsigned)line_uci_count,
424 opening_state.opponent_mode == OPENING_OPPONENT_PHYSICAL
425 ? "physical"
426 : "virtual");
427 return true;
428}

430bool game_opening_start(void) {
431 if (opening_state.line_uci_count == 0) {
432 return false;
433 }
434 if (!game_opening_physical_matches_start()) {
435 opening_state.setup_phase = true;
436 opening_state.active = false;
437 STAGING_LOGI(TAG, "start blocked — physical mismatch line=%s",
438 opening_state.line_id);
439 return false;
440 }
441 if (game_is_board_setup_tutorial_active()) {
442 game_exit_board_setup_tutorial(false);
443 }
444 game_puzzle_cancel();
445 game_reset_game();
446
447 player_t fen_player = PLAYER_WHITE;
448 if (!game_load_position_from_fen(opening_state.start_fen, &fen_player)) {
449 ESP_LOGE(TAG, "failed to load start FEN");
450 return false;
451 }
452
453 current_player = fen_player;
454 current_game_state = GAME_STATE_ACTIVE;
455 game_active = true;
456 opening_state.setup_phase = false;
457 opening_state.active = true;
458 opening_state.ply_index = 0;
459 opening_state.player_ply_index = 0;
460 opening_state.awaiting_checkpoint_ack = false;
461 opening_state.awaiting_opponent_physical = false;
462 opening_state.feedback = OPENING_FEEDBACK_NONE;
463 opening_state.last_opponent_uci[0] = '\0';
464 opening_state.last_wrong_uci[0] = '\0';
465 opening_state.last_wrong_uci[0] = '\0';
466 opening_state.wrong_move_count = 0;
467 led_clear_board_only();
468
469 opening_run_until_player_turn();
470 STAGING_LOGI(TAG, "started line=%s ply=%u", opening_state.line_id,
471 (unsigned)opening_state.ply_index);
472 return true;
473}

503bool game_opening_validate_expected_move(uint8_t from_row, uint8_t from_col,
504 uint8_t to_row, uint8_t to_col) {
505 if (!opening_state.active || opening_state.awaiting_checkpoint_ack) {
506 return false;
507 }
508 if (opening_state.ply_index >= opening_state.line_uci_count) {
509 return false;
510 }
511 char from_notation[3] = {0};
512 char to_notation[3] = {0};
513 snprintf(from_notation, sizeof(from_notation), "%c%d", 'a' + from_col, from_row + 1);
514 snprintf(to_notation, sizeof(to_notation), "%c%d", 'a' + to_col, to_row + 1);
515 if (opening_state.awaiting_opponent_physical) {
516 return (strcmp(from_notation, opening_state.expected_from) == 0 &&
517 strcmp(to_notation, opening_state.expected_to) == 0);
518 }
519 if (!opening_is_player_ply(opening_state.ply_index)) {
520 return false;
521 }
522 return (strcmp(from_notation, opening_state.expected_from) == 0 &&
523 strcmp(to_notation, opening_state.expected_to) == 0);
524}

572 opening_state.feedback = OPENING_FEEDBACK_CHECKPOINT;

582 !opening_state.awaiting_opponent_physical) {
583 opening_state.feedback = OPENING_FEEDBACK_NONE;

597bool game_opening_on_wrong_player_move(void) {
598 if (!opening_state.active) {
599 return false;
600 }
601 if (opening_state.wrong_move_count < 255) {
602 opening_state.wrong_move_count++;
