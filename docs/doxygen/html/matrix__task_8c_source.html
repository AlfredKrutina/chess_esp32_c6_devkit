242static void matrix_scan_row_internal(uint8_t row) {
243 if (row >= 8)
244 return;
245
246 // IMPROVED: P≈ôepni v≈°echny ≈ô√°dky na INPUT p≈ôed aktivac√≠ aktu√°ln√≠ho ≈ô√°dku
247 // Toto je bezpeƒçnƒõj≈°√≠ ne≈æ jen nastaven√≠ HIGH, proto≈æe zabra≈àuje konflikt≈Øm
248 // p≈ôi p≈ôep√≠n√°n√≠ mezi ≈ô√°dky (jako v referenƒçn√≠m k√≥du)
249 for (int i = 0; i < 8; i++) {
250 gpio_config_t io_conf = {.pin_bit_mask = (1ULL << matrix_row_pins[i]),
251 .mode = GPIO_MODE_INPUT,
252 .pull_up_en = GPIO_PULLUP_DISABLE,
253 .pull_down_en = GPIO_PULLDOWN_DISABLE,
254 .intr_type = GPIO_INTR_DISABLE};
255 gpio_config(&io_conf);
256 }
257
258 // IMPROVED: Pou≈æij esp_rom_delay_us m√≠sto vTaskDelay (timer callback
259 // context)
260 extern void esp_rom_delay_us(uint32_t us);
261 esp_rom_delay_us(20); // 20us settling time (reduced from 1ms)
262
263 // Nastav aktu√°ln√≠ ≈ô√°dek na OUTPUT a LOW (aktivace)
264 // Toto st√°hne column pin na LOW pokud je na tomto k≈ô√≠≈æen√≠ figurka (reed
265 // switch uzav≈ôen√Ω)
266 gpio_config_t io_conf = {.pin_bit_mask = (1ULL << matrix_row_pins[row]),
267 .mode = GPIO_MODE_OUTPUT,
268 .pull_up_en = GPIO_PULLUP_DISABLE,
269 .pull_down_en = GPIO_PULLDOWN_DISABLE,
270 .intr_type = GPIO_INTR_DISABLE};
271 gpio_config(&io_conf);
272 gpio_set_level(matrix_row_pins[row], 0); // LOW = aktivn√≠
273
274 // GPIO settling time: 200us for signal stabilization (long wires support)
275 // For digital signals with long cables (capacitance), we need more time
276 // for the signal to fall to logic LOW (grounding).
277 esp_rom_delay_us(200);
278
279 // Read all column pins for this row
280 // Column piny maj√≠ pull-up, tak≈æe bez figury budou HIGH
281 // Pokud je figurka (reed switch uzav≈ôen√Ω), column pin bude LOW
282
283 // Read all column pins for this row
284 // Column piny maj√≠ pull-up, tak≈æe bez figury budou HIGH
285 // Pokud je figurka (reed switch uzav≈ôen√Ω), column pin bude LOW
286
287 for (int col = 0; col < 8; col++) {
288 // Invert column index to fix GPIO physical wiring (H‚ÜíA instead of
289 // A‚ÜíH) GPIO pins are wired in reverse: pin 0 = column H, pin 7 = column A
290 // So we invert: col 0 ‚Üí index 7 (H), col 1 ‚Üí index 6 (G), ..., col 7 ‚Üí
291 // index 0 (A)
292 int inverted_col = 7 - col;
293 int index = row * 8 + inverted_col;
294
295 // DEBUG: Check if this is GPIO17 (column 7)
296 bool is_gpio17 = (matrix_col_pins[col] == GPIO_NUM_17);
297
298 int pin_level = gpio_get_level(matrix_col_pins[col]);
299
300 // In simulation mode, use simulated values
301 if (simulation_mode) {
302 matrix_state[index] = simulation_patterns[current_pattern][index];
303 } else {
304 // Real hardware: reed switch closed = piece present
305 // pin_level == 0 znamen√°, ≈æe column pin je sta≈æen√Ω na LOW (figurka je
306 // p≈ô√≠tomna)
307 // NOTE: GPIO17 is configured with pull-down for debug testing
308 // Normal logic still applies, but we'll log values for analysis
309 matrix_state[index] = (pin_level == 0) ? 1 : 0;
310
311 // DEBUG: Minimal logging for GPIO17 (user testing with multimeter)
312 // Only log first state change to confirm scanning works
313 static int last_gpio17_level = -1;
314 static bool gpio17_logged_once = false;
315 if (is_gpio17 && !gpio17_logged_once) {
316 ESP_LOGI(TAG, "üîç GPIO17 scan confirmed: row=%d, level=%d (pull-down mode)", row, pin_level);
317 gpio17_logged_once = true;
318 last_gpio17_level = pin_level;
319 } else if (is_gpio17 && last_gpio17_level == -1) {
320 last_gpio17_level = pin_level;
321 }
322 }
323 }
324
325 // IMPROVED: P≈ôepni ≈ô√°dek zpƒõt na INPUT po scanov√°n√≠ (bezpeƒçnƒõj≈°√≠)
326 io_conf.pin_bit_mask = (1ULL << matrix_row_pins[row]);
327 io_conf.mode = GPIO_MODE_INPUT;
328 gpio_config(&io_conf);
329
330 // Mal√© ƒçek√°n√≠ p≈ôed dal≈°√≠m ≈ô√°dkem pro stabilizaci
331 esp_rom_delay_us(20); // 20us delay (reduced from 2ms)
332}

336 if (row >= 8)

340 if (matrix_mutex != NULL) {
341 if (xSemaphoreTake(matrix_mutex, pdMS_TO_TICKS(100)) == pdTRUE) {

345 ESP_LOGW(TAG, "Failed to acquire matrix mutex for scan row");
346 }
347 } else {
348 // Fallback if mutex not available
349 matrix_scan_row_internal(row);

353void matrix_scan_all(void) {
354 uint32_t current_time = esp_timer_get_time() / 1000;
355
356 // NOTE: Don't call esp_task_wdt_reset() here because this function
357 // is called from timer callbacks which run in timer service task
358 // that is not registered with TWDT
359
360 // CRITICAL: Protect matrix state with mutex for change detection
361 if (matrix_mutex != NULL) {
362 if (xSemaphoreTake(matrix_mutex, pdMS_TO_TICKS(100)) == pdTRUE) {
363 // Scan each row (use internal function since we already hold mutex)
364 for (int row = 0; row < 8; row++) {
365 // NOTE: Don't call WDT reset in timer callback context
366 matrix_scan_row_internal(row);
367 }
368
369 // Detect changes
370 for (int i = 0; i < 64; i++) {
371 if (matrix_state[i] != matrix_previous[i]) {
372 matrix_changes[i] = 1;
373 } else {
374 matrix_changes[i] = 0;
375 }
376
377 // Demo reporting: If anything changed, report activity
378 if (matrix_changes[i]) {
379 demo_report_activity();
380 }
381 }
382
383 // CRITICAL: Detect moves BEFORE updating previous state
384 // This must be done while we still hold the mutex and before
385 // matrix_previous is overwritten, so matrix_detect_moves() can
386 // compare matrix_previous with matrix_state
387 matrix_detect_moves();
388
389 // Update previous state (AFTER move detection)
390 memcpy(matrix_previous, matrix_state, sizeof(matrix_state));
391
392 last_scan_time = current_time;
393 scan_count++;
394
395 // NOTE: Don't call WDT reset in timer callback context
396
397 xSemaphoreGive(matrix_mutex);
398 } else {
399 ESP_LOGW(TAG, "Failed to acquire matrix mutex for scan all");
400 return;
401 }
402 } else {
403 // Fallback if mutex not available
404 for (int row = 0; row < 8; row++) {
405 matrix_scan_row_internal(row);
406 }
407
408 for (int i = 0; i < 64; i++) {
409 if (matrix_state[i] != matrix_previous[i]) {
410 matrix_changes[i] = 1;
411 } else {
412 matrix_changes[i] = 0;
413 }
414
415 // Demo reporting: If anything changed, report activity
416 if (matrix_changes[i]) {
417 demo_report_activity();
418 }
419 }
420
421 // CRITICAL: Detect moves BEFORE updating previous state
422 // This must be done before matrix_previous is overwritten
423 matrix_detect_moves();
424
425 memcpy(matrix_previous, matrix_state, sizeof(matrix_state));
426 last_scan_time = current_time;
427 scan_count++;
428 }
429
430 if (simulation_mode) {
431 ESP_LOGD(TAG, "Matrix scan completed: pattern=%d, changes=%d",
432 current_pattern, 0); // Would count changes
433 }
434}

551void matrix_detect_moves(void) {
552 // Check for move timeout - change from RESET to WARNING + EXTEND
553 if (last_piece_lifted != 255) {
554 uint32_t current_time = esp_timer_get_time() / 1000;
555 if (current_time > move_detection_timeout) {
556 ESP_LOGW(TAG,
557 "‚è∞ Move taking longer than 5s - piece from %d still in air",
558 last_piece_lifted);
559 // DON'T reset last_piece_lifted - allow longer moves
560 // Just extend timeout to give player more time
561 move_detection_timeout = current_time + 5000; // Give another 5 seconds
562 }
563 }
564
565 // Look for piece lifted (1->0 transition)
566 uint8_t piece_lifted = 255;
567 for (int i = 0; i < 64; i++) {
568 if (matrix_previous[i] == 1 && matrix_state[i] == 0) {
569 piece_lifted = i;
570 break;
571 }
572 }
573
574 // Look for piece placed (0->1 transition)
575 uint8_t piece_placed = 255;
576 for (int i = 0; i < 64; i++) {
577 if (matrix_previous[i] == 0 && matrix_state[i] == 1) {
578 piece_placed = i;
579 break;
580 }
581 }
582
583 // Process move detection
584 if (piece_lifted != 255) {
585 last_piece_lifted = piece_lifted;
586 move_detection_timeout =
587 esp_timer_get_time() / 1000 + 5000; // 5 second timeout
588
589 // Convert square index to row/col
590 uint8_t from_row = piece_lifted / 8;
591 uint8_t from_col = piece_lifted % 8;
592
593 ESP_LOGI(TAG, "Piece lifted from square %d (%c%d)", piece_lifted,
594 'a' + from_col, from_row + 1);
595
596 // UNIFIED FLOW: Send PICKUP command to game_command_queue (same as UART)
597 matrix_send_pickup_command(piece_lifted);
598 }
599
600 if (piece_placed != 255) {
601 last_piece_placed = piece_placed;
602
603 // Convert square index to row/col
604 uint8_t to_row = piece_placed / 8;
605 uint8_t to_col = piece_placed % 8;
606
607 ESP_LOGI(TAG, "Piece placed on square %d (%c%d)", piece_placed,
608 'a' + to_col, to_row + 1);
609
610 // Check if this completes a move
611 if (last_piece_lifted != 255) {
612 if (last_piece_lifted == piece_placed) {
613 // Return to same square - send with from=to
614 ESP_LOGI(TAG,
615 "Piece returned to same square - sending DROP with from=to");
616 matrix_send_drop_command_with_from(last_piece_lifted, piece_placed);
617 } else {
618 // Complete move - send with from/to
619 matrix_send_drop_command_with_from(last_piece_lifted, piece_placed);
620 }
621 last_piece_lifted = 255; // Reset
622 } else {
623 // Piece placed without lift - IGNORE
624 ESP_LOGW(TAG, "Piece placed without previous lift (square %d) - ignoring",
625 piece_placed);
626 }
627 }
628}

648void matrix_square_to_notation(uint8_t square, char *notation) {
649 if (square >= 64 || notation == NULL)
650 return;
651
652 int row = square / 8;
653 int col = square % 8;
654
655 notation[0] = 'a' + col;
656 notation[1] = '1' + row;
657 notation[2] = '\0';
658}

660uint8_t matrix_notation_to_square(const char *notation) {
661 if (notation == NULL || strlen(notation) != 2)
662 return 255;
663
664 char file = tolower(notation[0]);
665 char rank = notation[1];
666
667 if (file < 'a' || file > 'h' || rank < '1' || rank > '8') {
668 return 255;
669 }
670
671 int col = file - 'a';
672 int row = rank - '1';
673
674 return row * 8 + col;
675}

755void matrix_simulate_move(const char *from, const char *to) {
756 uint8_t from_square = matrix_notation_to_square(from);
757 uint8_t to_square = matrix_notation_to_square(to);
758
759 if (from_square == 255 || to_square == 255) {
760 ESP_LOGE(TAG, "Invalid chess notation: %s -> %s", from, to);
761 return;
762 }
763
764 ESP_LOGI(TAG, "Simulating move: %s (%d) -> %s (%d)", from, from_square, to,
765 to_square);
766
767 // Update matrix state
768 matrix_state[from_square] = 0; // Remove piece from source
769 matrix_state[to_square] = 1; // Place piece at destination
