28static bool demo_is_castling_move(uint8_t from_row, uint8_t from_col,
29 uint8_t to_row, uint8_t to_col,
30 piece_t board[8][8]) {
31 // Must be on same rank
32 if (from_row != to_row) {
33 return false;
34 }
35
36 // Check if piece is a king
37 piece_t piece = board[from_row][from_col];
38 if (piece != PIECE_WHITE_KING && piece != PIECE_BLACK_KING) {
39 return false;
40 }
41
42 // Check if king moves 2 squares (castling signature)
43 int col_diff = abs((int)to_col - (int)from_col);
44 return (col_diff == 2);
45}

58static void
59demo_get_castling_rook_squares(uint8_t king_from_row, uint8_t king_from_col,
60 uint8_t king_to_col, uint8_t *rook_from_row,
61 uint8_t *rook_from_col, uint8_t *rook_to_row,
62 uint8_t *rook_to_col) {
63 *rook_from_row = king_from_row;
64 *rook_to_row = king_from_row;
65
66 // Kingside castling (king moves to g-file, col 6)
67 if (king_to_col == 6) {
68 *rook_from_col = 7; // h-file
69 *rook_to_col = 5; // f-file
70 }
71 // Queenside castling (king moves to c-file, col 2)
72 else if (king_to_col == 2) {
73 *rook_from_col = 0; // a-file
74 *rook_to_col = 3; // d-file
75 } else {
76 // Should not happen if is_castling_move returned true
77 ESP_LOGE("DEMO", "Invalid castling target column: %d", king_to_col);
78 *rook_from_col = 0;
79 *rook_to_col = 0;
80 }
81}

92static bool demo_is_promotion_move(uint8_t from_row, uint8_t from_col,
93 uint8_t to_row, piece_t board[8][8]) {
94 piece_t piece = board[from_row][from_col];
95
96 // Check if piece is a pawn
97 if (piece != PIECE_WHITE_PAWN && piece != PIECE_BLACK_PAWN) {
98 return false;
99 }
100
101 // White pawn reaching rank 8 (row 7)
102 if (piece == PIECE_WHITE_PAWN && to_row == 7) {
103 return true;
104 }
105
106 // Black pawn reaching rank 1 (row 0)
107 if (piece == PIECE_BLACK_PAWN && to_row == 0) {
108 return true;
109 }
110
111 return false;
112}

123static void demo_ensure_safe_timing(uint32_t *last_move_time_ms,
124 uint32_t interval_ms) {
125 uint32_t now_ms = xTaskGetTickCount() * portTICK_PERIOD_MS;
126 uint32_t elapsed = now_ms - *last_move_time_ms;
