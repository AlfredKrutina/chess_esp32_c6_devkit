609void execute_demo_move(void) {
610 // Reset watchdog at function entry.
611 // This function can block for 10+ seconds (castling), causing main task
612 // timeout
613 main_task_wdt_reset_safe();
614
615 if (!demo_mode_enabled || demo_move_index >= demo_moves_count) {
616 return;
617 }
618
619 const char *move = current_demo_moves[demo_move_index];
620
621 // DEBUG LOG - Critical to verify function entry
622 ESP_LOGI(TAG, "ü§ñ EXECUTE DEMO MOVE: Game='%s', Ind=%d/%d, Move=%s",
623 DEMO_GAME_NAMES[current_demo_game], demo_move_index + 1,
624 demo_moves_count, move);
625
626 // Use PICKUP/DROP commands for realistic demo (like up/dn)
627 if (strlen(move) == 4 && game_command_queue != NULL) {
628 chess_move_command_t cmd = {0};
629
630 // Step 1: PICKUP command (lift piece from source square)
631 cmd.type = GAME_CMD_PICKUP;
632 cmd.is_demo_mode = true; // Skip resignation timer in demo mode.
633 strncpy(cmd.from_notation, move, 2); // e.g. "e2"
634
635 ESP_LOGI(TAG, " ‚¨ÜÔ∏è Lifting piece from %c%c...", move[0], move[1]);
636
637 // Set response queue for feedback
638 cmd.response_queue = uart_response_queue;
639
640 if (xQueueSend(game_command_queue, &cmd, pdMS_TO_TICKS(100)) != pdTRUE) {
641 ESP_LOGE(TAG, "‚ùå Failed to send GAME_CMD_PICKUP");
642 return;
643 }
644
645 // Longer delay for Pickup animation
646 vTaskDelay(pdMS_TO_TICKS(1500)); // 1.5 seconds
647 main_task_wdt_reset_safe(); // Reset after delay.
648
649 // Step 2: DROP command (place piece on destination square)
650 cmd.type = GAME_CMD_DROP;
651 cmd.is_demo_mode = true; // Skip error recovery in demo mode.
652 strncpy(cmd.to_notation, move + 2, 2); // e.g. "e4"
653
654 ESP_LOGI(TAG, " ‚¨áÔ∏è Placing piece on %c%c...", move[2], move[3]);
655
656 // Set response queue for feedback
657 cmd.response_queue = uart_response_queue;
658
659 if (xQueueSend(game_command_queue, &cmd, pdMS_TO_TICKS(100)) != pdTRUE) {
660 ESP_LOGE(TAG, "‚ùå Failed to send GAME_CMD_DROP");
661 return;
662 }
663
664 // Wait for DROP animation
665 vTaskDelay(pdMS_TO_TICKS(800)); // 0.8 seconds
666 main_task_wdt_reset_safe(); // Reset after delay.
667
668 // Step 3: CASTLING Handling
669 // If move is castling, we must also move the rook!
670 char rook_from[3] = {0};
671 char rook_to[3] = {0};
672 bool is_castling = false;
673
674 if (strcmp(move, "e1g1") == 0) { // White King-side
675 strcpy(rook_from, "h1");
676 strcpy(rook_to, "f1");
677 is_castling = true;
678 } else if (strcmp(move, "e1c1") == 0) { // White Queen-side
679 strcpy(rook_from, "a1");
680 strcpy(rook_to, "d1");
681 is_castling = true;
682 } else if (strcmp(move, "e8g8") == 0) { // Black King-side
683 strcpy(rook_from, "h8");
684 strcpy(rook_to, "f8");
685 is_castling = true;
686 } else if (strcmp(move, "e8c8") == 0) { // Black Queen-side
687 strcpy(rook_from, "a8");
688 strcpy(rook_to, "d8");
689 is_castling = true;
690 }
691
692 if (is_castling) {
693 ESP_LOGI(TAG, " ‚ôú Castling detected! Moving rook %s -> %s", rook_from,
694 rook_to);
695 vTaskDelay(pdMS_TO_TICKS(500)); // Pause before moving rook
696 main_task_wdt_reset_safe(); // Reset during castling.
697
698 // Rook Pickup
699 memset(&cmd, 0, sizeof(cmd));
700 cmd.type = GAME_CMD_PICKUP;
701 cmd.is_demo_mode = true; // Skip resignation timer for rook.
702 strncpy(cmd.from_notation, rook_from, 2);
703 cmd.response_queue = uart_response_queue; // Add feedback.
704 xQueueSend(game_command_queue, &cmd, pdMS_TO_TICKS(100));
705 vTaskDelay(pdMS_TO_TICKS(1000));
706 main_task_wdt_reset_safe(); // Reset after rook pickup delay.
707
708 // Rook Drop
709 memset(&cmd, 0, sizeof(cmd));
710 cmd.type = GAME_CMD_DROP;
711 cmd.is_demo_mode = true; // Skip error recovery for rook.
712 strncpy(cmd.to_notation, rook_to, 2);
713 cmd.response_queue = uart_response_queue; // Add feedback.
714 xQueueSend(game_command_queue, &cmd, pdMS_TO_TICKS(100));
715 vTaskDelay(pdMS_TO_TICKS(800));
716 main_task_wdt_reset_safe(); // Reset after rook drop delay.
717 }
718 }
719
720 demo_move_index++;
721
722 // Check if all moves were played (endgame reached).
723 if (demo_move_index >= demo_moves_count) {
724 ESP_LOGI(TAG, "üèÅ Demo game '%s' complete! Endgame animations playing...",
725 DEMO_GAME_NAMES[current_demo_game]);
726 ESP_LOGI(TAG, "‚è±Ô∏è Waiting 5 seconds for endgame animation to complete...");
727
728 // Wait 5 seconds for endgame animation to play
729 // Split into shorter delays with watchdog resets.
730 for (int i = 0; i < 5; i++) {
731 vTaskDelay(pdMS_TO_TICKS(1000));
732 main_task_wdt_reset_safe();
733 }
734
735 // Rotate to the next demo game.
736 current_demo_game = (current_demo_game + 1) % DEMO_GAMES_COUNT;
737 current_demo_moves = DEMO_GAMES[current_demo_game];
738 demo_moves_count = DEMO_GAME_LENGTHS[current_demo_game];
739 demo_move_index = 0;
740
741 ESP_LOGI(TAG, "üîÑ Starting new demo game: %s (%d moves)",
742 DEMO_GAME_NAMES[current_demo_game], demo_moves_count);
743
744 // Send reset command to start fresh game
745 if (game_command_queue != NULL) {
746 chess_move_command_t cmd = {.type = GAME_CMD_NEW_GAME};
747 cmd.player = (demo_move_index % 2 == 0) ? PLAYER_WHITE : PLAYER_BLACK;
748
749 // Set response queue so game_task sends confirmation (like UART commands).
750 cmd.response_queue = uart_response_queue;
751
752 xQueueSend(game_command_queue, &cmd, pdMS_TO_TICKS(100));
753
754 // Wait for the game reset to complete before starting the first move.
755 ESP_LOGI(TAG, "‚è±Ô∏è Waiting 1 second for game reset to complete...");
756 vTaskDelay(pdMS_TO_TICKS(1000));
757 main_task_wdt_reset_safe(); // Reset after game reset delay.
758 }
759 }
