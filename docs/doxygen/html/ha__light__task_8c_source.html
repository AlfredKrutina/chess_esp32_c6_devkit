163 }
164

168 return false;

173 return false;

189static void ha_light_monitor_game_activity(void) {
190 // Activity monitoring is done via ha_light_report_activity() calls from:
191 // - game_task: when processing game commands
192 // - matrix_task: when detecting piece movement
193 // We just check if we need to switch modes based on activity timeout
194}
195
199void ha_light_report_activity(const char *activity_type) {
200 if (!task_running) {
201 return;
202 }
203
204 // Keep this function lightweight: it may be called from input paths.
205 // Mode switching and LED refresh must happen in HA task context.
206 last_activity_time_ms = esp_timer_get_time() / 1000;
207
208 // CHECK: Rate limiting for activity reporting (max 1 message per 500ms)
209 static uint32_t last_report_time = 0;
210 uint32_t current_time = esp_timer_get_time() / 1000;
211 if (current_time - last_report_time < 500) {
212 return; // Skip reporting if too frequent
213 }
214 last_report_time = current_time;
215
216 // Send simple command to queue - minimal overhead for caller
217 ha_light_command_t cmd;
218 cmd.type = 1; // 1 = Activity Report (magic number but internal)
219 // We don't need data for simple activity reset, but if we wanted to pass
220 // string: We can pass a static string pointer since most activity strings are
221 // literals. Warning: Do not pass stack buffers!
222 cmd.data = (void *)activity_type;
223
224 if (ha_light_cmd_queue != NULL) {
225 xQueueSend(ha_light_cmd_queue, &cmd, 0); // Non-blocking send
226 }
227}

232static void ha_light_check_activity_timeout(void) {
233 if (!task_running || current_mode == HA_MODE_HA) {
234 return; // Already in HA mode or task not running
235 }
236
237 if (last_activity_time_ms == 0) {
238 // First run - initialize timer
239 last_activity_time_ms = esp_timer_get_time() / 1000;
240 return;
241 }
242
243 uint32_t current_time_ms = esp_timer_get_time() / 1000;
244 uint32_t elapsed_ms = current_time_ms - last_activity_time_ms;
245
246 if (elapsed_ms >= HA_ACTIVITY_TIMEOUT_AUTO_MS) {
247 // 10 minutes passed - auto-switch to HA mode
248 if (ha_light_check_wifi_sta_connected()) {
249 ESP_LOGI(TAG, "10 minute timeout reached - auto-switching to HA mode");
250 ha_light_switch_to_ha_mode();
251 } else {
252 ESP_LOGD(TAG, "5 minute timeout reached but WiFi STA not connected - "
253 "staying in GAME mode");
254 }
255 }
256}

267static void ha_light_switch_to_ha_mode(void) {
268 bool mode_changed = (current_mode != HA_MODE_HA);
269
270 if (mode_changed) {
271 ESP_LOGI(TAG, "Switching to HA MODE");
272 }
273 current_mode = HA_MODE_HA;
274
275 // Apply HA color/state to all 64 board LEDs
276 if (ha_light_state.state) {
277 // ON: Apply RGB color with brightness
278 led_set_ha_color(ha_light_state.r, ha_light_state.g, ha_light_state.b,
279 ha_light_state.brightness);
280 } else {
281 // OFF: Turn all LEDs off (black)
282 led_set_ha_color(0, 0, 0, 0);
283 }
284
285 // Publish state update
286 ha_light_publish_state();
287}

294static void ha_light_switch_to_game_mode(void) {
295 if (current_mode == HA_MODE_GAME) {
296 return; // Already in game mode
297 }
298
299 ESP_LOGI(TAG, "Switching to GAME MODE");
300 current_mode = HA_MODE_GAME;
301
302 // Restore game LED state (refresh based on current game status)
303 game_refresh_leds();
304
305 // Obnovit tlačítka (zelená/modrá/červená podle dostupnosti a stisku)
306 led_refresh_all_button_leds();
307
308 // Publish state update
309 ha_light_publish_state();
310}

329static esp_err_t mqtt_load_config_from_nvs(char *host, size_t host_len,
330 uint16_t *port, char *username,
331 size_t username_len, char *password,
332 size_t password_len) {
333 if (host == NULL || port == NULL || username == NULL || password == NULL ||
334 host_len == 0 || username_len == 0 || password_len == 0) {
335 ESP_LOGE(TAG, "Invalid parameters for mqtt_load_config_from_nvs");
336 return ESP_ERR_INVALID_ARG;
337 }
338
339 nvs_handle_t nvs_handle;
340 esp_err_t ret = nvs_open(MQTT_NVS_NAMESPACE, NVS_READONLY, &nvs_handle);
341 if (ret != ESP_OK) {
342 ESP_LOGD(TAG, "MQTT config not found in NVS, using defaults");
343 // Vrátit default hodnoty
344 strncpy(host, MQTT_DEFAULT_HOST, host_len - 1);
345 host[host_len - 1] = '\0';
346 *port = MQTT_DEFAULT_PORT;
347 strncpy(username, MQTT_DEFAULT_USERNAME, username_len - 1);
348 username[username_len - 1] = '\0';
349 strncpy(password, MQTT_DEFAULT_PASSWORD, password_len - 1);
350 password[password_len - 1] = '\0';
351 return ESP_ERR_NOT_FOUND; // Není v NVS, ale vrátil defaults
352 }
353
354 // Načíst host
355 size_t required_size = host_len;
356 ret = nvs_get_str(nvs_handle, MQTT_NVS_KEY_HOST, host, &required_size);
357 if (ret != ESP_OK) {
358 ESP_LOGW(TAG, "Failed to get MQTT host from NVS, using default");
359 strncpy(host, MQTT_DEFAULT_HOST, host_len - 1);
360 host[host_len - 1] = '\0';
361 }
362
363 // Načíst port
364 uint32_t port_val = MQTT_DEFAULT_PORT;
365 ret = nvs_get_u32(nvs_handle, MQTT_NVS_KEY_PORT, &port_val);
366 if (ret != ESP_OK || port_val == 0 || port_val > 65535) {
367 ESP_LOGW(TAG, "Failed to get MQTT port from NVS, using default");
368 port_val = MQTT_DEFAULT_PORT;
369 }
370 *port = (uint16_t)port_val;
371
372 // Načíst username (volitelné - může být prázdné)
373 required_size = username_len;
374 ret =
375 nvs_get_str(nvs_handle, MQTT_NVS_KEY_USERNAME, username, &required_size);
376 if (ret != ESP_OK) {
377 ESP_LOGD(TAG, "MQTT username not in NVS, using empty");
378 strncpy(username, MQTT_DEFAULT_USERNAME, username_len - 1);
379 username[username_len - 1] = '\0';
380 }
381
382 // Načíst password (volitelné - může být prázdné)
383 required_size = password_len;
384 ret =
385 nvs_get_str(nvs_handle, MQTT_NVS_KEY_PASSWORD, password, &required_size);
386 if (ret != ESP_OK) {
387 ESP_LOGD(TAG, "MQTT password not in NVS, using empty");
388 strncpy(password, MQTT_DEFAULT_PASSWORD, password_len - 1);
389 password[password_len - 1] = '\0';
390 }
391
392 nvs_close(nvs_handle);
393 ESP_LOGI(TAG, "MQTT config loaded from NVS: host=%s, port=%d, username=%s",
394 host, *port, username[0] ? username : "(empty)");
395
396 return ESP_OK;
397}

408esp_err_t mqtt_save_config_to_nvs(const char *host, uint16_t port,
409 const char *username, const char *password) {
410 if (host == NULL || port == 0) {
411 ESP_LOGE(TAG, "Invalid parameters: host is NULL or port is 0");
412 return ESP_ERR_INVALID_ARG;
413 }
414
415 size_t host_len = strlen(host);
416 if (host_len == 0 || host_len > 127) {
417 ESP_LOGE(TAG, "Invalid host length: %zu (must be 1-127)", host_len);
418 return ESP_ERR_INVALID_ARG;
419 }
420
421 // Validovat username/password pokud jsou nastavené
422 if (username != NULL) {
423 size_t username_len = strlen(username);
424 if (username_len > 63) {
425 ESP_LOGE(TAG, "Invalid username length: %zu (max 63)", username_len);
426 return ESP_ERR_INVALID_ARG;
427 }
428 }
429
430 if (password != NULL) {
431 size_t password_len = strlen(password);
432 if (password_len > 63) {
433 ESP_LOGE(TAG, "Invalid password length: %zu (max 63)", password_len);
434 return ESP_ERR_INVALID_ARG;
435 }
436 }
437
438 nvs_handle_t nvs_handle;
439 esp_err_t ret = nvs_open(MQTT_NVS_NAMESPACE, NVS_READWRITE, &nvs_handle);
440 if (ret != ESP_OK) {
441 ESP_LOGE(TAG, "Failed to open NVS: %s", esp_err_to_name(ret));
442 return ret;
443 }
444
445 // Uložit host
446 ret = nvs_set_str(nvs_handle, MQTT_NVS_KEY_HOST, host);
447 if (ret != ESP_OK) {
448 ESP_LOGE(TAG, "Failed to set MQTT host in NVS: %s", esp_err_to_name(ret));
449 nvs_close(nvs_handle);
450 return ret;
451 }
452
453 // Uložit port
454 ret = nvs_set_u32(nvs_handle, MQTT_NVS_KEY_PORT, (uint32_t)port);
455 if (ret != ESP_OK) {
456 ESP_LOGE(TAG, "Failed to set MQTT port in NVS: %s", esp_err_to_name(ret));
457 nvs_close(nvs_handle);
458 return ret;
459 }
460
461 // Uložit username (pokud je nastavený, jinak uložit prázdný string)
462 const char *username_to_save =
463 (username != NULL && strlen(username) > 0) ? username : "";
464 ret = nvs_set_str(nvs_handle, MQTT_NVS_KEY_USERNAME, username_to_save);
465 if (ret != ESP_OK) {
466 ESP_LOGE(TAG, "Failed to set MQTT username in NVS: %s",
467 esp_err_to_name(ret));
468 nvs_close(nvs_handle);
469 return ret;
470 }
471
472 // Uložit password (pokud je nastavený, jinak uložit prázdný string)
473 const char *password_to_save =
474 (password != NULL && strlen(password) > 0) ? password : "";
475 ret = nvs_set_str(nvs_handle, MQTT_NVS_KEY_PASSWORD, password_to_save);
476 if (ret != ESP_OK) {
477 ESP_LOGE(TAG, "Failed to set MQTT password in NVS: %s",
478 esp_err_to_name(ret));
479 nvs_close(nvs_handle);
480 return ret;
481 }
482
483 ret = nvs_commit(nvs_handle);
484 if (ret != ESP_OK) {
485 ESP_LOGE(TAG, "Failed to commit NVS: %s", esp_err_to_name(ret));
486 nvs_close(nvs_handle);
487 return ret;
488 }
489
490 nvs_close(nvs_handle);
491 ESP_LOGI(TAG, "MQTT config saved to NVS: host=%s, port=%d, username=%s", host,
492 port, username_to_save[0] ? username_to_save : "(empty)");
493
494 return ESP_OK;
495}

504static void ha_light_publish_discovery(void) {
505 if (!mqtt_connected || mqtt_client == NULL) {
506 return;
507 }
508
509 // Get MAC address for unique ID
510 uint8_t mac_raw[6];
511 esp_read_mac(mac_raw, ESP_MAC_WIFI_STA);
512 char mac_str[13];
513 snprintf(mac_str, sizeof(mac_str), "%02x%02x%02x%02x%02x%02x", mac_raw[0],
514 mac_raw[1], mac_raw[2], mac_raw[3], mac_raw[4], mac_raw[5]);
515
516 // Construct unique ID and name
517 char unique_id[64];
518 snprintf(unique_id, sizeof(unique_id), "esp32_chess_light_%s", mac_str);
519
520 // Construct discovery topic: homeassistant/light/[node_id]/config
521 char discovery_topic[128];
522 snprintf(discovery_topic, sizeof(discovery_topic), "%s/%s/%s/config",
523 HA_DISCOVERY_PREFIX, HA_COMPONENT_LIGHT, unique_id);
524
525 ESP_LOGI(TAG, "Publishing HA Discovery to %s", discovery_topic);
526
527 // Create JSON payload
528 cJSON *json = cJSON_CreateObject();
529 cJSON_AddStringToObject(json, "name", "CzechMate");
530 cJSON_AddStringToObject(json, "unique_id", unique_id);
531 cJSON_AddStringToObject(json, "command_topic", HA_TOPIC_LIGHT_COMMAND);
532 cJSON_AddStringToObject(json, "state_topic", HA_TOPIC_LIGHT_STATE);
533 cJSON_AddStringToObject(json, "availability_topic",
534 HA_TOPIC_LIGHT_AVAILABILITY);
535 cJSON_AddStringToObject(json, "payload_available", HA_MQTT_PAYLOAD_ONLINE);
536 cJSON_AddStringToObject(json, "payload_not_available",
537 HA_MQTT_PAYLOAD_OFFLINE);
538 cJSON_AddStringToObject(json, "schema", "json");
539 cJSON_AddBoolToObject(json, "brightness", true);
540 cJSON_AddBoolToObject(json, "color_mode", true);
541 cJSON *color_modes = cJSON_CreateArray();
542 cJSON_AddItemToArray(color_modes, cJSON_CreateString("rgb"));
543 cJSON_AddItemToObject(json, "supported_color_modes", color_modes);
544
545 // Effect support
546 cJSON_AddBoolToObject(json, "effect", true);
547 cJSON *effects = cJSON_CreateArray();
548 cJSON_AddItemToArray(effects, cJSON_CreateString("rainbow"));
549 cJSON_AddItemToArray(effects, cJSON_CreateString("pulse"));
550 cJSON_AddItemToArray(effects, cJSON_CreateString("static"));
551 cJSON_AddItemToObject(json, "effect_list", effects);
552
553 // Device info
554 cJSON *device = cJSON_CreateObject();
555 cJSON_AddStringToObject(device, "identifiers", unique_id);
556 cJSON_AddStringToObject(device, "name", "ESP32 Chess System");
557 cJSON_AddStringToObject(device, "manufacturer", HA_DEVICE_MANUFACTURER);
558 cJSON_AddStringToObject(device, "model", HA_DEVICE_MODEL);
559 cJSON_AddStringToObject(device, "sw_version", HA_DEVICE_SW_VERSION);
560 cJSON_AddItemToObject(json, "device", device);
561
562 char *json_str = cJSON_PrintUnformatted(json);
563 if (json_str != NULL) {
564 // Publish with retain=true so HA picks it up anytime
565 esp_mqtt_client_publish(mqtt_client, discovery_topic, json_str, 0, 1, 1);
566 free(json_str);
567 }
568 cJSON_Delete(json);
569}

574static void ha_light_mqtt_event_handler(void *handler_args,
575 esp_event_base_t base, int32_t event_id,
576 void *event_data) {
577 esp_mqtt_event_handle_t event = (esp_mqtt_event_handle_t)event_data;
578 esp_mqtt_client_handle_t client = event->client;
579
580 switch ((esp_mqtt_event_id_t)event_id) {
581 case MQTT_EVENT_CONNECTED:
582 ESP_LOGI(TAG, "MQTT Connected");
583 mqtt_connected = true;
584
585 // Subscribe to light command topic
586 esp_mqtt_client_subscribe(client, HA_TOPIC_LIGHT_COMMAND, 0);
587 ESP_LOGI(TAG, "Subscribed to %s", HA_TOPIC_LIGHT_COMMAND);
588
589 // Publish Auto-Discovery Config
590 ha_light_publish_discovery();
591
592 // Publish availability: online
593 esp_mqtt_client_publish(client, HA_TOPIC_LIGHT_AVAILABILITY,
594 HA_MQTT_PAYLOAD_ONLINE, 0, 1, 1);
595
596 // Publish initial state
597 ha_light_publish_state();
598 break;
599
600 case MQTT_EVENT_DISCONNECTED:
601 ESP_LOGI(TAG, "MQTT Disconnected");
602 mqtt_connected = false;
603 break;
604
605 case MQTT_EVENT_DATA:
606 ESP_LOGI(TAG, "MQTT DATA: topic=%.*s, data=%.*s", event->topic_len,
607 event->topic, event->data_len, event->data);
608
609 // Handle command
610 char topic[128] = {0};
611 char data[512] = {0};
612
613 if (event->topic_len < sizeof(topic) && event->data_len < sizeof(data)) {
614 memcpy(topic, event->topic, event->topic_len);
615 memcpy(data, event->data, event->data_len);
616 ha_light_handle_mqtt_command(topic, data, event->data_len);
617 }
618 break;
619
620 case MQTT_EVENT_ERROR:
621 ESP_LOGW(TAG, "MQTT Error");
622 break;
623
624 default:
625 break;
626 }
627}

632static void ha_light_handle_mqtt_command(const char *topic, const char *data,
633 int data_len) {
634 if (strcmp(topic, HA_TOPIC_LIGHT_COMMAND) != 0) {
635 return; // Not our topic
636 }
637
638 // Check if we can apply HA command during GAME mode
639 bool can_apply_command = true;
640 if (current_mode == HA_MODE_GAME) {
641 uint32_t current_time_ms = esp_timer_get_time() / 1000;
642 uint32_t idle_ms = current_time_ms - last_activity_time_ms;
643
644 if (idle_ms >= HA_ACTIVITY_TIMEOUT_COMMAND_MS ||
645 game_get_move_count() == 0) {
646 // Idle ≥ 2min OR no moves yet → allow switch to HA mode
647 if (game_get_move_count() == 0) {
648 ESP_LOGI(TAG,
649 "HA command accepted immediately (game not started, moves=0)");
650 } else {
651 ESP_LOGI(
652 TAG,
653 "HA command after %lu ms idle (≥2min) - will switch to HA mode",
654 idle_ms);
655 }
656 can_apply_command = true;
657 } else {
658 // Idle < 2min AND game in progress → ignore command (game has priority)
659 ESP_LOGI(
660 TAG,
661 "HA command IGNORED (game active, moves=%lu, idle=%lu ms < 2min)",
662 game_get_move_count(), idle_ms);
663 can_apply_command = false;
664 }
665 }
666
667 // Parse JSON
668 cJSON *json = cJSON_ParseWithLength(data, data_len);
669 if (json == NULL) {
670 ESP_LOGE(TAG, "Failed to parse MQTT JSON");
671 return;
672 }
673
674 // Parse state (on/off)
675 cJSON *state_item = cJSON_GetObjectItem(json, "state");
676 if (cJSON_IsString(state_item)) {
677 if (strcasecmp(state_item->valuestring, "ON") == 0) {
678 ha_light_state.state = true;
679 } else if (strcasecmp(state_item->valuestring, "OFF") == 0) {
680 ha_light_state.state = false;
681 }
682 }
683
684 // Parse brightness
685 cJSON *brightness_item = cJSON_GetObjectItem(json, "brightness");
686 if (cJSON_IsNumber(brightness_item)) {
687 int brightness_val = (int)cJSON_GetNumberValue(brightness_item);
688 if (brightness_val < 0)
689 brightness_val = 0;
690 if (brightness_val > 255)
691 brightness_val = 255;
692 ha_light_state.brightness = (uint8_t)brightness_val;
693 }
694
695 // Parse color (JSON schema uses "color": {"r": 255, "g": 255, "b": 255})
696 cJSON *color_item = cJSON_GetObjectItem(json, "color");
697 if (cJSON_IsObject(color_item)) {
698 cJSON *r = cJSON_GetObjectItem(color_item, "r");
699 cJSON *g = cJSON_GetObjectItem(color_item, "g");
700 cJSON *b = cJSON_GetObjectItem(color_item, "b");
701

839static esp_err_t ha_light_init_mqtt(void) {
840 // Prevent multiple clients/memory leaks
841 if (mqtt_client != NULL) {
842 ESP_LOGI(TAG, "MQTT client already initialized, skipping re-init");
843 return ESP_OK;
844 }
845
846 // Ensure config is loaded from NVS (first time only)
847 mqtt_ensure_config_loaded();
848
849 // Build MQTT broker URI (buffer size 512 to avoid format-truncation warning)
850 char mqtt_broker_uri[512];
851 if (mqtt_config.username[0] != '\0' && mqtt_config.password[0] != '\0') {
852 // MQTT URI with username/password: mqtt://user:pass@host:port
853 snprintf(mqtt_broker_uri, sizeof(mqtt_broker_uri), "mqtt://%s:%s@%s:%d",
854 mqtt_config.username, mqtt_config.password, mqtt_config.host,
855 mqtt_config.port);
856 } else {
857 // MQTT URI without auth: mqtt://host:port
858 snprintf(mqtt_broker_uri, sizeof(mqtt_broker_uri), "mqtt://%s:%d",
859 mqtt_config.host, mqtt_config.port);
860 }
861
862 // Generate unique Client ID combined from Base + MAC
863 uint8_t mac_raw[6];
864 esp_read_mac(mac_raw, ESP_MAC_WIFI_STA);
865 char client_id[64];
866 snprintf(client_id, sizeof(client_id), "%s-%02x%02x%02x%02x%02x%02x",
867 HA_MQTT_CLIENT_ID, mac_raw[0], mac_raw[1], mac_raw[2], mac_raw[3],
868 mac_raw[4], mac_raw[5]);
869
870 ESP_LOGI(TAG, "Initializing MQTT client: %s (ID: %s)", mqtt_broker_uri,
871 client_id);
872
873 esp_mqtt_client_config_t mqtt_cfg = {
874 .broker.address.uri = mqtt_broker_uri,
875 .credentials.client_id = client_id,
876 .session.last_will.topic = HA_TOPIC_LIGHT_AVAILABILITY,
877 .session.last_will.msg = HA_MQTT_PAYLOAD_OFFLINE,
878 .session.last_will.qos = 1,
879 .session.last_will.retain = true,
880 .session.keepalive = 120, // Increase keepalive to 120s for stability
881 .network.disable_auto_reconnect =
882 false, // Ensure auto-reconnect is enabled
883 };
884
885 mqtt_client = esp_mqtt_client_init(&mqtt_cfg);
886 if (mqtt_client == NULL) {
887 ESP_LOGE(TAG, "Failed to initialize MQTT client");
888 return ESP_FAIL;
889 }
890
891 esp_mqtt_client_register_event(mqtt_client, ESP_EVENT_ANY_ID,
892 ha_light_mqtt_event_handler, NULL);
893
894 esp_err_t ret = esp_mqtt_client_start(mqtt_client);
895 if (ret != ESP_OK) {
896 ESP_LOGE(TAG, "Failed to start MQTT client: %s", esp_err_to_name(ret));
897 return ret;
898 }
899
900 ESP_LOGI(TAG, "MQTT client initialized and started");
901 return ESP_OK;
902}
