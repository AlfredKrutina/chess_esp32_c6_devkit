30uint8_t chess_pos_to_led_index(uint8_t row, uint8_t col)
31{
32 if (row >= 8 || col >= 8) {
33 ESP_LOGE(TAG, "Invalid chess position: row=%d, col=%d", row, col);
34 return 0;
35 }
36
37 // Sachovnice je otocena po Y ose (a1 je na h1 pozici)
38 // Takze col=0 (a) musi byt mapovano na LED pozici 7 (h)
39 uint8_t mapped_col = 7 - col;
40
41 if (row % 2 == 0) {
42 // Sude radky (0,2,4,6): normalni poradi a->h (ale s otocenymi sloupci)
43 return row * 8 + mapped_col;
44 } else {
45 // Liche radky (1,3,5,7): obracene poradi h->a (ale s otocenymi sloupci)
46 return row * 8 + (7 - mapped_col);
47 }
48}

60void led_index_to_chess_pos(uint8_t led_index, uint8_t* row, uint8_t* col)
61{
62 if (led_index >= 64 || row == NULL || col == NULL) {
63 ESP_LOGE(TAG, "Invalid LED index: %d", led_index);
64 if (row) *row = 0;
65 if (col) *col = 0;
66 return;
67 }
68
69 *row = led_index / 8;
70 uint8_t pos_in_row = led_index % 8;
71
72 if (*row % 2 == 0) {
73 // Sude radky: normalni poradi (ale s otocenymi sloupci)
74 *col = 7 - pos_in_row;
75 } else {
76 // Liche radky: obracene poradi (ale s otocenymi sloupci)
77 *col = pos_in_row;
78 }
79}

90uint8_t chess_notation_to_led_index(const char* notation)
91{
92 if (!notation || strlen(notation) < 2) {
93 ESP_LOGE(TAG, "Invalid notation: %s", notation ? notation : "NULL");
94 return 0;
95 }
96
97 char file = tolower(notation[0]);
98 char rank = notation[1];
99
100 if (file < 'a' || file > 'h' || rank < '1' || rank > '8') {
101 ESP_LOGE(TAG, "Invalid notation: %s", notation);
102 return 0;
103 }
104
105 uint8_t col = file - 'a'; // a=0, b=1, ..., h=7
106 uint8_t row = rank - '1'; // 1=0, 2=1, ..., 8=7
107
108 return chess_pos_to_led_index(row, col);
109}
