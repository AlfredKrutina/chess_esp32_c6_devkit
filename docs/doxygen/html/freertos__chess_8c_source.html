164// Enhanced GPIO validation function
165static esp_err_t validate_gpio_pin(gpio_num_t pin, const char *pin_name) {
166 // Check if pin exists
167 if (pin < 0 || pin > 30) {
168 ESP_LOGE(TAG, "Invalid GPIO pin %d for %s (ESP32-C6 has GPIO 0-30)", pin,
169 pin_name);
170 return ESP_ERR_INVALID_ARG;
171 }
172
173 // Check SPI Flash reserved pins (GPIO 24-26 are reserved, but 27-30 can be
174 // used for I/O)
175 if (pin >= 24 && pin <= 26) {
176 ESP_LOGE(TAG, "GPIO %d (%s) is reserved for SPI Flash on ESP32-C6", pin,
177 pin_name);
178 return ESP_ERR_INVALID_ARG;
179 }
180
181 // Warn about SPI Flash pins that can be used for I/O
182 if (pin >= 27 && pin <= 30) {
183 ESP_LOGW(TAG,
184 "GPIO %d (%s) is SPI Flash pin but can be used for I/O - use with "
185 "caution",
186 pin, pin_name);
187 }
188
189 // Warn about strapping pins
190 if (pin == 4 || pin == 5 || pin == 8 || pin == 9 || pin == 15) {
191 ESP_LOGW(TAG, "GPIO %d (%s) is a strapping pin - use with caution", pin,
192 pin_name);
193 }
194
195 // Warn about USB-JTAG pins
196 if (pin == 12 || pin == 13) {
197 ESP_LOGW(TAG, "GPIO %d (%s) is used for USB-JTAG debugging", pin, pin_name);
198 }
199
200 return ESP_OK;
201}

207esp_err_t chess_gpio_init(void) {
208 ESP_LOGI(TAG, "Initializing GPIO pins...");
209
210 // DEBUG: Ověření definice pinů
211 ESP_LOGI(TAG, "DEBUG: STATUS_LED_PIN = GPIO%d, BUTTON_RESET = GPIO%d",
212 STATUS_LED_PIN, BUTTON_RESET);
213 ESP_LOGI(TAG, "DEBUG: LED_DATA_PIN = GPIO%d", LED_DATA_PIN);
214
215 // GPIO initialization - no WDT reset needed during init
216
217 // Enhanced GPIO validation using the new validation function
218 ESP_LOGI(TAG, "Validating GPIO pin assignments...");
219
220 // Validate LED pins
221 esp_err_t ret = validate_gpio_pin(LED_DATA_PIN, "LED_DATA_PIN");
222 if (ret != ESP_OK)
223 return ret;
224
225 ret = validate_gpio_pin(STATUS_LED_PIN, "STATUS_LED_PIN");
226 if (ret != ESP_OK)
227 return ret;
228
229 // Validate matrix row pins
230 for (int i = 0; i < 8; i++) {
231 // GPIO validation - no WDT reset needed during init
232 ret = validate_gpio_pin(matrix_row_pins[i], "MATRIX_ROW");
233 if (ret != ESP_OK)
234 return ret;
235 }
236
237 // Validate matrix column pins
238 for (int i = 0; i < 8; i++) {
239 // GPIO validation - no WDT reset needed during init
240 ret = validate_gpio_pin(matrix_col_pins[i], "MATRIX_COL");
241 if (ret != ESP_OK)
242 return ret;
243 }
244
245 // Validate reset button pin
246 ret = validate_gpio_pin(BUTTON_RESET, "BUTTON_RESET");
247 if (ret != ESP_OK)
248 return ret;
249
250 ESP_LOGI(TAG, "✓ GPIO safety checks passed");
251
252 // GPIO configuration - no WDT reset needed during init
253
254 // Configure matrix row pins as outputs
255 for (int i = 0; i < 8; i++) {
256 // GPIO config - no WDT reset needed during init
257
258 // Bitová maska s explicitním přetypováním
259 uint32_t pin_number = (uint32_t)matrix_row_pins[i];
260 uint64_t pin_mask = (1ULL << pin_number);
261
262 ESP_LOGI(TAG, "Configuring MATRIX_ROW_%d (GPIO%d, mask=0x%llx)...", i,
263 pin_number, pin_mask);
264
265 gpio_config_t io_conf = {.pin_bit_mask = pin_mask,
266 .mode = GPIO_MODE_OUTPUT,
267 .pull_up_en = GPIO_PULLUP_DISABLE,
268 .pull_down_en = GPIO_PULLDOWN_DISABLE,
269 .intr_type = GPIO_INTR_DISABLE};
270 ret = gpio_config(&io_conf);
271 ESP_LOGI(TAG, "gpio_config returned %s", esp_err_to_name(ret));
272
273 if (ret != ESP_OK) {
274 ESP_LOGE(TAG, "Failed to configure matrix row pin %d (GPIO%d): %s", i,
275 pin_number, esp_err_to_name(ret));
276 return ret; // Return error instead of continuing
277 } else {
278 gpio_set_level(matrix_row_pins[i],
279 1); // Set all rows HIGH initially (inactive state)
280 ESP_LOGI(TAG, "gpio_set_level done for GPIO%d", pin_number);
281 }
282 }
283
284 // Configure matrix column pins as inputs with pull-up
285 ESP_LOGI(TAG, "DEBUG: Starting matrix column configuration loop");
286 for (int i = 0; i < 8; i++) {
287 // WDT reset odstraněn během inicializace
288
289 // Bitová maska s explicitním přetypováním
290 uint32_t pin_number = (uint32_t)matrix_col_pins[i];
291 uint64_t pin_mask = (1ULL << pin_number);
292
293 ESP_LOGI(TAG, "Configuring MATRIX_COL_%d (GPIO%d, mask=0x%llx)...", i,
294 pin_number, pin_mask);
295
296 // DEBUG: Special logging for GPIO17 (column 7)
297 if (pin_number == 17) {
298 ESP_LOGI(TAG, "🔍 DEBUG: Configuring GPIO17 (MATRIX_COL_7, index=%d)", i);
299 }
300
301 // CRITICAL: Skip strapping pins to avoid system reset
302 if (pin_number == 9) {
303 ESP_LOGW(TAG, "Skipping GPIO%d (strapping pin) to avoid system reset",
304 pin_number);
305 ESP_LOGI(TAG, "Matrix column pin %d skipped (strapping pin)", i);
306 ESP_LOGI(TAG, "DEBUG: About to continue to next iteration");
307 continue; // Přeskočení strapping pinu
308 }
309
310 ESP_LOGI(TAG, "DEBUG: Proceeding with GPIO%d configuration", pin_number);
311
312 // Explicitly reset GPIO17 before configuration
313 // GPIO17 is UART0 RX pin on ESP32-C6 and may have special properties
314 // that need to be cleared before using it as GPIO input
315 if (pin_number == 17) {
316 ESP_LOGI(TAG, "🔍 DEBUG: Resetting GPIO17 before configuration (UART0 RX pin)");
317 gpio_reset_pin(GPIO_NUM_17);
318 // Small delay to ensure reset is complete
319 vTaskDelay(pdMS_TO_TICKS(10));
320
321 // CRITICAL: Explicitly switch GPIO17 from UART function to GPIO function
322 // GPIO17 is UART0 RX pin on ESP32-C6 and may be stuck in UART mode
323 // even if UART driver is not active. We need to force it to GPIO mode.
324 // Strategy: Call gpio_set_direction() BEFORE gpio_config() to ensure
325 // IOMUX is switched to GPIO function before configuration
326 gpio_set_direction(GPIO_NUM_17, GPIO_MODE_INPUT);
327 ESP_LOGI(TAG, "🔧 DEBUG: GPIO17 direction set to INPUT (forcing IOMUX switch from UART to GPIO)");
328
329 // Small delay to ensure IOMUX switch is complete
330 vTaskDelay(pdMS_TO_TICKS(5));
331 }
332
333 // WDT reset odstraněn během inicializace
334
335 // Configure all matrix column pins as INPUT with pull-up (standard configuration)
336 gpio_config_t io_conf = {.pin_bit_mask = pin_mask,
337 .mode = GPIO_MODE_INPUT,
338 .pull_up_en = GPIO_PULLUP_ENABLE,
339 .pull_down_en = GPIO_PULLDOWN_DISABLE,
340 .intr_type = GPIO_INTR_DISABLE};
341 ret = gpio_config(&io_conf);
342 ESP_LOGI(TAG, "gpio_config returned %s", esp_err_to_name(ret));
343
344 if (ret != ESP_OK) {
345 ESP_LOGE(TAG, "Failed to configure matrix column pin %d (GPIO%d): %s", i,
346 pin_number, esp_err_to_name(ret));
347 return ret; // Return error instead of continuing
348 } else {
349 ESP_LOGI(TAG, "Matrix column pin %d configured successfully", i);
350
351 // Verify GPIO17 configuration (UART0 RX pin - needs explicit GPIO mode)
352 if (pin_number == 17) {
353 // Explicitly set GPIO17 to GPIO INPUT mode (not UART mode)
354 // GPIO17 is UART0 RX pin on ESP32-C6 and may default to UART mode
355 // even if UART is not active. Force it to GPIO INPUT mode.
356 gpio_set_direction(GPIO_NUM_17, GPIO_MODE_INPUT);
357 gpio_set_pull_mode(GPIO_NUM_17, GPIO_PULLUP_ONLY);
358
359 // Verify configuration
360 int test_level = gpio_get_level(GPIO_NUM_17);
361 ESP_LOGI(TAG, "🔧 GPIO17: Configured as INPUT with pull-up, initial level=%d", test_level);
362 }
363 }
364 }
365 ESP_LOGI(TAG, "DEBUG: Matrix column configuration loop completed");
366
367 ESP_LOGI(TAG, "DEBUG: Matrix column configuration loop completed");
368
369 // Configure status LED pin
370 // WDT reset removed during initialization // Reset před status LED config
371 ESP_LOGI(TAG, "DEBUG: About to configure STATUS_LED");
372
373 // Správná bitová maska s explicitním castem
374 uint32_t status_led_pin = (uint32_t)STATUS_LED_PIN;
375 uint64_t status_led_mask = (1ULL << status_led_pin);
376
377 ESP_LOGI(TAG, "Configuring STATUS_LED (GPIO%d, mask=0x%llx)...",
378 status_led_pin, status_led_mask);
379
380 // Reset před gpio_config()
381 // WDT reset removed during initialization
382
383 gpio_config_t status_led_conf = {.pin_bit_mask = status_led_mask,
384 .mode = GPIO_MODE_OUTPUT,
385 .pull_up_en = GPIO_PULLUP_DISABLE,
386 .pull_down_en = GPIO_PULLDOWN_DISABLE,
387 .intr_type = GPIO_INTR_DISABLE};
388 ret = gpio_config(&status_led_conf);
389 ESP_LOGI(TAG, "gpio_config returned %s", esp_err_to_name(ret));
390
391 if (ret != ESP_OK) {
392 ESP_LOGE(TAG, "Failed to configure status LED pin (GPIO%d): %s",
393 status_led_pin, esp_err_to_name(ret));
394 return ret; // Return error instead of continuing
395 } else {
396 gpio_set_level(STATUS_LED_PIN, 0); // LED off initially
397 ESP_LOGI(TAG, "Status LED configured successfully");
398 }
399
400 // Configure reset button pin
401 // WDT reset removed during initialization // Reset před reset button config
402
403 // Správná bitová maska s explicitním castem
404 uint32_t reset_button_pin = (uint32_t)BUTTON_RESET;
405 uint64_t reset_button_mask = (1ULL << reset_button_pin);
406
407 ESP_LOGI(TAG, "Configuring RESET_BUTTON (GPIO%d, mask=0x%llx)...",
408 reset_button_pin, reset_button_mask);
409
410 // Reset před gpio_config()
411 // WDT reset removed during initialization
412
413 gpio_config_t reset_button_conf = {.pin_bit_mask = reset_button_mask,
414 .mode = GPIO_MODE_INPUT,
415 .pull_up_en = GPIO_PULLUP_ENABLE,
416 .pull_down_en = GPIO_PULLDOWN_DISABLE,
417 .intr_type = GPIO_INTR_DISABLE};
418 ret = gpio_config(&reset_button_conf);
419 ESP_LOGI(TAG, "gpio_config returned %s", esp_err_to_name(ret));
420
421 if (ret != ESP_OK) {
422 ESP_LOGE(TAG, "Failed to configure reset button pin (GPIO%d): %s",
423 reset_button_pin, esp_err_to_name(ret));
424 return ret; // Return error instead of continuing
425 } else {
426 ESP_LOGI(TAG, "Reset button configured successfully");
427 }
428
429 // PŘIDAT: Finální reset
430 // WDT reset removed during initialization
431
432 ESP_LOGI(TAG, "✓ GPIO pins initialized successfully");
433
434 // Dokončení GPIO konfigurace
435 // WDT reset removed during initialization
436
437 // Fallback: If any GPIO configuration failed, log warning but continue
438 // ESP_LOGW(TAG, " Some GPIO pins may not be configured correctly - running
439 // in simulation mode"); ESP_LOGW(TAG, " Hardware functionality will be
440 // limited - check GPIO pin assignments");
441
442 return ESP_OK;
443}

804static void coordinated_multiplex_task(void *pvParameters) {
805 (void)pvParameters;
806
807 ESP_LOGI(TAG,
808 "✅ Coordinated multiplexing TASK started (25ms cycle, stack=%dKB)",
809 COORDINATED_MUX_TASK_STACK_SIZE / 1024);
810
811 // Register with TWDT (optional; safe wrapper pattern is used elsewhere)
812 esp_err_t wdt_ret = esp_task_wdt_add(NULL);
813 if (wdt_ret != ESP_OK && wdt_ret != ESP_ERR_INVALID_ARG) {
814 ESP_LOGW(TAG, "Multiplex task WDT registration failed: %s",
815 esp_err_to_name(wdt_ret));
816 }
817
818 TickType_t last_wake_time = xTaskGetTickCount();
819
820 for (;;) {
821 // Reset WDT (if registered)
822 esp_task_wdt_reset();
823
824 // PHASE 1: MATRIX scan
825 extern bool matrix_scanning_enabled;
826 if (matrix_scanning_enabled) {
827 extern void matrix_scan_all(void);
828 matrix_scan_all();
829 }
830
831 // PHASE 2: release matrix pins for button scan
832 extern void matrix_release_pins(void);
833 matrix_release_pins();
834
835 // PHASE 3: BUTTON scan
836 extern void button_scan_all(void);
837 button_scan_all();
838
839 // PHASE 4: re-acquire matrix pins
840 extern void matrix_acquire_pins(void);
841 matrix_acquire_pins();
842
843 // Full multiplexing cycle
844 vTaskDelayUntil(&last_wake_time, pdMS_TO_TICKS(TOTAL_CYCLE_TIME_MS));
845 }
846}
