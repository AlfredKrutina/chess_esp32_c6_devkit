101__attribute__((unused)) static rgb_color_t interpolate_color(rgb_color_t from, rgb_color_t to, float progress) {
102 rgb_color_t result;
103 result.r = from.r + (to.r - from.r) * progress;
104 result.g = from.g + (to.g - from.g) * progress;
105 result.b = from.b + (to.b - from.b) * progress;
106 return result;
107}

124static float get_board_distance(uint8_t pos1, uint8_t pos2) {
125 uint8_t row1, col1, row2, col2;
126 led_index_to_chess_pos(pos1, &row1, &col1);
127 led_index_to_chess_pos(pos2, &row2, &col2);
128
129 float dx = col2 - col1;
130 float dy = row2 - row1;
131 return sqrtf(dx*dx + dy*dy);
132}

141static void endgame_animation_victory_wave(uint32_t frame) {
142 if (wave_state.frame == 0) {
143 // Inicializace vlny
144 wave_state.center_pos = winning_king_position;
145 wave_state.max_radius = 10.0f;
146 wave_state.current_radius = 0.0f;
147 wave_state.wave_speed = 0.3f;
148 wave_state.active_waves = 3;
149
150 for (int i = 0; i < MAX_WAVES; i++) {
151 wave_state.waves[i].radius = -2.0f * i; // Postupn√© spou≈°tƒõn√≠ vln
152 wave_state.waves[i].active = true;
153 }
154
155 ESP_LOGI(TAG, "üåä Victory Wave animation started from position %d", winning_king_position);
156 }
157
158 // Aktualizace v≈°ech vln
159 for (int wave_idx = 0; wave_idx < wave_state.active_waves; wave_idx++) {
160 if (!wave_state.waves[wave_idx].active) continue;
161
162 wave_state.waves[wave_idx].radius += wave_state.wave_speed;
163
164 // Vlna je aktivn√≠ pouze pokud m√° pozitivn√≠ r√°dius
165 if (wave_state.waves[wave_idx].radius < 0) continue;
166
167 // Aplikace vlny na LED
168 for (int led = 0; led < 64; led++) {
169 float distance = get_board_distance(wave_state.center_pos, led);
170 float wave_radius = wave_state.waves[wave_idx].radius;
171
172 // Kontrola zda je LED v dosahu vlny
173 if (distance >= wave_radius - 0.5f && distance <= wave_radius + 0.5f) {
174 rgb_color_t wave_color;
175
176 // TODO: Zde by mƒõla b√Ωt kontrola zda je na pozici nep≈ô√°telsk√° figurka
177 bool enemy_piece_here = false; // Placeholder - implementovat podle game stavu
178
179 if (enemy_piece_here) {
180 // Pou≈æij ƒçervenou paletu pro nep≈ô√°telsk√© figurky
181 int palette_idx = wave_idx % (sizeof(enemy_red_palette) / sizeof(enemy_red_palette[0]));
182 wave_color = enemy_red_palette[palette_idx];
183 } else {
184 // Pou≈æij modrou paletu pro norm√°ln√≠ vlnu
185 int palette_idx = wave_idx % (sizeof(wave_blue_palette) / sizeof(wave_blue_palette[0]));
186 wave_color = wave_blue_palette[palette_idx];
187 }
188
189 // Fade efekt pro hezƒç√≠ vlnu
190 float fade = 1.0f - fabsf(distance - wave_radius) / 0.5f;
191 wave_color.r *= fade;
192 wave_color.g *= fade;
193 wave_color.b *= fade;
194
195 apply_color_safe(led, wave_color);
196 }
197 }
198
199 // Deaktivace vlny pokud je mimo board
200 if (wave_state.waves[wave_idx].radius > wave_state.max_radius) {
201 wave_state.waves[wave_idx].active = false;
202 }
203 }
204
205 wave_state.frame++;
206
207 // Restart animace kdy≈æ v≈°echny vlny dorazily na okraj
208 bool any_active = false;
209 for (int i = 0; i < wave_state.active_waves; i++) {
210 if (wave_state.waves[i].active) {
211 any_active = true;
212 break;
213 }
214 }
215
216 if (!any_active) {
217 // Restart s mal√Ωm zpo≈ædƒõn√≠m
218 wave_state.frame = 0;
219 vTaskDelay(pdMS_TO_TICKS(1000));
220 }

308static void endgame_animation_victory_fireworks(uint32_t frame) {
309 static firework_t fireworks[MAX_FIREWORKS];
310 static bool fireworks_initialized = false;
311
312 if (!fireworks_initialized) {
313 // Inicializace oh≈àostroj≈Ø
314 for (int i = 0; i < MAX_FIREWORKS; i++) {
315 fireworks[i].center_x = rand() % 8;
316 fireworks[i].center_y = rand() % 8;
317 fireworks[i].radius = 0.0f;
318 fireworks[i].max_radius = 2.0f + (rand() % 3);
319 fireworks[i].color_idx = rand() % 3;
320 fireworks[i].active = (i == 0); // Spus≈• prvn√≠ oh≈àostroj
321 fireworks[i].delay = i * 10; // Postupn√© spou≈°tƒõn√≠
322 }
323 fireworks_initialized = true;
324 }
325
326 // Vyƒçisti board
327 led_clear_all_safe();
328
329 // Aktualizace oh≈àostroj≈Ø
330 for (int f = 0; f < MAX_FIREWORKS; f++) {
331 if (!fireworks[f].active) {
332 // Aktivace s zpo≈ædƒõn√≠m
333 if (fireworks[f].delay > 0) {
334 fireworks[f].delay--;
335 } else if (fireworks[f].radius == 0) {
336 fireworks[f].active = true;
337 }
338 continue;
339 }
340
341 fireworks[f].radius += 0.15f;
342
343 // Vykreslen√≠ oh≈àostroje
344 for (int led = 0; led < 64; led++) {
345 uint8_t led_x, led_y;
346 led_index_to_chess_pos(led, &led_y, &led_x);
347
348 float dx = led_x - fireworks[f].center_x;
349 float dy = led_y - fireworks[f].center_y;
350 float distance = sqrtf(dx*dx + dy*dy);
351
352 if (fabsf(distance - fireworks[f].radius) < 0.8f) {
353 rgb_color_t firework_color;
354 switch (fireworks[f].color_idx) {
355 case 0: firework_color = COLOR_RED; break;
356 case 1: firework_color = COLOR_GREEN; break;
357 case 2: firework_color = COLOR_BLUE; break;
358 default: firework_color = COLOR_WHITE; break;
359 }
360
361 // Fade efekt
362 float fade = 1.0f - (fireworks[f].radius / fireworks[f].max_radius);
363 firework_color.r *= fade;
364 firework_color.g *= fade;
365 firework_color.b *= fade;
366
367 apply_color_safe(led, firework_color);
368 }
369 }
370
371 // Deaktivace kdy≈æ oh≈àostroj doraz√≠ na maximum
372 if (fireworks[f].radius > fireworks[f].max_radius) {
373 fireworks[f].active = false;
374 fireworks[f].radius = 0.0f;
375
376 // Restart s nov√Ωmi parametry
377 fireworks[f].center_x = rand() % 8;
378 fireworks[f].center_y = rand() % 8;
379 fireworks[f].max_radius = 2.0f + (rand() % 3);
380 fireworks[f].color_idx = rand() % 3;
381 fireworks[f].delay = rand() % 30;
382 }
383 }

482static void apply_subtle_animation(uint8_t led_index, subtle_animation_state_t* anim) {
483 if (!anim->active) return;
484
485 rgb_color_t result_color = anim->base_color;
486 float wave_progress = sinf(anim->frame * 0.1f);
487
488 switch (anim->type) {
489 case SUBTLE_ANIM_GENTLE_WAVE:
490 // Jemn√° vlna - m√≠rn√© zmƒõny v sytosti
491 result_color.r = anim->base_color.r * (0.9f + 0.1f * wave_progress);
492 result_color.g = anim->base_color.g * (0.9f + 0.1f * wave_progress);
493 result_color.b = anim->base_color.b * (0.9f + 0.1f * wave_progress);
494 break;
495
496 case SUBTLE_ANIM_WARM_GLOW:
497 // Tepl√© z√°≈ôen√≠ - smƒõs se ≈ælutou/oran≈æovou
498 float glow_intensity = (wave_progress + 1.0f) / 2.0f * 0.15f;
499 result_color.r = anim->base_color.r + 40 * glow_intensity;
500 result_color.g = anim->base_color.g + 20 * glow_intensity;
501 // Modr√° z≈Øst√°v√° stejn√°
502 break;
503
504 case SUBTLE_ANIM_COOL_PULSE:
505 // Chladn√© pulzov√°n√≠ - smƒõs s modrou/fialovou
506 float pulse_intensity = (wave_progress + 1.0f) / 2.0f * 0.1f;
507 result_color.b = anim->base_color.b + 30 * pulse_intensity;
508 result_color.r = anim->base_color.r * (1.0f - pulse_intensity * 0.2f);
509 break;
510
511 case SUBTLE_ANIM_WHITE_WINS:
512 // B√≠l√Ω v√≠tƒõz√≠ - b√≠l√° animace s jemn√Ωm pulzov√°n√≠m
513 float white_intensity = (wave_progress + 1.0f) / 2.0f * 0.2f;
514 result_color.r = 255 * (0.8f + white_intensity);
515 result_color.g = 255 * (0.8f + white_intensity);
516 result_color.b = 255 * (0.8f + white_intensity);
517 break;
518
519 case SUBTLE_ANIM_BLACK_WINS:
520 // ƒåern√Ω v√≠tƒõz√≠ - ƒçern√° animace s jemn√Ωm pulzov√°n√≠m
521 float black_intensity = (wave_progress + 1.0f) / 2.0f * 0.1f;
522 result_color.r = 50 * (0.5f + black_intensity);
523 result_color.g = 50 * (0.5f + black_intensity);
524 result_color.b = 50 * (0.5f + black_intensity);
525 break;
526
527 case SUBTLE_ANIM_DRAW:
528 // Rem√≠za - neutr√°ln√≠ animace s ≈°edou
529 float draw_intensity = (wave_progress + 1.0f) / 2.0f * 0.15f;
530 result_color.r = 128 * (0.8f + draw_intensity);
531 result_color.g = 128 * (0.8f + draw_intensity);
532 result_color.b = 128 * (0.8f + draw_intensity);
533 break;
534 }
535
536 // Omez barvy na validn√≠ rozsah
537 result_color.r = fminf(255, fmaxf(0, result_color.r));
538 result_color.g = fminf(255, fmaxf(0, result_color.g));
539 result_color.b = fminf(255, fmaxf(0, result_color.b));
540
541 apply_color_safe(led_index, result_color);
542 anim->frame++;
543}

574 static uint32_t frame_counter = 0;

576 switch (current_endgame_animation) {

598 ESP_LOGW(TAG, "Unknown endgame animation type: %d", current_endgame_animation);

601
602 frame_counter++;
603}
