50esp_err_t config_load_from_nvs(system_config_t* config)
51{
52 if (config == NULL) {
53 return ESP_ERR_INVALID_ARG;
54 }
55
56 ESP_LOGI(TAG, "Opening NVS handle for chess_config...");
57 esp_task_wdt_reset(); // Reset watchdog before NVS operations
58
59 nvs_handle_t nvs_handle;
60 esp_err_t ret = nvs_open("chess_config", NVS_READONLY, &nvs_handle);
61 if (ret != ESP_OK) {
62 ESP_LOGW(TAG, "Failed to open NVS handle, using defaults");
63 memcpy(config, &default_config, sizeof(system_config_t));
64 return ESP_OK;
65 }
66
67 ESP_LOGI(TAG, "NVS handle opened successfully, loading configuration...");
68
69 // Load configuration values
70
71 // Load verbose_mode
72 esp_task_wdt_reset(); // Reset watchdog before NVS read
73 uint8_t verbose_temp;
74 ret = nvs_get_u8(nvs_handle, "verbose", &verbose_temp);
75 if (ret != ESP_OK) {
76 config->verbose_mode = default_config.verbose_mode;
77 } else {
78 config->verbose_mode = (bool)verbose_temp;
79 }
80
81 // Load quiet_mode
82 esp_task_wdt_reset(); // Reset watchdog before NVS read
83 uint8_t quiet_temp;
84 ret = nvs_get_u8(nvs_handle, "quiet", &quiet_temp);
85 if (ret != ESP_OK) {
86 config->quiet_mode = default_config.quiet_mode;
87 } else {
88 config->quiet_mode = (bool)quiet_temp;
89 }
90
91 // Load log_level
92 esp_task_wdt_reset(); // Reset watchdog before NVS read
93 ret = nvs_get_u8(nvs_handle, "log_level", &config->log_level);
94 if (ret != ESP_OK) {
95 config->log_level = default_config.log_level;
96 }
97
98
99
100 // Load command_timeout_ms
101 esp_task_wdt_reset(); // Reset watchdog before NVS read
102 ret = nvs_get_u32(nvs_handle, "timeout", &config->command_timeout_ms);
103 if (ret != ESP_OK) {
104 config->command_timeout_ms = default_config.command_timeout_ms;
105 }
106
107 esp_task_wdt_reset(); // Reset watchdog before NVS close
108 nvs_close(nvs_handle);
109
110 ESP_LOGI(TAG, "Configuration loaded from NVS successfully");
111 ESP_LOGI(TAG, " Verbose: %s", config->verbose_mode ? "ON" : "OFF");
112 ESP_LOGI(TAG, " Quiet: %s", config->quiet_mode ? "ON" : "OFF");
113 ESP_LOGI(TAG, " Log Level: %d", config->log_level);
114 ESP_LOGI(TAG, " Timeout: %lu ms", config->command_timeout_ms);
115
116 return ESP_OK;
117}

119esp_err_t config_save_to_nvs(const system_config_t* config)
120{
121 if (config == NULL) {
122 return ESP_ERR_INVALID_ARG;
123 }
124
125 nvs_handle_t nvs_handle;
126 esp_err_t ret = nvs_open("chess_config", NVS_READWRITE, &nvs_handle);
127 if (ret != ESP_OK) {
128 ESP_LOGE(TAG, "Failed to open NVS handle for writing");
129 return ret;
130 }
131
132 // Save configuration values
133 ret = nvs_set_u8(nvs_handle, "verbose", (uint8_t)config->verbose_mode);
134 if (ret != ESP_OK) {
135 ESP_LOGE(TAG, "Failed to save verbose mode");
136 nvs_close(nvs_handle);
137 return ret;
138 }
139
140 ret = nvs_set_u8(nvs_handle, "quiet", (uint8_t)config->quiet_mode);
141 if (ret != ESP_OK) {
142 ESP_LOGE(TAG, "Failed to save quiet mode");
143 nvs_close(nvs_handle);
144 return ret;
145 }
146
147 ret = nvs_set_u8(nvs_handle, "log_level", config->log_level);
148 if (ret != ESP_OK) {
149 ESP_LOGE(TAG, "Failed to save log level");
150 nvs_close(nvs_handle);
151 return ret;
152 }
153
154
155
156 ret = nvs_set_u32(nvs_handle, "timeout", config->command_timeout_ms);
157 if (ret != ESP_OK) {
158 ESP_LOGE(TAG, "Failed to save timeout");
159 nvs_close(nvs_handle);
160 return ret;
161 }
162
163 // Commit changes
164 ret = nvs_commit(nvs_handle);
165 if (ret != ESP_OK) {
166 ESP_LOGE(TAG, "Failed to commit NVS changes");
167 nvs_close(nvs_handle);
168 return ret;
169 }
170
171 nvs_close(nvs_handle);
172
173 ESP_LOGI(TAG, "Configuration saved to NVS successfully");
174 return ESP_OK;
175}

177esp_err_t config_apply_settings(const system_config_t* config)
178{
179 if (config == NULL) {
180 return ESP_ERR_INVALID_ARG;
181 }
182
183 ESP_LOGI(TAG, "Applying configuration settings...");
184
185 // Apply log level
186 if (config->quiet_mode) {
187 esp_log_level_set("*", ESP_LOG_NONE);
188 ESP_LOGI(TAG, "Log level set to NONE (quiet mode)");
189 } else if (config->verbose_mode) {
190 esp_log_level_set("*", ESP_LOG_INFO);
191 esp_log_level_set("UART_TASK", ESP_LOG_DEBUG);
192 ESP_LOGI(TAG, "Log level set to INFO (verbose mode)");
193 } else {
194 esp_log_level_set("*", config->log_level);
195 ESP_LOGI(TAG, "Log level set to %d", config->log_level);
196 }
197
198 ESP_LOGI(TAG, "Configuration settings applied successfully");
199 return ESP_OK;
200}
