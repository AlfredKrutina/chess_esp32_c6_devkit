481void led_set_pixel_internal(uint8_t led_index, uint8_t red, uint8_t green,
482 uint8_t blue) {
483 if (led_index >= CHESS_LED_COUNT_TOTAL) {
484 ESP_LOGE(TAG, "Invalid LED index: %d (max: %d)", led_index,
485 CHESS_LED_COUNT_TOTAL - 1);
486 return;
487 }
488
489 // Check if LED component is enabled
490 if (!led_component_enabled) {
491 ESP_LOGD(TAG, "LED component disabled - ignoring LED command for LED %d",
492 led_index);
493 return;
494 }
495
496 // If matrix scanning is disabled and this is a board LED (0-63), limit
497 // animations
498 if (!matrix_scanning_enabled && led_index < 64) {
499 // Only allow static colors, no animations
500 if (red > 0 || green > 0 || blue > 0) {
501 ESP_LOGD(
502 TAG,
503 "Matrix scanning disabled - limiting LED animation for board LED %d",
504 led_index);
505 }
506 }
507
508 // ✅ OPRAVA: Unified mutex protection
509 if (led_unified_mutex != NULL) {
510 if (xSemaphoreTake(led_unified_mutex, LED_TASK_MUTEX_TIMEOUT_TICKS) !=
511 pdTRUE) {
512 ESP_LOGW(TAG,
513 "Failed to take LED unified mutex - skipping LED operation");
514 return;
515 }
516 }
517
518 // ✅ BATCH UPDATE SYSTEM - Just collect change, don't commit immediately
519 uint32_t color = (red << 16) | (green << 8) | blue;
520
521 // Update internal state
522 led_states[led_index] = color;
523
524 // Mark change as pending for next batch commit
525 if (led_initialized && led_strip != NULL && !simulation_mode) {
526 led_pending_changes[led_index] = color;
527 led_changed_flags[led_index] = true; // ✅ Mark this LED as changed
528 led_changes_pending = true;
529 }
530
531 if (simulation_mode) {
532 ESP_LOGI(TAG, "LED[%d] = RGB(%d,%d,%d) = 0x%06X", led_index, red, green,
533 blue, led_states[led_index]);
534 }
535
536 // ✅ OPRAVA: Release unified mutex
537 if (led_unified_mutex != NULL) {
538 xSemaphoreGive(led_unified_mutex);
539 }
540}
