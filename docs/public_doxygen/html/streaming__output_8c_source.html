61esp_err_t streaming_output_init(void)
62{
63 if (system_initialized) {
64 ESP_LOGW(TAG, "Streaming output already initialized");
65 return ESP_OK;
66 }
67
68 ESP_LOGI(TAG, "Initializing streaming output system...");
69
70 // Create mutex for thread safety
71 streaming_mutex = xSemaphoreCreateMutex();
72 if (streaming_mutex == NULL) {
73 ESP_LOGE(TAG, "Failed to create streaming mutex");
74 return ESP_ERR_NO_MEM;
75 }
76
77 // Initialize default output to UART
78 current_output.type = STREAM_UART;
79 current_output.uart_port = 0;
80 current_output.web_client = NULL;
81 current_output.queue = NULL;
82 current_output.auto_flush = true;
83 current_output.line_ending = STREAM_LF;
84
85 // Reset statistics
86 memset(&stats, 0, sizeof(stats));
87
88 system_initialized = true;
89 ESP_LOGI(TAG, "✓ Streaming output system initialized");
90 return ESP_OK;
91}

178esp_err_t stream_printf(const char* format, ...)
179{
180 if (!system_initialized || format == NULL) {
181 return ESP_ERR_INVALID_ARG;
182 }
183
184 va_list args;
185 va_start(args, format);
186
187 char buffer[STREAM_LINE_BUFFER_SIZE];
188 int len = vsnprintf(buffer, sizeof(buffer), format, args);
189 va_end(args);
190
191 if (len < 0) {
192 stats.write_errors++;
193 return ESP_ERR_INVALID_ARG;
194 }
195
196 if (len >= sizeof(buffer)) {
197 ESP_LOGW(TAG, "Stream printf truncated: %d -> %d chars", len, sizeof(buffer) - 1);
198 len = sizeof(buffer) - 1;
199 stats.truncated_writes++;
200 }
201
202 return stream_write(buffer, len);
203}

205esp_err_t stream_write(const char* data, size_t len)
206{
207 if (!system_initialized || data == NULL || len == 0) {
208 return ESP_ERR_INVALID_ARG;
209 }
210
211 if (xSemaphoreTake(streaming_mutex, pdMS_TO_TICKS(1000)) != pdTRUE) {
212 stats.mutex_timeouts++;
213 return ESP_ERR_TIMEOUT;
214 }
215
216 esp_err_t result = ESP_OK;
217
218 switch (current_output.type) {
219 case STREAM_UART:
220 result = stream_write_uart(data, len);
221 break;
222
223 case STREAM_WEB:
224 result = stream_write_web(data, len);
225 break;
226
227 case STREAM_QUEUE:
228 result = stream_write_queue(data, len);
229 break;
230
231 default:
232 result = ESP_ERR_NOT_SUPPORTED;
233 stats.write_errors++;
234 break;
235 }
236
237 if (result == ESP_OK) {
238 stats.total_bytes_written += len;
239 stats.total_writes++;
240 } else {
241 stats.write_errors++;
242 }
243
244 xSemaphoreGive(streaming_mutex);
245 return result;
246}

248esp_err_t stream_writeln(const char* data)
249{
250 if (data == NULL) {
251 return ESP_ERR_INVALID_ARG;
252 }
253
254 size_t len = strlen(data);
255 esp_err_t result = stream_write(data, len);
256
257 if (result == ESP_OK) {
258 // Add line ending
259 const char* line_end = (current_output.line_ending == STREAM_CRLF) ? "\r\n" : "\n";
260 result = stream_write(line_end, strlen(line_end));
261 }
262
263 return result;
264}

280static esp_err_t stream_write_uart(const char* data, size_t len)
281{
282 // Write directly to stdout for USB Serial JTAG or UART
283 size_t written = fwrite(data, 1, len, stdout);
284
285 if (current_output.auto_flush) {
286 fflush(stdout);
287 }
288
289 // Reset watchdog after each write
290 esp_task_wdt_reset();
291
292 return (written == len) ? ESP_OK : ESP_FAIL;
293}

328static esp_err_t stream_write_queue(const char* data, size_t len)
329{
330 if (current_output.queue == NULL) {
331 return ESP_ERR_INVALID_STATE;
332 }
333
334 // CRITICAL FIX: Actually send data to queue instead of just logging
335 // Create a proper game_response_t message for the queue
336 game_response_t response = {
337 .type = GAME_RESPONSE_SUCCESS,
338 .command_type = GAME_CMD_SHOW_BOARD, // Default command type
339 .error_code = 0,
340 .message = "Streaming data chunk",
341 .timestamp = esp_timer_get_time() / 1000
342 };
343
344 // Copy data to response buffer (truncate if too long)
345 size_t max_data_size = sizeof(response.data) - 1;
346 size_t copy_len = (len > max_data_size) ? max_data_size : len;
347 memcpy(response.data, data, copy_len);
348 response.data[copy_len] = '\0'; // Null terminate
349
350 // Send to queue with timeout
351 BaseType_t result = xQueueSend(current_output.queue, &response, pdMS_TO_TICKS(100));
352
353 if (result != pdTRUE) {
354 ESP_LOGW(TAG, "Failed to send streaming data to queue");
355 stats.write_errors++;
356 return ESP_ERR_TIMEOUT;
357 }
358
359 // Update statistics
360 stats.total_bytes_written += copy_len;
361 stats.total_writes++;
362
363 // Reset watchdog after each write
364 esp_task_wdt_reset();
365
366 return ESP_OK;
367}

482esp_err_t stream_led_board_row(int row, const uint32_t* led_colors)
483{
484 if (row < 0 || row > 7 || led_colors == NULL) {
485 return ESP_ERR_INVALID_ARG;
486 }
487
488 esp_err_t result = stream_printf("%d │ ", row + 1);
489
490 for (int col = 0; col < 8; col++) {
491 int led_index = chess_pos_to_led_index(row, col);
492 uint32_t color = led_colors[led_index];
493
494 const char* led_char = "⚫"; // Default: off
495
496 if (color != 0) {
497 uint8_t red = (color >> 16) & 0xFF;
498 uint8_t green = (color >> 8) & 0xFF;
499 uint8_t blue = color & 0xFF;
500
501 // Determine LED state based on color
502 if (red > 200 && green > 200 && blue < 100) {
503 led_char = "🟡"; // Yellow (lifted piece)
504 } else if (red < 100 && green > 200 && blue < 100) {
505 led_char = "🟢"; // Green (possible move)
506 } else if (red > 200 && green > 100 && blue < 100) {
507 led_char = "🟠"; // Orange (capture)
508 } else if (red < 100 && green < 100 && blue > 200) {
509 led_char = "🔵"; // Blue (placed)
510 } else {
511 led_char = "🔴"; // Other color
512 }
513 }
514
515 result |= stream_printf("%s│", led_char);
516
517 // Reset watchdog periodically
518 if (col % 4 == 3) {
519 esp_task_wdt_reset();
520 }
521 }
522
523 result |= stream_printf(" │%d\n", row + 1);
524
525 if (row > 0) {
526 result |= stream_writeln(" +---+---+---+---+---+---+---+---+");
527 }
528
529 return result;

573bool streaming_is_healthy(void)
574{
575 if (!system_initialized) {
576 return false;
577 }
578
579 // Health checks
580 bool healthy = true;
581
582 // Check error rate
583 if (stats.total_writes > 100 &&
584 stats.write_errors > (stats.total_writes * 0.1)) {
585 ESP_LOGW(TAG, "High write error rate: %d/%d",
586 stats.write_errors, stats.total_writes);
587 healthy = false;
588 }
589
590 // Check for excessive truncation
591 if (stats.truncated_writes > (stats.total_writes * 0.05)) {
592 ESP_LOGW(TAG, "High truncation rate: %d/%d",
593 stats.truncated_writes, stats.total_writes);
594 healthy = false;
595 }
596
597 // Check for mutex timeouts
598 if (stats.mutex_timeouts > 0) {
599 ESP_LOGW(TAG, "Mutex timeout issues: %d", stats.mutex_timeouts);
600 healthy = false;
601 }
602
603 return healthy;
