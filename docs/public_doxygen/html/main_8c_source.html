558void execute_demo_move(void) {
559 if (!demo_mode_enabled || demo_move_index >= demo_moves_count) {
560 return;
561 }
562
563 const char *move = current_demo_moves[demo_move_index];
564
565 // DEBUG LOG - Critical to verify function entry
566 ESP_LOGI(TAG, "ü§ñ EXECUTE DEMO MOVE: Game='%s', Ind=%d/%d, Move=%s",
567 DEMO_GAME_NAMES[current_demo_game], demo_move_index + 1,
568 demo_moves_count, move);
569
570 // Use PICKUP/DROP commands for realistic demo (like up/dn)
571 if (strlen(move) == 4 && game_command_queue != NULL) {
572 chess_move_command_t cmd = {0};
573
574 // Step 1: PICKUP command (lift piece from source square)
575 cmd.type = GAME_CMD_PICKUP;
576 strncpy(cmd.from_notation, move, 2); // e.g. "e2"
577
578 ESP_LOGI(TAG, " ‚¨ÜÔ∏è Lifting piece from %c%c...", move[0], move[1]);
579
580 // Set response queue for feedback
581 cmd.response_queue = uart_response_queue;
582
583 if (xQueueSend(game_command_queue, &cmd, pdMS_TO_TICKS(100)) != pdTRUE) {
584 ESP_LOGE(TAG, "‚ùå Failed to send GAME_CMD_PICKUP");
585 return;
586 }
587
588 // Longer delay for Pickup animation
589 vTaskDelay(pdMS_TO_TICKS(1500)); // 1.5 seconds
590
591 // Step 2: DROP command (place piece on destination square)
592 cmd.type = GAME_CMD_DROP;
593 strncpy(cmd.to_notation, move + 2, 2); // e.g. "e4"
594
595 ESP_LOGI(TAG, " ‚¨áÔ∏è Placing piece on %c%c...", move[2], move[3]);
596
597 // Set response queue for feedback
598 cmd.response_queue = uart_response_queue;
599
600 if (xQueueSend(game_command_queue, &cmd, pdMS_TO_TICKS(100)) != pdTRUE) {
601 ESP_LOGE(TAG, "‚ùå Failed to send GAME_CMD_DROP");
602 return;
603 }
604
605 // Wait for DROP animation
606 vTaskDelay(pdMS_TO_TICKS(800)); // 0.8 seconds
607
608 // Step 3: CASTLING Handling
609 // If move is castling, we must also move the rook!
610 char rook_from[3] = {0};
611 char rook_to[3] = {0};
612 bool is_castling = false;
613
614 if (strcmp(move, "e1g1") == 0) { // White King-side
615 strcpy(rook_from, "h1");
616 strcpy(rook_to, "f1");
617 is_castling = true;
618 } else if (strcmp(move, "e1c1") == 0) { // White Queen-side
619 strcpy(rook_from, "a1");
620 strcpy(rook_to, "d1");
621 is_castling = true;
622 } else if (strcmp(move, "e8g8") == 0) { // Black King-side
623 strcpy(rook_from, "h8");
624 strcpy(rook_to, "f8");
625 is_castling = true;
626 } else if (strcmp(move, "e8c8") == 0) { // Black Queen-side
627 strcpy(rook_from, "a8");
628 strcpy(rook_to, "d8");
629 is_castling = true;
630 }
631
632 if (is_castling) {
633 ESP_LOGI(TAG, " ‚ôú Castling detected! Moving rook %s -> %s", rook_from,
634 rook_to);
635 vTaskDelay(pdMS_TO_TICKS(500)); // Pause before moving rook
636
637 // Rook Pickup
638 memset(&cmd, 0, sizeof(cmd));
639 cmd.type = GAME_CMD_PICKUP;
640 strncpy(cmd.from_notation, rook_from, 2);
641 cmd.response_queue = uart_response_queue; // ‚úÖ Add feedback
642 xQueueSend(game_command_queue, &cmd, pdMS_TO_TICKS(100));
643 vTaskDelay(pdMS_TO_TICKS(1000));
644
645 // Rook Drop
646 memset(&cmd, 0, sizeof(cmd));
647 cmd.type = GAME_CMD_DROP;
648 strncpy(cmd.to_notation, rook_to, 2);
649 cmd.response_queue = uart_response_queue; // ‚úÖ Add feedback
650 xQueueSend(game_command_queue, &cmd, pdMS_TO_TICKS(100));
651 vTaskDelay(pdMS_TO_TICKS(800));
652 }
653 }
654
655 demo_move_index++;
656
657 // ‚úÖ Check if we've played all moves (endgame reached)
658 if (demo_move_index >= demo_moves_count) {
659 ESP_LOGI(TAG, "üèÅ Demo game '%s' complete! Endgame animations playing...",
660 DEMO_GAME_NAMES[current_demo_game]);
661 ESP_LOGI(TAG, "‚è±Ô∏è Waiting 5 seconds for endgame animation to complete...");
662
663 // Wait 5 seconds for endgame animation to play
664 vTaskDelay(pdMS_TO_TICKS(5000));
665
666 // ‚úÖ Rotate to next demo game
667 current_demo_game = (current_demo_game + 1) % DEMO_GAMES_COUNT;
668 current_demo_moves = DEMO_GAMES[current_demo_game];
669 demo_moves_count = DEMO_GAME_LENGTHS[current_demo_game];
670 demo_move_index = 0;
671
672 ESP_LOGI(TAG, "üîÑ Starting new demo game: %s (%d moves)",
673 DEMO_GAME_NAMES[current_demo_game], demo_moves_count);
674
675 // Send reset command to start fresh game
676 if (game_command_queue != NULL) {
677 chess_move_command_t cmd = {.type = GAME_CMD_NEW_GAME};
678 cmd.player = (demo_move_index % 2 == 0) ? PLAYER_WHITE : PLAYER_BLACK;
679
680 // ‚úÖ Set response queue so game_task sends confirmation (like UART
681 // commands) This ensures consistent behavior and logging
682 cmd.response_queue = uart_response_queue;
683
684 xQueueSend(game_command_queue, &cmd, pdMS_TO_TICKS(100));
685 }
686 }
