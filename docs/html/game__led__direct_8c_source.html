43void game_show_move_direct(uint8_t from_row, uint8_t from_col, uint8_t to_row, uint8_t to_col) {
44 uint8_t from_led = chess_pos_to_led_index(from_row, from_col);
45 uint8_t to_led = chess_pos_to_led_index(to_row, to_col);
46
47 // Vymazání předchozích zvýraznění
48 game_clear_highlights_direct();
49
50 // Animace tahu s progresivním osvětlením
51 // Step 1: Light source (yellow)
52 led_set_pixel_safe(from_led, 255, 255, 0); // Žlutá source
53 vTaskDelay(pdMS_TO_TICKS(300)); // 300ms delay
54
55 // Step 2: Fade source to orange
56 led_set_pixel_safe(from_led, 255, 165, 0); // Orange source
57 vTaskDelay(pdMS_TO_TICKS(200)); // 200ms delay
58
59 // Step 3: Light destination (green)
60 led_set_pixel_safe(to_led, 0, 255, 0); // Zelená destination
61 vTaskDelay(pdMS_TO_TICKS(300)); // 300ms delay
62
63 // Step 4: Fade source to red (capture indication)
64 led_set_pixel_safe(from_led, 255, 0, 0); // Red source (capture)
65 vTaskDelay(pdMS_TO_TICKS(200)); // 200ms delay
66
67 // Step 5: Final state - clear source, keep destination
68 led_set_pixel_safe(from_led, 0, 0, 0); // Clear source
69 led_set_pixel_safe(to_led, 0, 255, 0); // Keep destination green
70
71 ESP_LOGI(TAG, "Move animation complete: %d,%d -> %d,%d (LEDs %d -> %d)",
72 from_row, from_col, to_row, to_col, from_led, to_led);
73}

84void game_show_piece_lift_direct(uint8_t row, uint8_t col) {
85 uint8_t led_index = chess_pos_to_led_index(row, col);
86
87 // Jemný pulzující žlutý efekt
88 static uint8_t pulse_phase = 0;
89 pulse_phase = (pulse_phase + 1) % 64;
90
91 // Výpočet jemné pulsace (220-255)
92 uint8_t intensity = 220 + (pulse_phase > 32 ? (64 - pulse_phase) : pulse_phase);
93 uint8_t orange_mix = intensity * 0.9; // Jemný orange mix
94
95 led_set_pixel_safe(led_index, intensity, orange_mix, 0); // Žlutooranžová
96 ESP_LOGI(TAG, "Piece lift shown with subtle animation: %d,%d (LED %d)", row, col, led_index);
97}

162void game_show_state_direct(piece_t *board) {
163 // Přímé volání LED funkcí pro zobrazení stavu
164 for (int row = 0; row < 8; row++) {
165 for (int col = 0; col < 8; col++) {
166 uint8_t led_index = chess_pos_to_led_index(row, col);
167 piece_t piece = board[row * 8 + col];
168
169 if (piece != PIECE_EMPTY) {
170 if (piece >= PIECE_WHITE_PAWN && piece <= PIECE_WHITE_KING) {
171 led_set_pixel_safe(led_index, 255, 255, 255); // Bílá pro white pieces
172 } else {
173 led_set_pixel_safe(led_index, 0, 0, 255); // Modrá pro black pieces
174 }
175 } else {
176 led_set_pixel_safe(led_index, 0, 0, 0); // Černá pro empty
177 }
178 }
179 }
180
181 ESP_LOGI(TAG, "Game state shown");
182}

213void game_show_player_change_direct(player_t current_player) {
214 // Vymazání předchozích zvýraznění
215 game_clear_highlights_direct();
216
217 // Vlnová animace od středu
218 uint8_t r = (current_player == PLAYER_WHITE) ? 255 : 0;
219 uint8_t g = (current_player == PLAYER_WHITE) ? 255 : 255;
220 uint8_t b = (current_player == PLAYER_WHITE) ? 255 : 0;
221
222 // Wave animation - progressive lighting from center
223 for (int wave = 0; wave < 4; wave++) {
224 for (int i = 0; i < 64; i++) {
225 int row = i / 8;
226 int col = i % 8;
227 int center_dist = abs(row - 3) + abs(col - 3); // Distance from center
228
229 if (center_dist == wave) {
230 led_set_pixel_safe(i, r, g, b);
231 }
232 }
233 vTaskDelay(pdMS_TO_TICKS(100)); // 100ms delay between waves
234 }
235
236 ESP_LOGI(TAG, "Player change animation: %s - WAVE", (current_player == PLAYER_WHITE) ? "WHITE" : "BLACK");
237}

318void game_show_castling_guidance(uint8_t king_row, uint8_t king_col, uint8_t rook_row, uint8_t rook_col, bool is_kingside) {
319 uint8_t king_led = chess_pos_to_led_index(king_row, king_col);
320 uint8_t rook_led = chess_pos_to_led_index(rook_row, rook_col);
321
322 // Vymazání předchozích zvýraznění
323 game_clear_highlights_direct();
324
325 // Postupná animace rošády
326 // Step 1: Show king and rook
327 led_set_pixel_safe(king_led, 255, 255, 0); // Yellow king
328 led_set_pixel_safe(rook_led, 255, 255, 0); // Yellow rook
329 vTaskDelay(pdMS_TO_TICKS(1000));
330
331 // Step 2: Show king move path
332 uint8_t king_to_col = is_kingside ? king_col + 2 : king_col - 2;
333 uint8_t king_path_led = chess_pos_to_led_index(king_row, king_col + (is_kingside ? 1 : -1));
334 uint8_t king_final_led = chess_pos_to_led_index(king_row, king_to_col);
335 led_set_pixel_safe(king_path_led, 0, 255, 0); // Green path
336 led_set_pixel_safe(king_final_led, 0, 255, 0); // Green final
337 vTaskDelay(pdMS_TO_TICKS(1000));
338
339 // Step 3: Show rook move path
340 uint8_t rook_to_col = is_kingside ? rook_col - 2 : rook_col + 2;
341 uint8_t rook_path_led = chess_pos_to_led_index(rook_row, rook_col + (is_kingside ? -1 : 1));
342 uint8_t rook_final_led = chess_pos_to_led_index(rook_row, rook_to_col);
343 led_set_pixel_safe(rook_path_led, 0, 255, 0); // Green path
344 led_set_pixel_safe(rook_final_led, 0, 255, 0); // Green final
345 vTaskDelay(pdMS_TO_TICKS(1000));
346
347 // Step 4: Show final positions
348 led_clear_board_only();
349 led_set_pixel_safe(king_final_led, 255, 0, 255); // Magenta final king
350 led_set_pixel_safe(rook_final_led, 255, 0, 255); // Magenta final rook
351
352 ESP_LOGI(TAG, "Castling guidance shown: %s side (King %d,%d, Rook %d,%d)",
353 is_kingside ? "KINGSIDE" : "QUEENSIDE", king_row, king_col, rook_row, rook_col);
354}
