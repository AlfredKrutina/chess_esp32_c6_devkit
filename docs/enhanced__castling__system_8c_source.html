134esp_err_t enhanced_castling_start(player_t player, bool is_kingside)
135{
136 if (castling_system.active) {
137 ESP_LOGW(TAG, "‚ö†Ô∏è Castling already active - cancelling previous");
138 enhanced_castling_cancel();
139 }
140
141 ESP_LOGI(TAG, "üè∞ Starting castling: %s %s",
142 player == PLAYER_WHITE ? "White" : "Black",
143 is_kingside ? "kingside" : "queenside");
144
145 // Initialize system
146 castling_system.active = true;
147 castling_system.player = player;
148 castling_system.is_kingside = is_kingside;
149 castling_system.error_count = 0;
150
151 // Calculate positions
152 castling_calculate_positions(player, is_kingside);
153
154 // Start with king lift phase
155 enhanced_castling_update_phase(CASTLING_STATE_KING_LIFTED);
156
157 // Show initial guidance
158 castling_show_king_guidance();
159
160 castling_log_state_change("Castling started");
161 return ESP_OK;

167esp_err_t enhanced_castling_handle_king_lift(uint8_t row, uint8_t col)
168{
169 if (!castling_system.active) {
170 return ESP_ERR_INVALID_STATE;
171 }
172
173 if (castling_system.phase != CASTLING_STATE_KING_LIFTED) {
174 ESP_LOGW(TAG, "‚ö†Ô∏è King lift in wrong phase: %d", castling_system.phase);
175 enhanced_castling_handle_error(CASTLING_ERROR_INVALID_SEQUENCE, row, col);
176 return ESP_ERR_INVALID_STATE;
177 }
178
179 // Validate king position
180 if (row != castling_system.positions.king_from_row ||
181 col != castling_system.positions.king_from_col) {
182 ESP_LOGE(TAG, "‚ùå Wrong king position: %d,%d (expected %d,%d)",
183 row, col, castling_system.positions.king_from_row, castling_system.positions.king_from_col);
184 enhanced_castling_handle_error(CASTLING_ERROR_WRONG_KING_POSITION, row, col);
185 return ESP_ERR_INVALID_ARG;
186 }
187
188 ESP_LOGI(TAG, "‚úÖ King lifted from correct position: %c%d",
189 'a' + col, row + 1);
190
191 castling_log_state_change("King lifted");
192 return ESP_OK;
193}

198esp_err_t enhanced_castling_handle_king_drop(uint8_t row, uint8_t col)
199{
200 if (!castling_system.active) {
201 return ESP_ERR_INVALID_STATE;
202 }
203
204 if (castling_system.phase != CASTLING_STATE_KING_LIFTED) {
205 ESP_LOGW(TAG, "‚ö†Ô∏è King drop in wrong phase: %d", castling_system.phase);
206 enhanced_castling_handle_error(CASTLING_ERROR_INVALID_SEQUENCE, row, col);
207 return ESP_ERR_INVALID_STATE;
208 }
209
210 // Validate king move
211 if (!castling_validate_king_move(castling_system.positions.king_from_row,
212 castling_system.positions.king_from_col,
213 row, col)) {
214 ESP_LOGE(TAG, "‚ùå Invalid king move: %c%d -> %c%d",
215 'a' + castling_system.positions.king_from_col, castling_system.positions.king_from_row + 1,
216 'a' + col, row + 1);
217 enhanced_castling_handle_error(CASTLING_ERROR_WRONG_KING_POSITION, row, col);
218 return ESP_ERR_INVALID_ARG;
219 }
220
221 ESP_LOGI(TAG, "‚úÖ King moved to correct position: %c%d", 'a' + col, row + 1);
222
223 // Update phase to waiting for rook
224 enhanced_castling_update_phase(CASTLING_STATE_KING_MOVED_WAITING_ROOK);
225
226 // Show rook guidance
227 castling_show_rook_guidance();
228
229 castling_log_state_change("King moved, waiting for rook");
230 return ESP_OK;
231}

272 if (!castling_system.active) {

276 if (castling_system.phase != CASTLING_STATE_ROOK_LIFTED) {
277 ESP_LOGW(TAG, "‚ö†Ô∏è Rook drop in wrong phase: %d", castling_system.phase);
278 enhanced_castling_handle_error(CASTLING_ERROR_INVALID_SEQUENCE, row, col);
279 return ESP_ERR_INVALID_STATE;
280 }
281
282 // Validate rook move
283 if (!castling_validate_rook_move(castling_system.positions.rook_from_row,

287 'a' + castling_system.positions.rook_from_col, castling_system.positions.rook_from_row + 1,
288 'a' + col, row + 1);
289 enhanced_castling_handle_error(CASTLING_ERROR_WRONG_ROOK_POSITION, row, col);
290 return ESP_ERR_INVALID_ARG;

293 ESP_LOGI(TAG, "‚úÖ Rook moved to correct position: %c%d", 'a' + col, row + 1);
294
295 // Complete castling
296 enhanced_castling_update_phase(CASTLING_STATE_COMPLETING);
297
298 // Show completion celebration
299 castling_show_completion_celebration();
300
301 // Complete castling
302 enhanced_castling_update_phase(CASTLING_STATE_COMPLETED);
303

305 if (castling_system.completion_callback) {
306 castling_system.completion_callback(true);
307 }
308
309 // Reset system

311
312 castling_log_state_change("Castling completed successfully");
313 return ESP_OK;

361void enhanced_castling_update_phase(castling_phase_t new_phase)
362{
363 castling_system.phase = new_phase;
364 castling_system.phase_start_time = esp_timer_get_time() / 1000;
365
366 // Set timeout according to phase
367 switch (new_phase) {
368 case CASTLING_STATE_KING_LIFTED:
369 castling_system.phase_timeout_ms = 30000; // 30s na p≈ôem√≠stƒõn√≠ kr√°le
370 break;
371 case CASTLING_STATE_KING_MOVED_WAITING_ROOK:
372 castling_system.phase_timeout_ms = 60000; // 60s na vƒõ≈æ
373 break;
374 case CASTLING_STATE_ROOK_LIFTED:
375 castling_system.phase_timeout_ms = 30000; // 30s na p≈ôem√≠stƒõn√≠ vƒõ≈æe
376 break;
377 default:
378 castling_system.phase_timeout_ms = 10000; // V√Ωchoz√≠ timeout
379 break;
380 }
381
382 ESP_LOGI(TAG, "Phase changed to: %d, timeout: %lums",
383 new_phase, castling_system.phase_timeout_ms);
384}

389void enhanced_castling_handle_error(castling_error_t error, uint8_t row, uint8_t col)
390{
391 castling_system.error_count++;
392
393 ESP_LOGE(TAG, "‚ùå Castling error %d at position %d,%d (count: %d/%d)",
394 error, row, col, castling_system.error_count, castling_system.max_errors);
395
396 // Show error indication
397 castling_show_error_indication(error);
398
399 // Execute recovery action if available
400 if (error < sizeof(error_info) / sizeof(error_info[0]) &&
401 error_info[error].recovery_action) {
402 error_info[error].recovery_action();
403 }
404
405 // Check if max errors exceeded
406 if (castling_system.error_count >= castling_system.max_errors) {
407 ESP_LOGE(TAG, "‚ùå Maximum errors exceeded - cancelling castling");
408 enhanced_castling_handle_error(CASTLING_ERROR_MAX_ERRORS_EXCEEDED, row, col);
409 enhanced_castling_cancel();
410 }

416void castling_calculate_positions(player_t player, bool is_kingside)
417{
418 if (player == PLAYER_WHITE) {
419 castling_system.positions.king_from_row = 0;
420 castling_system.positions.king_from_col = 4;
421 castling_system.positions.king_to_row = 0;
422 castling_system.positions.king_to_col = is_kingside ? 6 : 2;
423
424 castling_system.positions.rook_from_row = 0;
425 castling_system.positions.rook_from_col = is_kingside ? 7 : 0;
426 castling_system.positions.rook_to_row = 0;
427 castling_system.positions.rook_to_col = is_kingside ? 5 : 3;
428 } else {
429 castling_system.positions.king_from_row = 7;
430 castling_system.positions.king_from_col = 4;
431 castling_system.positions.king_to_row = 7;
432 castling_system.positions.king_to_col = is_kingside ? 6 : 2;
433
434 castling_system.positions.rook_from_row = 7;
435 castling_system.positions.rook_from_col = is_kingside ? 7 : 0;
436 castling_system.positions.rook_to_row = 7;
437 castling_system.positions.rook_to_col = is_kingside ? 5 : 3;
438 }
439
440 ESP_LOGI(TAG, "Calculated positions: King %c%d->%c%d, Rook %c%d->%c%d",
441 'a' + castling_system.positions.king_from_col, castling_system.positions.king_from_row + 1,
442 'a' + castling_system.positions.king_to_col, castling_system.positions.king_to_row + 1,
443 'a' + castling_system.positions.rook_from_col, castling_system.positions.rook_from_row + 1,
444 'a' + castling_system.positions.rook_to_col, castling_system.positions.rook_to_row + 1);
445}

484bool castling_is_timeout_expired(void)
485{
486 if (!castling_system.active) {
487 return false;
488 }
489
490 uint32_t current_time = esp_timer_get_time() / 1000;
491 uint32_t elapsed = current_time - castling_system.phase_start_time;
492

625void castling_show_completion_celebration(void)
626{
627 if (!castling_system.active) {
628 return;
629 }
630
631 ESP_LOGI(TAG, "üéâ Showing castling completion celebration");
632
633 // Clear previous indications
634 castling_clear_all_indications();
635
636 // Show celebration animation
637 for (int cycle = 0; cycle < 3; cycle++) {
638 // Rainbow effect on king and rook positions
639 uint8_t king_led = chess_pos_to_led_index(castling_system.positions.king_to_row,
640 castling_system.positions.king_to_col);
641 uint8_t rook_led = chess_pos_to_led_index(castling_system.positions.rook_to_row,
642 castling_system.positions.rook_to_col);
643
644 // Cycle through colors
645 uint8_t colors[][3] = {
646 {255, 0, 0}, // Red
647 {0, 255, 0}, // Green
648 {0, 0, 255}, // Blue
649 {255, 255, 0}, // Yellow
650 {255, 0, 255}, // Magenta
651 {0, 255, 255} // Cyan
652 };
653
654 for (int i = 0; i < 6; i++) {
655 led_set_pixel_safe(king_led, colors[i][0], colors[i][1], colors[i][2]);
656 led_set_pixel_safe(rook_led, colors[i][0], colors[i][1], colors[i][2]);
657 vTaskDelay(pdMS_TO_TICKS(100));
658 }
659 }
660
661 // Final state - show completed positions
662 uint8_t king_led = chess_pos_to_led_index(castling_system.positions.king_to_row,
663 castling_system.positions.king_to_col);
664 uint8_t rook_led = chess_pos_to_led_index(castling_system.positions.rook_to_row,
665 castling_system.positions.rook_to_col);
666
667 led_set_pixel_safe(king_led, 0, 255, 0); // Green for success
668 led_set_pixel_safe(rook_led, 0, 255, 0); // Green for success
669}
