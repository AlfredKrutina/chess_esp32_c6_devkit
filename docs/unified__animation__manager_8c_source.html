57// Smooth interpolation functions
58static float ease_in_out_cubic(float t) {
59 return t < 0.5f ? 4.0f * t * t * t : 1.0f - powf(-2.0f * t + 2.0f, 3.0f) / 2.0f;
60}

62static void interpolate_color(uint8_t from_r, uint8_t from_g, uint8_t from_b,
63 uint8_t to_r, uint8_t to_g, uint8_t to_b,
64 float progress, uint8_t* out_r, uint8_t* out_g, uint8_t* out_b) {
65 float eased_progress = ease_in_out_cubic(progress);
66
67 *out_r = (uint8_t)(from_r + (to_r - from_r) * eased_progress);
68 *out_g = (uint8_t)(from_g + (to_g - from_g) * eased_progress);
69 *out_b = (uint8_t)(from_b + (to_b - from_b) * eased_progress);
70}

72esp_err_t animation_manager_init(const animation_config_t* config) {
73 if (manager_initialized) {
74 ESP_LOGW(TAG, "Animation manager already initialized");
75 return ESP_ERR_INVALID_STATE;
76 }
77
78 if (!config) {
79 ESP_LOGE(TAG, "Invalid configuration provided");
80 return ESP_ERR_INVALID_ARG;
81 }
82
83 // Initialize configuration
84 memcpy(&current_config, config, sizeof(animation_config_t));
85
86 // Initialize animation states
87 memset(animations, 0, sizeof(animations));
88 for (int i = 0; i < 16; i++) {
89 animations[i].id = 0; // Mark as inactive
90 animations[i].active = false;
91 }
92
93 next_animation_id = 1;
94 last_update_time = esp_timer_get_time() / 1000; // Convert to milliseconds
95 manager_initialized = true;
96
97 ESP_LOGI(TAG, "Unified Animation Manager initialized");
98 ESP_LOGI(TAG, " Max concurrent animations: %d", current_config.max_concurrent_animations);
99 ESP_LOGI(TAG, " Update frequency: %d Hz", current_config.update_frequency_hz);
100 ESP_LOGI(TAG, " Smooth interpolation: %s", current_config.enable_smooth_interpolation ? "enabled" : "disabled");
101 ESP_LOGI(TAG, " Trail effects: %s", current_config.enable_trail_effects ? "enabled" : "disabled");
102
103 return ESP_OK;
104}

477void animation_manager_update(void) {
478 if (!manager_initialized) {
479 return;
480 }
481
482 uint32_t current_time = esp_timer_get_time() / 1000;
483 uint32_t delta_time = current_time - last_update_time;
484
485 // Update frequency limiting
486 if (delta_time < (1000 / current_config.update_frequency_hz)) {
487 return;
488 }
489
490 last_update_time = current_time;
491
492 // Update all active animations
493 for (int i = 0; i < current_config.max_concurrent_animations; i++) {
494 animation_state_t* anim = &animations[i];
495 if (!anim->active) {
496 continue;
497 }
498
499 // Calculate progress
500 uint32_t elapsed = current_time - anim->start_time;
501
502 // Handle infinite animations (duration_ms = 0)
503 if (anim->duration_ms == 0) {
504 anim->progress = 0.0f; // Keep at 0 for infinite animations
505 } else {
506 anim->progress = (float)elapsed / (float)anim->duration_ms;
507
508 // Check if animation is complete
509 if (anim->progress >= 1.0f) {
510 animation_cleanup(anim);
511 continue;
512 }
513 }
514
515 // Update animation
516 if (anim->update_func) {
517 anim->update_func(anim);
518 }
519 }
520}

548static bool animation_update_smooth_interpolation(animation_state_t* anim) {
549 if (!current_config.enable_smooth_interpolation) {
550 return true;
551 }
552
553 uint8_t r, g, b;
554 interpolate_color(anim->color_start.r, anim->color_start.g, anim->color_start.b,
555 anim->color_end.r, anim->color_end.g, anim->color_end.b,
556 anim->progress, &r, &g, &b);
557
558 // Apply to LED
559 led_set_pixel_safe(anim->to_led, r, g, b);
560
561 // Trail effect
562 if (current_config.enable_trail_effects && anim->trail_length > 0) {
563 float trail_progress = anim->progress * anim->trail_length;
564 for (int i = 0; i < anim->trail_length; i++) {
565 if (trail_progress > i) {
566 float trail_intensity = 1.0f - (float)i / anim->trail_length;
567 uint8_t trail_r = (uint8_t)(r * trail_intensity);
568 uint8_t trail_g = (uint8_t)(g * trail_intensity);
569 uint8_t trail_b = (uint8_t)(b * trail_intensity);
570
571 // Calculate trail position (simplified)
572 uint8_t trail_led = anim->from_led + (anim->to_led - anim->from_led) * i / anim->trail_length;
573 led_set_pixel_safe(trail_led, trail_r, trail_g, trail_b);
574 }
575 }
576 }
577 return true;
578}

589static bool animation_update_pulsing(animation_state_t* anim) {
590 float pulse = (sinf(anim->progress * 4.0f * M_PI) + 1.0f) / 2.0f; // 0.0 to 1.0
591 float intensity = 0.3f + 0.7f * pulse; // 0.3 to 1.0
592
593 uint8_t r = (uint8_t)(anim->color_start.r * intensity);
594 uint8_t g = (uint8_t)(anim->color_start.g * intensity);
595 uint8_t b = (uint8_t)(anim->color_start.b * intensity);
596
597 led_set_pixel_safe(anim->to_led, r, g, b);
598 return true;
599}

609static bool animation_update_flashing(animation_state_t* anim) {
610 float flash = (sinf(anim->progress * 8.0f * M_PI) + 1.0f) / 2.0f; // 0.0 to 1.0
611 float intensity = flash > 0.5f ? 1.0f : 0.0f; // Binary flashing
612
613 uint8_t r = (uint8_t)(anim->color_start.r * intensity);
614 uint8_t g = (uint8_t)(anim->color_start.g * intensity);
615 uint8_t b = (uint8_t)(anim->color_start.b * intensity);
616
617 led_set_pixel_safe(anim->to_led, r, g, b);
618 return true;

632static bool animation_update_rainbow(animation_state_t* anim) {
633 float hue = anim->progress * 360.0f; // 0 to 360 degrees
634 float saturation = 1.0f;
635 float value = 0.8f + 0.2f * sinf(anim->progress * 4.0f * M_PI); // Pulsing brightness
636
637 // HSV to RGB conversion
638 float c = value * saturation;
