70static bool game_find_legal_attackers_to_square(uint8_t target_row,
71 uint8_t target_col,
72 piece_t target_piece) {
73 guided_capture_state.attacker_count = 0;
74
75 piece_t current_player_king =
76 (current_player == PLAYER_WHITE) ? PIECE_WHITE_KING : PIECE_BLACK_KING;
77
78 for (uint8_t row = 0; row < 8; row++) {
79 for (uint8_t col = 0; col < 8; col++) {
80 piece_t candidate_piece = board[row][col];
81 if (candidate_piece == PIECE_EMPTY ||
82 !game_is_same_color(candidate_piece, current_player_king)) {
83 continue;
84 }
85
86 chess_move_t capture_attempt = {.from_row = row,
87 .from_col = col,
88 .to_row = target_row,
89 .to_col = target_col,
90 .piece = candidate_piece,
91 .captured_piece = target_piece,
92 .timestamp = esp_timer_get_time() / 1000};
93
94 if (game_is_valid_move(&capture_attempt) == MOVE_ERROR_NONE &&
95 guided_capture_state.attacker_count < GUIDED_CAPTURE_MAX_ATTACKERS) {
96 uint8_t idx = guided_capture_state.attacker_count++;
97 guided_capture_state.attacker_rows[idx] = row;
98 guided_capture_state.attacker_cols[idx] = col;
99 guided_capture_state.attacker_pieces[idx] = candidate_piece;
100 }
101 }
102 }
103
104 return guided_capture_state.attacker_count > 0;
105}

382 (QueueHandle_t)cmd->response_queue);
383 return;
384 } else {

541

549 // má polož it)

551 led_set_pixel_safe(
552 chess_pos_to_led_index(lifted_piece_row, lifted_piece_col), 255,
553 255, 0); // Yellow
554 led_set_pixel_safe(chess_pos_to_led_index(from_row, from_col), 128, 0,
555 128); // Purple = target
556
557 // Poslat info message
558 char info_msg[128];
559 snprintf(info_msg, sizeof(info_msg),
560 "♟️ Step 2/3: Opponent piece removed. Place %s on PURPLE "
561 "square (%s) to capture",

572 }

576 (strlen(cmd->from_notation) > 0) ? cmd->from_notation : "??";

579 !game_guided_capture_has_mode_conflict() &&
580 game_find_legal_attackers_to_square(from_row, from_col, piece)) {

920void game_process_drop_command(const chess_move_command_t *cmd) {
921 if (!cmd)
922 return;

924 // BUG FIX 1: Během čekání na rozestavení blokovat tahy
925 if (current_game_state == GAME_STATE_WAITING_FOR_BOARD_SETUP) {
926 game_send_response_to_uart(

929 true, (QueueHandle_t)cmd->response_queue);
930 return;

937 return;

943 "⏸️ Čeká se na srovnání fyzické desky (matrix guard). "
944 "DROP z UART/Web je blokován.",
945 true, (QueueHandle_t)cmd->response_queue);
946 return;
947 }
948
949 ESP_LOGI(TAG, "🎯 Processing DROP command: %s (START)", cmd->to_notation);
950
951 // MOVED: New game detection moved to end of function to avoid false
952 // positives during move sequence
953
954 // Convert notation to coordinates
955 uint8_t to_row, to_col;
956 if (!convert_notation_to_coords(cmd->to_notation, &to_row, &to_col)) {
957 ESP_LOGE(TAG, "❌ Invalid notation: %s", cmd->to_notation);
958 game_send_response_to_uart("❌ Invalid square notation", true,
959 (QueueHandle_t)cmd->response_queue);
960 return;
961 }
962
963 // Při každém položení figurky (DROP) vždy přerušit červené blikání – ať už
964 // uživatel položil během animace nebo po ní; zvednutí–položení rychle za
965 // sebou tak vždy zastaví blink před dalším překreslením.
966 game_stop_error_blink();
967
968 if (guided_capture_state.active) {
969 if (game_guided_capture_has_mode_conflict()) {
970 STAGING_LOGI(TAG, "Guided capture cancelled due to mode conflict (drop)");
971 game_exit_guided_capture(true);
972 game_send_response_to_uart(
973 "⚠️ Guided capture cancelled due to higher-priority game state.", true,
974 (QueueHandle_t)cmd->response_queue);
975 return;
976 }
977
978 if (!game_refresh_guided_attackers()) {
979 STAGING_LOGI(TAG, "Guided capture cancelled (no legal attackers at drop)");
980 game_exit_guided_capture(true);
981 game_send_response_to_uart(
982 "⚠️ Guided capture cancelled - no legal capture available now.", true,
983 (QueueHandle_t)cmd->response_queue);
984 return;
985 }
986
987 if (!piece_lifted || lifted_piece == PIECE_EMPTY) {
988 // Escape/cancel path: opponent piece dropped back to target square.
989 if (to_row == guided_capture_state.target_row &&
990 to_col == guided_capture_state.target_col) {
991 game_exit_guided_capture(true);
992 game_send_response_to_uart(
993 "↩️ Guided capture cancelled. Opponent piece returned to target square.",
994 false, (QueueHandle_t)cmd->response_queue);
995 return;
996 }
997 game_show_guided_capture_leds();
998 game_send_response_to_uart(
999 game_get_guided_capture_hints_enabled()
1000 ? "⚠️ Guided capture active: lift highlighted YELLOW piece first."
1001 : "⚠️ Guided capture active: lift your piece first, then drop on PURPLE.",
1002 true,
1003 (QueueHandle_t)cmd->response_queue);
1004 return;
1005 }
1006
1007 bool is_allowed_attacker = false;
1008 for (uint8_t i = 0; i < guided_capture_state.attacker_count; i++) {
1009 if (guided_capture_state.attacker_rows[i] == lifted_piece_row &&
1010 guided_capture_state.attacker_cols[i] == lifted_piece_col &&
1011 guided_capture_state.attacker_pieces[i] == lifted_piece) {
1012 is_allowed_attacker = true;
1013 break;
1014 }
1015 }
1016
1017 if (!is_allowed_attacker) {
1018 // Prevent stale "piece still lifted" state when attacker set changed.
1019 piece_lifted = false;
1020 lifted_piece = PIECE_EMPTY;
1021 lifted_piece_row = 0;
1022 lifted_piece_col = 0;
1023 game_show_guided_capture_leds();
1024 game_send_response_to_uart(
1025 game_get_guided_capture_hints_enabled()
1026 ? "⚠️ This piece is not allowed. Use highlighted YELLOW attacker."
1027 : "⚠️ This piece is not allowed. Use a piece that can capture.",
1028 true,
1029 (QueueHandle_t)cmd->response_queue);
1030 return;
1031 }
1032
1033 if (to_row != guided_capture_state.target_row ||
1034 to_col != guided_capture_state.target_col) {
1035 // User put the attacker back on source square -> graceful cancel.
1036 if (to_row == lifted_piece_row && to_col == lifted_piece_col) {
1037 game_exit_guided_capture(true);
1038 game_send_response_to_uart(
1039 "↩️ Guided capture cancelled. Attacker returned to source square.",
1040 false, (QueueHandle_t)cmd->response_queue);
1041 return;
1042 }
1043
1044 // Wrong destination was used; clear lifted state to avoid ghost-lift.
1045 piece_lifted = false;
1046 lifted_piece = PIECE_EMPTY;
1047 lifted_piece_row = 0;
1048 lifted_piece_col = 0;
1049 game_show_guided_capture_leds();
1050 game_send_response_to_uart(
1051 "⚠️ Wrong destination. Lift attacker again and drop on PURPLE target.",
1052 true,
1053 (QueueHandle_t)cmd->response_queue);
1054 return;
1055 }
1056
1057 uint8_t source_row = lifted_piece_row;
1058 uint8_t source_col = lifted_piece_col;
1059 chess_move_t guided_capture_move = {.from_row = source_row,
1060 .from_col = source_col,
1061 .to_row = to_row,
1062 .to_col = to_col,
1063 .piece = lifted_piece,
1064 .captured_piece =
1065 guided_capture_state.target_piece,
1066 .timestamp =
1067 esp_timer_get_time() / 1000};
1068
1069 if (!game_execute_move(&guided_capture_move)) {
1070 STAGING_LOGI(TAG, "Guided capture rejected %c%d -> %c%d", 'a' + source_col,
1071 source_row + 1, 'a' + to_col, to_row + 1);
1072 game_show_guided_capture_leds();
1073 game_send_response_to_uart(
1074 "❌ Guided capture rejected by legality checks. Try another attacker.",
1075 true, (QueueHandle_t)cmd->response_queue);
1076 return;
1077 }
1078
1079 STAGING_LOGI(TAG, "Guided capture completed %c%d -> %c%d", 'a' + source_col,
1080 source_row + 1, 'a' + to_col, to_row + 1);
1081
1082 game_exit_guided_capture(true);
1083 game_send_response_to_uart("✅ Guided capture complete!", false,
1084 (QueueHandle_t)cmd->response_queue);
1085 return;
1086 }
1087
1088 // CASTLING INTERCEPTION: Handle rook moves during castling phase
1089 // This logic must be BEFORE standard validation to avoid "Blocked by King"
1090 // errors
1091 if (game_is_castle_animation_active()) {
1092 // 1. Check for completion (Drop on destination)
1093 // Use lifted_piece position if available, otherwise assume auto-recovery
1094 // from rook_from
1095 uint8_t effective_from_row =
1096 piece_lifted ? lifted_piece_row : rook_from_row;
1097 uint8_t effective_from_col =
1098 piece_lifted ? lifted_piece_col : rook_from_col;
1099
1100 if (game_complete_castle_animation(effective_from_row, effective_from_col,
1101 to_row, to_col)) {
1102 piece_lifted = false;
1103 lifted_piece = PIECE_EMPTY;
1104 game_send_response_to_uart("✅ Castling completed", false,
1105 (QueueHandle_t)cmd->response_queue);
1106 return;
1107 }
1108

1110 if (to_row == rook_from_row && to_col == rook_from_col) {
1111 ESP_LOGI(TAG, "🏰 Rook put back on start square - waiting for lift");
1112 piece_lifted = false;
1113 lifted_piece = PIECE_EMPTY;
1114 game_send_response_to_uart(
1115 "🏰 Rook put back - lift it again to complete castling", false,
1116 (QueueHandle_t)cmd->response_queue);
1117 return; // Keep castling active
1118 }
1119
1120 // 3. Intermediate Move (Dynamic Source) - Robustness for "fumbled" moves
1121 // If dropped on an empty square that is NOT the destination, move the rook
1122 // there and update the animation source!
1123 if (board[to_row][to_col] == PIECE_EMPTY) {
1124 ESP_LOGI(TAG, "🏰 Intermediate rook move to %c%d (updating source)",
1125 'a' + to_col, to_row + 1);
1126
1127 // Move piece in board simulation
1128 board[to_row][to_col] = board[rook_from_row][rook_from_col];
1129 board[rook_from_row][rook_from_col] = PIECE_EMPTY;
1130
1131 // Update Static Vars for animation source
1132 rook_from_row = to_row;
1133 rook_from_col = to_col;
1134
1135 // Update Lifted State
1136 piece_lifted = false;
1137 lifted_piece = PIECE_EMPTY;
1138
1139 // Calculate LED indices for immediate feedback
1140 uint8_t from_led = chess_pos_to_led_index(rook_from_row, rook_from_col);
1141 uint8_t to_led = chess_pos_to_led_index(rook_to_row, rook_to_col);
1142 led_set_pixel_safe(from_led, 255, 255, 0); // Yellow (New Source)
1143 led_set_pixel_safe(to_led, 0, 255, 0); // Green (Target)
1144
1145 char msg[128];
1146 snprintf(msg, sizeof(msg),
1147 "🏰 Rook moved to %c%d - now move to GREEN pulsing square",
1148 'a' + to_col, to_row + 1);
1149 game_send_response_to_uart(msg, false,
1150 (QueueHandle_t)cmd->response_queue);
1151 return;
1152 }
1153
1154 // 4. Invalid move (Occupied square)
1155 ESP_LOGW(TAG, "❌ Invalid rook move during castling to %c%d (Occupied)",
1156 'a' + to_col, to_row + 1);
1157 game_send_response_to_uart(
1158 "❌ Square occupied - move rook to flashing green square", true,
1159 (QueueHandle_t)cmd->response_queue);
1160 // Do not clear state, allow user to try again
1161 return;
1162 }

1165 // - Povolit fyzický DN pouze na promočním poli (bez běžného drop/move flow).
1166 // - Mimo promoční pole ignorovat a poslat krátkou hlášku.

1173 "👑 Promotion pending: DROP on promo square %c%d allowed "

1175 'a' + to_col, to_row + 1);
