318esp_err_t wifi_load_config_from_nvs(char *ssid, size_t ssid_len, char *password,
319 size_t password_len) {
320 if (ssid == NULL || password == NULL || ssid_len == 0 || password_len == 0) {
321 ESP_LOGE(TAG, "Invalid parameters for wifi_load_config_from_nvs");
322 return ESP_ERR_INVALID_ARG;
323 }
324
325 nvs_handle_t nvs_handle;
326 esp_err_t ret = nvs_open(WIFI_NVS_NAMESPACE, NVS_READONLY, &nvs_handle);
327 if (ret != ESP_OK) {
328 if (ret == ESP_ERR_NVS_NOT_FOUND) {
329 ESP_LOGD(TAG, "WiFi NVS namespace missing (no STA credentials saved yet)");
330 } else {
331 ESP_LOGE(TAG, "Failed to open NVS: %s", esp_err_to_name(ret));
332 }
333 return ret;
334 }
335
336 // Nacist SSID
337 size_t required_size = ssid_len;
338 ret = nvs_get_str(nvs_handle, WIFI_NVS_KEY_SSID, ssid, &required_size);
339 if (ret != ESP_OK) {
340 if (ret == ESP_ERR_NVS_NOT_FOUND) {
341 ESP_LOGD(TAG, "WiFi SSID not in NVS");
342 } else {
343 ESP_LOGE(TAG, "Failed to get SSID from NVS: %s", esp_err_to_name(ret));
344 }
345 nvs_close(nvs_handle);
346 return ret;
347 }
348
349 // Nacist password
350 required_size = password_len;
351 ret =
352 nvs_get_str(nvs_handle, WIFI_NVS_KEY_PASSWORD, password, &required_size);
353 if (ret != ESP_OK) {
354 if (ret == ESP_ERR_NVS_NOT_FOUND) {
355 ESP_LOGD(TAG, "WiFi password not in NVS");
356 } else {
357 ESP_LOGE(TAG, "Failed to get password from NVS: %s", esp_err_to_name(ret));
358 }
359 nvs_close(nvs_handle);
360 return ret;
361 }
362
363 nvs_close(nvs_handle);
364 ESP_LOGI(TAG, "WiFi config loaded from NVS: SSID=%s", ssid);
365
366 return ESP_OK;
367}

400esp_err_t wifi_save_config_to_nvs(const char *ssid, const char *password) {
401 if (ssid == NULL || password == NULL) {
402 ESP_LOGE(TAG, "Invalid parameters: ssid or password is NULL");
403 return ESP_ERR_INVALID_ARG;
404 }
405
406 size_t ssid_len = strlen(ssid);
407 size_t password_len = strlen(password);
408
409 if (ssid_len == 0 || ssid_len > 32) {
410 ESP_LOGE(TAG, "Invalid SSID length: %zu (must be 1-32)", ssid_len);
411 return ESP_ERR_INVALID_ARG;
412 }
413
414 if (password_len > 64) {
415 ESP_LOGE(TAG, "Invalid password length: %zu (must be 0-64)", password_len);
416 return ESP_ERR_INVALID_ARG;
417 }
418
419 nvs_handle_t nvs_handle;
420 esp_err_t ret = nvs_open(WIFI_NVS_NAMESPACE, NVS_READWRITE, &nvs_handle);
421 if (ret != ESP_OK) {
422 ESP_LOGE(TAG, "Failed to open NVS: %s", esp_err_to_name(ret));
423 return ret;
424 }
425
426 ret = nvs_set_str(nvs_handle, WIFI_NVS_KEY_SSID, ssid);
427 if (ret != ESP_OK) {
428 ESP_LOGE(TAG, "Failed to set SSID in NVS: %s", esp_err_to_name(ret));
429 nvs_close(nvs_handle);
430 return ret;
431 }
432
433 ret = nvs_set_str(nvs_handle, WIFI_NVS_KEY_PASSWORD, password);
434 if (ret != ESP_OK) {
435 ESP_LOGE(TAG, "Failed to set password in NVS: %s", esp_err_to_name(ret));
436 nvs_close(nvs_handle);
437 return ret;
438 }
439
440 ret = nvs_commit(nvs_handle);
441 if (ret != ESP_OK) {
442 ESP_LOGE(TAG, "Failed to commit NVS: %s", esp_err_to_name(ret));
443 nvs_close(nvs_handle);
444 return ret;
445 }
446
447 nvs_close(nvs_handle);
448 ESP_LOGI(TAG, "WiFi config saved to NVS: SSID=%s", ssid);
449
450 // Pokud je pripojeny a SSID se zmenil, odpojit
451 if (sta_connected && strcmp(sta_ssid, ssid) != 0) {
452 ESP_LOGI(TAG, "SSID changed from '%s' to '%s', disconnecting...", sta_ssid,
453 ssid);
454 wifi_disconnect_sta();
455 }
456
457 return ESP_OK;
458}

460esp_err_t wifi_connect_sta(void) {
461 // Kontrola: pokud uz je pripojeny, vratit uspech
462 if (sta_connected) {
463 ESP_LOGI(TAG, "Already connected to WiFi: %s (IP: %s)", sta_ssid, sta_ip);
464 return ESP_OK;
465 }
466
467 // Kontrola: pokud se prave pripojuje, vratit chybu (race condition
468 // protection)
469 if (sta_connecting) {
470 ESP_LOGW(TAG, "WiFi connection already in progress");
471 return ESP_ERR_INVALID_STATE;
472 }
473
474 char ssid[33] = {0};
475 char password[65] = {0};
476
477 // Nacist konfiguraci z NVS
478 esp_err_t ret =
479 wifi_load_config_from_nvs(ssid, sizeof(ssid), password, sizeof(password));
480 if (ret != ESP_OK) {
481 if (ret == ESP_ERR_NVS_NOT_FOUND) {
482 ESP_LOGI(TAG, "WiFi STA: no credentials in NVS (use AP or BLE to configure)");
483 } else {
484 ESP_LOGE(TAG, "Failed to load WiFi config from NVS: %s",
485 esp_err_to_name(ret));
486 }
487 return ret;
488 }
489
490 ESP_LOGI(TAG, "Connecting to WiFi: SSID=%s", ssid);
491 sta_manual_disconnect = false; // Povolit automaticke prepojeni pri ztrate
492 sta_connecting = true; // Nastavit flag pripojovani
493
494 // Nastavit WiFi STA konfiguraci
495 wifi_config_t wifi_config = {0};
496 strncpy((char *)wifi_config.sta.ssid, ssid, sizeof(wifi_config.sta.ssid) - 1);
497 strncpy((char *)wifi_config.sta.password, password,
498 sizeof(wifi_config.sta.password) - 1);
499 wifi_config.sta.threshold.authmode = WIFI_AUTH_WPA2_PSK;
500 wifi_config.sta.pmf_cfg.capable = true;
501 wifi_config.sta.pmf_cfg.required = false;
502
503 ret = esp_wifi_set_config(WIFI_IF_STA, &wifi_config);
504 if (ret != ESP_OK) {
505 ESP_LOGE(TAG, "Failed to set WiFi STA config: %s", esp_err_to_name(ret));
506 return ret;
507 }
508
509 // Spustit pripojeni
510 ret = esp_wifi_connect();
511 if (ret != ESP_OK) {
512 ESP_LOGE(TAG, "Failed to start WiFi connection: %s", esp_err_to_name(ret));
513 return ret;
514 }
515
516 // Cekat na pripojeni (max 30 sekund)
517 int retry_count = 0;
518 const int max_retries = 30; // 30 sekund (1 sekunda per retry)
519
520 while (retry_count < max_retries) {
521 // Resetovat WDT aby se zabranilo timeoutu (funkce muze blokovat az 30
522 // sekund)
523 web_server_task_wdt_reset_safe();
524
525 vTaskDelay(pdMS_TO_TICKS(1000));
526
527 if (sta_connected) {
528 ESP_LOGI(TAG, "WiFi connected successfully! IP: %s", sta_ip);
529 sta_connecting = false; // Resetovat flag
530 return ESP_OK;
531 }
532
533 retry_count++;
534 if (retry_count % 5 == 0) {
535 ESP_LOGI(TAG, "Waiting for WiFi connection... (%d/%d)", retry_count,
536 max_retries);
537 }
538 }
539
540 ESP_LOGE(TAG, "WiFi connection timeout after %d seconds", max_retries);
541 esp_wifi_disconnect();
542 sta_connecting = false; // Resetovat flag pri timeout
543
544 // Vratit specificky error code podle disconnection reason
545 // WIFI_REASON_NO_AP_FOUND = 201
546 // WIFI_REASON_AUTH_FAIL = 202
547 // WIFI_REASON_ASSOC_FAIL = 203
548 // WIFI_REASON_HANDSHAKE_TIMEOUT = 204
549 if (last_disconnect_reason == 201) {
550 return ESP_ERR_NOT_FOUND; // Network not found
551 } else if (last_disconnect_reason == 202 || last_disconnect_reason == 203 ||
552 last_disconnect_reason == 204) {
553 return ESP_ERR_INVALID_RESPONSE; // Authentication/association failed (wrong
554 // password)
555 }
556
557 return ESP_ERR_TIMEOUT; // General timeout
558}

560esp_err_t wifi_disconnect_sta(void) {
561 // Pokud neni pripojeny a nepripojuje se, vratit uspech
562 if (!sta_connected && !sta_connecting) {
563 ESP_LOGI(TAG, "WiFi already disconnected");
564 return ESP_OK;
565 }
566
567 ESP_LOGI(TAG, "Disconnecting from WiFi...");
568
569 sta_manual_disconnect = true;
570 if (sta_reconnect_timer != NULL) {
571 esp_timer_stop(sta_reconnect_timer); // Zrusit naplanovany reconnect
572 }
573 if (sta_connecting) {
574 ESP_LOGW(TAG, "Cancelling WiFi connection in progress");
575 sta_connecting = false;
576 }
577
578 esp_err_t ret = esp_wifi_disconnect();
579 if (ret != ESP_OK) {
580 ESP_LOGE(TAG, "Failed to disconnect WiFi: %s", esp_err_to_name(ret));
581 return ret;
582 }
583
584 // Cekat na odpojeni (max 5 sekund)
585 int retry_count = 0;
586 const int max_retries = 5;
587
588 while (retry_count < max_retries && sta_connected) {
589 // Resetovat WDT aby se zabranilo timeoutu
590 web_server_task_wdt_reset_safe();
591 vTaskDelay(pdMS_TO_TICKS(1000));
592 retry_count++;
593 }
594
595 if (!sta_connected) {
596 ESP_LOGI(TAG, "WiFi disconnected successfully");
597 } else {
598 ESP_LOGW(TAG, "WiFi disconnect timeout, but continuing");
599 }
600
601 return ESP_OK;
602}

1032static bool wifi_ap_ssid_matches_czechmate(const uint8_t *ssid_raw,
1033 size_t ssid_len, int *out_slot) {
1034 const char *base = WIFI_AP_SSID_BASE;
1035 const size_t bl = strlen(base);
1036 *out_slot = -1;
1037 if (ssid_len < bl) {
1038 return false;
1039 }
1040 if (memcmp(ssid_raw, base, bl) != 0) {
1041 return false;
1042 }
1043 if (ssid_len == bl) {
1044 *out_slot = 0;
1045 return true;
1046 }
1047 if (ssid_raw[bl] != (uint8_t)'_') {
1048 return false;
1049 }
1050 unsigned n = 0;
1051 for (size_t i = bl + 1; i < ssid_len; i++) {
1052 uint8_t c = ssid_raw[i];
1053 if (c < (uint8_t)'0' || c > (uint8_t)'9') {
1054 return false;
1055 }
1056 n = n * 10u + (unsigned)(c - (uint8_t)'0');
1057 }
1058 if (n < 1u || n > (unsigned)WIFI_AP_SSID_MAX_SLOTS) {
1059 return false;
1060 }
1061 *out_slot = (int)n;
1062 return true;
1063}
