38static uint8_t game_count_mask_bits(uint32_t low, uint32_t high) {
39 uint8_t count = 0;
40 while (low != 0) {
41 count += (uint8_t)(low & 1U);
42 low >>= 1;
43 }
44 while (high != 0) {
45 count += (uint8_t)(high & 1U);
46 high >>= 1;
47 }
48 return count;
49}

115void game_matrix_guard_render_leds(void) {
116 if (!matrix_guard_pause_state.active) {
117 return;
118 }
119 if (!chess_policy_matrix_guard_led_enabled()) {
120 return;
121 }
122
123 led_clear_board_only();
124 uint8_t phys[64] = {0};
125 matrix_get_state(phys);
126
127 for (uint8_t square = 0; square < 64; square++) {
128 uint8_t row = square / 8;
129 uint8_t col = square % 8;
130 piece_t piece = board[row][col];
131 uint8_t expected_occ = (piece == PIECE_EMPTY) ? 0 : 1;
132 if (phys[square] == expected_occ) {
133 continue;
134 }
135
136 if (piece >= PIECE_WHITE_PAWN && piece <= PIECE_WHITE_KING) {
137 if (chess_policy_mg_led_white_yellow()) {
138 led_set_pixel_safe(chess_pos_to_led_index(row, col), 255, 255, 0);
139 }
140 } else if (piece >= PIECE_BLACK_PAWN && piece <= PIECE_BLACK_KING) {
141 if (chess_policy_mg_led_black_blue()) {
142 led_set_pixel_safe(chess_pos_to_led_index(row, col), 0, 0, 255);
143 }
144 } else if (phys[square] != 0) {
145 if (chess_policy_mg_led_ghost_orange()) {
146 led_set_pixel_safe(chess_pos_to_led_index(row, col), 255, 140, 0);
147 }
148 } else if (chess_policy_mg_led_missing_white()) {
149 led_set_pixel_safe(chess_pos_to_led_index(row, col), 220, 220, 220);
150 }
151 }

154void game_matrix_guard_handle_command(const chess_move_command_t *cmd) {
155 if (cmd == NULL || cmd->type != GAME_CMD_MATRIX_GUARD) {
156 return;
157 }
158
159 uint8_t action = cmd->timer_data.matrix_guard.action;
160 if (action == 0) {
161 game_matrix_guard_restore_after_clear(true);
162 STAGING_LOGI(TAG, "Matrix guard cleared");
163 return;
164 }
165
166 if (!chess_policy_matrix_guard_enabled()) {
167 STAGING_LOGI(TAG, "Matrix guard command ignored (disabled in menuconfig)");
168 matrix_abort_ambiguous_guard_baseline();
169 return;
170 }
171
172 if (game_task_matrix_guard_mode_conflict_active()) {
173 STAGING_LOGI(TAG,
174 "Matrix guard ignored (special mode) — clearing matrix local "
175 "guard so UP/DN continues");
176 if (matrix_guard_pause_state.active) {
177 memset(&matrix_guard_pause_state, 0, sizeof(matrix_guard_pause_state));
178 resync_required_after_restore = false;
179 }
180 matrix_abort_ambiguous_guard_baseline();
181 return;
182 }
183
184 matrix_guard_pause_state.active = true;
185 matrix_guard_pause_state.lifted_mask_low =
186 cmd->timer_data.matrix_guard.lifted_mask_low;
187 matrix_guard_pause_state.lifted_mask_high =
188 cmd->timer_data.matrix_guard.lifted_mask_high;
189 matrix_guard_pause_state.dropped_mask_low =
190 cmd->timer_data.matrix_guard.dropped_mask_low;
191 matrix_guard_pause_state.dropped_mask_high =
192 cmd->timer_data.matrix_guard.dropped_mask_high;
193 matrix_guard_pause_state.conflict_count = game_count_mask_bits(
194 matrix_guard_pause_state.lifted_mask_low |
195 matrix_guard_pause_state.dropped_mask_low,
196 matrix_guard_pause_state.lifted_mask_high |
197 matrix_guard_pause_state.dropped_mask_high);
198
199 if (chess_policy_matrix_guard_should_freeze()) {
200 game_task_matrix_guard_freeze_move_flow();
201 }
202
203 uint8_t expected[64];
204 game_board_to_occupancy(expected);
205 matrix_guard_apply_expected_occupancy(expected);
206
207 STAGING_LOGI(TAG, "Matrix guard active, conflict_count=%u",
208 matrix_guard_pause_state.conflict_count);
209 game_matrix_guard_render_leds();
210}

212void game_matrix_guard_check_resync_after_restore(void) {
213 if (!chess_policy_matrix_guard_enabled() ||
214 !chess_policy_matrix_guard_nvs_resync_enabled()) {
215 return;
216 }
217 if (game_is_opening_trainer_active() || game_is_opening_trainer_setup_active()) {
218 return;
219 }
220 uint8_t matrix_state_now[64] = {0};
221 matrix_get_state(matrix_state_now);
222
223 uint32_t lifted_low = 0, lifted_high = 0;
224 uint8_t conflicts = 0;
225 for (uint8_t i = 0; i < 64; i++) {
226 uint8_t expected = (((uint8_t)board[i / 8][i % 8]) == PIECE_EMPTY) ? 0 : 1;
227 if (matrix_state_now[i] != expected) {
228 if (i < 32) {
229 lifted_low |= (1UL << i);
230 } else {
231 lifted_high |= (1UL << (i - 32));
232 }
233 conflicts++;
234 }
235 }
236
237 if (conflicts == 0) {
238 resync_required_after_restore = false;
239 return;
240 }
241
242 matrix_guard_pause_state.active = true;
243 matrix_guard_pause_state.lifted_mask_low = lifted_low;
244 matrix_guard_pause_state.lifted_mask_high = lifted_high;
245 matrix_guard_pause_state.dropped_mask_low = 0;
246 matrix_guard_pause_state.dropped_mask_high = 0;
247 matrix_guard_pause_state.conflict_count = conflicts;
248 resync_required_after_restore = true;
249
250 uint8_t expected[64];
251 game_board_to_occupancy(expected);
252 matrix_guard_apply_expected_occupancy(expected);
253
254 game_matrix_guard_render_leds();
255}
