35void game_detect_and_handle_castling(const chess_move_t *move) {
36 piece_t piece = move->piece;
37
38 // Je to král a pohybuje se o 2 pole?
39 if ((piece == PIECE_WHITE_KING || piece == PIECE_BLACK_KING) &&
40 abs((int)move->to_col - (int)move->from_col) == 2) {
41
42 ESP_LOGI(TAG, "🏰 CASTLING DETECTED! King moved 2 squares");
43
44 // Nastavit stav rošády
45 castling_state.in_progress = true;
46 castling_state.player = current_player;
47
48 // Vypočítat pozice věže
49 bool is_kingside = (move->to_col == 6);
50 castling_state.rook_from_row = move->to_row;
51 castling_state.rook_from_col = is_kingside ? 7 : 0;
52 castling_state.rook_to_row = move->to_row;
53 castling_state.rook_to_col = is_kingside ? 5 : 3;
54
55 // Zobrazit LED guidance pro věž
56 game_show_castling_rook_guidance();
57
58 // NEZMĚNIT HRÁČE - čekat na dokončení rošády
59 ESP_LOGI(TAG, "⏳ Waiting for rook move to complete castling...");
60 }
61}

66void game_show_castling_rook_guidance() {
67 // Vyčistit board
68 led_clear_board_only();
69
70 // Postupná animace od věže k cíli
71 uint8_t rook_from_led = chess_pos_to_led_index(castling_state.rook_from_row,
72 castling_state.rook_from_col);
73 uint8_t rook_to_led = chess_pos_to_led_index(castling_state.rook_to_row,
74 castling_state.rook_to_col);
75
76 // Zvýraznit věž (žlutá)
77 led_set_pixel_safe(rook_from_led, 255, 255, 0); // Žlutá - zvedni odtud
78
79 // Postupně rozsvítit pole od věže k cíli
80 int from_col = castling_state.rook_from_col;
81 int to_col = castling_state.rook_to_col;
82 int step = (to_col > from_col) ? 1 : -1;
83
84 for (int col = from_col + step; col != to_col + step; col += step) {
85 uint8_t led_index =
86 chess_pos_to_led_index(castling_state.rook_from_row, col);
87 led_set_pixel_safe(led_index, 0, 255, 0); // Zelená - cesta
88 vTaskDelay(pdMS_TO_TICKS(300));
89 }
90
91 // Zvýraznit cíl (zelená)
92 led_set_pixel_safe(rook_to_led, 0, 255, 0); // Zelená - polož sem
93
94 ESP_LOGI(TAG, "💡 Move rook from %c%d to %c%d to complete castling",
95 'a' + castling_state.rook_from_col, castling_state.rook_from_row + 1,
96 'a' + castling_state.rook_to_col, castling_state.rook_to_row + 1);
97}

104bool game_check_castling_completion(const chess_move_t *move) {
105 if (!castling_state.in_progress)
106 return false;
107
108 // Je to správný tah věže?
109 if (move->from_row == castling_state.rook_from_row &&
110 move->from_col == castling_state.rook_from_col &&
111 move->to_row == castling_state.rook_to_row &&
112 move->to_col == castling_state.rook_to_col) {
113
114 ESP_LOGI(TAG, "✅ CASTLING COMPLETED! Rook moved correctly");
115
116 // Provedeme posun věže v board[][]
117 board[move->to_row][move->to_col] = move->piece;
118 board[castling_state.rook_from_row][castling_state.rook_from_col] =
119 PIECE_EMPTY;
120
121 // Aktualizovat příznaky pro krále a věž
122 if (castling_state.player == PLAYER_WHITE) {
123 white_king_moved = true;
124 if (castling_state.rook_from_col == 7)
125 white_rook_h_moved = true;
126 else
127 white_rook_a_moved = true;
128 } else {
129 black_king_moved = true;
130 if (castling_state.rook_from_col == 7)
131 black_rook_h_moved = true;
132 else
133 black_rook_a_moved = true;
134 }
135
136 // Zlatá animace dokončení rošády
137 show_castling_completion_animation();
138
139 // Změnit hráče TEPRVE TEĎ
140 current_player =
141 (current_player == PLAYER_WHITE) ? PLAYER_BLACK : PLAYER_WHITE;
142
143 // KRITICKÉ: Endgame kontrola PŘED player change animací!
144 // Pokud je endgame, player change se NESPOUŠTÍ
145 game_state_t end_game_result = game_check_end_game_conditions();
146 if (end_game_result == GAME_STATE_FINISHED) {
147 current_game_state = GAME_STATE_FINISHED;
148 game_active = false;
149
150 // Najít pozici krále vítěze pro endgame animaci
151 player_t winner =
152 (current_player == PLAYER_WHITE) ? PLAYER_BLACK : PLAYER_WHITE;
153 uint8_t king_pos = 28; // default e4
154 for (int i = 0; i < 64; i++) {
155 piece_t piece = board[i / 8][i % 8];
156 if ((winner == PLAYER_WHITE && piece == PIECE_WHITE_KING) ||
157 (winner == PLAYER_BLACK && piece == PIECE_BLACK_KING)) {
158 king_pos = i;
159 break;
160 }
161 }
162
163 ESP_LOGI(TAG,
164 "🎯 Game finished after castling completion! Starting endgame "
165 "animation at position %d",
166 king_pos);
167
168 // Spustit endgame animaci (wave z krále vítěze)
169 led_command_t endgame_cmd = {.type = LED_CMD_ANIM_ENDGAME,
170 .led_index = king_pos,
171 .red = 255,
172 .green = 255,
173 .blue = 0, // Yellow
174 .duration_ms = 0, // Endless
175 .data = NULL};
176 led_execute_command_new(&endgame_cmd);
177
178 ESP_LOGI(
179 TAG,
180 "✅ Endgame animation started - player change animation SKIPPED");
181 } else {
182 // Není endgame - spustit player change animaci
183 uint8_t player_color =
184 (current_player == PLAYER_WHITE) ? 1 : 0; // 1=white, 0=black
185 led_command_t player_change_cmd = {
186 .type = LED_CMD_ANIM_PLAYER_CHANGE, // Použít
187 // ANIM_PLAYER_CHANGE místo
188 // PLAYER_CHANGE
189 .led_index = 0,
190 .red = 0,
191 .green = 0,
192 .blue = 0,
193 .duration_ms = 0,
194 .data = &player_color // Předat barvu hráče
195 };
196 led_execute_command_new(&player_change_cmd);
197
198 // Vyčistit stav rošády
199 memset(&castling_state, 0, sizeof(castling_state));
200
201 // Zobrazit pohyblivé figury pro nového hráče
202 led_clear_board_only();
203 game_highlight_movable_pieces();
204
205 ESP_LOGI(TAG, "🏰 Castling completed! %s to move",
206 (current_player == PLAYER_WHITE) ? "White" : "Black");
207
208 // Zkontrolovat, zda je nový hráč v šachu
209 bool in_check = game_is_king_in_check(current_player);
210 if (in_check) {
211 // Najít pozici krále
212 int king_row = -1, king_col = -1;
213 piece_t king_piece = (current_player == PLAYER_WHITE)
214 ? PIECE_WHITE_KING
215 : PIECE_BLACK_KING;
216 for (int row = 0; row < 8; row++) {
217 for (int col = 0; col < 8; col++) {
218 if (board[row][col] == king_piece) {
219 king_row = row;
220 king_col = col;
221 break;
222 }
223 }
224 if (king_row != -1)
225 break;
226 }
227
228 if (king_row != -1 && king_col != -1) {
229 if (game_led_guidance_show_check_anim()) {
230 uint8_t king_led_index = chess_pos_to_led_index(king_row, king_col);
231 // Spustit check animaci - růžové svícení na králi
232 led_command_t check_cmd = {.type = LED_CMD_ANIM_CHECK,
233 .led_index = king_led_index,
234 .red = 0,
235 .green = 0,
236 .blue = 0,
237 .duration_ms =
238 0, // Trvalé až do dalšího tahu
239 .data = NULL};
240 led_execute_command_new(&check_cmd);
241 ESP_LOGI(TAG, "⚠️ CHECK! %s is in check",
242 current_player == PLAYER_WHITE ? "White" : "Black");
243 }
244 }
245 }
246 }
247
248 return true;
249 } else {
250 // Nesprávný tah věže - ukázat znovu guidance
251 ESP_LOGI(TAG, "❌ Incorrect rook move for castling");
252 game_show_castling_rook_guidance();
253 return false;
254 }

266void game_start_castle_animation(const chess_move_extended_t *move) {
267 if (!move)
268 return;
269
270 ESP_LOGI(TAG, "🏰 Starting castle animation for %s",
271 (move->move_type == MOVE_TYPE_CASTLE_KING) ? "kingside"
272 : "queenside");
273
274 // Calculate rook positions
275 uint8_t king_row = (current_player == PLAYER_WHITE) ? 0 : 7;
276
277 if (move->move_type == MOVE_TYPE_CASTLE_KING) {
278 // Kingside: rook from h-file to f-file
279 rook_from_row = king_row;
280 rook_from_col = 7; // h-file
281 rook_to_row = king_row;
282 rook_to_col = 5; // f-file
283 } else {
284 // Queenside: rook from a-file to d-file
285 rook_from_row = king_row;
286 rook_from_col = 0; // a-file
287 rook_to_row = king_row;
288 rook_to_col = 3; // d-file
289 }
290
291 // Store castle move for completion
292 pending_castle_move = *move;
293 castle_animation_active = true;
294
295 // Clear board and start repeating rook move animation
296 led_clear_board_only();
297
298 // Spustit opakující se animaci pohybu věže
299 game_start_repeating_rook_animation();
300
301 // Show initial animation immediately
302 uint8_t from_led = chess_pos_to_led_index(rook_from_row, rook_from_col);
303 uint8_t to_led = chess_pos_to_led_index(rook_to_row, rook_to_col);
304 led_set_pixel_safe(from_led, 255, 255, 0); // Yellow for rook position
305 led_set_pixel_safe(to_led, 0, 255, 0); // Green for destination
306
307 ESP_LOGI(TAG, "🏰 Rook at %c%d needs to move to %c%d", 'a' + rook_from_col,
308 rook_from_row + 1, 'a' + rook_to_col, rook_to_row + 1);
309}

315void rook_animation_timer_callback(TimerHandle_t xTimer) {
316 if (!rook_animation_active || !castle_animation_active) {

327 // Show rook position in yellow (pulsing)

330

332 // Bright yellow for rook position
333 led_set_pixel_safe(from_led, 255, 255, 0);

344 rook_from_row + 1, 'a' + rook_to_col, rook_to_row + 1,
345 pulse_state ? "bright" : "dim");

351void game_start_repeating_rook_animation(void) {
352 if (rook_animation_timer == NULL) {
353 // Create timer for rook animation
354 rook_animation_timer =
355 xTimerCreate("rook_anim_timer",
356 pdMS_TO_TICKS(500), // 500ms interval (snappier)
357 pdTRUE, // Auto-reload
358 NULL, rook_animation_timer_callback);
359

363 }

365
366 rook_animation_active = true;
367
368 // Start the timer
369 if (xTimerStart(rook_animation_timer, 0) != pdPASS) {
370 ESP_LOGE(TAG, "❌ Failed to start rook animation timer");
371 rook_animation_active = false;

424bool game_complete_castle_animation(uint8_t from_row, uint8_t from_col,
425 uint8_t to_row, uint8_t to_col) {
426 if (!castle_animation_active) {
427 return false;
428 }
429
430 // Check if rook was moved to correct position
431 if (from_row == rook_from_row && from_col == rook_from_col &&
432 to_row == rook_to_row && to_col == rook_to_col) {
433
434 ESP_LOGI(TAG, "✅ Castle animation completed! Rook moved correctly.");
435
436 // Skutečně přesunout věž až když hráč udělá správný tah
437 piece_t rook_piece = board[rook_from_row][rook_from_col];
438 board[rook_from_row][rook_from_col] = PIECE_EMPTY;
439 board[rook_to_row][rook_to_col] = rook_piece;
440
441 // Stop repeating rook animation
442 game_stop_repeating_rook_animation();
443
444 // Complete the castle move
445 castle_animation_active = false;
446
447 // Switch players now that castle is complete
448 player_t previous_player = current_player;
449 current_player =
450 (current_player == PLAYER_WHITE) ? PLAYER_BLACK : PLAYER_WHITE;
451
452 // Po úspěšné rošádě spustit animaci změny hráče
453 game_show_player_change_animation(previous_player, current_player);
454
455 // Clear board and show movable pieces for new player
456 led_clear_board_only();
457 game_highlight_movable_pieces();
458
459 ESP_LOGI(TAG, "🏰 Castling completed! %s to move",
460 (current_player == PLAYER_WHITE) ? "White" : "Black");
461
462 return true;
463 } else {
464 ESP_LOGW(TAG,
465 "❌ Invalid rook move for castling: %c%d -> %c%d (expected: %c%d "
466 "-> %c%d)",
467 'a' + from_col, from_row + 1, 'a' + to_col, to_row + 1,
468 'a' + rook_from_col, rook_from_row + 1, 'a' + rook_to_col,
469 rook_to_row + 1);
470
471 // Keep castle animation active, wait for correct move
472 return false;
473 }

482void game_refresh_leds(void) {
483 ESP_LOGI(TAG, "🔄 Refreshing LED state based on game status...");
484
485 // If game is not active, just show the board pattern
486 if (!game_active && current_game_state != GAME_STATE_PROMOTION) {
487 // If we are in "setup" or "idle" mode
488 led_show_chess_board();
489 return;
490 }
491
492 // Clear board first (to remove HA colors)
493 led_clear_board_only();
494
495 if (game_is_matrix_guard_active()) {
496 ESP_LOGI(TAG, " - Restoring MATRIX GUARD LEDs (physical vs logic)");
497 game_matrix_guard_render_leds();
498 game_check_promotion_needed();
499 return;
500 }
501
502 // 1. Check for Error Recovery State
503 if (error_recovery_state.waiting_for_move_correction) {
504 ESP_LOGI(TAG, " - Restoring ERROR state LEDs");
505 // Show validation error
506 game_show_invalid_move_error_with_blink(error_recovery_state.invalid_row,
507 error_recovery_state.invalid_col);
508
509 // Also highlight the original valid position (Blue)
510 led_set_pixel_safe(
511 chess_pos_to_led_index(error_recovery_state.original_valid_row,
512 error_recovery_state.original_valid_col),
513 0, 0, 255);
514
515 return;
516 }
517
518 // 2. Check for Lifted Piece State
519 if (piece_lifted) {
520 ESP_LOGI(TAG, " - Restoring LIFTED piece state LEDs at %c%d",
521 'a' + lifted_piece_col, lifted_piece_row + 1);
522
523 // Highlight source square (Yellow)
524 led_set_pixel_safe(
525 chess_pos_to_led_index(lifted_piece_row, lifted_piece_col), 255, 255,
526 0);
527
528 // Show valid moves for this piece
529 move_suggestion_t suggestions[64];
530 uint32_t valid_moves = game_get_available_moves(
531 lifted_piece_row, lifted_piece_col, suggestions, 64);
532
533 if (valid_moves > 0) {
534 for (uint32_t i = 0; i < valid_moves; i++) {
535 uint8_t dest_row = suggestions[i].to_row;
536 uint8_t dest_col = suggestions[i].to_col;
537 uint8_t led_index = chess_pos_to_led_index(dest_row, dest_col);
538
539 // Check if destination has opponent's piece
540 piece_t dest_piece = board[dest_row][dest_col];
541 bool is_opponent_piece =
542 (current_player == PLAYER_WHITE && dest_piece >= PIECE_BLACK_PAWN &&
543 dest_piece <= PIECE_BLACK_KING) ||
544 (current_player == PLAYER_BLACK && dest_piece >= PIECE_WHITE_PAWN &&
545 dest_piece <= PIECE_WHITE_KING);
546
547 if (is_opponent_piece) {
548 led_set_pixel_safe(led_index, 255, 165, 0); // Orange for capture
549 } else {
550 led_set_pixel_safe(led_index, 0, 255, 0); // Green for move
551 }
552 }
553 }
554 return;
555 }
556
557 // 3. Check for Check state
558 if (game_is_king_in_check(current_player)) {
559 // Find King
560 int king_row = -1, king_col = -1;
561 piece_t king_piece =
562 (current_player == PLAYER_WHITE) ? PIECE_WHITE_KING : PIECE_BLACK_KING;
563 for (int row = 0; row < 8; row++) {
564 for (int col = 0; col < 8; col++) {
565 if (board[row][col] == king_piece) {
566 king_row = row;
567 king_col = col;
568 break;
569 }
570 }
571 if (king_row != -1)
572 break;
573 }
574
575 if (king_row != -1) {
576 if (game_led_guidance_show_check_anim()) {
577 ESP_LOGI(TAG, " - Restoring CHECK state LEDs");
578 uint8_t king_led_index = chess_pos_to_led_index(king_row, king_col);
579 led_command_t check_cmd = {.type = LED_CMD_ANIM_CHECK,
580 .led_index = king_led_index,
581 .red = 0,
582 .green = 0,
583 .blue = 0,
584 .duration_ms = 0,
585 .data = NULL};
586 led_execute_command_new(&check_cmd);
587 }
588 }
589 }
590
591 // 4. Default Active State - Highlight movable pieces
592 ESP_LOGI(TAG, " - Restoring ACTIVE state LEDs (movable pieces)");
593 game_highlight_movable_pieces();
594
595 // Update button LEDs (64-71) to match current game state (promotion green/blue)
596 // so they are correct after switching back to game mode (e.g. from HA).
597 game_check_promotion_needed();
598}
