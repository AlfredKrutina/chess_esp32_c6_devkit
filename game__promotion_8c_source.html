32void game_update_promotion_anchor_led(void) {
33 if (!promotion_state.pending) {
34 return;
35 }
36
37 uint8_t row = promotion_state.square_row;
38 uint8_t col = promotion_state.square_col;
39 uint8_t led_index = chess_pos_to_led_index(row, col);
40
41 // Pulse between purple (pending) and gold (choice/upgrade) for clarity
42 uint32_t now_ms = (uint32_t)(esp_timer_get_time() / 1000ULL);
43 float t = (float)(now_ms % 1000U) / 1000.0f; // 0..1
44 float phase = t * 2.0f * 3.14159f;
45 float mix = 0.5f + 0.5f * sinf(phase); // 0..1
46
47 // Purple base
48 float pr = 128.0f, pg = 0.0f, pb = 128.0f;
49 // Gold target
50 float gr = 255.0f, gg = 215.0f, gb = 0.0f;
51
52 uint8_t r = (uint8_t)(pr * (1.0f - mix) + gr * mix);
53 uint8_t g = (uint8_t)(pg * (1.0f - mix) + gg * mix);
54 uint8_t b = (uint8_t)(pb * (1.0f - mix) + gb * mix);
55
56 led_set_pixel_safe(led_index, r, g, b);
57}

219void game_process_promotion_button(uint8_t button_id) {
220 // #2 & #3: Atomic check with mutex
221 // Použít recursive mutex funkce (promotion_mutex je recursive mutex)
222 if (xSemaphoreTakeRecursive(promotion_mutex, pdMS_TO_TICKS(100)) != pdTRUE) {
223 ESP_LOGE(TAG, "❌ Failed to acquire promotion mutex");
224 return;
225 }
226
227 // Check if promotion is pending
228 if (!promotion_state.pending) {
229 xSemaphoreGiveRecursive(promotion_mutex);
230 ESP_LOGW(TAG, "⚠️ Promotion already processed");
231 return;
232 }
233
234 // Map physical button ID (0-3) to promotion choice
235 // Physical buttons are shared between players:
236 // - Button 0: QUEEN (both white and black)
237 // - Button 1: ROOK (both white and black)
238 // - Button 2: BISHOP (both white and black)
239 // - Button 3: KNIGHT (both white and black)
240 promotion_choice_t choice;
241 const char *piece_name;
242
243 if (button_id <= 3) {
244 // Valid physical button (0-3)
245 choice = (promotion_choice_t)button_id;
246 } else {
247 xSemaphoreGiveRecursive(promotion_mutex); // Release mutex on invalid button
248 ESP_LOGW(TAG, "⚠️ Invalid promotion button ID: %d (expected 0-3)",
249 button_id);
250 return;
251 }
252
253 ESP_LOGI(TAG, "👑 Physical promotion button pressed: button=%d, choice=%d",
254 button_id, choice);
255
256 // Get piece name
257 switch (choice) {
258 case PROMOTION_QUEEN:
259 piece_name = "Queen";
260 break;
261 case PROMOTION_ROOK:
262 piece_name = "Rook";
263 break;
264 case PROMOTION_BISHOP:
265 piece_name = "Bishop";
266 break;
267 case PROMOTION_KNIGHT:
268 piece_name = "Knight";
269 break;
270 default:
271 piece_name = "Unknown";
272 break;
273 }
274
275 ESP_LOGI(TAG, "👑 Processing promotion button %d: %s", button_id, piece_name);
276
277 // Execute promotion
278 if (game_execute_promotion(choice)) {
279 // Clear promotion state (mutex already held from beginning)
280 promotion_state.pending = false;
281
282 // Log success
283 printf("\r\n");
284 printf("═══════════════════════════════════════════════════════════════\r"
285 "\n");
286 printf("👑 PAWN PROMOTION SUCCESSFUL!\r\n");
287 printf(" • Square: %c%d\r\n", 'a' + promotion_state.square_col,
288 promotion_state.square_row + 1);
289 printf(" • Promoted to: %s\r\n", piece_name);
290 printf(" • Player: %s\r\n",
291 promotion_state.player == PLAYER_WHITE ? "White" : "Black");
292 printf("═══════════════════════════════════════════════════════════════\r"
293 "\n");
294 printf("\r\n");
295
296 // Switch to next player after promotion
297 player_t previous_player = current_player;
298 current_player =
299 (current_player == PLAYER_WHITE) ? PLAYER_BLACK : PLAYER_WHITE;
300
301 // Po dokončení promoce už žádná figurka není zvednutá – aby se zobrazily
302 // movable pieces (game_highlight_movable_pieces nesmí skipnout kvůli
303 // piece_lifted).
304 piece_lifted = false;
305 lifted_piece_row = 0;
306 lifted_piece_col = 0;
307 lifted_piece = PIECE_EMPTY;
308
309 // #3: Timer switch AFTER button promotion choice
310 ESP_LOGI(
311 TAG,
312 "🔄 Timer switch (button promotion): ending for %s, starting for %s",
313 previous_player == PLAYER_WHITE ? "White" : "Black",
314 current_player == PLAYER_WHITE ? "White" : "Black");
315 game_end_timer_move();
316 game_start_timer_move(current_player == PLAYER_WHITE);
317
318 // KRITICKÉ: Endgame kontrola PŘED player change animací!
319 // Pokud je endgame, player change se NESPOUŠTÍ
320 game_state_t end_game_result = game_check_end_game_conditions();
321 if (end_game_result == GAME_STATE_FINISHED) {
322 current_game_state = GAME_STATE_FINISHED;
323 game_active = false;
324
325 // Najít pozici krále vítěze pro endgame animaci
326 player_t winner =
327 (current_player == PLAYER_WHITE) ? PLAYER_BLACK : PLAYER_WHITE;
328 uint8_t king_pos = 28; // default e4
329 for (int i = 0; i < 64; i++) {
330 piece_t piece = board[i / 8][i % 8];
331 if ((winner == PLAYER_WHITE && piece == PIECE_WHITE_KING) ||
332 (winner == PLAYER_BLACK && piece == PIECE_BLACK_KING)) {
333 king_pos = i;
334 break;
335 }
336 }
337
338 ESP_LOGI(TAG,
339 "🎯 Game finished after promotion! Starting endgame animation "
340 "at position %d",
341 king_pos);
342
343 // Spustit endgame animaci (wave z krále vítěze)
344 led_command_t endgame_cmd = {.type = LED_CMD_ANIM_ENDGAME,
345 .led_index = king_pos,
346 .red = 255,
347 .green = 255,
348 .blue = 0, // Yellow
349 .duration_ms = 0, // Endless
350 .data = NULL};
351 led_execute_command_new(&endgame_cmd);
352
353 ESP_LOGI(
354 TAG,
355 "✅ Endgame animation started - player change animation SKIPPED");
356 } else {
357 // Není endgame - spustit player change animaci
358 uint8_t player_color =
359 (current_player == PLAYER_WHITE) ? 1 : 0; // 1=white, 0=black
360 led_command_t player_change_cmd = {
361 .type = LED_CMD_ANIM_PLAYER_CHANGE, // Použít
362 // ANIM_PLAYER_CHANGE místo
363 // PLAYER_CHANGE
364 .led_index = 0,
365 .red = 0,
366 .green = 0,
367 .blue = 0,
368 .duration_ms = 0,
369 .data = &player_color // Předat barvu hráče
370 };
371 led_execute_command_new(&player_change_cmd);
372
373 // Update LED indications
374 game_check_promotion_needed();
375 game_highlight_movable_pieces();
376 led_force_immediate_update(); // Commit highlight so movable pieces show
377 // right after player change animation
378 }
379
380 // Zkontrolovat, zda je nový hráč v šachu
381 bool in_check = game_is_king_in_check(current_player);
382 if (in_check) {
383 // Najít pozici krále
384 int king_row = -1, king_col = -1;
385 piece_t king_piece = (current_player == PLAYER_WHITE) ? PIECE_WHITE_KING
386 : PIECE_BLACK_KING;
387 for (int row = 0; row < 8; row++) {
388 for (int col = 0; col < 8; col++) {
389 if (board[row][col] == king_piece) {
390 king_row = row;
391 king_col = col;
392 break;
393 }
394 }
395 if (king_row != -1)
396 break;
397 }
398
399 if (king_row != -1 && king_col != -1) {
400 if (game_led_guidance_show_check_anim()) {
401 uint8_t king_led_index = chess_pos_to_led_index(king_row, king_col);
402 // Spustit check animaci - růžové svícení na králi
403 led_command_t check_cmd = {.type = LED_CMD_ANIM_CHECK,
404 .led_index = king_led_index,
405 .red = 0,
406 .green = 0,
407 .blue = 0,
408 .duration_ms =
409 0, // Trvalé až do dalšího tahu
410 .data = NULL};
411 led_execute_command_new(&check_cmd);
412 ESP_LOGI(TAG, "⚠️ CHECK! %s is in check",
413 current_player == PLAYER_WHITE ? "White" : "Black");
414 }
415 }
416 }
417
418 ESP_LOGI(TAG, "✅ Promotion completed successfully");
419 } else {
420 ESP_LOGE(TAG, "❌ Promotion execution failed");
421 }
422 // Release mutex at the end of the function to ensure it's always released
423 xSemaphoreGiveRecursive(promotion_mutex);
424}
