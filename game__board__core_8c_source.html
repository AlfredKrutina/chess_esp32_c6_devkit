68static bool game_parse_piece_from_fen(char c, piece_t *out_piece) {
69 if (out_piece == NULL) {
70 return false;
71 }
72 switch (c) {
73 case 'P':
74 *out_piece = PIECE_WHITE_PAWN;
75 return true;
76 case 'N':
77 *out_piece = PIECE_WHITE_KNIGHT;
78 return true;
79 case 'B':
80 *out_piece = PIECE_WHITE_BISHOP;
81 return true;
82 case 'R':
83 *out_piece = PIECE_WHITE_ROOK;
84 return true;
85 case 'Q':
86 *out_piece = PIECE_WHITE_QUEEN;
87 return true;
88 case 'K':
89 *out_piece = PIECE_WHITE_KING;
90 return true;
91 case 'p':
92 *out_piece = PIECE_BLACK_PAWN;
93 return true;
94 case 'n':
95 *out_piece = PIECE_BLACK_KNIGHT;
96 return true;
97 case 'b':
98 *out_piece = PIECE_BLACK_BISHOP;
99 return true;
100 case 'r':
101 *out_piece = PIECE_BLACK_ROOK;
102 return true;
103 case 'q':
104 *out_piece = PIECE_BLACK_QUEEN;
105 return true;
106 case 'k':
107 *out_piece = PIECE_BLACK_KING;
108 return true;
109 default:
110 return false;
111 }

114bool game_load_position_from_fen(const char *fen, player_t *active_player) {
115 if (fen == NULL || active_player == NULL) {
116 return false;
117 }
118
119 memset(board, 0, sizeof(board));
120
121 int row = 7;
122 int col = 0;
123 const char *p = fen;
124 while (*p != '\0' && *p != ' ') {
125 if (*p == '/') {
126 if (col != 8) {
127 return false;
128 }
129 row--;
130 col = 0;
131 p++;
132 continue;
133 }
134 if (*p >= '1' && *p <= '8') {
135 col += (*p - '0');
136 if (col > 8) {
137 return false;
138 }
139 p++;
140 continue;
141 }
142 piece_t piece = PIECE_EMPTY;
143 if (!game_parse_piece_from_fen(*p, &piece)) {
144 return false;
145 }
146 if (row < 0 || row > 7 || col < 0 || col > 7) {
147 return false;
148 }
149 board[row][col] = piece;
150 col++;
151 p++;
152 }
153
154 if (row != 0 || col != 8 || *p != ' ') {
155 return false;
156 }
157
158 p++;
159 if (*p == 'w') {
160 *active_player = PLAYER_WHITE;
161 } else if (*p == 'b') {
162 *active_player = PLAYER_BLACK;
163 } else {
164 return false;
165 }
166 return true;
167}

199bool game_is_same_color(piece_t piece1, piece_t piece2) {
200 if (piece1 == PIECE_EMPTY || piece2 == PIECE_EMPTY) {
201 return false;
202 }
203 return (game_is_white_piece(piece1) && game_is_white_piece(piece2)) ||
204 (game_is_black_piece(piece1) && game_is_black_piece(piece2));
205}
