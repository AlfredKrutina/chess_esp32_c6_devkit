240void ha_light_report_activity(const char *activity_type) {
241 if (!task_running) {
242 return;
243 }
244
245 // Keep this function lightweight: it may be called from input paths.
246 // Mode switching and LED refresh must happen in HA task context.
247 last_activity_time_ms = esp_timer_get_time() / 1000;
248
249 /* Rate limit: max 1 queued activity / 500 ms (šetří frontu a MQTT).
250 * Výjimka: POST /api/light/game_mode — musí vždy projít, jinak uživatel
251 * nevrátí desku z režimu lampy po rychlé sérii jiných událostí. */
252 static uint32_t last_report_time = 0;
253 uint32_t current_time = esp_timer_get_time() / 1000;
254 const bool force_queue =
255 (activity_type != NULL && strcmp(activity_type, "web_game_mode") == 0);
256 if (!force_queue && current_time - last_report_time < 500) {
257 return;
258 }
259 last_report_time = current_time;
260
261 ha_light_command_t cmd;
262 cmd.type = HA_CMD_ACTIVITY;
263 cmd.u.data = (void *)activity_type;
264
265 if (ha_light_cmd_queue != NULL) {
266 xQueueSend(ha_light_cmd_queue, &cmd, 0);
267 }
268}

273static void ha_light_check_activity_timeout(void) {
274 if (!task_running || current_mode == HA_MODE_HA) {
275 return; // Already in HA mode or task not running
276 }
277
278 if (last_activity_time_ms == 0) {
279 // First run - initialize timer
280 last_activity_time_ms = esp_timer_get_time() / 1000;
281 return;
282 }
283
284 uint32_t current_time_ms = esp_timer_get_time() / 1000;
285 uint32_t elapsed_ms = current_time_ms - last_activity_time_ms;
286 uint32_t timeout_ms = (uint32_t)activity_timeout_auto_sec * 1000;
287
288 if (elapsed_ms >= timeout_ms) {
289 if (ha_light_check_wifi_sta_connected()) {
290 ESP_LOGI(TAG,
291 "Activity timeout (%lu s) reached - auto-switching to HA mode",
292 (unsigned long)activity_timeout_auto_sec);
293 ha_light_switch_to_ha_mode();
294 } else {
295 ESP_LOGD(
296 TAG,
297 "Timeout reached but WiFi STA not connected - staying in GAME mode");
298 }
299 }

311static void ha_light_switch_to_ha_mode(void) {
312 ha_light_wdt_feed_if_own_task();
313 bool mode_changed = (current_mode != HA_MODE_HA);
314
315 if (mode_changed) {
316 ESP_LOGI(TAG, "Switching to HA MODE");
317 }
318 current_mode = HA_MODE_HA;
319
320 uint8_t r, g, b, brightness;
321 bool state;
322 if (ha_light_state_mutex != NULL &&
323 xSemaphoreTake(ha_light_state_mutex, pdMS_TO_TICKS(50)) == pdTRUE) {
324 r = ha_light_state.r;
325 g = ha_light_state.g;
326 b = ha_light_state.b;
327 brightness = ha_light_state.brightness;
328 state = ha_light_state.state;
329 xSemaphoreGive(ha_light_state_mutex);
330 } else {
331 r = g = b = 255;
332 brightness = 255;
333 state = true;
334 }
335
336 // Apply HA color/state to all 64 board LEDs
337 if (state) {
338 led_set_ha_color(r, g, b, brightness);
339 } else {
340 led_set_ha_color(0, 0, 0, 0);
341 }
342 ha_light_wdt_feed_if_own_task();
343
344 ha_light_publish_state();
345 ha_light_wdt_feed_if_own_task();
346}

353static void ha_light_switch_to_game_mode(void) {
354 if (current_mode == HA_MODE_GAME) {
355 return; // Already in game mode
356 }
357
358 ESP_LOGI(TAG, "Switching to GAME MODE");
359 ha_light_wdt_feed_if_own_task();

368 ha_light_wdt_feed_if_own_task();
369
370 // Publish state update
371 ha_light_publish_state();
372 ha_light_wdt_feed_if_own_task();
373}

392static esp_err_t mqtt_load_config_from_nvs(char *host, size_t host_len,
393 uint16_t *port, char *username,
394 size_t username_len, char *password,
395 size_t password_len) {
396 if (host == NULL || port == NULL || username == NULL || password == NULL ||
397 host_len == 0 || username_len == 0 || password_len == 0) {
398 ESP_LOGE(TAG, "Invalid parameters for mqtt_load_config_from_nvs");
399 return ESP_ERR_INVALID_ARG;
400 }
401
402 nvs_handle_t nvs_handle;
403 esp_err_t ret = nvs_open(MQTT_NVS_NAMESPACE, NVS_READONLY, &nvs_handle);
404 if (ret != ESP_OK) {
405 ESP_LOGD(TAG, "MQTT config not found in NVS, using defaults");
406 // Vrátit default hodnoty
407 strncpy(host, MQTT_DEFAULT_HOST, host_len - 1);
408 host[host_len - 1] = '\0';
409 *port = MQTT_DEFAULT_PORT;
410 strncpy(username, MQTT_DEFAULT_USERNAME, username_len - 1);
411 username[username_len - 1] = '\0';
412 strncpy(password, MQTT_DEFAULT_PASSWORD, password_len - 1);
413 password[password_len - 1] = '\0';
414 return ESP_ERR_NOT_FOUND; // Není v NVS, ale vrátil defaults
415 }
416
417 // Načíst host
418 size_t required_size = host_len;
419 ret = nvs_get_str(nvs_handle, MQTT_NVS_KEY_HOST, host, &required_size);
420 if (ret != ESP_OK) {
421 ESP_LOGW(TAG, "Failed to get MQTT host from NVS, using default");
422 strncpy(host, MQTT_DEFAULT_HOST, host_len - 1);
423 host[host_len - 1] = '\0';
424 }
425
426 // Načíst port
427 uint32_t port_val = MQTT_DEFAULT_PORT;
428 ret = nvs_get_u32(nvs_handle, MQTT_NVS_KEY_PORT, &port_val);
429 if (ret != ESP_OK || port_val == 0 || port_val > 65535) {
430 ESP_LOGW(TAG, "Failed to get MQTT port from NVS, using default");
431 port_val = MQTT_DEFAULT_PORT;
432 }
433 *port = (uint16_t)port_val;
434
435 // Načíst username (volitelné - může být prázdné)
436 required_size = username_len;
437 ret =
438 nvs_get_str(nvs_handle, MQTT_NVS_KEY_USERNAME, username, &required_size);
439 if (ret != ESP_OK) {
440 ESP_LOGD(TAG, "MQTT username not in NVS, using empty");
441 strncpy(username, MQTT_DEFAULT_USERNAME, username_len - 1);
442 username[username_len - 1] = '\0';
443 }
444
445 // Načíst password (volitelné - může být prázdné)
446 required_size = password_len;
447 ret =
448 nvs_get_str(nvs_handle, MQTT_NVS_KEY_PASSWORD, password, &required_size);
449 if (ret != ESP_OK) {
450 ESP_LOGD(TAG, "MQTT password not in NVS, using empty");
451 strncpy(password, MQTT_DEFAULT_PASSWORD, password_len - 1);
452 password[password_len - 1] = '\0';
453 }
454
455 nvs_close(nvs_handle);
456 ESP_LOGI(TAG, "MQTT config loaded from NVS: host=%s, port=%d, username=%s",
457 host, *port, username[0] ? username : "(empty)");
458
459 return ESP_OK;
460}

471esp_err_t mqtt_save_config_to_nvs(const char *host, uint16_t port,
472 const char *username, const char *password) {
473 if (host == NULL || port == 0) {
474 ESP_LOGE(TAG, "Invalid parameters: host is NULL or port is 0");
475 return ESP_ERR_INVALID_ARG;
476 }
477
478 size_t host_len = strlen(host);
479 if (host_len == 0 || host_len > 127) {
480 ESP_LOGE(TAG, "Invalid host length: %zu (must be 1-127)", host_len);
481 return ESP_ERR_INVALID_ARG;
482 }
483
484 // Validovat username/password pokud jsou nastavené
485 if (username != NULL) {
486 size_t username_len = strlen(username);
487 if (username_len > 63) {
488 ESP_LOGE(TAG, "Invalid username length: %zu (max 63)", username_len);
489 return ESP_ERR_INVALID_ARG;
490 }
491 }
492
493 if (password != NULL) {
494 size_t password_len = strlen(password);
495 if (password_len > 63) {
496 ESP_LOGE(TAG, "Invalid password length: %zu (max 63)", password_len);
497 return ESP_ERR_INVALID_ARG;
498 }
499 }
500
501 nvs_handle_t nvs_handle;
502 esp_err_t ret = nvs_open(MQTT_NVS_NAMESPACE, NVS_READWRITE, &nvs_handle);
503 if (ret != ESP_OK) {
504 ESP_LOGE(TAG, "Failed to open NVS: %s", esp_err_to_name(ret));
505 return ret;
506 }
507
508 // Uložit host
509 ret = nvs_set_str(nvs_handle, MQTT_NVS_KEY_HOST, host);
510 if (ret != ESP_OK) {
511 ESP_LOGE(TAG, "Failed to set MQTT host in NVS: %s", esp_err_to_name(ret));
512 nvs_close(nvs_handle);
513 return ret;
514 }
515
516 // Uložit port
517 ret = nvs_set_u32(nvs_handle, MQTT_NVS_KEY_PORT, (uint32_t)port);
518 if (ret != ESP_OK) {
519 ESP_LOGE(TAG, "Failed to set MQTT port in NVS: %s", esp_err_to_name(ret));
520 nvs_close(nvs_handle);
521 return ret;
522 }
523
524 // Uložit username (pokud je nastavený, jinak uložit prázdný string)
525 const char *username_to_save =
526 (username != NULL && strlen(username) > 0) ? username : "";
527 ret = nvs_set_str(nvs_handle, MQTT_NVS_KEY_USERNAME, username_to_save);
528 if (ret != ESP_OK) {
529 ESP_LOGE(TAG, "Failed to set MQTT username in NVS: %s",
530 esp_err_to_name(ret));
531 nvs_close(nvs_handle);
532 return ret;
533 }
534
535 // Uložit password (pokud je nastavený, jinak uložit prázdný string)
536 const char *password_to_save =
537 (password != NULL && strlen(password) > 0) ? password : "";
538 ret = nvs_set_str(nvs_handle, MQTT_NVS_KEY_PASSWORD, password_to_save);
539 if (ret != ESP_OK) {
540 ESP_LOGE(TAG, "Failed to set MQTT password in NVS: %s",
541 esp_err_to_name(ret));
542 nvs_close(nvs_handle);
543 return ret;
544 }
545
546 ret = nvs_commit(nvs_handle);
547 if (ret != ESP_OK) {
548 ESP_LOGE(TAG, "Failed to commit NVS: %s", esp_err_to_name(ret));
549 nvs_close(nvs_handle);
550 return ret;
551 }
552
553 nvs_close(nvs_handle);
554 ESP_LOGI(TAG, "MQTT config saved to NVS: host=%s, port=%d, username=%s", host,
555 port, username_to_save[0] ? username_to_save : "(empty)");
556
557 return ESP_OK;
558}

567static void ha_light_publish_discovery(void) {
568 if (!mqtt_connected || mqtt_client == NULL) {
569 return;
570 }
571 if (esp_get_free_heap_size() < HA_MQTT_STOP_HEAP_BYTES) {
572 ESP_LOGD(TAG, "[STAGING] ha_light_publish_discovery: skip (low heap)");
573 return;
574 }
575
576 // Get MAC address for unique ID
577 uint8_t mac_raw[6];
578 esp_read_mac(mac_raw, ESP_MAC_WIFI_STA);
579 char mac_str[13];
580 snprintf(mac_str, sizeof(mac_str), "%02x%02x%02x%02x%02x%02x", mac_raw[0],
581 mac_raw[1], mac_raw[2], mac_raw[3], mac_raw[4], mac_raw[5]);
582
583 // Construct unique ID and name
584 char unique_id[64];
585 snprintf(unique_id, sizeof(unique_id), "esp32_chess_light_%s", mac_str);
586
587 // Construct discovery topic: homeassistant/light/[node_id]/config
588 char discovery_topic[128];
589 snprintf(discovery_topic, sizeof(discovery_topic), "%s/%s/%s/config",
590 HA_DISCOVERY_PREFIX, HA_COMPONENT_LIGHT, unique_id);
591
592 ESP_LOGI(TAG, "Publishing HA Discovery to %s", discovery_topic);
593
594 // Create JSON payload
595 cJSON *json = cJSON_CreateObject();
596 cJSON_AddStringToObject(json, "name", "CzechMate");
597 cJSON_AddStringToObject(json, "unique_id", unique_id);
598 cJSON_AddStringToObject(json, "command_topic", HA_TOPIC_LIGHT_COMMAND);
599 cJSON_AddStringToObject(json, "state_topic", HA_TOPIC_LIGHT_STATE);
600 cJSON_AddStringToObject(json, "availability_topic",
601 HA_TOPIC_LIGHT_AVAILABILITY);
602 cJSON_AddStringToObject(json, "payload_available", HA_MQTT_PAYLOAD_ONLINE);
603 cJSON_AddStringToObject(json, "payload_not_available",
604 HA_MQTT_PAYLOAD_OFFLINE);
605 cJSON_AddStringToObject(json, "schema", "json");
606 cJSON_AddBoolToObject(json, "brightness", true);
607 cJSON_AddBoolToObject(json, "color_mode", true);
608 cJSON *color_modes = cJSON_CreateArray();
609 cJSON_AddItemToArray(color_modes, cJSON_CreateString("rgb"));
610 cJSON_AddItemToObject(json, "supported_color_modes", color_modes);
611
612 // Effect support
613 cJSON_AddBoolToObject(json, "effect", true);
614 cJSON *effects = cJSON_CreateArray();
615 cJSON_AddItemToArray(effects, cJSON_CreateString("rainbow"));
616 cJSON_AddItemToArray(effects, cJSON_CreateString("pulse"));
617 cJSON_AddItemToArray(effects, cJSON_CreateString("static"));
618 cJSON_AddItemToObject(json, "effect_list", effects);
619
620 // Device info
621 cJSON *device = cJSON_CreateObject();
622 cJSON_AddStringToObject(device, "identifiers", unique_id);
623 cJSON_AddStringToObject(device, "name", "ESP32 Chess System");
624 cJSON_AddStringToObject(device, "manufacturer", HA_DEVICE_MANUFACTURER);
625 cJSON_AddStringToObject(device, "model", HA_DEVICE_MODEL);
626 cJSON_AddStringToObject(device, "sw_version", HA_DEVICE_SW_VERSION);
627 cJSON_AddItemToObject(json, "device", device);
628
629 char *json_str = cJSON_PrintUnformatted(json);
630 if (json_str != NULL) {
631 // Publish with retain=true so HA picks it up anytime
632 esp_mqtt_client_publish(mqtt_client, discovery_topic, json_str, 0, 1, 1);
633 free(json_str);
634 }
635 cJSON_Delete(json);
636}

641static void ha_light_mqtt_event_handler(void *handler_args,
642 esp_event_base_t base, int32_t event_id,
643 void *event_data) {
644 esp_mqtt_event_handle_t event = (esp_mqtt_event_handle_t)event_data;
645 esp_mqtt_client_handle_t client = event->client;
646
647 switch ((esp_mqtt_event_id_t)event_id) {
648 case MQTT_EVENT_CONNECTED:
649 ESP_LOGI(TAG, "MQTT Connected");
650 mqtt_connected = true;
651
652 // Subscribe to light command topic
653 esp_mqtt_client_subscribe(client, HA_TOPIC_LIGHT_COMMAND, 0);
654 ESP_LOGI(TAG, "Subscribed to %s", HA_TOPIC_LIGHT_COMMAND);
655
656 // Publish Auto-Discovery Config
657 ha_light_publish_discovery();
658
659 // Publish availability: online
660 esp_mqtt_client_publish(client, HA_TOPIC_LIGHT_AVAILABILITY,
661 HA_MQTT_PAYLOAD_ONLINE, 0, 1, 1);
662
663 // Publish initial state
664 ha_light_publish_state();
665 break;
666
667 case MQTT_EVENT_DISCONNECTED:
668 ESP_LOGI(TAG, "MQTT Disconnected");
669 mqtt_connected = false;
670 break;
671
672 case MQTT_EVENT_DATA:
673 ESP_LOGI(TAG, "MQTT DATA: topic=%.*s, data=%.*s", event->topic_len,
674 event->topic, event->data_len, event->data);
675
676 // Handle command
677 char topic[128] = {0};
678 char data[512] = {0};
679
680 if (event->topic_len < sizeof(topic) && event->data_len < sizeof(data)) {
681 memcpy(topic, event->topic, event->topic_len);
682 memcpy(data, event->data, event->data_len);
683 ha_light_handle_mqtt_command(topic, data, event->data_len);
684 }
685 break;
686
687 case MQTT_EVENT_ERROR:

699static void ha_light_handle_mqtt_command(const char *topic, const char *data,
700 int data_len) {
701 if (strcmp(topic, HA_TOPIC_LIGHT_COMMAND) != 0) {
702 return; // Not our topic
703 }
704
705 // Check if we can apply HA command during GAME mode
706 bool can_apply_command = true;
707 if (current_mode == HA_MODE_GAME) {
708 uint32_t current_time_ms = esp_timer_get_time() / 1000;
709 uint32_t idle_ms = current_time_ms - last_activity_time_ms;
710
711 if (idle_ms >= HA_ACTIVITY_TIMEOUT_COMMAND_MS ||
712 game_get_move_count() == 0) {
713 // Idle ≥ 2min OR no moves yet → allow switch to HA mode
714 if (game_get_move_count() == 0) {
715 ESP_LOGI(TAG,
716 "HA command accepted immediately (game not started, moves=0)");
717 } else {
718 ESP_LOGI(
719 TAG,
720 "HA command after %lu ms idle (≥2min) - will switch to HA mode",
721 idle_ms);
722 }
723 can_apply_command = true;
724 } else {
725 // Idle < 2min AND game in progress → ignore command (game has priority)
726 ESP_LOGI(
727 TAG,
728 "HA command IGNORED (game active, moves=%lu, idle=%lu ms < 2min)",
729 game_get_move_count(), idle_ms);
730 can_apply_command = false;
731 }
732 }
733
734 // Parse JSON
735 cJSON *json = cJSON_ParseWithLength(data, data_len);
736 if (json == NULL) {
737 ESP_LOGE(TAG, "Failed to parse MQTT JSON");
738 return;
739 }
740
741 bool new_state = ha_light_state.state;
742 uint8_t new_r = ha_light_state.r, new_g = ha_light_state.g,
743 new_b = ha_light_state.b;
744 uint8_t new_brightness = ha_light_state.brightness;
745
746 cJSON *state_item = cJSON_GetObjectItem(json, "state");
747 if (cJSON_IsString(state_item)) {
748 if (strcasecmp(state_item->valuestring, "ON") == 0) {
749 new_state = true;
750 } else if (strcasecmp(state_item->valuestring, "OFF") == 0) {
751 new_state = false;
752 }
753 }
754
755 cJSON *brightness_item = cJSON_GetObjectItem(json, "brightness");
756 if (cJSON_IsNumber(brightness_item)) {
757 int brightness_val = (int)cJSON_GetNumberValue(brightness_item);
758 if (brightness_val < 0)
759 brightness_val = 0;
760 if (brightness_val > 255)
761 brightness_val = 255;
762 new_brightness = (uint8_t)brightness_val;
763 }
764
765 cJSON *color_item = cJSON_GetObjectItem(json, "color");
766 if (cJSON_IsObject(color_item)) {
767 cJSON *r = cJSON_GetObjectItem(color_item, "r");
768 cJSON *g = cJSON_GetObjectItem(color_item, "g");
769 cJSON *b = cJSON_GetObjectItem(color_item, "b");
770 if (cJSON_IsNumber(r) && cJSON_IsNumber(g) && cJSON_IsNumber(b)) {
771 new_r = (uint8_t)cJSON_GetNumberValue(r);
772 new_g = (uint8_t)cJSON_GetNumberValue(g);
773 new_b = (uint8_t)cJSON_GetNumberValue(b);
774 }
775 }
776
777 cJSON *rgb_item = cJSON_GetObjectItem(json, "rgb_color");
778 if (cJSON_IsArray(rgb_item) && cJSON_GetArraySize(rgb_item) >= 3) {
779 new_r = (uint8_t)cJSON_GetNumberValue(cJSON_GetArrayItem(rgb_item, 0));
780 new_g = (uint8_t)cJSON_GetNumberValue(cJSON_GetArrayItem(rgb_item, 1));
781 new_b = (uint8_t)cJSON_GetNumberValue(cJSON_GetArrayItem(rgb_item, 2));
782 }
783
784 char new_effect[32] = "solid";
785 cJSON *effect_item = cJSON_GetObjectItem(json, "effect");
786 if (cJSON_IsString(effect_item) && effect_item->valuestring) {
787 strncpy(new_effect, effect_item->valuestring, sizeof(new_effect) - 1);
788 new_effect[sizeof(new_effect) - 1] = '\0';
789 }
790
791 if (ha_light_state_mutex != NULL &&
792 xSemaphoreTake(ha_light_state_mutex, pdMS_TO_TICKS(100)) == pdTRUE) {
793 ha_light_state.state = new_state;
794 ha_light_state.r = new_r;
795 ha_light_state.g = new_g;
796 ha_light_state.b = new_b;
797 ha_light_state.brightness = new_brightness;
798 strncpy(ha_light_state.effect, new_effect,
799 sizeof(ha_light_state.effect) - 1);
800 ha_light_state.effect[sizeof(ha_light_state.effect) - 1] = '\0';
801 xSemaphoreGive(ha_light_state_mutex);
802 }
803
804 cJSON_Delete(json);
805
806 // Apply changes based on mode and idle time
807 if (can_apply_command) {
808 if (current_mode == HA_MODE_GAME) {
809 // Switch from GAME to HA mode
810 ESP_LOGI(TAG, "Switching from GAME to HA mode (HA command accepted)");
811 ha_light_switch_to_ha_mode();
812 } else {
813 // Already in HA mode, just re-apply
814 ha_light_switch_to_ha_mode(); // Re-apply with new settings
815 }
816 }
817
818 // Always publish state update (virtual state)
819 ha_light_publish_state();
820}

827 return;

876 time_until_timeout = HA_ACTIVITY_TIMEOUT_AUTO_MS - elapsed;
877 }

880
881 char *json_str = cJSON_Print(json);
882 if (json_str != NULL) {
883 esp_mqtt_client_publish(mqtt_client, HA_TOPIC_LIGHT_STATE, json_str, 0, 1,
884 0);
885 free(json_str);
886 }
887 cJSON_Delete(json);

930static esp_err_t ha_light_init_mqtt(void) {
931 // Prevent multiple clients/memory leaks
932 if (mqtt_client != NULL) {
933 ESP_LOGI(TAG, "MQTT client already initialized, skipping re-init");
934 return ESP_OK;
935 }
936
937 size_t free_heap = esp_get_free_heap_size();
938 if (free_heap < HA_MQTT_MIN_FREE_HEAP_BYTES) {
939 ESP_LOGW(TAG,
940 "MQTT deferred: free heap %zu B < %d B (need RAM for mqtt task)",
941 free_heap, HA_MQTT_MIN_FREE_HEAP_BYTES);
942 return ESP_ERR_NO_MEM;
943 }
944
945 // Ensure config is loaded from NVS (first time only)
946 mqtt_ensure_config_loaded();
947
948 // Build MQTT broker URI (buffer size 512 to avoid format-truncation warning)
949 char mqtt_broker_uri[512];
950 if (mqtt_config.username[0] != '\0' && mqtt_config.password[0] != '\0') {
951 // MQTT URI with username/password: mqtt://user:pass@host:port
952 snprintf(mqtt_broker_uri, sizeof(mqtt_broker_uri), "mqtt://%s:%s@%s:%d",
953 mqtt_config.username, mqtt_config.password, mqtt_config.host,
954 mqtt_config.port);
955 } else {
956 // MQTT URI without auth: mqtt://host:port
957 snprintf(mqtt_broker_uri, sizeof(mqtt_broker_uri), "mqtt://%s:%d",
958 mqtt_config.host, mqtt_config.port);
959 }
960
961 // Generate unique Client ID combined from Base + MAC
962 uint8_t mac_raw[6];
963 esp_read_mac(mac_raw, ESP_MAC_WIFI_STA);
964 char client_id[64];
965 snprintf(client_id, sizeof(client_id), "%s-%02x%02x%02x%02x%02x%02x",
966 HA_MQTT_CLIENT_ID, mac_raw[0], mac_raw[1], mac_raw[2], mac_raw[3],
967 mac_raw[4], mac_raw[5]);
968
969 ESP_LOGI(TAG, "Initializing MQTT client: %s (ID: %s)", mqtt_broker_uri,
970 client_id);
971
972 esp_mqtt_client_config_t mqtt_cfg = {
973 .broker.address.uri = mqtt_broker_uri,
974 .credentials.client_id = client_id,
975 .session.last_will.topic = HA_TOPIC_LIGHT_AVAILABILITY,
976 .session.last_will.msg = HA_MQTT_PAYLOAD_OFFLINE,
977 .session.last_will.qos = 1,
978 .session.last_will.retain = true,
979 .session.keepalive = 120, // Increase keepalive to 120s for stability
980 .network.disable_auto_reconnect =
981 false, // Ensure auto-reconnect is enabled
982 };
983
984 mqtt_client = esp_mqtt_client_init(&mqtt_cfg);
985 if (mqtt_client == NULL) {
986 ESP_LOGE(TAG, "Failed to initialize MQTT client");
987 return ESP_FAIL;
988 }
989
990 esp_mqtt_client_register_event(mqtt_client, ESP_EVENT_ANY_ID,
991 ha_light_mqtt_event_handler, NULL);
992
993 esp_err_t ret = esp_mqtt_client_start(mqtt_client);
994 if (ret != ESP_OK) {
995 ESP_LOGE(TAG, "Failed to start MQTT client: %s", esp_err_to_name(ret));
996 esp_mqtt_client_destroy(mqtt_client);
997 mqtt_client = NULL;
998 return ret;
999 }
1000
1001 ESP_LOGI(TAG, "MQTT client initialized and started");
1002 return ESP_OK;
1003}

1243void ha_light_task_start(void *pvParameters) {
1244 ESP_LOGI(TAG, "Starting HA Light Task...");
1245 s_ha_light_task_hdl = xTaskGetCurrentTaskHandle();
1246
1247 // Register with WDT
1248 esp_err_t ret = esp_task_wdt_add(NULL);
1249 if (ret != ESP_OK && ret != ESP_ERR_INVALID_ARG) {
1250 ESP_LOGW(TAG, "Failed to register with WDT: %s", esp_err_to_name(ret));
1251 }
1252
1253 task_running = true;
1254 current_mode = HA_MODE_GAME;
1255 last_activity_time_ms = esp_timer_get_time() / 1000;
1256
1257 ha_light_state_mutex = xSemaphoreCreateMutex();
1258 if (ha_light_state_mutex == NULL) {
1259 ESP_LOGE(TAG, "Failed to create ha_light_state mutex");
1260 }
1261
1262 ha_light_cmd_queue = xQueueCreate(10, sizeof(ha_light_command_t));
1263 if (ha_light_cmd_queue == NULL) {
1264 ESP_LOGE(TAG, "Failed to create HA command queue");
1265 }
1266
1267 lamp_nvs_load();
1268
1269 // Main loop
1270 // TickType_t last_wake_time = xTaskGetTickCount(); // Unused in queue-based
1271 // loop
1272
1273 for (;;) {
1274 // =========================================================================
1275 // 🛑 BOOT ANIMATION PROTECTION - BLOKUJE HA OPERACE BĚHEM BOOT
1276 // =========================================================================
1277 // Pokud běží boot animace, HA task nesmí posílat LED příkazy!
1278 // (např. "zhasni světlo" při rychlém WiFi připojení)
1279 if (led_is_booting()) {
1280 ha_light_task_wdt_reset_safe();
1281 vTaskDelay(pdMS_TO_TICKS(10)); // 10ms - rychlé probuzení po fade_out!
1282 continue;
1283 }
1284 // =========================================================================
1285
1286 // Reset WDT
1287 ha_light_task_wdt_reset_safe();
1288
1289 // Process queue with timeout (100ms) - replaces vTaskDelay
1290 ha_light_command_t cmd;
1291 if (ha_light_cmd_queue &&
1292 xQueueReceive(ha_light_cmd_queue, &cmd, pdMS_TO_TICKS(100)) == pdTRUE) {
1293 if (cmd.type == HA_CMD_WEB_LAMP) {
1294 if (ha_light_state_mutex != NULL &&
1295 xSemaphoreTake(ha_light_state_mutex, pdMS_TO_TICKS(100)) ==
1296 pdTRUE) {
