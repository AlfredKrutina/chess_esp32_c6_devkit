122static inline bool chess_gpio_pin_is_stm32_nrst_output(int gpio_num) {
123#if CONFIG_CHESS_STM32_I2C_BL_ENABLE
124 if (gpio_num < 0) {
125 return false;

127 return (CONFIG_CHESS_STM32_BL_NRST_GPIO_SEG0 >= 0 &&
128 gpio_num == CONFIG_CHESS_STM32_BL_NRST_GPIO_SEG0) ||
129 (CONFIG_CHESS_STM32_BL_NRST_GPIO_SEG1 >= 0 &&
130 gpio_num == CONFIG_CHESS_STM32_BL_NRST_GPIO_SEG1) ||
131 (CONFIG_CHESS_STM32_BL_NRST_GPIO_SEG2 >= 0 &&

133 (CONFIG_CHESS_STM32_BL_NRST_GPIO_SEG3 >= 0 &&
134 gpio_num == CONFIG_CHESS_STM32_BL_NRST_GPIO_SEG3);
135#else
136 (void)gpio_num;
137 return false;
