35static const char *skip_leading_ws(const char *s) {

116static void cli_web_command(const char *sub_args) {
117 char verb[16];
118 if (sscanf(sub_args, "%15s", verb) != 1) {
119 uart_send_error("CLI WEB START|STOP|STATUS");
120 return;
121 }
122
123 if (!strcasecmp(verb, "START")) {
124 esp_err_t e = web_server_enqueue_command(WEB_CMD_START_SERVER);
125 if (e == ESP_OK) {
126 uart_send_success("WEB START zařazeno (web_server task)");
127 } else {
128 uart_send_formatted("WEB START chyba: %s", esp_err_to_name(e));
129 }
130 return;
131 }
132 if (!strcasecmp(verb, "STOP")) {
133 esp_err_t e = web_server_enqueue_command(WEB_CMD_STOP_SERVER);
134 if (e == ESP_OK) {
135 uart_send_success("WEB STOP zařazeno");
136 } else {
137 uart_send_formatted("WEB STOP chyba: %s", esp_err_to_name(e));
138 }
139 return;
140 }
141 if (!strcasecmp(verb, "STATUS")) {
142 esp_err_t e = web_server_enqueue_command(WEB_CMD_GET_STATUS);
143 if (e != ESP_OK) {
144 uart_send_formatted("WEB STATUS fronta: %s", esp_err_to_name(e));
145 }
146 uart_send_formatted(" active=%d task_run=%d clients=%lu uptime_ms=%lu err=%s",
147 web_server_is_active() ? 1 : 0,
148 web_server_is_task_running() ? 1 : 0,
149 (unsigned long)web_server_get_client_count(),
150 (unsigned long)web_server_get_uptime(),
151 esp_err_to_name(web_server_get_last_http_error()));
152 uart_send_formatted(" AP SSID=%s web_locked=%d", web_server_get_ap_ssid(),
153 web_is_locked() ? 1 : 0);
154 return;
155 }
156 uart_send_error("CLI WEB START|STOP|STATUS");
157}

171static void cli_lock_command(const char *sub_args) {

209 if (e != ESP_OK || json == NULL || len == 0) {
210 uart_send_formatted("SNAPSHOT chyba: %s", esp_err_to_name(e));

213 uart_send_colored_line(COLOR_INFO, "Snapshot JSON");
214 const size_t chunk = 240;
215 for (size_t i = 0; i < len; i += chunk) {
216 size_t n = len - i;
217 if (n > chunk) {
218 n = chunk;

220 char line[256];
221 if (n >= sizeof(line)) {
222 n = sizeof(line) - 1;
223 }
224 memcpy(line, json + i, n);
225 line[n] = '\0';
226 uart_send_formatted("%s", line);
227 }

243command_result_t uart_cmd_cli(const char *args) {
244 char line[512];
245 if (args) {
246 strncpy(line, args, sizeof(line) - 1);
247 line[sizeof(line) - 1] = '\0';
248 } else {
249 line[0] = '\0';
250 }
251
252 char cmd[32];
253 const char *s = skip_leading_ws(line);
254 size_t ci = 0;
255 while (*s && *s != ' ' && *s != '\t' && ci + 1 < sizeof(cmd)) {
256 cmd[ci++] = (char)toupper((unsigned char)*s++);
257 }
258 cmd[ci] = '\0';
259
260 if (cmd[0] == '\0' || !strcasecmp(cmd, "HELP") || !strcasecmp(cmd, "?")) {
261 uart_cli_print_help();
262 return CMD_SUCCESS;
263 }
264
265 char tail[480];
266 skip_first_word(line, tail, sizeof(tail));
267
268 if (!strcasecmp(cmd, "PARTITIONS") || !strcasecmp(cmd, "PART")) {
269 cli_dump_partitions();
270 return CMD_SUCCESS;
271 }
272 if (!strcasecmp(cmd, "FLASH")) {
273 cli_flash_info();
274 return CMD_SUCCESS;
275 }
276 if (!strcasecmp(cmd, "CHIP")) {
277 cli_chip_info();
278 return CMD_SUCCESS;
279 }
280 if (!strcasecmp(cmd, "APP")) {
281 cli_app_info();
282 return CMD_SUCCESS;
283 }
284 if (!strcasecmp(cmd, "HEAP")) {
285 uart_send_formatted("heap_free=%zu heap_min=%zu internal_free=%zu",
286 (size_t)esp_get_free_heap_size(),
287 (size_t)esp_get_minimum_free_heap_size(),
288 (size_t)heap_caps_get_free_size(MALLOC_CAP_INTERNAL));
289 return CMD_SUCCESS;
290 }
291 if (!strcasecmp(cmd, "WEB")) {
292 if (tail[0] == '\0') {
293 uart_send_error("CLI WEB START|STOP|STATUS");
294 return CMD_ERROR_INVALID_SYNTAX;
295 }
296 cli_web_command(tail);
297 return CMD_SUCCESS;
298 }
299 if (!strcasecmp(cmd, "LOCK")) {
300 char sub[16];
301 sscanf(tail, "%15s", sub);
302 if (sub[0] == '\0') {
303 uart_send_error("CLI LOCK ON|OFF");
304 return CMD_ERROR_INVALID_SYNTAX;
305 }
306 cli_lock_command(sub);
307 return CMD_SUCCESS;
308 }
309 if (!strcasecmp(cmd, "OTA")) {
310 const char *p = skip_leading_ws(tail);
311 if (p[0] == '\0') {
312 cli_ota_info();
313 return CMD_SUCCESS;
314 }
315 if (!strncasecmp(p, "INFO", 4) &&
316 (p[4] == '\0' || p[4] == ' ' || p[4] == '\t')) {
317 cli_ota_info();
318 return CMD_SUCCESS;
319 }
320 esp_err_t oe = ota_update_try_start_url(p);
321 if (oe == ESP_OK) {
322 uart_send_success("OTA stažení spuštěno na pozadí");
323 } else {
324 uart_send_formatted("OTA start: %s", esp_err_to_name(oe));
325 ESP_LOGW(TAG, "OTA URL rejected: %s", esp_err_to_name(oe));
326 }
327 return CMD_SUCCESS;
328 }
329 if (!strcasecmp(cmd, "BLE")) {
330 cli_ble_json(tail);
331 return CMD_SUCCESS;
332 }
333 if (!strcasecmp(cmd, "SNAP") || !strcasecmp(cmd, "SNAPSHOT")) {
334 cli_snapshot();
335 return CMD_SUCCESS;
336 }
337 if (!strcasecmp(cmd, "RESET")) {
338 uart_send_line("Restart…");
339 esp_restart();
340 return CMD_SUCCESS;
341 }
342
343 uart_send_error("Neznámý CLI podpříkaz — zadej CLI HELP");
344 return CMD_ERROR_INVALID_SYNTAX;
345}
