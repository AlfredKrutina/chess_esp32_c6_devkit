212static size_t cli_hex_to_bin(const char *hex, uint8_t *out, size_t out_max) {
213 size_t len = strlen(hex);
214 if (len < 2 || (len % 2) != 0) {
215 return 0;
216 }
217 size_t n = len / 2;
218 if (n > out_max) {
219 return 0;
220 }
221 for (size_t i = 0; i < n; i++) {
222 unsigned v = 0;
223 if (sscanf(hex + 2 * i, "%2x", &v) != 1) {
224 return 0;
225 }
226 out[i] = (uint8_t)v;
227 }
228 return n;
229}

231static command_result_t cli_stm32_bl_tail(const char *tail) {
232 const char *p = skip_leading_ws(tail);
233 char verb[24];
234 if (sscanf(p, "%23s", verb) != 1) {
235 uart_send_error("CLI STM32 HELP | PROBE | ERASE | WRITE | FLASH_PART");
236 return CMD_ERROR_INVALID_SYNTAX;
237 }
238
239 if (!strcasecmp(verb, "HELP") || !strcasecmp(verb, "?")) {
240 uart_send_line("STM32 I2C ROM bootloader (AN4221, addr z menuconfig)");
241 uart_send_line(" CLI STM32 PROBE <seg> [BOOT]");
242 uart_send_line(" CLI STM32 ERASE <seg> [BOOT] (mass erase 0xFFFF)");
243 uart_send_line(
244 " CLI STM32 WRITE <seg> <addr_hex> <data_hex> (max 256 B)");
245 uart_send_line(
246 " CLI STM32 FLASH_PART <seg> <partition_label> [BOOT]");
247 uart_send_line(
248 "NRST/BOOT0 GPIO + sdílený I2C s Hall viz menuconfig STM32 BL.");
249 return CMD_SUCCESS;
250 }
251
252 esp_err_t ie = stm32_i2c_bl_init();
253 if (ie != ESP_OK) {
254 uart_send_formatted("stm32_i2c_bl_init: %s", esp_err_to_name(ie));
255 return CMD_ERROR_SYSTEM_ERROR;
256 }
257
258 if (!strcasecmp(verb, "PROBE")) {
259 unsigned seg = 0;
260 char bflag[8] = "";
261 int n = sscanf(p, "%*s %u %7s", &seg, bflag);
262 if (n < 1 || seg > 3) {
263 uart_send_error("CLI STM32 PROBE <seg> [BOOT]");
264 return CMD_ERROR_INVALID_SYNTAX;
265 }
266 bool boot = (n >= 2 && !strcasecmp(bflag, "BOOT"));
267 uint16_t pid = 0;
268 esp_err_t e = stm32_i2c_bl_cmd_get_id((uint8_t)seg, boot, &pid);
269 if (e != ESP_OK) {
270 uart_send_formatted("GET_ID: %s", esp_err_to_name(e));
271 return CMD_ERROR_SYSTEM_ERROR;
272 }
273 uart_send_formatted("segment %u PID=0x%04X", seg, (unsigned)pid);
274 return CMD_SUCCESS;
275 }
276
277 if (!strcasecmp(verb, "ERASE")) {
278 unsigned seg = 0;
279 char bflag[8] = "";
280 int n = sscanf(p, "%*s %u %7s", &seg, bflag);
281 if (n < 1 || seg > 3) {
282 uart_send_error("CLI STM32 ERASE <seg> [BOOT]");
283 return CMD_ERROR_INVALID_SYNTAX;
284 }
285 bool boot = (n >= 2 && !strcasecmp(bflag, "BOOT"));
286 esp_err_t e = stm32_i2c_bl_erase_all((uint8_t)seg, boot);
287 if (e != ESP_OK) {
288 uart_send_formatted("ERASE: %s", esp_err_to_name(e));
289 return CMD_ERROR_SYSTEM_ERROR;
290 }
291 uart_send_success("Mass erase dokončen (nebo částečný OK)");
292 return CMD_SUCCESS;
293 }
294
295 if (!strcasecmp(verb, "WRITE")) {
296 unsigned seg = 0;
297 uint32_t addr = 0;
298 char hexbuf[520];
299 if (sscanf(p, "%*s %u %" SCNx32 " %519s", &seg, &addr, hexbuf) != 3 ||
300 seg > 3) {
301 uart_send_error("CLI STM32 WRITE <seg> <addr_hex> <data_hex>");
302 return CMD_ERROR_INVALID_SYNTAX;
303 }
304 uint8_t raw[256];
305 size_t ln = cli_hex_to_bin(hexbuf, raw, sizeof(raw));
306 if (ln == 0) {
307 uart_send_error("špatná hex data (sudý počet znaků, max 512 hex znaků)");
308 return CMD_ERROR_INVALID_SYNTAX;
309 }
310 esp_err_t e =
311 stm32_i2c_bl_write_memory((uint8_t)seg, false, addr, raw, ln);
312 if (e != ESP_OK) {
313 uart_send_formatted("WRITE: %s", esp_err_to_name(e));
314 return CMD_ERROR_SYSTEM_ERROR;
315 }
316 uart_send_success("WRITE OK");
317 return CMD_SUCCESS;
318 }
319
320 if (!strcasecmp(verb, "FLASH_PART")) {
321 unsigned seg = 0;
322 char label[32];
323 char bflag[8] = "";
324 int n = sscanf(p, "%*s %u %31s %7s", &seg, label, bflag);
325 bool boot = (n == 3 && !strcasecmp(bflag, "BOOT"));
326 if (n < 2 || seg > 3) {
327 uart_send_error("CLI STM32 FLASH_PART <seg> <label> [BOOT]");
328 return CMD_ERROR_INVALID_SYNTAX;
329 }
330 const esp_partition_t *part =
331 esp_partition_find_first(ESP_PARTITION_TYPE_ANY,
332 ESP_PARTITION_SUBTYPE_ANY, label);
333 if (!part) {
334 uart_send_formatted("oddíl '%s' nenalezen", label);
335 return CMD_ERROR_INVALID_PARAMETER;
336 }
337 size_t psz = part->size;
338 if (psz > 512 * 1024) {
339 uart_send_error("oddíl příliš velký (>512 KiB) — bezpečnostní limit");
340 return CMD_ERROR_INVALID_PARAMETER;
341 }
342 uint8_t *buf = (uint8_t *)malloc(psz);
343 if (!buf) {
344 uart_send_error("malloc bin buffer");
345 return CMD_ERROR_SYSTEM_ERROR;
346 }
347 esp_err_t le =
348 esp_partition_read(part, 0, buf, part->size);
349 if (le != ESP_OK) {
350 free(buf);
351 uart_send_formatted("read partition: %s", esp_err_to_name(le));
352 return CMD_ERROR_SYSTEM_ERROR;
353 }
354 esp_err_t fe = stm32_i2c_bl_flash_binary((uint8_t)seg, boot,
355 0x08000000, buf, psz);
356 free(buf);
357 if (fe != ESP_OK) {
358 uart_send_formatted("FLASH_PART: %s", esp_err_to_name(fe));
359 return CMD_ERROR_SYSTEM_ERROR;
360 }
361 uart_send_success("FLASH_PART hotovo");
362 return CMD_SUCCESS;
363 }
364
365 uart_send_error("neznámý STM32 příkaz — CLI STM32 HELP");
366 return CMD_ERROR_INVALID_SYNTAX;
367}

411command_result_t uart_cmd_cli(const char *args) {
412 char line[512];
413 if (args) {
414 strncpy(line, args, sizeof(line) - 1);
415 line[sizeof(line) - 1] = '\0';
416 } else {
417 line[0] = '\0';
418 }
419
420 char cmd[32];
421 const char *s = skip_leading_ws(line);
422 size_t ci = 0;
423 while (*s && *s != ' ' && *s != '\t' && ci + 1 < sizeof(cmd)) {
424 cmd[ci++] = (char)toupper((unsigned char)*s++);
425 }
426 cmd[ci] = '\0';
427
428 if (cmd[0] == '\0' || !strcasecmp(cmd, "HELP") || !strcasecmp(cmd, "?")) {
429 uart_cli_print_help();
430 return CMD_SUCCESS;
431 }
432
433 char tail[480];
434 skip_first_word(line, tail, sizeof(tail));
435
436 if (!strcasecmp(cmd, "PARTITIONS") || !strcasecmp(cmd, "PART")) {
437 cli_dump_partitions();
438 return CMD_SUCCESS;
439 }
440 if (!strcasecmp(cmd, "FLASH")) {
441 cli_flash_info();
442 return CMD_SUCCESS;
443 }
444 if (!strcasecmp(cmd, "CHIP")) {
445 cli_chip_info();
446 return CMD_SUCCESS;
447 }
448 if (!strcasecmp(cmd, "APP")) {
449 cli_app_info();
450 return CMD_SUCCESS;
451 }
452 if (!strcasecmp(cmd, "HEAP")) {
453 uart_send_formatted("heap_free=%zu heap_min=%zu internal_free=%zu",
454 (size_t)esp_get_free_heap_size(),
455 (size_t)esp_get_minimum_free_heap_size(),
456 (size_t)heap_caps_get_free_size(MALLOC_CAP_INTERNAL));
457 return CMD_SUCCESS;
458 }
459 if (!strcasecmp(cmd, "WEB")) {
460 if (tail[0] == '\0') {
461 uart_send_error("CLI WEB START|STOP|STATUS");
462 return CMD_ERROR_INVALID_SYNTAX;
463 }
464 cli_web_command(tail);
465 return CMD_SUCCESS;
466 }
467 if (!strcasecmp(cmd, "LOCK")) {
468 char sub[16];
469 sscanf(tail, "%15s", sub);
470 if (sub[0] == '\0') {
471 uart_send_error("CLI LOCK ON|OFF");
472 return CMD_ERROR_INVALID_SYNTAX;
473 }
474 cli_lock_command(sub);
475 return CMD_SUCCESS;
476 }
477 if (!strcasecmp(cmd, "OTA")) {
478 const char *p = skip_leading_ws(tail);
479 if (p[0] == '\0') {
480 cli_ota_info();
481 return CMD_SUCCESS;
482 }
483 if (!strncasecmp(p, "INFO", 4) &&
484 (p[4] == '\0' || p[4] == ' ' || p[4] == '\t')) {
485 cli_ota_info();
486 return CMD_SUCCESS;
487 }
488 esp_err_t oe = ota_update_try_start_url(p);
489 if (oe == ESP_OK) {
490 uart_send_success("OTA stažení spuštěno na pozadí");
491 } else {
492 uart_send_formatted("OTA start: %s", esp_err_to_name(oe));
493 ESP_LOGW(TAG, "OTA URL rejected: %s", esp_err_to_name(oe));
494 }
495 return CMD_SUCCESS;
496 }
497 if (!strcasecmp(cmd, "BLE")) {
498 cli_ble_json(tail);
499 return CMD_SUCCESS;
500 }
501 if (!strcasecmp(cmd, "SNAP") || !strcasecmp(cmd, "SNAPSHOT")) {
502 cli_snapshot();
503 return CMD_SUCCESS;
504 }
505 if (!strcasecmp(cmd, "STM32")) {
506#if CONFIG_CHESS_STM32_I2C_BL_ENABLE
507 return cli_stm32_bl_tail(tail);
508#else
509 uart_send_error(
510 "CLI STM32 — zapni CHESS_STM32_I2C_BL_ENABLE v menuconfig");
511 return CMD_ERROR_INVALID_SYNTAX;
512#endif
513 }
514 if (!strcasecmp(cmd, "RESET")) {
515 uart_send_line("Restart…");
516 esp_restart();
517 return CMD_SUCCESS;
518 }
519
520 uart_send_error("Neznámý CLI podpříkaz — zadej CLI HELP");
521 return CMD_ERROR_INVALID_SYNTAX;
522}
