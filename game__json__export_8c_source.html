41 // Thread-safe access to board
42 if (game_mutex != NULL) {
43 if (xSemaphoreTake(game_mutex, pdMS_TO_TICKS(1000)) != pdTRUE) {
44 return ESP_ERR_TIMEOUT;
45 }
46 }

48 // Build JSON string manually (more efficient than cJSON)
49 int offset = 0;
50
51 offset += snprintf(buffer + offset, size - offset, "{\"board\":[");
52
53 for (int row = 0; row < 8; row++) {
54 offset += snprintf(buffer + offset, size - offset, "[");
55 for (int col = 0; col < 8; col++) {
56 char piece_char = piece_to_char(board[row][col]);
57 offset += snprintf(buffer + offset, size - offset, "\"%c\"", piece_char);

91static esp_err_t game_status_json_append(size_t *offset, char *buffer, size_t size,
92 const char *fmt, ...) {
93 va_list ap;
94 va_start(ap, fmt);
95 if (*offset >= size) {
96 va_end(ap);
97 ESP_LOGE(TAG, "game_get_status_json: offset invalid %zu", *offset);
98 return ESP_ERR_NO_MEM;
99 }
100 size_t rem = size - *offset;
101 if (rem <= 1) {
102 va_end(ap);
103 return ESP_ERR_NO_MEM;
104 }
105 int n = vsnprintf(buffer + *offset, rem, fmt, ap);
106 va_end(ap);
107 if (n < 0 || (size_t)n >= rem) {
108 ESP_LOGE(TAG,
109 "game_get_status_json: append overflow need=%d rem=%zu off=%zu", n,
110 rem, *offset);
111 return ESP_ERR_NO_MEM;
112 }
113 *offset += (size_t)n;
114 return ESP_OK;
115}
