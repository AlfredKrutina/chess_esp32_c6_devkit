118command_result_t uart_cmd_promote(const char *args) {
119 SAFE_WDT_RESET();
120
121 if (!args || strlen(args) == 0) {
122 uart_send_error("❌ Usage: PROMOTE <square>=<piece>");
123 uart_send_formatted("💡 Example: PROMOTE e8=Q (promote pawn to Queen)");
124 return CMD_ERROR_INVALID_SYNTAX;
125 }
126
127 uart_send_colored_line(COLOR_INFO, "👑 Pawn Promotion");
128 uart_send_formatted(
129 "═══════════════════════════════════════════════════════════════");
130
131 // Parse promotion notation (e.g., "e8=Q")
132 char square[4] = {0};
133 char piece[2] = {0};
134
135 // Try parsing (support both "e8=Q" and just "Q" if square is implicit/ignored
136 // by game logic)
137 if (sscanf(args, "%3[^=]=%1s", square, piece) != 2) {
138 // Fallback: maybe just piece is provided "Q"
139 if (strlen(args) == 1) {
140 strcpy(piece, args);
141 strcpy(square, "??"); // Square handled by game state
142 } else {
143 uart_send_error(
144 "❌ Invalid promotion format. Use: <square>=<piece> or just <piece>");
145 uart_send_formatted("💡 Example: PROMOTE e8=Q or PROMOTE Q");
146 return CMD_ERROR_INVALID_SYNTAX;
147 }
148 }
149
150 promotion_choice_t choice;
151 if (strcmp(piece, "Q") == 0 || strcmp(piece, "q") == 0)
152 choice = PROMOTION_QUEEN;
153 else if (strcmp(piece, "R") == 0 || strcmp(piece, "r") == 0)
154 choice = PROMOTION_ROOK;
155 else if (strcmp(piece, "B") == 0 || strcmp(piece, "b") == 0)
156 choice = PROMOTION_BISHOP;
157 else if (strcmp(piece, "N") == 0 || strcmp(piece, "n") == 0)
158 choice = PROMOTION_KNIGHT;
159 else {
160 uart_send_error("❌ Invalid piece for promotion");
161 uart_send_formatted(
162 "💡 Valid pieces: Q (Queen), R (Rook), B (Bishop), N (Knight)");
163 return CMD_ERROR_INVALID_SYNTAX;
164 }
165
166 uart_send_formatted("🎯 Promotion Command:");
167 uart_send_formatted(" • Promote to: %s (%s)", piece,
168 choice == PROMOTION_QUEEN ? "Queen"
169 : choice == PROMOTION_ROOK ? "Rook"
170 : choice == PROMOTION_BISHOP ? "Bishop"
171 : "Knight");
172
173 // Construct command
174 chess_move_command_t cmd = {0};
175 cmd.type = GAME_CMD_PROMOTION;
176 cmd.promotion_choice = choice;
177 cmd.response_queue = NULL;
178
179 // Send to queue
180 if (game_command_queue == NULL) {
181 uart_send_error("❌ internal error: game_command_queue not initialized");
182 return CMD_ERROR_SYSTEM_ERROR;
183 }
184
185 if (xQueueSend(game_command_queue, &cmd, pdMS_TO_TICKS(100)) == pdTRUE) {
186 uart_send_formatted("✅ Promotion command sent to game task");
187 return CMD_SUCCESS;
188 } else {
189 uart_send_error("❌ Failed to send command to queue (full?)");
190 return CMD_ERROR_SYSTEM_ERROR;
191 }
192}

268command_result_t uart_cmd_move(const char *args) {
269 if (!args || strlen(args) < 4) {
270 uart_send_error("❌ Missing arguments");
271 return CMD_ERROR_INVALID_SYNTAX;
272 }
273
274 char from_square[3] = {0};
275 char to_square[3] = {0};
276
277 // Parse move notation
278 if (parse_move_notation(args, from_square, to_square)) {
279 // Validate chess squares
280 if (validate_chess_squares(from_square, to_square)) {
281 uart_send_colored_line(COLOR_INFO, "🔄 Starting move sequence");
282
283 // Step 1: Call UP command directly
284 uart_send_colored_line(COLOR_INFO, "🔄 Lifting piece...");
285 command_result_t up_result = uart_cmd_up(from_square);
286 if (up_result != CMD_SUCCESS) {
287 uart_send_error("❌ Failed to lift piece");
288 return up_result;
289 }
290
291 // Step 2: Wait 500ms for animations
292 uart_send_colored_line(COLOR_INFO, "⏳ Waiting for animations...");
293 vTaskDelay(pdMS_TO_TICKS(500));
294
295 // Step 3: Send DROP command WITH validation via EXISTING response queue
296 uart_send_colored_line(COLOR_INFO, "🔄 Placing piece...");
297
298 // Použití existující globální uart_response_queue
299 chess_move_command_t drop_cmd = {.type = GAME_CMD_DROP,
300 .player = 0,
301 .response_queue =
302 (QueueHandle_t)uart_response_queue};
303 strcpy(drop_cmd.from_notation, "");
304 strncpy(drop_cmd.to_notation, to_square,
305 sizeof(drop_cmd.to_notation) - 1);
306 drop_cmd.to_notation[sizeof(drop_cmd.to_notation) - 1] = '\0';
307
308 // Send to game task
309 if (!send_to_game_task(&drop_cmd)) {
310 uart_send_error("❌ Failed to send DROP command");
311 return CMD_ERROR_SYSTEM_ERROR;
312 }
313
314 // Wait for validation response from game_task
315 game_response_t response;
316 if (xQueueReceive(uart_response_queue, &response, pdMS_TO_TICKS(5000)) ==
317 pdTRUE) {
318 if (response.error_code != 0) {
319 // Invalid move!
320 uart_send_error(response.message);
321 uart_send_error("❌ Invalid move - piece must be returned");
322 return CMD_ERROR_INVALID_PARAMETER;
323 }
324
325 // Valid move!
326 uart_send_colored_line(
327 COLOR_INFO,
328 "💡 LEDs: Blue flash (piece placed), then Yellow (movable pieces)");
329 uart_send_success("✅ Move completed");
330 return CMD_SUCCESS;
331 } else {
332 uart_send_error("❌ Timeout waiting for move validation");
333 return CMD_ERROR_SYSTEM_ERROR;
334 }
335 } else {
336 uart_send_error("Invalid chess squares");
337 return CMD_ERROR_INVALID_PARAMETER;
338 }
339 } else {
340 uart_send_error("Invalid move format");
341 return CMD_ERROR_INVALID_SYNTAX;
342 }
343}

355command_result_t uart_cmd_up(const char *args) {
356 if (!args || strlen(args) < 2) {
357 uart_send_error("❌ Missing arguments");
358 uart_send_info("Usage: UP <square>");
359 uart_send_info("Examples: UP a2, UP e4");
360 return CMD_ERROR_INVALID_SYNTAX;
361 }
362
363 // Parse square notation (support both "a2" and "a 2" formats)
364 char square[4];
365 if (strlen(args) == 2) {
366 // Format: "a2"
367 strncpy(square, args, 2);
368 square[2] = '\0';
369 } else if (strlen(args) == 3 && args[1] == ' ') {
370 // Format: "a 2"
371 square[0] = args[0];
372 square[1] = args[2];
373 square[2] = '\0';
374 } else {
375 uart_send_error("❌ Invalid square format");
376 uart_send_info("Use format: a2 or a 2");
377 return CMD_ERROR_INVALID_SYNTAX;
378 }
379
380 // Validate square notation
381 if (!is_valid_square_notation(square)) {
382 uart_send_error("❌ Invalid square notation");
383 uart_send_info("Use format: a2, b3, c4, etc.");
384 return CMD_ERROR_INVALID_SYNTAX;
385 }
386
387 // Send pickup command to game task WITH response queue
388 chess_move_command_t cmd = {.type = GAME_CMD_PICKUP,
389 .player = 0, // Will be determined by game task
390 .response_queue =
391 (QueueHandle_t)uart_response_queue};
392 strncpy(cmd.from_notation, square, sizeof(cmd.from_notation) - 1);
393 cmd.from_notation[sizeof(cmd.from_notation) - 1] = '\0';
394 strcpy(cmd.to_notation, "");
395
396 if (!send_to_game_task(&cmd)) {
397 uart_send_error("Internal error: failed to lift piece");
398 return CMD_ERROR_SYSTEM_ERROR;
399 }
400
401 // Čekání na odpověď z game tasku (může obsahovat chybu)
402 game_response_t response;
403 if (xQueueReceive(uart_response_queue, &response, pdMS_TO_TICKS(5000)) ==
404 pdTRUE) {
405 if (response.error_code != 0) {
406 // Chyba při zvednutí!
407 uart_send_error(response.message);
408 return CMD_ERROR_INVALID_PARAMETER;
409 }
410
411 // Úspěch!
412 char msg[64];
413 snprintf(msg, sizeof(msg), "🔄 Piece lifted from %s", square);
414 uart_send_colored_line(COLOR_INFO, msg);
415 uart_send_colored_line(COLOR_INFO,
416 "💡 LEDs: Yellow square (lifted piece), Green "
417 "(possible moves), Orange (captures)");
418 return CMD_SUCCESS;
419 } else {
420 uart_send_error("❌ Timeout waiting for game response");
421 return CMD_ERROR_SYSTEM_ERROR;
422 }

495command_result_t uart_cmd_board(const char *args) {
496 (void)args; // Unused parameter
497
498 // Reset watchdog timeru před operacemi
499 SAFE_WDT_RESET();
500
501 uart_send_colored_line(COLOR_INFO, "🏁 Chess Board");
502 uart_send_formatted(
503 "═══════════════════════════════════════════════════════════════");
504
505 // MEMORY OPTIMIZATION: Use local board display instead of game task
506 // communication This eliminates the need for large response buffers and
507 // reduces stack usage
508 ESP_LOGI(TAG, "📡 Using local board display (no queue communication)");
509
510 // Display board header
511 uart_send_formatted(" a b c d e f g h");
512 uart_send_formatted(" +---+---+---+---+---+---+---+---+");
513
514 // Display board row by row to minimize stack usage
515 for (int row = 7; row >= 0; row--) {
516 // Reset watchdog timeru před každým řádkem
517 SAFE_WDT_RESET();
518
519 // Build row string in small buffer
520 char row_buffer[64];
521 int pos = snprintf(row_buffer, sizeof(row_buffer), "%d |", row + 1);
522
523 // Add pieces to row
524 for (int col = 0; col < 8; col++) {
525 piece_t piece = game_get_piece(row, col);
526 const char *symbol;
527 switch (piece) {
528 case PIECE_WHITE_PAWN:
529 symbol = "P";
530 break;
531 case PIECE_WHITE_KNIGHT:
532 symbol = "N";
533 break;
534 case PIECE_WHITE_BISHOP:
535 symbol = "B";
536 break;
537 case PIECE_WHITE_ROOK:
538 symbol = "R";
539 break;
540 case PIECE_WHITE_QUEEN:
541 symbol = "Q";
542 break;
543 case PIECE_WHITE_KING:
544 symbol = "K";
545 break;
546 case PIECE_BLACK_PAWN:
547 symbol = "p";
548 break;
549 case PIECE_BLACK_KNIGHT:
550 symbol = "n";
551 break;
552 case PIECE_BLACK_BISHOP:
553 symbol = "b";
554 break;
555 case PIECE_BLACK_ROOK:
556 symbol = "r";
557 break;
558 case PIECE_BLACK_QUEEN:
559 symbol = "q";
560 break;
561 case PIECE_BLACK_KING:
562 symbol = "k";
563 break;
564 default:
565 symbol = "·";
566 break;
567 }
568 pos +=
569 snprintf(row_buffer + pos, sizeof(row_buffer) - pos, " %s |", symbol);
570 }
571
572 // Add row number at the end
573 snprintf(row_buffer + pos, sizeof(row_buffer) - pos, " %d", row + 1);
574
575 // Send row
576 uart_send_formatted("%s", row_buffer);
577
578 // Send separator line (except after last row)
579 if (row > 0) {
580 uart_send_formatted(" +---+---+---+---+---+---+---+---+");
581 }
582 }
583
584 // Display board footer
585 uart_send_formatted(" +---+---+---+---+---+---+---+---+");
586 uart_send_formatted(" a b c d e f g h");
587
588 // Display game status
589 uart_send_formatted("");
590 uart_send_formatted("Current player: %s",
591 game_get_current_player() == PLAYER_WHITE ? "White"
592 : "Black");
593 uart_send_formatted("Move count: %" PRIu32, game_get_move_count());
594
595 // Reset watchdog timeru po dokončení
596 SAFE_WDT_RESET();
597
598 uart_send_formatted("");
599 uart_send_colored_line(
600 COLOR_INFO, "💡 Use 'UP <square>' to lift piece, 'DN <square>' to place");
601
602 ESP_LOGI(TAG, "✅ Board display completed successfully (local)");
603 return CMD_SUCCESS;
604}

641void uart_display_enhanced_board(void) {

643 SAFE_WDT_RESET();
644
645 bool mutex_taken = false;
646 if (uart_mutex != NULL) {
647 // Použití kratšího timeoutu pro prevenci WDT problémů
648 mutex_taken = (xSemaphoreTake(uart_mutex, pdMS_TO_TICKS(50)) == pdTRUE);
649 if (!mutex_taken) {
650 ESP_LOGW(TAG, "Mutex timeout in board display, continuing without mutex");
651 }

653
654 // CRITICAL: Reset WDT before any output operations
655 SAFE_WDT_RESET();
656
657 // Standardized 8x8 chess board with perfect alignment
658 if (color_enabled)
