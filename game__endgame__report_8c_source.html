27void game_update_endgame_statistics(game_result_type_t result_type) {
28 switch (result_type) {
29 case RESULT_WHITE_WINS:
30 white_wins++;
31 ESP_LOGI(TAG, "📊 Statistics: White wins! (Total: %" PRIu32 ")",
32 white_wins);
33 break;
34 case RESULT_BLACK_WINS:
35 black_wins++;
36 ESP_LOGI(TAG, "📊 Statistics: Black wins! (Total: %" PRIu32 ")",
37 black_wins);
38 break;
39 case RESULT_DRAW_STALEMATE:
40 case RESULT_DRAW_50_MOVE:
41 case RESULT_DRAW_REPETITION:
42 case RESULT_DRAW_INSUFFICIENT:
43 draws++;
44 ESP_LOGI(TAG, "📊 Statistics: Draw! (Total: %" PRIu32 ")", draws);
45 break;
46 }
47 total_games++;
48 ESP_LOGI(TAG, "📊 Total games: %" PRIu32, total_games);
49}

112void game_print_endgame_report_uart(game_result_type_t result_type) {
113 uint32_t current_time = esp_timer_get_time() / 1000;
114 uint32_t game_duration =
115 (game_start_time > 0) ? (current_time - game_start_time) : 0;
116 uint32_t minutes = game_duration / 60;
117 uint32_t seconds = game_duration % 60;
118
119 // Print endgame announcement
120 print_report_header("🏆 ENDGAME REPORT");
121
122 // Určit výsledek a typ ukončení podle current_endgame_reason
123 const char *winner = "";
124 const char *loser = "";
125 const char *end_reason = "";
126
127 // Nejprve určíme vítěze z result_type
128 if (result_type == RESULT_WHITE_WINS) {
129 winner = "White";
130 loser = "Black";
131 } else if (result_type == RESULT_BLACK_WINS) {
132 winner = "Black";
133 loser = "White";
134 } else {
135 winner = "Draw";
136 loser = "Draw";
137 }
138
139 // Pak určíme důvod konce hry a vytiskneme výsledek
140 switch (current_endgame_reason) {
141 // === CHECKMATE VARIATIONS ===
142 case ENDGAME_REASON_CHECKMATE:
143 printf("🎯 Game Result: %s WINS BY CHECKMATE\r\n",
144 (result_type == RESULT_WHITE_WINS) ? "WHITE" : "BLACK");
145 end_reason = "Checkmate";
146 break;
147 case ENDGAME_REASON_CHECKMATE_EN_PASSANT:
148 printf("⚔️ Game Result: %s WINS BY EN PASSANT CHECKMATE!\r\n",
149 (result_type == RESULT_WHITE_WINS) ? "WHITE" : "BLACK");
150 end_reason = "Checkmate (En Passant)";
151 break;
152 case ENDGAME_REASON_CHECKMATE_CASTLING:
153 printf("🏰 Game Result: %s WINS BY CASTLING CHECKMATE!\r\n",
154 (result_type == RESULT_WHITE_WINS) ? "WHITE" : "BLACK");
155 end_reason = "Checkmate (Castling)";
156 break;
157 case ENDGAME_REASON_CHECKMATE_PROMOTION:
158 printf("👑 Game Result: %s WINS BY PROMOTION CHECKMATE!\r\n",
159 (result_type == RESULT_WHITE_WINS) ? "WHITE" : "BLACK");
160 end_reason = "Checkmate (Promotion)";
161 break;
162 case ENDGAME_REASON_CHECKMATE_DISCOVERED:
163 printf("💥 Game Result: %s WINS BY DISCOVERED CHECK CHECKMATE!\r\n",
164 (result_type == RESULT_WHITE_WINS) ? "WHITE" : "BLACK");
165 end_reason = "Checkmate (Discovered Check)";
166 break;
167
168 // === PLAYER ACTIONS ===
169 case ENDGAME_REASON_RESIGNATION:
170 printf("🏳️ Game Result: %s WINS BY RESIGNATION\r\n",
171 (result_type == RESULT_WHITE_WINS) ? "WHITE" : "BLACK");
172 end_reason = "Resignation";
173 break;
174 case ENDGAME_REASON_TIMEOUT:
175 printf("⏰ Game Result: %s WINS BY TIMEOUT\r\n",
176 (result_type == RESULT_WHITE_WINS) ? "WHITE" : "BLACK");
177 end_reason = "Timeout";
178 break;
179
180 // === DRAW CONDITIONS ===
181 case ENDGAME_REASON_STALEMATE:
182 printf("🤝 Game Result: DRAW - Stalemate\r\n");
183 end_reason = "Stalemate";
184 break;
185 case ENDGAME_REASON_50_MOVE:
186 printf("🤝 Game Result: DRAW - 50-move rule\r\n");
187 end_reason = "50-move rule";
188 break;
189 case ENDGAME_REASON_REPETITION:
190 printf("🤝 Game Result: DRAW - Three-fold repetition\r\n");
191 end_reason = "Threefold repetition";
192 break;
193 case ENDGAME_REASON_INSUFFICIENT:
194 printf("🤝 Game Result: DRAW - Insufficient material\r\n");
195 end_reason = "Insufficient material";
196 break;
197 }
198
199 print_separator();
200 printf("📋 Game Summary:\r\n");
201 print_report_line("Winner", "%s", winner);
202 if (result_type != RESULT_DRAW_STALEMATE &&
203 result_type != RESULT_DRAW_50_MOVE &&
204 result_type != RESULT_DRAW_REPETITION &&
205 result_type != RESULT_DRAW_INSUFFICIENT) {
206 print_report_line("Loser", "%s", loser);
207 }
208 print_report_line("End Reason", "%s", end_reason);
209 print_report_line("Total Moves", "%" PRIu32, move_count);
210 print_report_line("Game Duration", "%" PRIu32 ":%02" PRIu32 " (mm:ss)",
211 minutes, seconds);
212
213 // Vypočítat material advantage
214 int white_material = 0, black_material = 0;
215 for (uint32_t i = 0; i < white_captured_count; i++) {
216 piece_t p = white_captured_pieces[i];
217 if (p == PIECE_BLACK_PAWN)
218 white_material += 1;
219 else if (p == PIECE_BLACK_KNIGHT || p == PIECE_BLACK_BISHOP)
220 white_material += 3;
221 else if (p == PIECE_BLACK_ROOK)
222 white_material += 5;
223 else if (p == PIECE_BLACK_QUEEN)
224 white_material += 9;
225 }
226 for (uint32_t i = 0; i < black_captured_count; i++) {
227 piece_t p = black_captured_pieces[i];
228 if (p == PIECE_WHITE_PAWN)
229 black_material += 1;
230 else if (p == PIECE_WHITE_KNIGHT || p == PIECE_WHITE_BISHOP)
231 black_material += 3;
232 else if (p == PIECE_WHITE_ROOK)
233 black_material += 5;
234 else if (p == PIECE_WHITE_QUEEN)
235 black_material += 9;
236 }
237 int material_diff = white_material - black_material;
238
239 print_separator();
240 printf("📊 Game Statistics:\r\n");
241 print_report_line("White Moves", "%" PRIu32, (move_count + 1) / 2);
242 print_report_line("Black Moves", "%" PRIu32, move_count / 2);
243 if (game_duration > 0 && move_count > 0) {
244 print_report_line("Avg Time per Move", "%" PRIu32 "s",
245 game_duration / move_count);
246 }
247
248 print_separator();
249 printf("⚔️ Combat Statistics:\r\n");
250 print_report_line("White Captures", "%" PRIu32 " pieces (%d points)",
251 white_captures, white_material);
252 print_report_line("Black Captures", "%" PRIu32 " pieces (%d points)",
253 black_captures, black_material);
254 if (material_diff > 0) {
255 print_report_line("Material Advantage", "White +%d", material_diff);
256 } else if (material_diff < 0) {
257 print_report_line("Material Advantage", "Black +%d", -material_diff);
258 } else {
259 print_report_line("Material Advantage", "Equal (0)");
260 }
261 print_report_line("White Checks", "%" PRIu32, white_checks);
262 print_report_line("Black Checks", "%" PRIu32, black_checks);
263
264 // Captured pieces vizuálně
265 if (white_captured_count > 0) {
266 printf(" • White Captured: ");
267 for (uint32_t i = 0; i < white_captured_count && i < GAME_TASK_MAX_CAPTURED_PIECES;
268 i++) {
269 char piece_char = piece_to_char(white_captured_pieces[i]);
270 printf("%c ", piece_char);
271 }
272 printf("\r\n");
273 }
274 if (black_captured_count > 0) {
275 printf(" • Black Captured: ");
276 for (uint32_t i = 0; i < black_captured_count && i < GAME_TASK_MAX_CAPTURED_PIECES;
277 i++) {
278 char piece_char = piece_to_char(black_captured_pieces[i]);
279 printf("%c ", piece_char);
280 }
281 printf("\r\n");
282 }
283
284 // Castling info
285 print_report_line("White Castling", "%s", white_castles > 0 ? "Yes" : "No");
286 print_report_line("Black Castling", "%s", black_castles > 0 ? "Yes" : "No");
287
288 print_separator();
289 printf("📈 Overall Statistics:\r\n");
290 print_report_line("Total Games Played", "%" PRIu32, total_games);
291 print_report_line("White Wins", "%" PRIu32 " (%.0f%%)", white_wins,
292 total_games > 0 ? (white_wins * 100.0f / total_games)
293 : 0.0f);
294 print_report_line("Black Wins", "%" PRIu32 " (%.0f%%)", black_wins,
295 total_games > 0 ? (black_wins * 100.0f / total_games)
296 : 0.0f);
297 print_report_line("Draws", "%" PRIu32 " (%.0f%%)", draws,
298 total_games > 0 ? (draws * 100.0f / total_games) : 0.0f);
299
300 printf("═══════════════════════════════════════════════════════════════\r\n");
301 printf("\r\n");
302
303 // Flush stdout pro okamžité zobrazení
304 fflush(stdout);
305}
