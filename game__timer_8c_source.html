129esp_err_t game_process_timer_command(const chess_move_command_t *cmd) {
130 if (cmd == NULL) {
131 ESP_LOGE(TAG, "Invalid command parameter");
132 return ESP_ERR_INVALID_ARG;
133 }
134
135 esp_err_t ret = ESP_OK;
136
137 switch (cmd->type) {
138 case GAME_CMD_SET_TIME_CONTROL: {
139 time_control_config_t config;
140 time_control_type_t type =
141 (time_control_type_t)cmd->timer_data.timer_config.time_control_type;
142
143 if (type == TIME_CONTROL_CUSTOM) {
144 // Vlastní časová kontrola
145 ret = timer_set_custom_time_control(
146 cmd->timer_data.timer_config.custom_minutes,
147 cmd->timer_data.timer_config.custom_increment);
148 } else {
149 // Předdefinovaná časová kontrola
150 ret = timer_get_config_by_type(type, &config);
151 if (ret == ESP_OK) {
152 ret = timer_set_time_control(&config);
153 }
154 }
155
156 if (ret == ESP_OK) {
157 ESP_LOGI(TAG, "Time control command processed successfully");
158 } else {
159 ESP_LOGE(TAG, "Failed to process time control command");
160 }
161 } break;
162
163 case GAME_CMD_PAUSE_TIMER:
164 ret = game_pause_timer();
165 break;
166
167 case GAME_CMD_RESUME_TIMER:
168 ret = game_resume_timer();
169 break;
170
171 case GAME_CMD_RESET_TIMER:
172 ret = game_reset_timer();
173 break;
174
175 case GAME_CMD_GET_TIMER_STATE:
176 // Timer state je získáván přes JSON API
177 ret = ESP_OK;
178 break;
179
180 case GAME_CMD_TIMER_TIMEOUT:
181 ret = game_handle_time_expiration();
182 break;
183
184 default:
185 ESP_LOGW(TAG, "Unknown timer command: %d", cmd->type);
186 ret = ESP_ERR_INVALID_ARG;
187 break;
188 }
189
190 return ret;
191}

193esp_err_t game_handle_time_expiration(void) {
194 ESP_LOGW(TAG, "⏰ Time expired! Handling timeout...");
195
196 // Získat stav timeru
197 chess_timer_t timer_data;
198 esp_err_t ret = timer_get_state(&timer_data);
199 if (ret != ESP_OK) {
200 ESP_LOGE(TAG, "Failed to get timer state");
201 return ret;
202 }
203
204 // Určit vítěze
205 player_t winner = timer_data.is_white_turn ? PLAYER_BLACK : PLAYER_WHITE;
206 const char *winner_name = (winner == PLAYER_WHITE) ? "White" : "Black";
207 const char *loser_name = (winner == PLAYER_WHITE) ? "Black" : "White";
208
209 ESP_LOGW(TAG, "🏆 %s wins by timeout! %s ran out of time.", winner_name,
210 loser_name);
211
212 // Ukončit hru
213 current_game_state = GAME_STATE_FINISHED;
214 game_result = GAME_STATE_FINISHED;
215 current_result_type =
216 (winner == PLAYER_WHITE) ? RESULT_WHITE_WINS : RESULT_BLACK_WINS;
217 current_endgame_reason = ENDGAME_REASON_TIMEOUT; // Označit jako timeout
218
219 // Požadavek na endgame report (stejný fix jako u resignation)
220 game_update_endgame_statistics(current_result_type);
221 endgame_report_requested = true;
222
223 // UNIFIED VICTORY ANIMATION
224 // Timeout = Winner determined above
225 game_trigger_victory_animation(winner);
226
227 // Zobrazit výsledek na LED (použít existující LED funkce)
228 // led_show_game_result(game_result); // Funkce neexistuje
229
230 // Pozastavit timer
231 timer_pause();
232
233 ESP_LOGI(TAG, "Game ended due to timeout. Winner: %s", winner_name);
234
235 return ESP_OK;
236}
