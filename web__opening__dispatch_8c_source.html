166esp_err_t web_server_opening_dispatch_json(cJSON *root) {
167 if (root == NULL) {
168 return ESP_ERR_INVALID_ARG;
169 }
170 if (web_is_locked()) {
171 return ESP_ERR_INVALID_STATE;
172 }
173
174 cJSON *action_j = cJSON_GetObjectItemCaseSensitive(root, "action");
175 if (!cJSON_IsString(action_j) || action_j->valuestring == NULL) {
176 return ESP_ERR_NOT_FOUND;
177 }
178
179 uint8_t action_code = opening_action_code(action_j->valuestring);
180 if (action_code == 255) {
181 return ESP_ERR_INVALID_ARG;
182 }
183
184 if (action_code == 1) {
185 esp_err_t load_err = opening_load_start_config(root);
186 if (load_err != ESP_OK) {
187 return load_err;
188 }
189 if (!game_opening_physical_matches_start()) {
190 if (!game_opening_enter_setup_phase()) {
191 return ESP_ERR_INVALID_ARG;
192 }
193 ESP_LOGI(TAG, "opening physical mismatch — setup_phase, start not queued");
194 return ESP_OK;
195 }
196 }
197
198 if (action_code == 2) {
199 if (!game_opening_hint()) {
200 return ESP_ERR_INVALID_STATE;
201 }
202 return ESP_OK;
203 }
204
205 if (action_code == 3) {
206 if (!game_opening_checkpoint_ack()) {
207 return ESP_ERR_NOT_ALLOWED;
208 }
209 return ESP_OK;
210 }
211
212 return opening_queue_action(action_code);
213}
