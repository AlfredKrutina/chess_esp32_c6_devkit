44static bool game_board_starting_occupancy_matches(const uint8_t state[64]) {
45 for (int row = 2; row < 6; row++) {
46 for (int col = 0; col < 8; col++) {
47 if (state[row * 8 + col] != 0) {
48 return false;
49 }
50 }
51 }
52 for (int row = 0; row < 2; row++) {
53 for (int col = 0; col < 8; col++) {
54 if (state[row * 8 + col] == 0) {
55 return false;
56 }
57 }
58 }
59 for (int row = 6; row < 8; row++) {
60 for (int col = 0; col < 8; col++) {
61 if (state[row * 8 + col] == 0) {
62 return false;
63 }
64 }
65 }
66 return true;
67}

69static void game_log_first_startpos_mismatch(const uint8_t state[64]) {

118bool game_is_board_in_starting_position(void) {
119 uint8_t votes[64] = {0};
120
121 for (int s = 0; s < GAME_START_POS_MATRIX_SAMPLES; s++) {
122 uint8_t snap[64];
123 matrix_get_state(snap);
124 for (int i = 0; i < 64; i++) {
125 votes[i] += snap[i];
126 }
127 if (s + 1 < GAME_START_POS_MATRIX_SAMPLES) {
128 vTaskDelay(pdMS_TO_TICKS(12));
129 }
130 }
131
132 uint8_t stable[64];
133 for (int i = 0; i < 64; i++) {
134 stable[i] = (votes[i] >= GAME_START_POS_MAJORITY_VOTES) ? 1 : 0;
135 }
136
137 if (game_board_starting_occupancy_matches(stable)) {
138 return true;
139 }
140 game_log_first_startpos_mismatch(stable);
141 return false;
142}
