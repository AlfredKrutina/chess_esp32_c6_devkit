45game_state_t game_analyze_position(player_t player) {
46 bool king_in_check = game_is_king_in_check(player);
47 uint32_t legal_moves = game_generate_legal_moves(player);
48
49 if (legal_moves == 0) {
50 if (king_in_check) {
51 // Checkmate
52 game_result = GAME_STATE_FINISHED;
53 game_result_type_t result_type =
54 player == PLAYER_WHITE ? RESULT_BLACK_WINS : RESULT_WHITE_WINS;
55 current_result_type = result_type; // Uložit pro web API
56 game_update_endgame_statistics(result_type);
57 game_print_endgame_report_uart(result_type);
58 ESP_LOGI(TAG, "🎯 CHECKMATE! %s wins!",
59 player == PLAYER_WHITE ? "Black" : "White");
60 return GAME_STATE_FINISHED;
61 } else {
62 // Stalemate
63 game_result = GAME_STATE_FINISHED;
64 current_result_type = RESULT_DRAW_STALEMATE; // Uložit pro web API
65 game_update_endgame_statistics(RESULT_DRAW_STALEMATE);
66 game_print_endgame_report_uart(RESULT_DRAW_STALEMATE);
67 ESP_LOGI(TAG, "🤝 STALEMATE! Game drawn!");
68 return GAME_STATE_FINISHED;
69 }
70 }
71
72 // Check for fifty-move rule
73 if (fifty_move_counter >= 100) { // 50 moves per side
74 game_result = GAME_STATE_FINISHED;
75 current_result_type = RESULT_DRAW_50_MOVE; // Uložit pro web API
76 game_update_endgame_statistics(RESULT_DRAW_50_MOVE);
77 game_print_endgame_report_uart(RESULT_DRAW_50_MOVE);
78 ESP_LOGI(TAG, "🤝 DRAW! Fifty-move rule!");
79 return GAME_STATE_FINISHED;
80 }
81
82 // Check for insufficient material (simplified)
83 int white_pieces = 0, black_pieces = 0;
84 bool white_has_major = false, black_has_major = false;
85
86 for (int row = 0; row < 8; row++) {
87 for (int col = 0; col < 8; col++) {
88 piece_t piece = board[row][col];
89 if (piece == PIECE_EMPTY)
90 continue;
91
92 if (game_is_white_piece(piece)) {
93 white_pieces++;
94 if (piece == PIECE_WHITE_QUEEN || piece == PIECE_WHITE_ROOK ||
95 piece == PIECE_WHITE_PAWN) {
96 white_has_major = true;
97 }
98 } else {
99 black_pieces++;
100 if (piece == PIECE_BLACK_QUEEN || piece == PIECE_BLACK_ROOK ||
101 piece == PIECE_BLACK_PAWN) {
102 black_has_major = true;
103 }
104 }
105 }
106 }
107
108 if (white_pieces <= 2 && black_pieces <= 2 && !white_has_major &&
109 !black_has_major) {
110 game_result = GAME_STATE_FINISHED;
111 current_result_type = RESULT_DRAW_INSUFFICIENT; // Uložit pro web API
112 game_update_endgame_statistics(RESULT_DRAW_INSUFFICIENT);
113 game_print_endgame_report_uart(RESULT_DRAW_INSUFFICIENT);
114 ESP_LOGI(TAG, "🤝 DRAW! Insufficient material!");
115 return GAME_STATE_FINISHED;
116 }
117
118 return GAME_STATE_ACTIVE;
119}

121bool game_execute_move(const chess_move_t *move) {
122 if (!move) {
123 return false;
124 }
125
126 const bool api_immediate_promo = s_uart_move_immediate_promotion;
127 const promotion_choice_t api_promo_choice = s_uart_move_immediate_promotion_piece;
128 s_uart_move_immediate_promotion = false;
129
130 move_error_t error = game_is_valid_move(move);
131 if (error != MOVE_ERROR_NONE) {
132 ESP_LOGW(TAG, "Invalid move attempted: %d", error);
133 return false;
134 }
135
136 ESP_LOGI(TAG, "Executing move: %c%d-%c%d", 'a' + move->from_col,
137 move->from_row + 1, 'a' + move->to_col, move->to_row + 1);
138
139 // Get pieces
140 piece_t source_piece = game_get_piece(move->from_row, move->from_col);
141 piece_t dest_piece = game_get_piece(move->to_row, move->to_col);
142
143 // Create extended move for proper handling of special cases
144 chess_move_extended_t extended_move = {.from_row = move->from_row,
145 .from_col = move->from_col,
146 .to_row = move->to_row,
147 .to_col = move->to_col,
148 .piece = source_piece,
149 .captured_piece = dest_piece,
150 .move_type = MOVE_TYPE_NORMAL,
151 .promotion_piece = PROMOTION_QUEEN,
152 .timestamp =
153 esp_timer_get_time() / 1000,
154 .is_check = false,
155 .is_checkmate = false,
156 .is_stalemate = false};
157
158 // Detect special move types
159
160 // Disable special move detection during error recovery
161 // During error recovery, ALL moves are treated as NORMAL moves to prevent
162 // state conflicts
163 if (!error_recovery_state.waiting_for_move_correction) {
164 // 1. Check for castling
165 if ((source_piece == PIECE_WHITE_KING ||
166 source_piece == PIECE_BLACK_KING) &&
167 abs(move->to_col - move->from_col) == 2) {
168 extended_move.move_type = (move->to_col > move->from_col)
169 ? MOVE_TYPE_CASTLE_KING
170 : MOVE_TYPE_CASTLE_QUEEN;
171 }
172
173 // 2. Check for en passant
174 else if ((source_piece == PIECE_WHITE_PAWN ||
175 source_piece == PIECE_BLACK_PAWN) &&
176 abs(move->to_col - move->from_col) == 1 &&
177 dest_piece == PIECE_EMPTY) {
178 if (game_is_en_passant_possible(move)) {
179 extended_move.move_type = MOVE_TYPE_EN_PASSANT;
180 }
181
182 }
183
184 // 3. Check for promotion
185 else if ((source_piece == PIECE_WHITE_PAWN && move->to_row == 7) ||
186 (source_piece == PIECE_BLACK_PAWN && move->to_row == 0)) {
187 extended_move.move_type = MOVE_TYPE_PROMOTION;
188
189 if (api_immediate_promo && api_promo_choice <= PROMOTION_KNIGHT) {
190 extended_move.promotion_piece = api_promo_choice;
191 ESP_LOGI(TAG,
192 "👑 API immediate promotion (choice=%d) — skip pending state",
193 (int)api_promo_choice);
194 } else {
195 // OKAMŽITĚ nastavit promotion_state před voláním
196 // game_execute_move_enhanced() Toto zajistí, že
197 // game_execute_move_enhanced() neprovádí auto-promoci Použít recursive
198 // mutex funkce (promotion_mutex je recursive mutex)
199 if (xSemaphoreTakeRecursive(promotion_mutex, pdMS_TO_TICKS(100)) ==
200 pdTRUE) {
201 promotion_state.pending = true;
202 promotion_state.square_row = move->to_row;
203 promotion_state.square_col = move->to_col;
204 promotion_state.player = current_player;
205 xSemaphoreGiveRecursive(promotion_mutex);
206
207 ESP_LOGI(TAG,
208 "⏸️ Promotion detected - setting pending state at %c%d for %s "
209 "player",
210 'a' + move->to_col, move->to_row + 1,
211 current_player == PLAYER_WHITE ? "White" : "Black");
212 } else {
213 ESP_LOGE(TAG,
214 "❌ Failed to acquire promotion mutex during move detection!");
215 }
216 }
217 }
218
219 // 4. Check for capture
220 else if (dest_piece != PIECE_EMPTY) {
221 extended_move.move_type = MOVE_TYPE_CAPTURE;
222 ESP_LOGI(TAG, "Capture: %s captures %s",
223 game_get_piece_name(source_piece),
224 game_get_piece_name(dest_piece));
225
226 // Add captured piece to tracking
227 game_add_captured_piece(dest_piece);
228 }
229 } else {
230 // During error recovery, treat everything as NORMAL move (or CAPTURE if
231 // opponent piece present)
232 if (dest_piece != PIECE_EMPTY) {
233 extended_move.move_type = MOVE_TYPE_CAPTURE;
234 ESP_LOGI(TAG, "Error recovery: treating as simple capture");
235 } else {
236 extended_move.move_type = MOVE_TYPE_NORMAL;
237 ESP_LOGI(TAG, "Error recovery: treating as normal move");
238 }
239 }
240
241 // Execute the enhanced move
242 bool success = game_execute_move_enhanced(&extended_move);
243
244 ESP_LOGI(TAG,
245 "🔍 game_execute_move: success=%d, current_player before change=%s",
246 success, current_player == PLAYER_WHITE ? "White" : "Black");
247
248 if (success) {
249 // Update last move tracking for future en passant detection
250 last_move_from_row = move->from_row;
251 last_move_from_col = move->from_col;
252 last_move_to_row = move->to_row;
253 last_move_to_col = move->to_col;
254 has_last_move = true;
255
256 // Rošáda = jeden tah v historii (král už byl přidán; tah věže nepřidávat)
257 bool is_castling_rook_completion =
258 (castling_state.in_progress &&
259 move->from_row == castling_state.rook_from_row &&
260 move->from_col == castling_state.rook_from_col &&
261 move->to_row == castling_state.rook_to_row &&
262 move->to_col == castling_state.rook_to_col);
263
264 if (!is_castling_rook_completion) {
265 // Add to move history
266 if (history_index < GAME_TASK_MAX_MOVES_HISTORY) {
267 move_history[history_index] = *move;
268 move_history[history_index].piece = source_piece;
269 move_history[history_index].captured_piece = dest_piece;
270 move_history[history_index].timestamp = esp_timer_get_time() / 1000;
271 move_history_kind[history_index] = extended_move.move_type;
272 history_index++;
273 }
274
275 // Update game state
276 move_count++;
277 last_move_time = esp_timer_get_time() / 1000;
278
279 // Record material advantage pro graf
280 game_record_material_advantage();
281 } else {
282 last_move_time = esp_timer_get_time() / 1000;
283 }
284
285 // ENHANCED CASTLING DETECTION
286 if (extended_move.move_type == MOVE_TYPE_CASTLE_KING ||
287 extended_move.move_type == MOVE_TYPE_CASTLE_QUEEN) {
288 ESP_LOGI(TAG, "🏰 CASTLING DETECTED! King moved 2 squares");
289
290 // Nastavit castling state pro tracking
291 castling_state.in_progress = true;
292 castling_state.player = current_player;
293 castling_state.is_kingside = (move->to_col > move->from_col);
294
295 // Vypočítat očekávané pozice věže
296 uint8_t rook_row = move->from_row;
297 if (castling_state.is_kingside) {
298 castling_state.rook_from_row = rook_row;
299 castling_state.rook_from_col = 7; // h-file
300 castling_state.rook_to_row = rook_row;
301 castling_state.rook_to_col = 5; // f-file
302 } else {
303 castling_state.rook_from_row = rook_row;
304 castling_state.rook_from_col = 0; // a-file
305 castling_state.rook_to_row = rook_row;
306 castling_state.rook_to_col = 3; // d-file
307 }
308
309 // Ukázat LED indikaci pro věž s pulzováním pro lepší viditelnost
310 led_clear_board_only();
311
312 uint8_t rook_from_led = chess_pos_to_led_index(
313 castling_state.rook_from_row, castling_state.rook_from_col);
314 uint8_t rook_to_led = chess_pos_to_led_index(castling_state.rook_to_row,
315 castling_state.rook_to_col);
316
317 // Pulzování pro lepší viditelnost (3 cykly s plynulým přechodem)
318 // Použít správný výpočet brightness pro plynulé pulzování
319 for (int pulse = 0; pulse < 3; pulse++) {
320 // Plynulé pulzování: 0.5 -> 1.0 -> 0.5
321 // Použít sin() s normalizací - sin() vrací -1 až 1, normalizujeme na
322 // 0-1, pak na 0.5-1.0
323 float phase = (float)pulse * 2.0f * 3.14159f / 3.0f; // 0, 2π/3, 4π/3
324 float brightness =
325 0.5f + 0.5f * (1.0f + sin(phase)) /
326 2.0f; // Normalizace: sin() -> 0-1 -> 0.5-1.0
327
328 // Stříbrná pro věž (source) s pulzováním
329 led_set_pixel_safe(rook_from_led, (uint8_t)(192 * brightness),
330 (uint8_t)(192 * brightness),
331 (uint8_t)(192 * brightness));
332
333 // Zelená pro cíl věže (destination) s pulzováním
334 led_set_pixel_safe(rook_to_led, 0, (uint8_t)(255 * brightness), 0);
335
336 vTaskDelay(pdMS_TO_TICKS(200));
337 }
338
339 // Finální statické zobrazení
340 led_set_pixel_safe(rook_from_led, 192, 192, 192); // Stříbrná pro věž
341 led_set_pixel_safe(rook_to_led, 0, 255, 0); // Zelená pro cíl věže
342
343 // NEMĚNIT HRÁČE pro castling!
344 ESP_LOGI(
345 TAG, "⏳ Waiting for rook move from %c%d to %c%d",
346 'a' + castling_state.rook_from_col, castling_state.rook_from_row + 1,
347 'a' + castling_state.rook_to_col, castling_state.rook_to_row + 1);
348 ESP_LOGI(TAG, "🏰 Castling in progress - player remains %s",
349 current_player == PLAYER_WHITE ? "White" : "Black");
350 game_snapshot_persist_after_valid_move();
351 return success; // Return success but don't change player - hráč se změní
352 // až po dokončení rošády
353 }
354
355 // Check for promotion - start promotion animation (jen když nebyla API okamžitá)
356 if (extended_move.move_type == MOVE_TYPE_PROMOTION) {
357 const bool api_promo_inline =
358 api_immediate_promo && api_promo_choice <= PROMOTION_KNIGHT;
359 if (!api_promo_inline) {
360 ESP_LOGI(TAG, "👑 PROMOTION DETECTED! Starting promotion animation...");
361
362 // #2: Mutex protection for promotion_state
363 // Použít recursive mutex funkce (promotion_mutex je recursive mutex)
364 if (xSemaphoreTakeRecursive(promotion_mutex, pdMS_TO_TICKS(100)) ==
365 pdTRUE) {
366 // Set promotion state to pending
367 // This blocks further moves and enables selection via buttons/UART/web
368 promotion_state.pending = true;
369 promotion_state.square_row = move->to_row;
370 promotion_state.square_col = move->to_col;
371 promotion_state.player = current_player;
372 xSemaphoreGiveRecursive(promotion_mutex);
373
374 ESP_LOGI(TAG, "⏸️ Promotion pending at %c%d for %s player",
375 'a' + move->to_col, move->to_row + 1,
376 current_player == PLAYER_WHITE ? "White" : "Black");
377 } else {
378 ESP_LOGE(TAG, "❌ Failed to acquire promotion mutex!");
379 return false;
380 }
381
382 // Start promotion animation
383 led_command_t promote_cmd = {
384 .type = LED_CMD_ANIM_PROMOTE,
385 .led_index = chess_pos_to_led_index(move->to_row, move->to_col),
386 .red = 255,
387 .green = 215,
388 .blue = 0, // Gold
389 .duration_ms = 2000,
390 .data = NULL};
391 led_execute_command_new(&promote_cmd);
392
393 // Update LED button indications (green for promotion buttons)
394 game_check_promotion_needed();
395
396 // CRITICAL: Return here WITHOUT changing player
397 // Player will change AFTER promotion selection in
398 // game_process_promotion_command()
399 ESP_LOGI(
400 TAG,
401 "🏁 Promotion flow started - awaiting selection (buttons/UART/web)");
402 game_snapshot_persist_after_valid_move();
403 return success; // Don't change player - wait for promotion choice
404 }
405 ESP_LOGI(TAG,
406 "👑 Inline API promotion — continuing (player switch below)");
407 }
408
409 // KONTROLA DOKONČENÍ ROŠÁDY
410 if (castling_state.in_progress) {
411 // Je to tah věže pro dokončení rošády?
412 if (move->from_row == castling_state.rook_from_row &&
413 move->from_col == castling_state.rook_from_col &&
414 move->to_row == castling_state.rook_to_row &&
415 move->to_col == castling_state.rook_to_col) {
416
417 ESP_LOGI(TAG, "✅ CASTLING COMPLETED! Rook moved correctly");
418
419 // Provedeme posun věže v board[][]
420 board[move->to_row][move->to_col] = move->piece;
421 board[castling_state.rook_from_row][castling_state.rook_from_col] =
422 PIECE_EMPTY;
423
424 // Počet rošád pro výukový přehled / API
425 if (castling_state.player == PLAYER_WHITE) {
426 white_castles++;
427 } else {
428 black_castles++;
429 }
430
431 // Aktualizovat příznaky pro krále a věž
432 if (castling_state.player == PLAYER_WHITE) {
433 white_king_moved = true;
434 if (castling_state.rook_from_col == 7)
435 white_rook_h_moved = true;
436 else
437 white_rook_a_moved = true;
438 } else {
439 black_king_moved = true;
440 if (castling_state.rook_from_col == 7)
441 black_rook_h_moved = true;
442 else
443 black_rook_a_moved = true;
444 }
445
446 // Zlatá animace dokončení rošády
447 show_castling_completion_animation();
448
449 /* Vždy ukončit stav rošády před přepnutím hráče / kontrolou konce hry.
450 * Dříve se in_progress vynulovalo jen v ne-endgame větvi — po matu/patu
451 * zůstalo true a validace dál odmítala tahy jako „čeká se na věž“. */
452 castling_state.in_progress = false;
453 STAGING_LOGI(TAG,
454 "castling: completion cleared in_progress (before endgame "
455 "check)");
456
457 // TEPRVE NYÍ změnit hráče po dokončení rošády
458 player_t previous_player = current_player;
459 current_player =
460 (current_player == PLAYER_WHITE) ? PLAYER_BLACK : PLAYER_WHITE;
461
462 ESP_LOGI(TAG, "🎉 Castling completed! Player changed to %s",
463 current_player == PLAYER_WHITE ? "White" : "Black");
464
465 // Timer integration: End timer for previous player and start for new
466 // player
467 ESP_LOGI(TAG,
468 "🔄 Timer switch (castling): ending for %s, starting for %s",
469 previous_player == PLAYER_WHITE ? "White" : "Black",
470 current_player == PLAYER_WHITE ? "White" : "Black");
471 game_end_timer_move();
472 game_start_timer_move(current_player == PLAYER_WHITE);
473
474 // Endgame kontrola PŘED player change animací!
475 // Pokud je endgame, player change se NESPOUŠTÍ
476 game_state_t end_game_result = game_check_end_game_conditions();
477 if (end_game_result == GAME_STATE_FINISHED) {
478 current_game_state = GAME_STATE_FINISHED;
479 game_active = false;
480
481 // Najít pozici krále vítěze pro endgame animaci
482 player_t winner =
483 (current_player == PLAYER_WHITE) ? PLAYER_BLACK : PLAYER_WHITE;
484 uint8_t king_pos = 28; // default e4
485 for (int i = 0; i < 64; i++) {
486 piece_t piece = board[i / 8][i % 8];
487 if ((winner == PLAYER_WHITE && piece == PIECE_WHITE_KING) ||
488 (winner == PLAYER_BLACK && piece == PIECE_BLACK_KING)) {
489 king_pos = i;
490 break;
491 }
492 }
493
494 ESP_LOGI(TAG,
495 "🎯 Game finished after castling! Starting endgame "
496 "animation at position %d",
497 king_pos);
498
499 // Spustit endgame animaci (wave z krále vítěze)
500 led_command_t endgame_cmd = {.type = LED_CMD_ANIM_ENDGAME,
501 .led_index = king_pos,
502 .red = 255,
503 .green = 255,
504 .blue = 0, // Yellow
505 .duration_ms = 0, // Endless
506 .data = NULL};
507 led_execute_command_new(&endgame_cmd);
508
509 ESP_LOGI(
510 TAG,
511 "✅ Endgame animation started - player change animation SKIPPED");
512 } else {
513 // Není endgame - spustit player change animaci
514 uint8_t player_color =
515 (current_player == PLAYER_WHITE) ? 1 : 0; // 1=white, 0=black
516 led_command_t player_change_cmd = {
517 .type = LED_CMD_ANIM_PLAYER_CHANGE, // Použít
518 // ANIM_PLAYER_CHANGE místo
519 // PLAYER_CHANGE
520 .led_index = 0,
521 .red = 0,
522 .green = 0,
523 .blue = 0,
524 .duration_ms = 0,
525 .data = &player_color // Předat barvu hráče
526 };
527 led_execute_command_new(&player_change_cmd);
528
529 // Zobrazit pohyblivé figury pro nového hráče
530 led_clear_board_only();
531 game_highlight_movable_pieces();
532
533 // Zkontrolovat, zda je nový hráč v šachu
534 bool in_check = game_is_king_in_check(current_player);
535 if (in_check) {
536 // Najít pozici krále
537 int king_row = -1, king_col = -1;
538 piece_t king_piece = (current_player == PLAYER_WHITE)
539 ? PIECE_WHITE_KING
540 : PIECE_BLACK_KING;
541 for (int row = 0; row < 8; row++) {
542 for (int col = 0; col < 8; col++) {
543 if (board[row][col] == king_piece) {
544 king_row = row;
545 king_col = col;
546 break;
547 }
548 }
549 if (king_row != -1)
550 break;
551 }
552
553 if (king_row != -1 && king_col != -1) {
554 if (game_led_guidance_show_check_anim()) {
555 uint8_t king_led_index =
556 chess_pos_to_led_index(king_row, king_col);
557 // Spustit check animaci - růžové svícení na králi
558 led_command_t check_cmd = {.type = LED_CMD_ANIM_CHECK,
559 .led_index = king_led_index,
560 .red = 0,
561 .green = 0,
562 .blue = 0,
563 .duration_ms =
564 0, // Trvalé až do dalšího tahu
565 .data = NULL};
566 led_execute_command_new(&check_cmd);
567 ESP_LOGI(TAG, "⚠️ CHECK! %s is in check",
568 current_player == PLAYER_WHITE ? "White" : "Black");
569 }
570 }
571 }
572 }
573
574 // RETURN po dokončení rošády - hráč se už změnil
575 game_snapshot_persist_after_valid_move();
576 return success;
577 } else {
578 // Wrong move during castling
579 // Check if user is aborting by moving King back or elsewhere
580 if (move->piece == PIECE_WHITE_KING ||
581 move->piece == PIECE_BLACK_KING) {
582 ESP_LOGW(TAG, "🚫 Castling ABORTED by King move %c%d-%c%d",

588 // Clear LEDs

591 // We return true (success) so the move is accepted update the board
592 // But we MUST ensure the player switches/unswitches correctly?

594 // But 'game_execute_move' logic for NORMAL moves switches player.

615 // through. The cleanest way: Set in_progress = false, then fall
616 // through? But the block wraps the return.
617
618 // I will Copy-Paste the 'Normal Move' logic essentially, or better:
619 // Just fall through! I will remove the 'return success' and 'else'
620 // structure slightly.
621
622 // REWRITE:
623 /*
624 if (castling_state.in_progress) {
625 if (is_rook_move) {
626 ...

628 }
629 // Not rook move - Abort?
630 castling_state.in_progress = false;

641 castling_state.rook_from_row + 1,

643 // Nezměnit hráče - stále čekáme na správný tah věže
644 game_snapshot_persist_after_valid_move();

646 }

653 // game_process_drop_command) Switch player BEFORE checking
654 // endgame conditions (podle starého projektu)
655 player_t previous_player = current_player;
656 ESP_LOGI(TAG, "🔄 Changing player from %s to %s",
657 previous_player == PLAYER_WHITE ? "White" : "Black",
658 (previous_player == PLAYER_WHITE) ? "Black" : "White");
659 current_player =
660 (current_player == PLAYER_WHITE) ? PLAYER_BLACK : PLAYER_WHITE;
661
662 ESP_LOGI(TAG,
663 "✅ Move executed successfully. %s to move (current_player=%s)",
664 (current_player == PLAYER_WHITE) ? "White" : "Black",
665 current_player == PLAYER_WHITE ? "White" : "Black");

668 // starého projektu)

670 if (end_game_result == GAME_STATE_FINISHED) {
671 ESP_LOGI(TAG, "🎯 Endgame detected after move execution");
672 current_game_state = GAME_STATE_FINISHED;
673 game_active = false;
674 }
675 }
676
677 // Update promotion LED indications after move
678 if (success) {
679 game_check_promotion_needed();
680 game_snapshot_persist_after_valid_move();
681 game_bump_revision_and_notify();

731bool game_execute_move_enhanced(chess_move_extended_t *move) {
732 if (move == NULL)
733 return false;
734
735 // Nastavit typ posledního tahu pro detekci speciálních šachmatů
736 switch (move->move_type) {
737 case MOVE_TYPE_EN_PASSANT:
738 // Remove the captured pawn
739 board[en_passant_victim_row][en_passant_victim_col] = PIECE_EMPTY;
740 last_move_type = LAST_MOVE_EN_PASSANT; // Označit jako en passant
741 ESP_LOGI(TAG, "⚔️ En passant move executed");
742 break;
743
744 case MOVE_TYPE_CASTLE_KING:
745 // Věž se nepřesunuje automaticky - hráč ji musí přesunout sám
746 // Věž zůstává na původní pozici, animace donutí hráče ji přesunout
747 last_move_type = LAST_MOVE_CASTLING; // Označit jako castling
748 ESP_LOGI(TAG, "🏰 Kingside castling - rook stays in place, waiting for "
749 "player to move it");
750 break;
751
752 case MOVE_TYPE_CASTLE_QUEEN:
753 // Věž se nepřesunuje automaticky - hráč ji musí přesunout sám
754 // Věž zůstává na původní pozici, animace donutí hráče ji přesunout
755 last_move_type = LAST_MOVE_CASTLING; // Označit jako castling
756 ESP_LOGI(TAG, "🏰 Queenside castling - rook stays in place, waiting for "
757 "player to move it");
758 break;
759
760 case MOVE_TYPE_PROMOTION:
761 // Handle promotion - the piece will be set below
762 last_move_type = LAST_MOVE_PROMOTION; // Označit jako promotion
763 break;
764
765 default:
766 // Normal move or capture - nothing special needed
767 last_move_type = LAST_MOVE_NORMAL; // Resetovat na normální tah
768 break;
769 }
770
771 // Make the basic move
772 board[move->to_row][move->to_col] = move->piece;
773 board[move->from_row][move->from_col] = PIECE_EMPTY;
774
775 // Handle promotion
776 if (move->move_type == MOVE_TYPE_PROMOTION) {
777 // Zkontrolovat zda je promotion_state.pending
778 // Pokud ano, pak uživatel ještě nevybral figurku - ponechat pěšce na místě
779 // Pokud ne, pak promotion_piece je nastaveno a můžeme provést promoci
780 if (promotion_state.pending) {
781 // Promotion je pending - uživatel ještě nevybral figurku
782 // Ponechat pěšce na promotion rank (už je tam přesunutý v předchozím
783 // kroku)
784 ESP_LOGI(TAG,
785 "⏸️ Promotion pending - keeping pawn at %c%d, waiting for user "
786 "selection",
787 'a' + move->to_col, move->to_row + 1);
788 } else {
789 // Promotion není pending - promotion_piece je nastaveno, provést promoci
790 piece_t promoted_piece =
791 game_piece_for_promotion_choice(move->piece, move->promotion_piece);
792 board[move->to_row][move->to_col] = promoted_piece;
793 ESP_LOGI(TAG, "✅ Promotion executed immediately with piece %d",
794 (int)move->promotion_piece);
795 }
796 }
797
798 // Update castling flags
799 if (move->piece == PIECE_WHITE_KING)
800 white_king_moved = true;
801 if (move->piece == PIECE_BLACK_KING)
802 black_king_moved = true;
803 if (move->piece == PIECE_WHITE_ROOK) {
804 if (move->from_col == 0)
805 white_rook_a_moved = true;
806 if (move->from_col == 7)
807 white_rook_h_moved = true;
808 }
809 if (move->piece == PIECE_BLACK_ROOK) {
810 if (move->from_col == 0)
811 black_rook_a_moved = true;
812 if (move->from_col == 7)
813 black_rook_h_moved = true;
814 }
815
816 // Update en passant state
817 en_passant_available = false;
818 if ((move->piece == PIECE_WHITE_PAWN || move->piece == PIECE_BLACK_PAWN) &&
819 abs(move->to_row - move->from_row) == 2) {
820 // Pawn made a double move - enable en passant
821 en_passant_available = true;
822 en_passant_target_row = (move->from_row + move->to_row) / 2;
823 en_passant_target_col = move->from_col;
824 en_passant_victim_row = move->to_row;
825 en_passant_victim_col = move->to_col;
826 }
827
828 // Update fifty-move counter
829 if (move->piece == PIECE_WHITE_PAWN || move->piece == PIECE_BLACK_PAWN ||
830 move->captured_piece != PIECE_EMPTY) {
831 fifty_move_counter = 0;
832 } else {
833 fifty_move_counter++;
834 }
835
836 // Update move counters and statistics
837 if (current_player == PLAYER_WHITE) {
838 white_moves_count++;
839 } else {
840 black_moves_count++;
841 }
842
843 // NEMĚNIT HRÁČE ZDE - hráč se mění v game_execute_move() po
844 // dokončení tahu Switch players - REMOVED: player is switched in
845 // game_execute_move() after move execution
846
847 return true;
848}
