97static uint32_t game_crc32_simple(const uint8_t *data, size_t len) {
98 uint32_t h = 2166136261u;
99 if (data == NULL) {
100 return h;
101 }
102 for (size_t i = 0; i < len; i++) {
103 h ^= data[i];
104 h *= 16777619u;
105 }
106 return h;
107}

335static void game_boot_tracker_update_on_move(void) {
336 game_boot_tracker_t trk = {0};
337 size_t len = sizeof(trk);
338 bool loaded =
339 (config_load_blob_from_nvs(CONFIG_NVS_KEY_BOOT_TRACKER, &trk, &len) == ESP_OK &&

342
343 if (!loaded) {
344 time_t now = time(NULL);
345 trk.version = GAME_SNAPSHOT_VERSION;
346 trk.first_boot_epoch_s = (now >= 1700000000) ? (uint32_t)now : 0u;
347 trk.boot_counter_window = 1u;
348 }
349 trk.moved_since_boot = 1u;
350 trk.crc32 = game_crc32_simple(((const uint8_t *)&trk) + 8, sizeof(trk) - 8);
351 (void)config_save_blob_to_nvs(CONFIG_NVS_KEY_BOOT_TRACKER, &trk, sizeof(trk));
352}

354static bool game_boot_tracker_should_force_new_game(void) {
355 time_t now = time(NULL);
356 const bool wall_time_ok = (now >= 1700000000);
357 const uint32_t now_s = wall_time_ok ? (uint32_t)now : 0u;
358
359 game_boot_tracker_t trk = {0};
360 size_t len = sizeof(trk);
361 bool loaded =
362 (config_load_blob_from_nvs(CONFIG_NVS_KEY_BOOT_TRACKER, &trk, &len) == ESP_OK &&
363 len == sizeof(trk) && trk.version == GAME_SNAPSHOT_VERSION &&
364 trk.crc32 == game_crc32_simple(((const uint8_t *)&trk) + 8, sizeof(trk) - 8));
365
366 bool in_window = false;
367 if (loaded) {
368 if (wall_time_ok) {
369 in_window = (now_s >= trk.first_boot_epoch_s &&
370 (now_s - trk.first_boot_epoch_s) <= BOOT_WINDOW_SECONDS);
371 } else {
372 in_window = true;
373 }
374 }
375
376 bool force_new_game = false;
377 if (wall_time_ok && loaded && trk.moved_since_boot == 0u && in_window &&
378 trk.boot_counter_window == 1u) {
379 force_new_game = true;
380 }
381
382 game_boot_tracker_t next = {0};
383 next.version = GAME_SNAPSHOT_VERSION;
384 next.moved_since_boot = 0u;
385 if (loaded && in_window) {
386 next.first_boot_epoch_s = trk.first_boot_epoch_s;
387 next.boot_counter_window = trk.boot_counter_window + 1u;
388 } else if (wall_time_ok) {
389 next.first_boot_epoch_s = now_s;
390 next.boot_counter_window = 1u;
391 } else {
392 next.first_boot_epoch_s = loaded ? trk.first_boot_epoch_s : 0u;
393 next.boot_counter_window = 1u;
394 }
395 next.crc32 = game_crc32_simple(((const uint8_t *)&next) + 8, sizeof(next) - 8);
396 (void)config_save_blob_to_nvs(CONFIG_NVS_KEY_BOOT_TRACKER, &next, sizeof(next));
397
398 return force_new_game;
399}

409void game_snapshot_restore_on_boot(void) {
410 bool has_valid_snapshot = game_snapshot_nvs_has_valid();
411 if (has_valid_snapshot) {
412 boot_new_game_triggered = false;
413 STAGING_LOGI(TAG,
414 "boot: valid NVS snapshot present — skipping force-new-game "
415 "heuristic");
416 } else {
417 boot_new_game_triggered = game_boot_tracker_should_force_new_game();
418 }
419
420 if (boot_new_game_triggered) {
421 ESP_LOGW(TAG, "Boot rule triggered: forcing new game (2 boots / 1 minute)");
422 game_start_new_game();
423 snapshot_loaded_on_boot = false;
424 } else {
425 esp_err_t restore_ret = game_load_snapshot_from_nvs();
426 if (restore_ret == ESP_OK) {
427 ESP_LOGI(TAG, "Game snapshot restored from NVS");
428 game_matrix_guard_check_resync_after_restore();
429 } else {
430 ESP_LOGI(TAG, "No valid snapshot found, using fresh board");
431 }
432 }
