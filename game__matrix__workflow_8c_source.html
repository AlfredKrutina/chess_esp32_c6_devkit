37bool game_detect_new_game_setup(void) {
38 if (board_setup_tutorial_active) {
39 return false;
40 }
41
42 // Zkontrolovat, jestli jsou řádky 1, 2, 7, 8 obsazené figurkami
43 // a řádky 3, 4, 5, 6 jsou prázdné
44
45 // Zkontrolovat řádky 0, 1 (bílé figurky) a 6, 7 (černé figurky)
46 bool rows_0_1_6_7_occupied = true;
47 bool rows_2_3_4_5_empty = true;
48
49 // Zkontrolovat řádky 0, 1, 6, 7 - musí být obsazené
50 for (int row = 0; row < 8; row++) {
51 bool row_has_pieces = false;
52 for (int col = 0; col < 8; col++) {
53 if (board[row][col] != PIECE_EMPTY) {
54 row_has_pieces = true;
55 break;
56 }
57 }
58
59 if ((row == 0 || row == 1 || row == 6 || row == 7)) {
60 // Tyto řádky musí být obsazené
61 if (!row_has_pieces) {
62 rows_0_1_6_7_occupied = false;
63 break;
64 }
65 } else if (row >= 2 && row <= 5) {
66 // Tyto řádky musí být prázdné
67 if (row_has_pieces) {
68 rows_2_3_4_5_empty = false;
69 break;
70 }
71 }
72 }
73
74 bool is_starting_setup = rows_0_1_6_7_occupied && rows_2_3_4_5_empty;
75
76 if (is_starting_setup) {
77 ESP_LOGI(TAG, "🎮 Starting position detected: rows 0,1,6,7 occupied, rows "
78 "2,3,4,5 empty");
79 }
80
81 return is_starting_setup;
82}

96void game_handle_piece_lifted(uint8_t row, uint8_t col) {
97 ESP_LOGI(TAG, "🖐️ Matrix: Piece lifted from %c%d", 'a' + col, row + 1);
98
99 // Check if there's a piece at the square
100 piece_t piece = board[row][col];
101 if (piece == PIECE_EMPTY) {
102 ESP_LOGW(TAG, "❌ No piece to lift at %c%d", 'a' + col, row + 1);
103 return;
104 }
105
106 // Check if it's the current player's piece
107 bool is_white_piece =
108 (piece >= PIECE_WHITE_PAWN && piece <= PIECE_WHITE_KING);
109 bool is_current_player = (current_player == PLAYER_WHITE && is_white_piece) ||
110 (current_player == PLAYER_BLACK && !is_white_piece);
111
112 if (!is_current_player) {
113 ESP_LOGW(TAG, "❌ Cannot lift opponent's piece at %c%d", 'a' + col,
114 row + 1);
115 return;
116 }
117
118 // Check if castle animation is active and this is a rook
119 if (game_is_castle_animation_active()) {
120 bool is_rook = (piece == PIECE_WHITE_ROOK || piece == PIECE_BLACK_ROOK);
121
122 if (is_rook) {
123 // Check if rook is being lifted from the correct position for castling
124 if (row == rook_from_row && col == rook_from_col) {
125 ESP_LOGI(TAG, "✅ Rook lifted from correct position for castling");
126 // Continue with normal move highlighting
127 } else {
128 // Rook lifted from wrong position - show error and valid moves from
129 // correct position
130 ESP_LOGW(TAG,
131 "❌ Rook lifted from wrong position for castling: %c%d "
132 "(expected: %c%d)",
133 'a' + col, row + 1, 'a' + rook_from_col, rook_from_row + 1);
134
135 // Show error animation - blink red
136 led_clear_board_only();
137 for (int i = 0; i < 3; i++) {
138 led_set_pixel_safe(chess_pos_to_led_index(row, col), 255, 0,
139 0); // Red
140 vTaskDelay(pdMS_TO_TICKS(200));
141 led_clear_board_only();
142 vTaskDelay(pdMS_TO_TICKS(200));
143 }
144
145 // Show valid moves from the correct rook position
146 led_set_pixel_safe(chess_pos_to_led_index(rook_from_row, rook_from_col),
147 255, 255,
148 0); // Yellow
149 led_set_pixel_safe(chess_pos_to_led_index(rook_to_row, rook_to_col), 0,
150 255, 0); // Green
151
152 return;
153 }
154 } else {
155 ESP_LOGI(TAG, "🏰 Castle animation active - only rook can be moved");
156 return;
157 }
158 }
159
160 // Show possible moves for this piece
161 move_suggestion_t suggestions[64];
162 uint32_t valid_moves = game_get_available_moves(row, col, suggestions, 64);
163
164 if (valid_moves > 0) {
165 ESP_LOGI(TAG, "💡 Found %lu valid moves for piece at %c%d", valid_moves,
166 'a' + col, row + 1);
167
168 // Send LED commands to highlight possible moves
169 // Highlight source square in yellow
170 // Highlight source square (yellow)
171 led_set_pixel_safe(chess_pos_to_led_index(row, col), 255, 255,
172 0); // Yellow for source
173
174 // Highlight possible destinations
175 if (game_led_guidance_show_destinations()) {
176 for (uint32_t i = 0; i < valid_moves; i++) {
177 uint8_t dest_row = suggestions[i].to_row;
178 uint8_t dest_col = suggestions[i].to_col;
179 uint8_t led_index = chess_pos_to_led_index(dest_row, dest_col);
180
181 // Check if destination has opponent's piece
182 piece_t dest_piece = board[dest_row][dest_col];
183 bool is_opponent_piece =
184 (current_player == PLAYER_WHITE && dest_piece >= PIECE_BLACK_PAWN &&
185 dest_piece <= PIECE_BLACK_KING) ||
186 (current_player == PLAYER_BLACK && dest_piece >= PIECE_WHITE_PAWN &&
187 dest_piece <= PIECE_WHITE_KING);
188
189 if (is_opponent_piece) {
190 // Orange for opponent's pieces (capture)
191 led_set_pixel_safe(led_index, 255, 165, 0);
192 } else {
193 // Green for empty squares
194 led_set_pixel_safe(led_index, 0, 255, 0);
195 }
196 }
197 }
198 } else {
199 ESP_LOGI(TAG, "💡 No valid moves for piece at %c%d", 'a' + col, row + 1);
200 }
201
202 // Po zvednutí figurky spustit animaci změny hráče a zobrazit
203 // pohyblivé figurky (pouze pokud není aktivní animace rosady a není
204 // očekávána rosada)
205 if (!game_is_castle_animation_active() && !game_is_castling_expected()) {
206 // After piece is lifted, show pieces that the opponent can move
207 // This completes the cycle as requested by the user
208 game_highlight_movable_pieces();
209 }

313void game_handle_matrix_move(uint8_t from_row, uint8_t from_col, uint8_t to_row,
314 uint8_t to_col) {
315 ESP_LOGI(TAG, "🎯 Matrix: Complete move %c%d -> %c%d", 'a' + from_col,
316 from_row + 1, 'a' + to_col, to_row + 1);
317
318 // Check if castle animation is active
319 if (game_is_castle_animation_active()) {
320 if (game_complete_castle_animation(from_row, from_col, to_row, to_col)) {
321 ESP_LOGI(TAG, "✅ Castle animation completed via matrix move");
322 return;
323 } else {
324 ESP_LOGW(TAG,
325 "❌ Invalid rook move for castling - waiting for correct move");
326 return;
327 }
328 }
329
330 // Convert matrix event to chess move
331 chess_move_t move = {
332 .from_row = from_row,
333 .from_col = from_col,
334 .to_row = to_row,
335 .to_col = to_col,
336
337 .piece = board[from_row][from_col], // Skutečná figurka ze zdrojového pole
338 .captured_piece =
339 board[to_row][to_col], // Skutečná figurka z cílového pole
340 .timestamp = esp_timer_get_time() / 1000};
341
342 // KRITICKÁ OPRAVA: Validace tahu před voláním game_execute_move (stejně
343 // jako UART flow)
344 move_error_t error = game_is_valid_move(&move);
345 if (error != MOVE_ERROR_NONE) {
346 ESP_LOGW(TAG, "❌ Invalid matrix move rejected: %c%d -> %c%d (error: %d)",
347 'a' + from_col, from_row + 1, 'a' + to_col, to_row + 1, error);
348
349 // Clear LEDs on invalid move
350 led_clear_board_only();
351
352 // Show error animation
353 led_set_pixel_safe(chess_pos_to_led_index(from_row, from_col), 255, 0,
354 0); // Red for source
355 led_set_pixel_safe(chess_pos_to_led_index(to_row, to_col), 255, 0,
356 0); // Red for destination
357
358 return; // Reject invalid move
359 }
360
361 // Detekce rosady před voláním game_execute_move
362 bool is_castling =
363 (move.piece == PIECE_WHITE_KING || move.piece == PIECE_BLACK_KING) &&
364 abs((int)move.to_col - (int)move.from_col) == 2;
365
366 if (is_castling) {
367 ESP_LOGI(TAG, "🏰 CASTLING DETECTED in matrix move: %c%d -> %c%d",
368 'a' + move.from_col, move.from_row + 1, 'a' + move.to_col,
369 move.to_row + 1);
370 }
371
372 // Execute move (already validated)
373 if (game_execute_move(&move)) {
374 ESP_LOGI(TAG, "✅ Matrix move executed successfully");
375
376 // KRITICKÁ OPRAVA: Kontrola checkmate/stalemate po každém tahu!
377 game_state_t end_game_result = game_check_end_game_conditions();
378 if (end_game_result == GAME_STATE_FINISHED) {
379 current_game_state = GAME_STATE_FINISHED;
380 game_active = false;
381 ESP_LOGI(TAG, "🎉 Game finished detected in matrix move!");
382 }
383
384 if (is_castling) {
385 // Pro rosadu nespouštět game_highlight_opponent_pieces
386 ESP_LOGI(TAG, "🏰 Castling move completed - waiting for rook animation");
387 } else {
388 // Po úspěšném tahu spustit animaci změny hráče a zobrazit
389 // pohyblivé figurky (pouze pokud není aktivní animace rosady a není
390 // očekávána rosada)
391 if (!game_is_castle_animation_active() && !game_is_castling_expected()) {
392 // After successful move, show pieces that the opponent can move
393 game_highlight_movable_pieces();
394 }
395 }
396 } else {
397 ESP_LOGW(TAG, "❌ Invalid matrix move rejected");
398
399 // Clear LEDs on invalid move
400 led_clear_board_only();
401 }
402}

450void game_process_promotion_command(const chess_move_command_t *cmd) {
451 if (!cmd) {
452 ESP_LOGE(TAG, "❌ Invalid promotion command");
453 return;
454 }
455
456 ESP_LOGI(TAG, "👑 Processing promotion command");
457
458 // #2 & #3: Atomic check with mutex to prevent double execution
459 // Použít recursive mutex funkce (promotion_mutex je recursive mutex)
460 if (xSemaphoreTakeRecursive(promotion_mutex, pdMS_TO_TICKS(100)) != pdTRUE) {
461 ESP_LOGE(TAG, "❌ Failed to acquire promotion mutex");
462 return;
463 }
464
465 // Check if promotion is still pending (might have been processed by another
466 // task)
467 if (!promotion_state.pending) {
468 xSemaphoreGiveRecursive(promotion_mutex);
469 ESP_LOGW(TAG, "⚠️ Promotion already processed by another task");
470 return;
471 }
472
473 // CRITICAL: Keep mutex held during ENTIRE execution to prevent race!
474 // DO NOT release here - release only after completion or on error
475
476 // Determine promotion choice (default to Queen if invalid or not specified)
477 promotion_choice_t choice = PROMOTION_QUEEN;
478
479 if (cmd->promotion_choice <= PROMOTION_KNIGHT) {
480 choice = (promotion_choice_t)cmd->promotion_choice;
481 ESP_LOGI(TAG, "👑 Promotion choice received: %d (%s)", choice,
482 choice == PROMOTION_QUEEN ? "Queen"
483 : choice == PROMOTION_ROOK ? "Rook"
484 : choice == PROMOTION_BISHOP ? "Bishop"
485 : "Knight");
486 } else {
487 ESP_LOGW(TAG,
488 "⚠️ Invalid or missing promotion choice (%d), defaulting to Queen",
489 cmd->promotion_choice);
490 }
491
492 // Execute the promotion with the selected piece
493 if (game_execute_promotion(choice)) {
494 ESP_LOGI(TAG, "✅ Promotion executed successfully");
495
496 // Clear promotion state (mutex already held from line 10419)
497 promotion_state.pending = false;
498
499 // Clear promotion state and return to active gameplay
500 current_game_state =
501 GAME_STATE_ACTIVE; // Fix: use constant instead of magic number 6
502
503 // Switch to next player
504 player_t previous_player = current_player;
505 current_player =
506 (current_player == PLAYER_WHITE) ? PLAYER_BLACK : PLAYER_WHITE;
507
508 // Po dokončení promoce už žádná figurka není zvednutá – aby se zobrazily
509 // movable pieces (game_highlight_movable_pieces nesmí skipnout kvůli
510 // piece_lifted).
511 piece_lifted = false;
512 lifted_piece_row = 0;
513 lifted_piece_col = 0;
514 lifted_piece = PIECE_EMPTY;
515
516 // #2: Timer switch AFTER promotion choice
517 ESP_LOGI(TAG, "🔄 Timer switch (promotion): ending for %s, starting for %s",
518 previous_player == PLAYER_WHITE ? "White" : "Black",
519 current_player == PLAYER_WHITE ? "White" : "Black");
520 game_end_timer_move();
521 game_start_timer_move(current_player == PLAYER_WHITE);
522
523 // KRITICKÉ: Endgame kontrola PŘED player change animací!
524 // Pokud je endgame, player change se NESPOUŠTÍ
525 game_state_t end_game_result = game_check_end_game_conditions();
526 if (end_game_result == GAME_STATE_FINISHED) {
527 current_game_state = GAME_STATE_FINISHED;
528 game_active = false;
529
530 // Najít pozici krále vítěze pro endgame animaci
531 player_t winner =
532 (current_player == PLAYER_WHITE) ? PLAYER_BLACK : PLAYER_WHITE;
533 uint8_t king_pos = 28; // default e4
534 for (int i = 0; i < 64; i++) {
535 piece_t piece = board[i / 8][i % 8];
536 if ((winner == PLAYER_WHITE && piece == PIECE_WHITE_KING) ||
537 (winner == PLAYER_BLACK && piece == PIECE_BLACK_KING)) {
538 king_pos = i;
539 break;
540 }
541 }
542
543 ESP_LOGI(TAG,
544 "🎯 Game finished after promotion command! Starting endgame "
545 "animation at position %d",
546 king_pos);
547
548 // Spustit endgame animaci (wave z krále vítěze)
549 led_command_t endgame_cmd = {.type = LED_CMD_ANIM_ENDGAME,
550 .led_index = king_pos,
551 .red = 255,
552 .green = 255,
553 .blue = 0, // Yellow
554 .duration_ms = 0, // Endless
555 .data = NULL};
556 led_execute_command_new(&endgame_cmd);
557
558 ESP_LOGI(
559 TAG,
560 "✅ Endgame animation started - player change animation SKIPPED");
561 } else {
562 // Není endgame - spustit player change animaci
563 uint8_t player_color =
564 (current_player == PLAYER_WHITE) ? 1 : 0; // 1=white, 0=black
565 led_command_t player_change_cmd = {
566 .type = LED_CMD_ANIM_PLAYER_CHANGE, // Použít
567 // ANIM_PLAYER_CHANGE místo
568 // PLAYER_CHANGE
569 .led_index = 0,
570 .red = 0,
571 .green = 0,
572 .blue = 0,
573 .duration_ms = 0,
574 .data = &player_color // Předat barvu hráče
575 };
576 led_execute_command_new(&player_change_cmd);
577
578 // Print updated board
579 game_print_board();
580
581 // Highlight pieces that the new current player can move
582 game_highlight_movable_pieces();
583 led_force_immediate_update(); // Commit highlight so movable pieces show
584 // right after player change animation
585
586 // Zkontrolovat, zda je nový hráč v šachu
587 bool in_check = game_is_king_in_check(current_player);
588 if (in_check) {
589 // Najít pozici krále
590 int king_row = -1, king_col = -1;
591 piece_t king_piece = (current_player == PLAYER_WHITE)
592 ? PIECE_WHITE_KING
593 : PIECE_BLACK_KING;
594 for (int row = 0; row < 8; row++) {
595 for (int col = 0; col < 8; col++) {
596 if (board[row][col] == king_piece) {
597 king_row = row;
598 king_col = col;
599 break;
600 }
601 }
602 if (king_row != -1)
603 break;
604 }
605
606 if (king_row != -1 && king_col != -1) {
607 if (game_led_guidance_show_check_anim()) {
608 uint8_t king_led_index = chess_pos_to_led_index(king_row, king_col);
609 // Spustit check animaci - růžové svícení na králi
610 led_command_t check_cmd = {.type = LED_CMD_ANIM_CHECK,
611 .led_index = king_led_index,
612 .red = 0,
613 .green = 0,
614 .blue = 0,
615 .duration_ms =
616 0, // Trvalé až do dalšího tahu
617 .data = NULL};
618 led_execute_command_new(&check_cmd);
619 ESP_LOGI(TAG, "⚠️ CHECK! %s is in check",
620 current_player == PLAYER_WHITE ? "White" : "Black");
621 }
622 }
623 }
624 }
625
626 } else {
627 ESP_LOGE(TAG, "❌ Failed to execute promotion");
628 }
629
630 // CRITICAL: Release mutex at the end of the function to ensure it's always
631 // released Použít recursive mutex funkce (promotion_mutex je recursive mutex)
632 xSemaphoreGiveRecursive(promotion_mutex);
633}

658bool game_execute_promotion(promotion_choice_t choice) {
659 ESP_LOGI(TAG, "👑 Executing pawn promotion: %d", choice);
660
661 // Find the pawn that needs to be promoted
662 // This should be set when a pawn reaches the promotion rank
663 // For now, we'll implement a basic version
664
665 // Convert promotion choice to piece type
666 piece_t promoted_piece;
667 // Use promotion_state.player instead of current_player
668 // current_player has already been switched to the opponent by
669 // game_execute_move() so we must use the player stored in the promotion
670 // state
671 if (promotion_state.player == PLAYER_WHITE) {
672 switch (choice) {
673 case PROMOTION_QUEEN:
674 promoted_piece = PIECE_WHITE_QUEEN;
675 break;
676 case PROMOTION_ROOK:
677 promoted_piece = PIECE_WHITE_ROOK;
678 break;
679 case PROMOTION_BISHOP:
680 promoted_piece = PIECE_WHITE_BISHOP;
681 break;
682 case PROMOTION_KNIGHT:
683 promoted_piece = PIECE_WHITE_KNIGHT;
684 break;
685 default:
686 return false;
687 }
688 } else {
689 switch (choice) {
690 case PROMOTION_QUEEN:
691 promoted_piece = PIECE_BLACK_QUEEN;
692 break;
693 case PROMOTION_ROOK:
694 promoted_piece = PIECE_BLACK_ROOK;
695 break;
696 case PROMOTION_BISHOP:
697 promoted_piece = PIECE_BLACK_BISHOP;
698 break;
699 case PROMOTION_KNIGHT:
700 promoted_piece = PIECE_BLACK_KNIGHT;
701 break;
702 default:
703 return false;
704 }
705 }
706
707 // BUG #3: Use stored position instead of scanning entire board
708 // The position was already found and stored in promotion_state by
709 // game_check_promotion_needed()
710 uint8_t row = promotion_state.square_row;
711 uint8_t col = promotion_state.square_col;
712 piece_t piece = board[row][col];
713
714 // Verify it's actually a pawn (safety check)
715 if (piece != PIECE_WHITE_PAWN && piece != PIECE_BLACK_PAWN) {
716 ESP_LOGE(TAG, "❌ Promotion failed: No pawn at stored position %c%d",
717 'a' + col, row + 1);
718 return false;
719 }
720
721 // Promote the pawn
722 board[row][col] = promoted_piece;
723 current_game_state = GAME_STATE_ACTIVE; // Restore game state to ACTIVE
724 ESP_LOGI(TAG, "✅ Promoted %s pawn at %c%d to %s",
725 current_player == PLAYER_WHITE ? "white" : "black", 'a' + col,
726 row + 1,
727 choice == PROMOTION_QUEEN ? "Queen"
728 : choice == PROMOTION_ROOK ? "Rook"
729 : choice == PROMOTION_BISHOP ? "Bishop"
730 : "Knight");
731
732 // Spustit promotion animaci
733 uint8_t promotion_led = chess_pos_to_led_index(row, col);
734 led_command_t promote_cmd = {.type = LED_CMD_ANIM_PROMOTE,
735 .led_index = promotion_led,
736 .red = 255,
737 .green = 215,
738 .blue = 0, // Gold
739 .duration_ms = 2000,
740 .data = NULL};
741 led_execute_command_new(&promote_cmd);
742
743 return true;
744}

749void game_highlight_movable_pieces(void) {
750 // Do not highlight pieces if system is booting
751 if (led_is_booting()) {
752 return;
753 }
754
755 // Do not highlight movable pieces if a piece is already
756 // lifted This prevents overwriting the "valid moves" (green) with "movable
757 // pieces" (yellow)
758 if (piece_lifted) {
759 ESP_LOGD(TAG, "Highlight skipped - piece is lifted");
760 return;
761 }
762
763 // Nespouštět highlight během rošády - rošáda má vlastní LED indikaci
764 if (castling_state.in_progress) {
765 ESP_LOGD(TAG, "Highlight skipped - castling in progress");
766 return;
767 }
768
769 if (!game_led_guidance_show_movable_yellow()) {
770 ESP_LOGD(TAG, "Movable piece highlights skipped (LED guidance level %u)",
771 (unsigned)game_get_led_guidance_level());
772 return;
773 }
774
775 ESP_LOGI(TAG, "🟡 Highlighting movable pieces for %s player",
776 current_player == PLAYER_WHITE ? "white" : "black");
777
778 // Endgame detection is handled in game_analyze_position() called from
779 // game_execute_move()
780
781 // LED command queue removed - using direct LED calls now
782
783 uint32_t highlighted_count = 0;
784
785 // Scan all squares for current player's pieces
786 for (uint8_t row = 0; row < 8; row++) {
787 for (uint8_t col = 0; col < 8; col++) {
788 piece_t piece = board[row][col];
789
790 // Check if piece belongs to current player
791 bool is_current_player_piece = false;
792 if (current_player == PLAYER_WHITE) {
793 is_current_player_piece =
794 (piece >= PIECE_WHITE_PAWN && piece <= PIECE_WHITE_KING);
795 } else {
796 is_current_player_piece =
797 (piece >= PIECE_BLACK_PAWN && piece <= PIECE_BLACK_KING);
798 }
799
800 if (is_current_player_piece) {
801 // Check if piece has valid moves
802 // OPTIMIZATION: Reduced buffer size from 64 to 1 to prevent stack
803 // overflow in Timer Task We only need to know if count > 0, so asking
804 // for 1 move is enough
805 move_suggestion_t suggestions[1];
806 uint32_t valid_moves =
807 game_get_available_moves(row, col, suggestions, 1);
808
809 if (valid_moves > 0) {
810 // Highlight movable piece in yellow
811 led_set_pixel_safe(chess_pos_to_led_index(row, col), 255, 255,
812 0); // Yellow
813 highlighted_count++;
814 }
815 }
816 }
817 }
818
819 ESP_LOGI(TAG, "🟡 Highlighted %lu movable pieces", highlighted_count);
820}
