207esp_err_t uart_task_wdt_reset_safe(void) {
208 esp_err_t ret = esp_task_wdt_reset();
209
210 if (ret == ESP_ERR_NOT_FOUND) {
211 // Log as WARNING instead of ERROR - task not registered yet
212 ESP_LOGW(
213 "UART_TASK",
214 "WDT reset: task not registered yet (this is normal during startup)");
215 return ESP_OK; // Treat as success for our purposes
216 } else if (ret != ESP_OK) {
217 ESP_LOGE("UART_TASK", "WDT reset failed: %s", esp_err_to_name(ret));
218 return ret;
219 }
220
221 return ESP_OK;
222}

655void uart_show_progress_bar(const char *label, uint32_t max_value,
656 uint32_t duration_ms) {
657 // Zamykání mutexu pro celou progress bar operaci
658 if (uart_mutex != NULL) {
659 xSemaphoreTake(uart_mutex, portMAX_DELAY);
660 }
661
662 const int bar_width = 20;
663 uint32_t step_delay = duration_ms / max_value;
664 if (step_delay < 5)
665 step_delay = 5; // Minimum 5ms delay for smooth animation
666
667 if (color_enabled)
668 uart_write_string_immediate("\033[1;32m"); // bold green
669 char buffer[256];
670 snprintf(buffer, sizeof(buffer), "%s: [", label);
671 uart_write_string_immediate(buffer);
672
673 for (int i = 0; i < bar_width; i++) {
674 uart_write_string_immediate(".");
675 }
676 uart_write_string_immediate("] 0%");
677 if (color_enabled)
678 uart_write_string_immediate("\033[0m"); // reset colors
679
680 for (uint32_t i = 0; i <= max_value; i++) {
681 int filled = (i * bar_width) / max_value;
682
683 // Reset watchdog before each progress update (only if registered)
684 esp_err_t wdt_ret = uart_task_wdt_reset_safe();
685 if (wdt_ret != ESP_OK && wdt_ret != ESP_ERR_NOT_FOUND) {
686 // Task not registered with TWDT yet - this is normal during startup
687 }
688
689 // Move cursor back to start of progress bar
690 if (color_enabled)
691 uart_write_string_immediate("\033[1;32m"); // bold green
692 uart_write_string_immediate("\r");
693 snprintf(buffer, sizeof(buffer), "%s: [", label);
694 uart_write_string_immediate(buffer);
695
696 // Show filled part (ASCII only)
697 for (int j = 0; j < filled; j++) {
698 uart_write_string_immediate("#");
699 }
700
701 // Show unfilled part
702 for (int j = filled; j < bar_width; j++) {

977void uart_queue_message(uart_msg_type_t type, bool add_newline,
978 const char *format, ...) {
979 if (uart_output_queue == NULL) {
980 // Queue not ready, fall back to direct output
981 va_list args;
982 va_start(args, format);
983 char buffer[512];
984 vsnprintf(buffer, sizeof(buffer), format, args);
985 va_end(args);
986 if (UART_ENABLED) {
987 uart_write_bytes(UART_PORT_NUM, buffer, strlen(buffer));
988 if (add_newline)
989 uart_write_bytes(UART_PORT_NUM, "\n", 1);
990 } else {
991 printf("%s", buffer);
992 if (add_newline)
993 printf("\n");
994 }
995 return;
996 }
997
998 uart_message_t msg;
999 msg.type = type;
1000 msg.add_newline = add_newline;
1001
1002 va_list args;
1003 va_start(args, format);
1004 vsnprintf(msg.message, sizeof(msg.message), format, args);
1005 va_end(args);
1006
1007 // Send to queue with timeout to prevent blocking
1008 if (xQueueSend(uart_output_queue, &msg, pdMS_TO_TICKS(10)) != pdTRUE) {
1009 // Queue full, fall back to direct output
1010 if (UART_ENABLED) {
1011 uart_write_bytes(UART_PORT_NUM, msg.message, strlen(msg.message));
1012 if (add_newline)
1013 uart_write_bytes(UART_PORT_NUM, "\n", 1);
1014 } else {
1015 printf("%s", msg.message);
1016 if (add_newline)
1017 printf("\n");
1018 }
1019 }
1020}

1025static void uart_process_output_queue(void) {
1026 uart_message_t msg;
1027
1028 // Process all pending messages
1029 while (xQueueReceive(uart_output_queue, &msg, 0) == pdTRUE) {
1030 // Take mutex for entire message output
1031 if (uart_mutex != NULL) {
1032 xSemaphoreTake(uart_mutex, portMAX_DELAY);
1033 }
1034
1035 // Apply color based on message type
1036 const char *color = COLOR_RESET;
1037 switch (msg.type) {
1038 case UART_MSG_ERROR:
1039 color = COLOR_ERROR;
1040 break;
1041 case UART_MSG_WARNING:
1042 color = COLOR_WARNING;
1043 break;
1044 case UART_MSG_SUCCESS:
1045 color = COLOR_SUCCESS;
1046 break;
1047 case UART_MSG_INFO:
1048 color = COLOR_INFO;
1049 break;
1050 case UART_MSG_DEBUG:
1051 color = COLOR_DEBUG;
1052 break;
1053 default:
1054 color = COLOR_RESET;
1055 break;
1056 }
1057
1058 if (UART_ENABLED) {
1059 if (color_enabled && msg.type != UART_MSG_NORMAL) {
1060 char buffer[512];
1061 snprintf(buffer, sizeof(buffer), "%s%s%s", color, msg.message,
1062 COLOR_RESET);
1063 uart_write_bytes(UART_PORT_NUM, buffer, strlen(buffer));
1064 } else {
1065 uart_write_bytes(UART_PORT_NUM, msg.message, strlen(msg.message));
1066 }
1067
1068 if (msg.add_newline) {
1069 uart_write_bytes(UART_PORT_NUM, "\n", 1);
1070 }
1071 } else {
1072 // For USB Serial JTAG, use printf
1073 if (color_enabled && msg.type != UART_MSG_NORMAL) {
1074 printf("%s%s%s", color, msg.message, COLOR_RESET);
1075 } else {
1076 printf("%s", msg.message);
1077 }
1078
1079 if (msg.add_newline) {
1080 printf("\n");
1081 }
1082 }
1083
1084 // No need for fflush with ESP-IDF UART driver
1085
1086 if (uart_mutex != NULL) {
1087 xSemaphoreGive(uart_mutex);
1088 }
1089 }
1090}

1188
1189 history->current = (history->current + 1) % history->max_size;
1190 if (history->count < history->max_size) {
1191 history->count++;
1192 }
1193}
