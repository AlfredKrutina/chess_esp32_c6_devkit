28esp_err_t http_ws_handler(httpd_req_t *req) {
29 if (req->method == HTTP_GET) {
30 ESP_LOGD(TAG, "WebSocket handshake OK (/ws)");
31 return ESP_OK;
32 }
33
34 httpd_ws_frame_t ws_pkt;
35 memset(&ws_pkt, 0, sizeof(ws_pkt));
36 ws_pkt.type = HTTPD_WS_TYPE_TEXT;
37 esp_err_t ret = httpd_ws_recv_frame(req, &ws_pkt, 0);
38 if (ret != ESP_OK) {
39 ESP_LOGD(TAG, "ws recv (len): %s", esp_err_to_name(ret));
40 return ret;
41 }
42 if (ws_pkt.len) {
43 uint8_t *buf = calloc(1, ws_pkt.len + 1);
44 if (buf == NULL) {
45 return ESP_ERR_NO_MEM;
46 }
47 ws_pkt.payload = buf;
48 ret = httpd_ws_recv_frame(req, &ws_pkt, ws_pkt.len);
49 free(buf);
50 if (ret != ESP_OK) {
51 ESP_LOGD(TAG, "ws recv payload: %s", esp_err_to_name(ret));
52 return ret;
53 }
54 }
55 return ESP_OK;
56}

58static bool ws_has_clients(void) {
59 if (web_server_get_httpd_handle() == NULL) {

76void ws_broadcast_snapshot(void) {
77 if (web_server_get_httpd_handle() == NULL || !web_server_is_active()) {
78 return;
79 }
80 if (!ws_has_clients()) {
81 return;
82 }
83 (void)web_server_task_wdt_reset_safe();
84 size_t len = 0;
85 if (build_snapshot_json(&len) != ESP_OK) {
86 ESP_LOGD(TAG, "WS broadcast: build_snapshot_json failed");
87 return;
88 }
89 uint8_t *payload = (uint8_t *)malloc(len);
90 if (payload == NULL) {
91 ESP_LOGE(TAG, "WS broadcast: malloc %u B failed", (unsigned)len);
92 return;
93 }
94 memcpy(payload, snapshot_buffer, len);
95 httpd_ws_frame_t ws_pkt = {.type = HTTPD_WS_TYPE_TEXT,
96 .payload = payload,
97 .len = len};
98 size_t n = WS_MAX_CLIENT_FDS;
99 int fds[WS_MAX_CLIENT_FDS];
100 if (httpd_get_client_list(web_server_get_httpd_handle(), &n, fds) != ESP_OK) {
101 free(payload);
102 return;
103 }
104 size_t ws_count = 0;
105 for (size_t i = 0; i < n; i++) {
106 if (httpd_ws_get_fd_info(web_server_get_httpd_handle(), fds[i]) !=
107 HTTPD_WS_CLIENT_WEBSOCKET) {
108 continue;
109 }
110 ws_count++;
111 (void)web_server_task_wdt_reset_safe();
112 esp_err_t err = httpd_ws_send_data(web_server_get_httpd_handle(), fds[i], &ws_pkt);
113 if (err != ESP_OK) {
114 ESP_LOGD(TAG, "WS send fd %d: %s", fds[i], esp_err_to_name(err));
115 }
116 }
117 ESP_LOGD(TAG, "[STAGING] WS broadcast → %u WS client(s), %u B payload",
118 (unsigned)ws_count, (unsigned)len);
119 free(payload);
120}

142void web_server_websocket_init(void) {
143#if CONFIG_HTTPD_WS_SUPPORT
144 if (ws_broadcast_timer != NULL) {
145 return;
146 }
147 const esp_timer_create_args_t args = {.callback = &ws_broadcast_timer_cb,
148 .name = "ws_snap"};
149 esp_err_t e = esp_timer_create(&args, &ws_broadcast_timer);
150 if (e != ESP_OK) {
151 ESP_LOGE(TAG, "WS timer create failed: %s", esp_err_to_name(e));
152 return;
153 }
154 e = esp_timer_start_periodic(ws_broadcast_timer, 3000 * 1000);
155 if (e != ESP_OK) {
156 ESP_LOGE(TAG, "WS timer start failed: %s", esp_err_to_name(e));
157 esp_timer_delete(ws_broadcast_timer);
158 ws_broadcast_timer = NULL;
159 return;
160 }
161 czechmate_ensure_snapshot_notify_queue();
162 ESP_LOGI(TAG,
163 "WebSocket: watchdog broadcast 3 s + okamžitý push při změně stavu");
164#else
165 ESP_LOGD(TAG, "WebSocket disabled (CONFIG_HTTPD_WS_SUPPORT)");
166#endif
167}
