23bool parse_move_notation(const char *input, char *from, char *to) {
24 if (!input || !from || !to)
25 return false;
26
27 // Remove leading/trailing whitespace
28 while (*input == ' ' || *input == '\t')
29 input++;
30
31 // Support formats: "e2 e4", "e2-e4", "e2e4", "E2 E4", "E2-E4", "E2E4"
32 if (strlen(input) < 4)
33 return false;
34
35 // Format: "e2 e4" or "E2 E4" (space separated)
36 if (strchr(input, ' ')) {
37 char *space = strchr(input, ' ');
38 size_t from_len = space - input;
39 if (from_len != 2)
40 return false;
41
42 strncpy(from, input, 2);
43 from[2] = '\0';
44 from[0] = tolower(from[0]); // Convert to lowercase
45
46 // Skip whitespace
47 while (*space == ' ' || *space == '\t')
48 space++;
49 if (strlen(space) != 2)
50 return false;
51
52 strncpy(to, space, 2);
53 to[2] = '\0';
54 to[0] = tolower(to[0]); // Convert to lowercase
55
56 return true;
57 }
58
59 // Format: "e2-e4" or "E2-E4" (dash separated)
60 if (strchr(input, '-')) {
61 char *dash = strchr(input, '-');
62 size_t from_len = dash - input;
63 if (from_len != 2)
64 return false;
65
66 strncpy(from, input, 2);
67 from[2] = '\0';
68 from[0] = tolower(from[0]); // Convert to lowercase
69
70 if (strlen(dash + 1) != 2)
71 return false;
72
73 strncpy(to, dash + 1, 2);
74 to[2] = '\0';
75 to[0] = tolower(to[0]); // Convert to lowercase
76
77 return true;
78 }
79
80 // Format: "e2e4" or "E2E4" (compact)
81 if (strlen(input) == 4) {
82 strncpy(from, input, 2);
83 from[2] = '\0';
84 from[0] = tolower(from[0]); // Convert to lowercase
85 strncpy(to, input + 2, 2);
86 to[2] = '\0';
87 to[0] = tolower(to[0]); // Convert to lowercase
88
89 return true;
90 }
91
92 return false;
93}

95bool validate_chess_squares(const char *from, const char *to) {
96 if (!from || !to)
97 return false;
98
99 // Check format: [a-h][1-8]
100 if (strlen(from) != 2 || strlen(to) != 2)
101 return false;
102
103 // Validate file (column) - case-insensitive
104 char from_file = tolower(from[0]);
105 char to_file = tolower(to[0]);
106 if (from_file < 'a' || from_file > 'h' || to_file < 'a' || to_file > 'h') {
107 return false;
108 }
109

112 return false;
113 }
114
115 // Check that from and to are different
116 if (strcmp(from, to) == 0)
117 return false;
118
119 return true;
120}

238const uart_command_t *find_command(const char *command) {
239 if (command == NULL)
240 return NULL;
241
242 // Convert input to uppercase for case-insensitive comparison
243 char cmd_upper[64];
244 strncpy(cmd_upper, command, sizeof(cmd_upper) - 1);
245 cmd_upper[sizeof(cmd_upper) - 1] = '\0';
246
247 for (int i = 0; cmd_upper[i]; i++) {
248 cmd_upper[i] = toupper(cmd_upper[i]);
249 }
250
251 for (int i = 0; uart_commands[i].name != NULL; i++) {
252 if (strcmp(uart_commands[i].name, cmd_upper) == 0) {
253 return &uart_commands[i];
254 }
255
256 // Check aliases (also case-insensitive)
257 for (int j = 0; j < 5 && uart_commands[i].aliases[j][0] != '\0'; j++) {
258 if (strcmp(uart_commands[i].aliases[j], cmd_upper) == 0) {
259 return &uart_commands[i];
260 }
261 }
262 }
263
264 return NULL;
265}

267command_result_t execute_command(const char *command, const char *args) {
268 if (command == NULL) {
269 return CMD_ERROR_INVALID_SYNTAX;
270 }
271
272 // Find command (case-insensitive)
273 const uart_command_t *cmd = find_command(command);
274 if (cmd == NULL) {
275 // Check if it's a direct move (like "e2e4")
276 if (strlen(command) == 4 && is_valid_move_notation(command)) {
277 ESP_LOGI(TAG, "Processing direct move: %s", command);
278
279 // Create move command
280 chess_move_command_t move_cmd = {.type = GAME_CMD_MAKE_MOVE,
281 .player = 0, // White to move (default)
282 .response_queue = 0};
283
284 // Copy move notation (e.g., "e2e4" -> from="e2", to="e4")
285 strncpy(move_cmd.from_notation, command, 2);
286 move_cmd.from_notation[2] = '\0';
287 strncpy(move_cmd.to_notation, command + 2, 2);
288 move_cmd.to_notation[2] = '\0';
289
290 // Send to game task
291 if (send_to_game_task(&move_cmd)) {
292 uart_send_formatted("Move requested: %s → %s", move_cmd.from_notation,
293 move_cmd.to_notation);
294 uart_send_formatted("Move sent to game engine for validation");
295 command_count++;
296 return CMD_SUCCESS;
297 } else {
298 uart_send_error("Internal error: failed to send move to game engine");
299 return CMD_ERROR_SYSTEM_ERROR;
300 }
301 }
302
303 uart_send_error("❌ Unknown command");
304 uart_send_formatted("Command '%s' not found", command);
305 uart_send_line("Type 'HELP' for available commands");
306 return CMD_ERROR_INVALID_SYNTAX;
307 }
308
309 // Check if args are required
310 if (cmd->requires_args && (args == NULL || strlen(args) == 0)) {
311 uart_send_error("❌ Missing arguments");
312 uart_send_formatted("Usage: %s", cmd->usage);
313 return CMD_ERROR_INVALID_SYNTAX;
314 }
315
316 // Execute command
317 ESP_LOGI(TAG, "Executing command: %s with args: %s", cmd->name,
318 args ? args : "none");
319
320 command_result_t result = cmd->handler(args);
321
322 if (result == CMD_SUCCESS) {
323 command_count++;
324 last_command_time = esp_timer_get_time() / 1000;
325 } else {
326 error_count++;
327 ESP_LOGE(TAG, "Command '%s' failed with result: %d", cmd->name, result);
328 }
329
330 return result;
331}

333void uart_parse_command(const char *input) {
334 if (input == NULL || strlen(input) == 0)
335 return;
336
337 // Skip leading whitespace
338 while (*input == ' ' || *input == '\t')
339 input++;
340
341 // Find command and arguments
342 char command[64] = {0};
343 char args[640] = {0};
344
345 const char *space = strchr(input, ' ');
346 if (space != NULL) {
347 // Command has arguments
348 size_t cmd_len = space - input;
349 strncpy(command, input, cmd_len);
350 command[cmd_len] = '\0';
351
352 // Skip whitespace after command
353 while (*space == ' ' || *space == '\t')
354 space++;
355 strncpy(args, space, sizeof(args) - 1);
356 args[sizeof(args) - 1] = '\0';
357 } else {
358 // Command without arguments
359 strncpy(command, input, sizeof(command) - 1);
360 command[sizeof(command) - 1] = '\0';
361 }
362
363 // Execute command
364 execute_command(command, args);
365}
